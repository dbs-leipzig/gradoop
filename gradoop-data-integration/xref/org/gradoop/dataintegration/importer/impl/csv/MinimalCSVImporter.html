161   /**
162    * Import each row of the file as a vertex and create a logical graph from it.
163    * If no column property names are set, read the first line of the file as header and set this
164    * values as column names.
165    *
166    * @return a logical graph with a vertex per csv line and no edges
167    * @throws IOException on failure
168    */
169   @Override
170   public LogicalGraph getLogicalGraph() throws IOException {
171     DataSet<EPGMVertex> vertices;
172 
173     if (columnNames == null) {
174       vertices = readCSVFile(readHeaderRow(), checkReoccurringHeader);
175     } else {
176       vertices = readCSVFile(columnNames, checkReoccurringHeader);
177     }
178 
179     return config.getLogicalGraphFactory().fromDataSets(vertices);
180   }
181 
182   @Override
183   public GraphCollection getGraphCollection() throws IOException {
184     LogicalGraph logicalGraph = getLogicalGraph();
185     return logicalGraph.getCollectionFactory().fromGraph(logicalGraph);
186   }

188   /**
189    * Reads the csv file specified by {@link MinimalCSVImporter#path} and converts each valid line
190    * to a {@link EPGMVertex}.
191    *
192    * @param propertyNames list of the property identifier names
193    * @param checkReoccurringHeader set to true if each row of the file should be checked for
194    *                               reoccurring of the column property names
195    * @return a {@link DataSet} of all vertices from one specific file
196    */
197   private DataSet<EPGMVertex> readCSVFile(List<String> propertyNames, boolean checkReoccurringHeader) {
198     Class<String>[] types = IntStream.range(0, propertyNames.size())
199       .mapToObj(i -> String.class)
200       .<Class<String>>toArray(Class[]::new);
201 
202     return getCSVReader(config.getExecutionEnvironment(), path, types)
203       .flatMap(new CsvRowToProperties<>(propertyNames, checkReoccurringHeader))
204       .filter(p -> !p.isEmpty())
205       .map(new PropertiesToVertex<>(config.getLogicalGraphFactory().getVertexFactory()))
206       .returns(config.getLogicalGraphFactory().getVertexFactory().getType());
207   }

209   /**
210    * Create CSVReader for the provided types.
211    * Flinks CSVReader builder does not support tuples of dynamic length.
212    *
213    * @param env execution environment
214    * @param filename path of csv file
215    * @param fieldTypes array of types
216    * @return csv reader returning a tuple of fieldTypes
217    */
218   private org.apache.flink.api.java.operators.DataSource<Tuple> getCSVReader(ExecutionEnvironment env,
219     String filename, Class<?>[] fieldTypes) {
220     TupleTypeInfo<Tuple> typeInfo = TupleTypeInfo.getBasicAndBasicValueTupleTypeInfo(fieldTypes);
221     TupleCsvInputFormat<Tuple> inputFormat = new TupleCsvInputFormat<>(
222       new org.apache.flink.core.fs.Path(filename), typeInfo);
223     inputFormat.setCharset(this.charset);
224     inputFormat.setFieldDelimiter(this.tokenSeparator);
225     inputFormat.setSkipFirstLineAsHeader(false);
226     if (this.parseQuotedStrings) {
227       inputFormat.enableQuotedStringParsing(this.quoteCharacter);
228     }
229     return new org.apache.flink.api.java.operators.DataSource<>(env, inputFormat, typeInfo,
230       Utils.getCallLocationName());
231   }

233   /**
234    * Reads the fist row of a csv file and creates a list including all column entries that
235    * will be used as property names.
236    *
237    * @return the property names
238    * @throws IOException if an error occurred while open the stream
239    */
240   private List<String> readHeaderRow() throws IOException {
241     Path filePath = new Path(path);
242     try (FileSystem fs = FileSystem.get(filePath.toUri(), new Configuration())) {
243       FSDataInputStream inputStream = fs.open(filePath);
244       BufferedReader lineReader = new BufferedReader(new InputStreamReader(inputStream, charset));
245       String headerLine = lineReader.readLine();
246       lineReader.close();
247       if (headerLine == null || headerLine.isEmpty()) {
248         throw new IOException("The csv file '" + path + "' does not contain any rows.");
249       }
250       return Arrays.asList(headerLine.split(tokenSeparator));
251     } catch (IOException ioe) {
252       throw new IOException("Error while opening a stream to '" + path + "'.", ioe);
253     }
254   }
