38  /**
39   * Read a csv file and import each row as a vertex in EPGM representation.
40   */
41  public class MinimalCSVImporter implements DataSource {
42  
43    /**
44     * Token delimiter of the CSV file.
45     */
46    private String tokenSeparator;
47  
48    /**
49     * Path to the csv file.
50     */
51    private String path;
52  
53    /**
54     * The charset used in the csv file, e.g., "UTF-8".
55     */
56    private String charset;
57  
58    /**
59     * The property names for all columns in the file. If {@code null}, the first line will be
60     * interpreted as header row.
61     */
62    private List<String> columnNames;
63  
64    /**
65     * Flag to specify if each row of the file should be checked for reoccurring of the column
66     * property names.
67     */
68    private boolean checkReoccurringHeader;
69  
70    /**
71     * Gradoop Flink configuration.
72     */
73    private GradoopFlinkConfig config;
74  
75    /**
76     * Create a new MinimalCSVImporter instance by the given parameters.
77     *
78     * @param path the path to the csv file
79     * @param tokenSeparator the token delimiter of the csv file
80     * @param config GradoopFlinkConfig configuration
81     * @param columnNames property identifiers for each column
82     * @param checkReoccurringHeader if each row of the file should be checked for reoccurring of
83     *                               the column property names
84     */
85    public MinimalCSVImporter(String path, String tokenSeparator, GradoopFlinkConfig config,
86                              List<String> columnNames, boolean checkReoccurringHeader) {
87      this(path, tokenSeparator, config, checkReoccurringHeader);
88      this.columnNames = Objects.requireNonNull(columnNames);
89    }
90  
91    /**
92     * Create a new MinimalCSVImporter instance by the given parameters.
93     * Use the UTF-8 charset as default. The first line of the file will be set as the property
94     * names for each column.
95     *
96     * @param path the path to the csv file
97     * @param tokenSeparator the token delimiter of the csv file
98     * @param config GradoopFlinkConfig configuration
99     * @param checkReoccurringHeader if each row of the file should be checked for reoccurring of
100    *                               the column property names.
101    */
102   public MinimalCSVImporter(String path, String tokenSeparator, GradoopFlinkConfig config,
103                             boolean checkReoccurringHeader) {
104     this(path, tokenSeparator, config, "UTF-8", checkReoccurringHeader);
105   }
106 
107   /**
108    * Create a new MinimalCSVImporter with a user set charset. The first line of the file
109    * will set as the property names for each column.
110    *
111    * @param path the path to the csv file
112    * @param tokenSeparator the token delimiter of the csv file
113    * @param config GradoopFlinkConfig configuration
114    * @param charset the charset used in the csv file
115    * @param checkReoccurringHeader if each row of the file should be checked for reoccurring of
116    *                               the column property names.
117    */
118   public MinimalCSVImporter(String path, String tokenSeparator, GradoopFlinkConfig config,
119                             String charset, boolean checkReoccurringHeader) {
120     this.path = Objects.requireNonNull(path);
121     this.tokenSeparator = Objects.requireNonNull(tokenSeparator);
122     this.config = Objects.requireNonNull(config);
123     this.charset = Objects.requireNonNull(charset);
124     this.checkReoccurringHeader = checkReoccurringHeader;
125   }
126 
127   /**
128    * Import each row of the file as a vertex and create a logical graph from it.
129    * If no column property names are set, read the first line of the file as header and set this
130    * values as column names.
131    *
132    * @return a logical graph with a vertex per csv line and no edges
133    * @throws IOException on failure
134    */
135   @Override
136   public LogicalGraph getLogicalGraph() throws IOException {
137     DataSet<EPGMVertex> vertices;
138 
139     if (columnNames == null) {
140       vertices = readCSVFile(readHeaderRow(), checkReoccurringHeader);
141     } else {
142       vertices = readCSVFile(columnNames, checkReoccurringHeader);
143     }
144 
145     return config.getLogicalGraphFactory().fromDataSets(vertices);
146   }
147 
148   @Override
149   public GraphCollection getGraphCollection() throws IOException {
150     return config.getGraphCollectionFactory().fromGraph(getLogicalGraph());
151   }
152 
153   /**
154    * Reads the csv file specified by {@link MinimalCSVImporter#path} and converts each valid line
155    * to a {@link EPGMVertex}.
156    *
157    * @param propertyNames list of the property identifier names
158    * @param checkReoccurringHeader set to true if each row of the file should be checked for
159    *                               reoccurring of the column property names
160    * @return a {@link DataSet} of all vertices from one specific file
161    */
162   private DataSet<EPGMVertex> readCSVFile(List<String> propertyNames, boolean checkReoccurringHeader) {
163     return config.getExecutionEnvironment()
164       .readTextFile(path)
165       .flatMap(new CsvRowToProperties(tokenSeparator, propertyNames, checkReoccurringHeader))
166       .map(new PropertiesToVertex<>(config.getLogicalGraphFactory().getVertexFactory()))
167       .returns(config.getLogicalGraphFactory().getVertexFactory().getType());
168   }
169 
170   /**
171    * Reads the fist row of a csv file and creates a list including all column entries that
172    * will be used as property names.
173    *
174    * @return the property names
175    * @throws IOException if an error occurred while open the stream
176    */
177   private List<String> readHeaderRow() throws IOException {
178     Path filePath = new Path(path);
179     try (FileSystem fs = FileSystem.get(filePath.toUri(), new Configuration())) {
180       FSDataInputStream inputStream = fs.open(filePath);
181       BufferedReader lineReader = new BufferedReader(new InputStreamReader(inputStream, charset));
182       String headerLine = lineReader.readLine();
183       lineReader.close();
184       if (headerLine == null || headerLine.isEmpty()) {
185         throw new IOException("The csv file '" + path + "' does not contain any rows.");
186       }
187       return Arrays.asList(headerLine.split(tokenSeparator));
188     } catch (IOException ioe) {
189       throw new IOException("Error while opening a stream to '" + path + "'.", ioe);
190     }
191   }
