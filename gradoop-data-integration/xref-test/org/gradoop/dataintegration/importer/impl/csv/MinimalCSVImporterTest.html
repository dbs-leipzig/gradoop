303   /**
304    * Checks if the csv file import is equals to the assert import.
305    *
306    * @param vertices list of the import vertices
307    */
308   private void checkImportedVertices(List<EPGMVertex> vertices) {
309     assertEquals(3, vertices.size());
310     // Check that IDs are unique.
311     assertEquals(3, vertices.stream().map(Vertex::getId).distinct().count());
312 
313     for (EPGMVertex v : vertices) {
314       switch (String.valueOf(v.getPropertyValue("name"))) {
315       case "foo":
316         assertNotNull(v.getProperties());
317         assertEquals(4, v.getProperties().size());
318         assertEquals("453", v.getPropertyValue("value1").getString());
319         assertEquals("true", v.getPropertyValue("value2").getString());
320         assertEquals("71.03", v.getPropertyValue("value3").getString());
321         break;
322       case "bar":
323         assertNotNull(v.getProperties());
324         assertEquals(4, v.getProperties().size());
325         assertEquals("76", v.getPropertyValue("value1").getString());
326         assertEquals("false", v.getPropertyValue("value2").getString());
327         assertEquals("33.4", v.getPropertyValue("value3").getString());
328         break;
329       case "bla":
330         assertNotNull(v.getProperties());
331         assertEquals(3, v.getProperties().size());
332         assertEquals("4568", v.getPropertyValue("value1").toString());
333         assertFalse(v.hasProperty("value2"));
334         assertEquals("9.42", v.getPropertyValue("value3").toString());
335         break;
336       default:
337         fail();
338         break;
339       }
340     }
341   }

343   /**
344    * Check if the correct exceptions are thrown if the file of the importer is empty.
345    *
346    * @param importer the importer to check
347    * @param emptyFile the file given to the importer
348    */
349   private void checkExceptions(DataSource importer, File emptyFile) {
350     String exceptionMessage = "Error while opening a stream to '" + emptyFile.getPath() + "'.";
351     String causeMessage = "The csv file '" + emptyFile.getPath() + "' does not contain any rows.";
352     try {
353       importer.getLogicalGraph();
354       fail("No expected IOException was thrown.");
355     } catch (IOException exception) {
356       assertEquals(exceptionMessage, exception.getMessage());
357       assertEquals(causeMessage, exception.getCause().getMessage());
358     }
