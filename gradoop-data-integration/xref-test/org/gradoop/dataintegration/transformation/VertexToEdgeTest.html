16  package org.gradoop.dataintegration.transformation;

18  import org.gradoop.flink.model.GradoopFlinkTestBase;
19  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
20  import org.gradoop.flink.util.FlinkAsciiGraphLoader;
21  import org.junit.Test;
22  
23  /**
24   * This class contains tests for the {@link VertexToEdge} transformation operator.
25   */
26  public class VertexToEdgeTest extends GradoopFlinkTestBase {
27  
28    /**
29     * Test the {@link VertexToEdge} transformation where one one edge is added.
30     *
31     * @throws Exception when the execution in Flink fails.
32     */
33    @Test
34    public void testWithEdgeCreation() throws Exception {
35      FlinkAsciiGraphLoader loader = getLoaderFromString("input:test[" +
36        "(v0:Blue {a : 3})" +
37        "(v1:Green {a : 2})" +
38        "(v2:Blue {a : 4})" +
39        "(v0)-[{b : 2}]->(v1)" +
40        "(v1)-[{b : 4}]->(v2)" +
41        "]" +
42        "expected:test[" +
43        "(v00:Blue {a : 3})" +
44        "(v01:Green {a : 2})" +
45        "(v02:Blue {a : 4})" +
46        "(v00)-[{b : 2}]->(v01)" +
47        "(v01)-[{b : 4}]->(v02)" +
48        "(v00)-[:foo {a : 2, originalVertexLabel: \"Green\"," +
49        "firstEdgeLabel: \"\", secondEdgeLabel: \"\"}]->(v02)" +
50        "]");
51      LogicalGraph input = loader.getLogicalGraphByVariable("input");
52      LogicalGraph expected = loader.getLogicalGraphByVariable("expected");
53  
54      VertexToEdge transformation = new VertexToEdge("Green", "foo");
55      LogicalGraph transformed = input.callForGraph(transformation);
56  
57      collectAndAssertTrue(transformed.equalsByData(expected));
58    }
59  }
