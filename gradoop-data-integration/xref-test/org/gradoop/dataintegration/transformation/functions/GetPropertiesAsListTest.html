31  import static org.junit.Assert.assertEquals;

33  /**
34   * Test for the {@link GetPropertiesAsList} function.
35   */
36  public class GetPropertiesAsListTest extends GradoopFlinkTestBase {
37  
38    /**
39     * A list of test property keys.
40     */
41    private final List<String> testKeys = Arrays.asList(KEY_a, KEY_b);
42  
43    /**
44     * A first test value.
45     */
46    private final PropertyValue valueA = PropertyValue.create(1L);
47  
48    /**
49     * A second test value.
50     */
51    private final PropertyValue valueB = PropertyValue.create("b");
52  
53    /**
54     * Test the function with all properties set.
55     *
56     * @throws IOException When accessing the property value list fails.
57     */
58    @Test
59    public void testWithAllSet() throws IOException {
60      Vertex testVertex = getConfig().getLogicalGraphFactory().getVertexFactory().createVertex();
61      testVertex.setProperty(KEY_a, valueA);
62      testVertex.setProperty(KEY_b, valueB);
63      List<PropertyValue> result = toList(new GetPropertiesAsList<>(testKeys).getKey(testVertex));
64      assertEquals(2, result.size());
65      assertEquals(valueA, result.get(0));
66      assertEquals(valueB, result.get(1));
67    }
68  
69    /**
70     * Test the function with some properties set.
71     *
72     * @throws IOException When accessing the property value list fails.
73     */
74    @Test
75    public void testWithSomeSet() throws IOException {
76      Vertex testVertex = getConfig().getLogicalGraphFactory().getVertexFactory().createVertex();
77      testVertex.setProperty(KEY_a, valueA);
78      List<PropertyValue> result = toList(new GetPropertiesAsList<>(testKeys).getKey(testVertex));
79      assertEquals(2, result.size());
80      assertEquals(valueA, result.get(0));
81      assertEquals(PropertyValue.NULL_VALUE, result.get(1));
82    }
83  
84    /**
85     * Test the function with no properties set.
86     *
87     * @throws IOException When accessing the property value list fails.
88     */
89    @Test
90    public void testWithAllUnset() throws IOException {
91      Vertex testVertex = getConfig().getLogicalGraphFactory().getVertexFactory().createVertex();
92      List<PropertyValue> result = toList(new GetPropertiesAsList<>(testKeys).getKey(testVertex));
93      assertEquals(2, result.size());
94      assertEquals(PropertyValue.NULL_VALUE, result.get(0));
95      assertEquals(PropertyValue.NULL_VALUE, result.get(1));
96    }
97  
98    /**
99     * Helper functions converting a {@link PropertyValueList} to a list of {@link PropertyValue}s.
100    *
101    * @param list The list object.
102    * @return The list of values.
103    */
104   private List<PropertyValue> toList(PropertyValueList list) {
105     List<PropertyValue> values = new ArrayList<>();
106     list.forEach(values::add);
107     return values;
108   }
