40  @RunWith(Parameterized.class)

43    /**
44     * Loader to get the test graphs from.

46    private final FlinkAsciiGraphLoader loader = getLoaderFromString("input[" +
47      "(i1:I1)-->(center:Center)-->(o1:O1)" +
48      "(center2:Center2)" +
49      "]" +

53      "incoming2[] outgoing2[] undirected2[]");

55    /**
56     * The variable name of the center vertex.
57     */
58    @Parameterized.Parameter(0)
59    public String centerVertex;

61    /**
62     * The variable name of the neighborhood graph.

64    @Parameterized.Parameter(1)
65    public String neighborhood;
66  
67    /**
68     * The direction of the neighborhood.
69     */
70    @Parameterized.Parameter(2)
71    public Neighborhood.EdgeDirection direction;

73    /**
74     * Parameters for this test.

77     */
78    @Parameterized.Parameters(name = "neighborhood of {0} ({2})")
79    public static Iterable<Object[]> parameters() {
80      return Arrays.asList(
81        new Object[]{"center", "incoming1", Neighborhood.EdgeDirection.INCOMING},
82        new Object[]{"center", "outgoing1", Neighborhood.EdgeDirection.OUTGOING},
83        new Object[]{"center", "undirected1", Neighborhood.EdgeDirection.UNDIRECTED},
84        new Object[]{"center2", "incoming2", Neighborhood.EdgeDirection.INCOMING},
85        new Object[]{"center2", "outgoing2", Neighborhood.EdgeDirection.OUTGOING},
86        new Object[]{"center2", "undirected2", Neighborhood.EdgeDirection.UNDIRECTED},
87        new Object[]{"nocenter", "incoming2", Neighborhood.EdgeDirection.INCOMING},
88        new Object[]{"nocenter", "outgoing2", Neighborhood.EdgeDirection.OUTGOING},
89        new Object[]{"nocenter", "undirected2", Neighborhood.EdgeDirection.UNDIRECTED});
90    }
91  
92    /**
93     * Test {@link Neighborhood} with different parameters.
94     *
95     * @throws Exception when the execution in Flink fails.
96     */
97    @Test
98    public void testNeighborhood() throws Exception {
99      LogicalGraph input = loader.getLogicalGraphByVariable("input");
100     EPGMVertex center = loader.getVertexByVariable(centerVertex);
101     Collection<EPGMVertex> neighborhoodVertices = loader.getVerticesByGraphVariables(neighborhood);
102     DataSet<EPGMVertex> centers = center == null ? getEmptyDataSet(new EPGMVertex()) :
103       getExecutionEnvironment().fromElements(center);
104     List<Tuple2<EPGMVertex, List<NeighborhoodVertex>>> neighborhoods =
105       Neighborhood.getPerVertex(input, centers, direction).collect();
106     // Make sure we only have 1 center vertex.
107     long centerCount = neighborhoods.stream().map(h -> h.f0.getId()).distinct().count();
108     assertEquals(center != null ? 1 : 0, centerCount);
109 
110     // Get all neighbor ids of that vertex.
111     Object[] neighbors = neighborhoods.stream().flatMap(h -> h.f1.stream())
112       .map(NeighborhoodVertex::getNeighborId)
113       .sorted()
114       .toArray();
115     Object[] neighborIdsExpected = neighborhoodVertices.stream().map(EPGMVertex::getId).sorted()
116         .toArray();
117     assertArrayEquals(neighborIdsExpected, neighbors);
118   }
