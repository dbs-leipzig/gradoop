51     * @throws Exception If the data could not be loaded or collected properly.

53    @Test
54    public void cityDeduplicationTest() throws Exception {
55      LogicalGraph social = getSocialNetworkLoader().getLogicalGraph();
56  
57      // this operation should create 3 vertices with the property values: Berlin, Dresden, Leipzig
58      // keep in mind that the default for condensation is 'true'
59      UnaryGraphToGraphOperator extOne = new ExtractPropertyFromVertex("Person", "city",
60        "City", "name");
61  
62      LogicalGraph extOneGraph = social.callForGraph(extOne);
63      List<EPGMVertex> city = extOneGraph.getVerticesByLabel("City").collect();
64      Set<String> cityNames = new HashSet<>();
65      for (EPGMVertex v : city) {
66        String cityName = v.getPropertyValue("name").getString();
67        Assert.assertTrue(cityName.equals("Dresden") ||
68          cityName.equals("Berlin") ||
69          cityName.equals("Leipzig"));
70        Assert.assertTrue(cityNames.add(cityName));
71      }
72      Assert.assertEquals(3, cityNames.size());
73  
74      // The number of edges should be the same as before.
75      Assert.assertEquals(social.getEdges().count(), extOneGraph.getEdges().count());
76    }

111   /**
112    * A private convenience method for easier testing of different setups in the edge creation process.
113    *
114    * @param graph     The input graph for the tests.
115    * @param direction The edge direction the graph is tested for.
116    * @throws Exception Is thrown if the process cant be executed properly.
117    */
118   private void testForEdgeDirection(LogicalGraph graph, EdgeDirection direction) throws Exception {
119     Set<String> cities = new HashSet<>(Arrays.asList("Dresden", "Berlin", "Leipzig"));
120     Set<String> persons = new HashSet<>(Arrays.asList("Eve", "Alice", "Frank", "Dave", "Bob", "Carol"));
121 
122     UnaryGraphToGraphOperator extract = new ExtractPropertyFromVertex("Person", "city",
123       "City", "name", direction, "newLabel");
124     LogicalGraph extractedGraph = graph.callForGraph(extract);
125 
126     long expectedEdgeCount = direction.equals(EdgeDirection.BIDIRECTIONAL) ? 12 : 6;
127     Assert.assertEquals(expectedEdgeCount, extractedGraph.getEdgesByLabel("newLabel").count());
128 
129     List<EPGMVertex> vertices = new ArrayList<>();
130     extractedGraph.getVertices()
131       .filter(new LabelIsIn<>("Person", "City"))
132       .output(new LocalCollectionOutputFormat<>(vertices));
133 
134     List<EPGMEdge> newEdges = new ArrayList<>();
135     extractedGraph
136       .getEdgesByLabel("newLabel")
137       .output(new LocalCollectionOutputFormat<>(newEdges));
138 
139     getConfig().getExecutionEnvironment().execute();
140 
141     Map<GradoopId, String> idMap = new HashMap<>();
142     vertices.forEach(v -> idMap.put(v.getId(), v.getPropertyValue("name").getString()));
143 
144     for (EPGMEdge e : newEdges) {
145       String sourceName = idMap.get(e.getSourceId());
146       String targetName = idMap.get(e.getTargetId());
147 
148       if (direction.equals(EdgeDirection.ORIGIN_TO_NEWVERTEX)) {
149         Assert.assertTrue("source: " + sourceName + " | target: " + targetName +
150             " | edge direction: " + direction.name(),
151           persons.contains(sourceName) && cities.contains(targetName));
152       } else if (direction.equals(EdgeDirection.NEWVERTEX_TO_ORIGIN)) {
153         Assert.assertTrue("source: " + sourceName + " | target: " + targetName +
154             " | edge direction: " + direction.name(),
155           cities.contains(sourceName) && persons.contains(targetName));
156       } else if (direction.equals(EdgeDirection.BIDIRECTIONAL)) {
157         boolean cityContainment = cities.contains(sourceName) || cities.contains(targetName);
158         boolean personContainment = persons.contains(sourceName) || persons.contains(targetName);
159 
160         Assert.assertTrue("vertex name 1: " + sourceName + " | vertex name 2: " +
161             targetName + " | edge direction: " + direction.name(),
162           cityContainment && personContainment);
163       }
164     }
165   }

169    *
170    * @throws Exception If the data could not be loaded or collected properly.

172   @Test
173   public void nonDeduplicationTest() throws Exception {
174     LogicalGraph social = getSocialNetworkLoader().getLogicalGraph();
175 
176     // this operation should create 6 vertices with the property values:
177     // 'Berlin', 3 times 'Dresden', 2 times 'Leipzig'
178     ExtractPropertyFromVertex extract = new ExtractPropertyFromVertex("Person", "city",
179       "City", "name");
180     extract.setCondensation(false);
181 
182     LogicalGraph extractedGraph = social.callForGraph(extract);
183 
184     List<EPGMVertex> vertices = new ArrayList<>();
185     extractedGraph
186       .getVerticesByLabel("City")
187       .output(new LocalCollectionOutputFormat<>(vertices));
188 
189     getConfig().getExecutionEnvironment().execute();
190 
191     Map<String, Integer> cityCountMap = new HashMap<>();
192     vertices.forEach(v -> cityCountMap.merge(v.getPropertyValue("name").getString(), 1,
193       Integer::sum));
194 
195     Assert.assertEquals(1, cityCountMap.get("Berlin").intValue());
196     Assert.assertEquals(3, cityCountMap.get("Dresden").intValue());
197     Assert.assertEquals(2, cityCountMap.get("Leipzig").intValue());
198   }

236    *
237    * @throws Exception If collect doesn't work as expected.
238    */
239   @Test
240   public void containGraphHeadIdTest() throws Exception {
241     LogicalGraph socialGraph = getSocialNetworkLoader().getLogicalGraph();
242 
243     // this operation should create 6 vertices with the property values:
244     // 'Berlin', 3 times 'Dresden', 2 times 'Leipzig'
245     ExtractPropertyFromVertex extract = new ExtractPropertyFromVertex("Person", "city",
246       "City", "name", EdgeDirection.ORIGIN_TO_NEWVERTEX, "livesIn");
247 
248     LogicalGraph extractedGraph = socialGraph.callForGraph(extract);
249 
250     List<EPGMVertex> createdVertices = new ArrayList<>();
251     extractedGraph
252       .getVerticesByLabel("City")
253       .output(new LocalCollectionOutputFormat<>(createdVertices));
254 
255     List<EPGMEdge> createdEdges = new ArrayList<>();
256     extractedGraph
257       .getEdgesByLabel("livesIn")
258       .output(new LocalCollectionOutputFormat<>(createdEdges));
259 
260     // get graph heads
261     List<EPGMGraphHead> graphHeads = new ArrayList<>();
262     socialGraph.getGraphHead().output(new LocalCollectionOutputFormat<>(graphHeads));
263 
264     getConfig().getExecutionEnvironment().execute();
265 
266     // check if the id of the original graph is assigned to the newly created elements
267     for (EPGMGraphHead graphHead: graphHeads) {
268       for (EPGMVertex vertex: createdVertices) {
269         Assert.assertTrue(vertex.getGraphIds().contains(graphHead.getId()));
270       }
271       for (EPGMEdge edge: createdEdges) {
272         Assert.assertTrue(edge.getGraphIds().contains(graphHead.getId()));
273       }
274     }
