16  package org.gradoop.common.model.api.strategies;
17  
18  import org.apache.flink.core.memory.DataInputView;
19  import org.apache.flink.core.memory.DataOutputView;
20  import org.gradoop.common.model.impl.properties.PropertyValue;
21  
22  import java.io.IOException;
23  
24  /**
25   * Interface defining the methods necessary to handle the (de-) serialization of a
26   * {@link PropertyValue}'s value.
27   *
28   * @param <T> The property value type.
29   */
30  public interface PropertyValueStrategy<T> {
31  
32    /**
33     * Writes the given value to the provided {@link DataOutputView}.
34     * The argument {@code value} can not be {@code null}.
35     *
36     * @param value      to be written to the {@link DataOutputView}.
37     * @param outputView that the value is written to.
38     * @throws IOException if write process fails.
39     */
40    void write(T value, DataOutputView outputView) throws IOException;
41  
42    /**
43     * Reads raw bytes from the given {@link DataInputView} and deserializes the contained object.
44     *
45     * @param inputView containing serialized object.
46     * @param typeByte  byte needed to indicate whether serialized object has a variable length.
47     * @return deserialized object.
48     * @throws IOException when reading or deserialization of the object fails.
49     */
50    T read(DataInputView inputView, byte typeByte) throws IOException;
51  
52    /**
53     * Compares two objects.
54     *
55     * @param value first object.
56     * @param other second object.
57     * @return a negative integer, zero, or a positive integer as first object is less than, equal to,
58     * or greater than the second object.
59     * @throws IllegalArgumentException when {@code other} is not comparable to {@code value}.
60     */
61    int compare(T value, Object other);
62  
63    /**
64     * Checks if given object is an instance of the data type the specific strategy handles.
65     *
66     * @param value to be checked.
67     * @return true if {@code value} is an instance of the data type this strategy handles.
68     * False otherwise.
69     */
70    boolean is(Object value);
71  
72    /**
73     * Gets the class of the data type the specific strategy handles.
74     *
75     * @return class of the type handled by this strategy.
76     */
77    Class<T> getType();
78  
79    /**
80     * Deserializes an object from the provided byte array.
81     *
82     * @param bytes representing a serialized object.
83     * @return an instance of the type handled by this strategy.
84     * @throws IOException on failure
85     */
86    T get(byte[] bytes) throws IOException;
87  
88    /**
89     * Gets a byte which represents the data type the specific strategy handles.
90     *
91     * @return a byte.
92     */
93    byte getRawType();
94  
95    /**
96     * Serializes the given object.
97     * The argument {@code value} can not be {@code null}.
98     *
99     * @param value the object to be serialized.
100    * @return byte array representation of the provided object.
101    * @throws IOException on failure
102    */
103   byte[] getRawBytes(T value) throws IOException;
104 }
