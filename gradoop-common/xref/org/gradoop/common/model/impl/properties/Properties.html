32  /**
33   * Represents the properties of an {@link EPGMElement}.
34   */
35  public class Properties implements Iterable<Property>, Value, Serializable {
36  
37    /**
38     * Default capacity for new property lists.
39     */
40    private static final int DEFAULT_CAPACITY = 10;
41  
42    /**
43     * Internal representation
44     */
45    private Map<String, PropertyValue> properties;
46  
47    /**
48     * Default constructor
49     */
50    public Properties() {
51      properties = new HashMap<>(DEFAULT_CAPACITY);
52    }
53  
54    /**
55     * Creates property list with given capacity.
56     *
57     * @param capacity initial capacity
58     */
59    private Properties(int capacity) {
60      properties = new HashMap<>(capacity);
61    }

63    /**
64     * Creates a new property list.
65     *
66     * @return PropertyList
67     */
68    public static Properties create() {
69      return new Properties();
70    }

72    /**
73     * Creates a new property list with the given initial capacity.
74     *
75     * @param capacity initial capacity
76     * @return PropertyList
77     */
78    public static Properties createWithCapacity(int capacity) {
79      return new Properties(capacity);
80    }

82    /**
83     * Creates a new property collection from a given map.
84     *
85     * If map is {@code null} an empty properties instance will be returned.
86     *
87     * @param map key value map
88     * @return PropertyList
89     */
90    public static Properties createFromMap(Map<String, Object> map) {
91      Properties properties;
92  
93      if (map == null) {
94        properties = Properties.createWithCapacity(0);
95      } else {
96        properties = Properties.createWithCapacity(map.size());
97  
98        for (Map.Entry<String, Object> entry : map.entrySet()) {
99          properties.set(entry.getKey(), PropertyValue.create(entry.getValue()));
100       }
101     }
102 
103     return properties;
104   }

115   /**
116    * Checks if a property with the given key is contained in the properties.

118    * @param key property key
119    * @return true, if there is a property with the given key
120    */
121   public boolean containsKey(String key) {
122     return get(key) != null;

125   /**
126    * Returns the value to the given key of {@code null} if the value does not

128    *
129    * @param key property key
130    * @return property value or {@code null} if key does not exist

132   public PropertyValue get(String key) {
133     Objects.requireNonNull(key);
134     return properties.get(key);
135   }

147   /**
148    * Sets the given property. If a property with the same property key already
149    * exists, it will be replaced by the given property.
150    *
151    * @param key   property key
152    * @param value property value
153    */
154   public void set(String key, PropertyValue value) {
155     Objects.requireNonNull(key);
156     Objects.requireNonNull(value);
157     properties.put(key, value);
158   }
159 
160   /**
161    * Sets the given property. If a property with the same property key already
162    * exists, it will be replaced by the given property.
163    *
164    * @param key   property key
165    * @param value property value
166    */
167   public void set(String key, Object value) {
168     PropertyValue propertyValue;
169 
170     if (value instanceof PropertyValue) {
171       propertyValue = (PropertyValue) value;
172     } else {
173       propertyValue = PropertyValue.create(value);
174     }
175 
176     set(key, propertyValue);
177   }

179   /**
180    * Removes the property of the given key from the list.
181    *
182    * @param key property key

184    *         <tt>null</tt> if there was no mapping for <tt>key</tt>.
185    */
186   public PropertyValue remove(String key) {
187     Objects.requireNonNull(key);
188     return properties.remove(key);
189   }
190 
191   /**
192    * Removes the given property from the list.
193    *
194    * @param property property
195    * @return the previous value associated with <tt>key</tt>, or
196    *         <tt>null</tt> if there was no mapping for <tt>key</tt>.
197    */
198   public PropertyValue remove(Property property) {
199     Objects.requireNonNull(property);
200     return remove(property.getKey());
201   }

228   /**
229    * Two properties collections are considered equal, if they contain the
230    * same properties in the same order.
231    *
232    * @param o other properties collection
233    * @return  true, iff the two property collections contain the same elements
234    *          in the same order
235    */
236   @Override
237   public boolean equals(Object o) {
238     if (this == o) {
239       return true;
240     }
241     if (o == null || getClass() != o.getClass()) {
242       return false;
243     }
244 
245     Properties that = (Properties) o;
246 
247     return Objects.equals(properties, that.properties);
248   }
249 
250   /**
251    * Two properties collections have identical hash codes, if they contain the
252    * same properties in the same order.
253    *
254    * @return hash code
255    */
256   @Override
257   public int hashCode() {
258     return properties != null ? properties.hashCode() : 0;
259   }

261   @Override
262   public Iterator<Property> iterator() {
263     return toList().iterator();
264   }

266   /**
267    * Returns a list of properties.
268    *
269    * @return List of properties
270    */
271   public List<Property> toList() {
272     return  properties.entrySet().stream()
273             .map(e -> Property.create(e.getKey(), e.getValue()))
274             .collect(Collectors.toList());
275   }
276 
277   @Override
278   public void write(DataOutputView outputView) throws IOException {
279     outputView.writeInt(properties.size());
280 
281     for (Map.Entry<String, PropertyValue> entry : properties.entrySet()) {
282       outputView.writeUTF(entry.getKey());
283       entry.getValue().write(outputView);
284     }
285   }
286 
287   @Override
288   public void read(DataInputView inputView) throws IOException {
289     int propertyCount = inputView.readInt();
290     this.properties = new HashMap<>(propertyCount);
291 
292     String key;
293     PropertyValue value;
294 
295     for (int i = 0; i < propertyCount; i++) {
296       key = inputView.readUTF();
297       value = new PropertyValue();
298       value.read(inputView);
299       properties.put(key, value);
300     }
301   }
302 
303   @Override
304   public String toString() {
305     return toList().stream()
306       .map(Property::toString)
307       .collect(Collectors.joining(","));
308   }
