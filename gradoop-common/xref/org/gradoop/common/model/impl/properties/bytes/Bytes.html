247     /**
248      * Constructor
249      */
250     ByteArrayComparator() {
251       super();
252     }

254     @Override
255     public int compare(byte [] left, byte [] right) {
256       return compareTo(left, right);
257     }

259     /**
260      * Compares two byte arrays with a specified offset and length.
261      *
262      * @param b1 left byte array
263      * @param s1 offset of left byte array
264      * @param l1 length of left byte array
265      * @param b2 right byte array
266      * @param s2 offset of right byte array
267      * @param l2 length of right byte array
268      * @return 0 if equal, &lt; 0 if left is less than right, etc.
269      */
270     public int compare(byte [] b1, int s1, int l1, byte [] b2, int s2, int l2) {
271       return LexicographicalComparerHolder.BEST_COMPARER.
272         compareTo(b1, s1, l1, b2, s2, l2);
273     }

402   /**
403    * Converts a byte array to a long value.
404    *
405    * @param bytes array of bytes
406    * @param offset offset into array
407    * @param length length of data (must be {@link #SIZEOF_LONG})
408    * @return the long value
409    * @throws IllegalArgumentException if length is not {@link #SIZEOF_LONG} or
410    * if there's not enough room in the array at the offset indicated.
411    */
412   public static long toLong(byte[] bytes, int offset, final int length) {
413     if (length != SIZEOF_LONG || offset + length > bytes.length) {
414       throw explainWrongLengthOrOffset(bytes, offset, length, SIZEOF_LONG);
415     }
416     if (UNSAFE_UNALIGNED) {
417       return toLongUnsafe(bytes, offset);
418     } else {
419       long l = 0;
420       for (int i = offset; i < offset + length; i++) {
421         l <<= 8;
422         l ^= bytes[i] & 0xFF;
423       }
424       return l;
425     }
426   }

428   /**
429    * Return Exception with fancy error message.
430    *
431    * @param bytes used byte array
432    * @param offset used offset
433    * @param length used length
434    * @param expectedLength expected length
435    * @return IllegalArgumentException with fancy error message
436    */
437   private static IllegalArgumentException explainWrongLengthOrOffset(final byte[] bytes,
438                                                                      final int offset,
439                                                                      final int length,
440                                                                      final int expectedLength) {
441     String reason;
442     if (length != expectedLength) {
443       reason = "Wrong length: " + length + ", expected " + expectedLength;
444     } else {
445       reason = "offset (" + offset + ") + length (" + length + ") exceed the capacity of the array: " +
446         bytes.length;
447     }
448     return new IllegalArgumentException(reason);
449   }

451   /**
452    * Put a long value out to the specified byte array position.
453    *
454    * @param bytes the byte array
455    * @param offset position in the array
456    * @param val long to write out
457    * @return incremented offset
458    * @throws IllegalArgumentException if the byte array given doesn't have
459    * enough room at the offset specified.
460    */
461   public static int putLong(byte[] bytes, int offset, long val) {
462     if (bytes.length - offset < SIZEOF_LONG) {
463       throw new IllegalArgumentException("Not enough room to put a long at" + " offset " + offset +
464         " in a " + bytes.length + " byte array");
465     }
466     if (UNSAFE_UNALIGNED) {
467       return putLongUnsafe(bytes, offset, val);
468     } else {
469       for (int i = offset + 7; i > offset; i--) {
470         bytes[i] = (byte) val;
471         val >>>= 8;
472       }
473       bytes[offset] = (byte) val;
474       return offset + SIZEOF_LONG;
475     }
476   }

478   /**
479    * Put a long value out to the specified byte array position (Unsafe).
480    *
481    * @param bytes the byte array
482    * @param offset position in the array
483    * @param val long to write out
484    * @return incremented offset
485    */
486   private static int putLongUnsafe(byte[] bytes, int offset, long val) {
487     if (LexicographicalComparerHolder.UnsafeComparer.LITTLE_ENDIAN) {
488       val = Long.reverseBytes(val);
489     }
490     LexicographicalComparerHolder.UnsafeComparer.UNSAFE.putLong(bytes, (long) offset +
491       LexicographicalComparerHolder.UnsafeComparer.BYTE_ARRAY_BASE_OFFSET, val);
492     return offset + SIZEOF_LONG;
493   }

604       return n;

724   /**
725    * Converts a byte array to a short value
726    *
727    * @param bytes byte array
728    * @param offset offset into array
729    * @param length length, has to be {@link #SIZEOF_SHORT}
730    * @return the short value
731    * @throws IllegalArgumentException if length is not {@link #SIZEOF_SHORT}
732    * or if there's not enough room in the array at the offset indicated.
733    */
734   public static short toShort(byte[] bytes, int offset, final int length) {
735     if (length != SIZEOF_SHORT || offset + length > bytes.length) {
736       throw explainWrongLengthOrOffset(bytes, offset, length, SIZEOF_SHORT);
737     }
738     if (UNSAFE_UNALIGNED) {
739       return toShortUnsafe(bytes, offset);
740     } else {
741       short n = 0;
742       n = (short) ((n ^ bytes[offset]) & 0xFF);
743       n = (short) (n << 8);
744       n = (short) ((n ^ bytes[offset + 1]) & 0xFF);
745       return n;
746     }
747   }

933      * Java comparer (without unsafe)
934      */
935     enum PureJavaComparer implements Comparer<byte[]> {
936       /**
937        * Instance
938        */
939       INSTANCE;
940 
941       @Override
942       public int compareTo(byte[] buffer1, int offset1, int length1,
943                            byte[] buffer2, int offset2, int length2) {
944         // Short circuit equal case
945         if (buffer1 == buffer2 &&
946           offset1 == offset2 &&
947           length1 == length2) {
948           return 0;
949         }
950         // Bring WritableComparator code local
951         int end1 = offset1 + length1;
952         int end2 = offset2 + length2;
953         for (int i = offset1, j = offset2; i < end1 && j < end2; i++, j++) {
954           int a = buffer1[i] & 0xff;
955           int b = buffer2[j] & 0xff;
956           if (a != b) {
957             return a - b;
958           }
959         }
960         return length1 - length2;
961       }
962     }

964     /**
965      * Unsafe comparer
966      */
967     enum UnsafeComparer implements Comparer<byte[]> {
968       /**
969        * Instance
970        */
971       INSTANCE;
972 
973       /**
974        * Unsafe
975        */
976       static final Unsafe UNSAFE;
977 
978       /** The offset to the first element in a byte array. */
979       static final int BYTE_ARRAY_BASE_OFFSET;
980 
981       static {
982         if (UNSAFE_UNALIGNED) {
983           UNSAFE = (Unsafe) AccessController.doPrivileged((PrivilegedAction<Object>) () -> {
984             try {
985               Field f = Unsafe.class.getDeclaredField("theUnsafe");
986               f.setAccessible(true);
987               return f.get(null);
988             } catch (ReflectiveOperationException e) {
989               LOG.warn("sun.misc.Unsafe is not accessible");
990               return null;
991             }
992           });
993         } else {
994           // It doesn't matter what we throw;
995           // it's swallowed in getBestComparer().
996           throw new Error();
997         }
998 
999         BYTE_ARRAY_BASE_OFFSET = UNSAFE.arrayBaseOffset(byte[].class);
1000 
1001         // sanity check - this should never fail
1002         if (UNSAFE.arrayIndexScale(byte[].class) != 1) {
1003           throw new AssertionError();
1004         }
1005       }
1006 
1007       /**
1008        * true, if byte order is little endian
1009        */
1010       static final boolean LITTLE_ENDIAN = ByteOrder.nativeOrder().equals(ByteOrder.LITTLE_ENDIAN);
1011 
1012       /**
1013        * Lexicographically compare two arrays.
1014        *
1015        * @param buffer1 left operand
1016        * @param buffer2 right operand
1017        * @param offset1 Where to start comparing in the left buffer
1018        * @param offset2 Where to start comparing in the right buffer
1019        * @param length1 How much to compare from the left buffer
1020        * @param length2 How much to compare from the right buffer
1021        * @return 0 if equal, < 0 if left is less than right, etc.
1022        */
1023       @Override
1024       public int compareTo(byte[] buffer1, int offset1, int length1,
1025                            byte[] buffer2, int offset2, int length2) {
1026 
1027         // Short circuit equal case
1028         if (buffer1 == buffer2 &&
1029           offset1 == offset2 &&
1030           length1 == length2) {
1031           return 0;
1032         }
1033         final int stride = 8;
1034         final int minLength = Math.min(length1, length2);
1035         int strideLimit = minLength & -stride;
1036         final long offset1Adj = (long) offset1 + BYTE_ARRAY_BASE_OFFSET;
1037         final long offset2Adj = (long) offset2 + BYTE_ARRAY_BASE_OFFSET;
1038         int i;
1039 
1040         /*
1041          * Compare 8 bytes at a time. Benchmarking on x86 shows a stride of 8 bytes is no slower
1042          * than 4 bytes even on 32-bit. On the other hand, it is substantially faster on 64-bit.
1043          */
1044         for (i = 0; i < strideLimit; i += stride) {
1045           long lw = UNSAFE.getLong(buffer1, offset1Adj + (long) i);
1046           long rw = UNSAFE.getLong(buffer2, offset2Adj + (long) i);
1047           if (lw != rw) {
1048             if (!LITTLE_ENDIAN) {
1049               return ((lw + Long.MIN_VALUE) < (rw + Long.MIN_VALUE)) ? -1 : 1;
1050             }
1051 
1052             /*
1053              * We want to compare only the first index where left[index] != right[index]. This
1054              * corresponds to the least significant nonzero byte in lw ^ rw, since lw and rw are
1055              * little-endian. Long.numberOfTrailingZeros(diff) tells us the least significant
1056              * nonzero bit, and zeroing out the first three bits of L.nTZ gives us the shift to get
1057              * that least significant nonzero byte. This comparison logic is based on UnsignedBytes
1058              * comparator from guava v21
1059              */
1060             int n = Long.numberOfTrailingZeros(lw ^ rw) & ~0x7;
1061             return ((int) ((lw >>> n) & 0xFF)) - ((int) ((rw >>> n) & 0xFF));
1062           }
1063         }
1064 
1065         // The epilogue to cover the last (minLength % stride) elements.
1066         for (; i < minLength; i++) {
1067           int a = buffer1[offset1 + i] & 0xFF;
1068           int b = buffer2[offset2 + i] & 0xFF;
1069           if (a != b) {
1070             return a - b;
1071           }
1072         }
1073         return length1 - length2;
1074       }
1075     }
1076   }
