376    * Returns the value as the specified type. If it is already of the same type as requested, we
377    * just return the value. If not, an {@link UnsupportedOperationException} is thrown.
378    *

384   @SuppressWarnings("unchecked")
385   public <T> T get(Class<T> clazz) throws UnsupportedOperationException {
386     PropertyValueStrategy strategy = PropertyValueStrategyFactory.get(clazz);
387     if (strategy.is(value)) {
388       return (T) value;
389     }
390     throw new UnsupportedOperationException("Value '" + value + "' of type " +
391       value.getClass().getSimpleName() + ", cannot be accessed as " + clazz.getSimpleName());
392   }

755   /**
756    * Byte representation:
757    *
758    * byte 1       : type info
759    *
760    * for dynamic length types (e.g. String and BigDecimal)
761    * byte 2       : length (short)
762    * byte 3       : length (short)
763    * byte 4 - end : value bytes
764    *
765    * If the size of the internal byte representation if larger than
766    * {@link #LARGE_PROPERTY_THRESHOLD} (i.e. if a {@code short} is too small to store the length),
767    * then the {@link #FLAG_LARGE} bit will be set in the first byte and the byte representation
768    * will be:
769    * byte 2       ; length (int)
770    * byte 3       : length (int)
771    * byte 4       : length (int)
772    * byte 5       : length (int)
773    * byte 6 - end : value bytes
774    *
775    * for fixed length types (e.g. int, long, float, ...)
776    * byte 2 - end : value bytes
777    *
778    * @param outputView data output to write data to
779    * @throws IOException if write to output view fails.
780    */
781   @Override
782   public void write(DataOutputView outputView) throws IOException {
783     PropertyValueStrategyFactory.get(value).write(value, outputView);
784   }
785 
786   @Override
787   public void read(DataInputView inputView) throws IOException {
788     // type
789     byte typeByte = inputView.readByte();
790     // Apply bitmask to get the actual type.
791     byte type = (byte) (~PropertyValue.FLAG_LARGE & typeByte);
792 
793     PropertyValueStrategy strategy = PropertyValueStrategyFactory.get(type);
794 
795     if (strategy == null) {
796       throw new UnsupportedTypeException("No strategy for type byte from input view found");
797     } else {
798       value = strategy.read(inputView, typeByte);
799     }
800   }
