16  package org.gradoop.common.model.impl.properties.strategies;

18  import org.apache.flink.core.memory.DataInputView;
19  import org.apache.flink.core.memory.DataInputViewStreamWrapper;
20  import org.apache.flink.core.memory.DataOutputViewStreamWrapper;
21  import org.gradoop.common.model.impl.properties.PropertyValue;
22  import org.gradoop.common.model.impl.properties.Type;

24  import java.io.ByteArrayInputStream;
25  import java.io.ByteArrayOutputStream;
26  import java.io.DataInputStream;
27  import java.io.DataOutputStream;
28  import java.io.IOException;
29  import java.util.ArrayList;
30  import java.util.List;
31  
32  /**
33   * Strategy class for handling {@code PropertyValue} operations with a value of the type
34   * {@code List}.
35   */
36  public class ListStrategy extends AbstractVariableSizedPropertyValueStrategy<List<PropertyValue>> {
37  
38    @Override
39    public List<PropertyValue> read(DataInputView inputView, byte typeByte) throws IOException {
40      byte[] rawBytes = readVariableSizedData(inputView, typeByte);
41  
42      PropertyValue item;
43      List<PropertyValue> list = new ArrayList<>();
44      DataInputViewStreamWrapper internalInputView = createInputView(rawBytes);
45  
46      try {
47        while (internalInputView.available() > 0) {
48          item = new PropertyValue();
49          item.read(internalInputView);
50  
51          list.add(item);
52        }
53      } catch (IOException e) {
54        throw new IOException("Error reading PropertyValue with ListStrategy.", e);
55      }
56  
57      return list;
58    }
59  
60    @Override
61    public int compare(List value, Object other) {
62      throw new UnsupportedOperationException("Method compare() is not supported for List.");
63    }
64  
65    @Override
66    public boolean is(Object value) {
67      if (!(value instanceof List)) {
68        return false;
69      }
70      for (Object item : (List) value) {
71        // List is not a valid property value if it contains any object
72        // that is not a property value itself.
73        if (!(item instanceof PropertyValue)) {
74          return false;
75        }
76      }
77  
78      return true;
79    }
80  
81    @Override
82    public Class<List<PropertyValue>> getType() {
83      return (Class) List.class;
84    }
85  
86    /**
87     * {@inheritDoc}
88     * @throws IOException if converting the byte array to a List fails.
89     */
90    @Override
91    public List<PropertyValue> get(byte[] bytes) throws IOException {
92      PropertyValue item;
93      List<PropertyValue> list = new ArrayList<>();
94  
95      try (ByteArrayInputStream byteStream = new ByteArrayInputStream(bytes);
96          DataInputStream inputStream = new DataInputStream(byteStream);
97          DataInputViewStreamWrapper inputView = new DataInputViewStreamWrapper(inputStream)) {
98  
99        if (inputView.skipBytes(PropertyValue.OFFSET) != PropertyValue.OFFSET) {
100         throw new IOException("Malformed entry in PropertyValue List.");
101       }
102       while (inputView.available() > 0) {
103         item = new PropertyValue();
104         item.read(inputView);
105 
106         list.add(item);
107       }
108     } catch (IOException e) {
109       throw new IOException("Error reading PropertyValue with ListStrategy.", e);
110     }
111 
112     return list;
113   }
114 
115   @Override
116   public byte getRawType() {
117     return Type.LIST.getTypeByte();
118   }
119 
120   /**
121    * {@inheritDoc}
122    * @throws IOException if converting the value to a byte array fails.
123    */
124   @Override
125   public byte[] getRawBytes(List<PropertyValue> value) throws IOException {
126     int size = value.stream().mapToInt(PropertyValue::byteSize).sum() +
127       PropertyValue.OFFSET;
128 
129     try (ByteArrayOutputStream byteStream = new ByteArrayOutputStream(size);
130         DataOutputStream outputStream = new DataOutputStream(byteStream);
131         DataOutputViewStreamWrapper outputView = new DataOutputViewStreamWrapper(outputStream)) {
132 
133       outputStream.write(getRawType());
134       for (PropertyValue entry : value) {
135         entry.write(outputView);
136       }
137 
138       return byteStream.toByteArray();
139     } catch (IOException e) {
140       throw new IOException("Error writing PropertyValue with ListStrategy.", e);
141     }
142   }
143 }
