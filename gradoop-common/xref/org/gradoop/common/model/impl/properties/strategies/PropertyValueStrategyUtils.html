16  package org.gradoop.common.model.impl.properties.strategies;

18  import org.gradoop.common.exceptions.UnsupportedTypeException;
19  import java.math.BigDecimal;
20  import static com.google.common.base.Preconditions.checkNotNull;

22  /**
23   * Utility class which mainly handles comparing any values of the type {@code Numerical}.
24   */
25  class PropertyValueStrategyUtils {
26  
27    /**
28     * Short type.
29     */
30    private static final int SHORT = 0;
31    /**
32     * Integer type.
33     */
34    private static final int INT = 1;
35    /**
36     * Long type.
37     */
38    private static final int LONG = 2;
39    /**
40     * Float type.
41     */
42    private static final int FLOAT = 3;
43    /**
44     * Double type.
45     */
46    private static final int DOUBLE = 4;
47    /**
48     * Big decimal type.
49     */
50    private static final int BIG_DECIMAL = 5;
51  
52    /**
53     * Compares a numerical to an object. The object needs to be an instance of {@code Number}.
54     *
55     * @param aValue first value
56     * @param other second value
57     * @return a negative integer, zero, or a positive integer as aValue is less than, equal to, or
58     * greater than bValue.
59     * @throws IllegalArgumentException if other is not an instance of number.
60     */
61    static int compareNumerical(Number aValue, Object other) {
62      if (!(other instanceof Number)) {
63        throw new IllegalArgumentException(String.format(
64          "Incompatible types: %s, %s", aValue.getClass(), other.getClass()));
65      }
66  
67      Number bValue = (Number) other;
68      int aType = getType(aValue);
69      int bType = getType(bValue);
70  
71      boolean sameType = aType == bType;
72  
73      int maxType = Math.max(aType, bType);
74  
75      int result;
76  
77      if (maxType == SHORT) {
78        result = Short.compare(aValue.shortValue(), bValue.shortValue());
79  
80      } else if (maxType == INT) {
81        int a;
82        int b;
83  
84        if (sameType) {
85          a = aValue.intValue();
86          b = bValue.intValue();
87        } else {
88          a = aType == INT ? aValue.intValue() : aValue.shortValue();
89          b = bType == INT ? bValue.intValue() : bValue.shortValue();
90        }
91  
92        result = Integer.compare(a, b);
93  
94      } else if (maxType == FLOAT) {
95        float a;
96        float b;
97  
98        if (sameType) {
99          a = aValue.floatValue();
100         b = bValue.floatValue();
101       } else {
102         a = aType == FLOAT ? aValue.floatValue() : floatValue(aValue, aType);
103         b = bType == FLOAT ? bValue.floatValue() : floatValue(bValue, bType);
104       }
105 
106       result = Float.compare(a, b);
107 
108     } else if (maxType == LONG) {
109       long a;
110       long b;
111 
112       if (sameType) {
113         a = aValue.longValue();
114         b = bValue.longValue();
115       } else {
116         a = aType == LONG ? aValue.longValue() : longValue(aValue, aType);
117         b = bType == LONG ? bValue.longValue() : longValue(bValue, bType);
118       }
119 
120       result = Long.compare(a, b);
121 
122     } else if (maxType == DOUBLE) {
123       double a;
124       double b;
125 
126       if (sameType) {
127         a = aValue.doubleValue();
128         b = bValue.doubleValue();
129       } else {
130         a = aType == DOUBLE ? aValue.doubleValue() : doubleValue(aValue, aType);
131         b = bType == DOUBLE ? bValue.doubleValue() : doubleValue(bValue, bType);
132       }
133 
134       result = Double.compare(a, b);
135 
136     } else {
137       BigDecimal a;
138       BigDecimal b;
139 
140       if (sameType) {
141         a = (BigDecimal) aValue;
142         b = (BigDecimal) bValue;
143       } else {
144         a = aType == BIG_DECIMAL ? (BigDecimal) aValue :
145         bigDecimalValue(aValue, aType);
146         b = bType == BIG_DECIMAL ? (BigDecimal) bValue :
147         bigDecimalValue(bValue, bType);
148       }
149 
150       result = a.compareTo(b);
151     }
152 
153     return result;
154   }

156   /**
157    * Gets the number type of the provided object.
158    *
159    * @param value some number object.
160    * @return integer representation of the objects data type.
161    */
162   private static int getType(Number value) {
163     checkNotNull(value);
164 
165     int type;
166 
167     if (value instanceof Short) {
168       type = SHORT;
169     } else if (value instanceof Integer) {
170       type = INT;
171     } else if (value instanceof Long) {
172       type = LONG;
173     } else if (value instanceof Float) {
174       type = FLOAT;
175     } else if (value instanceof Double) {
176       type = DOUBLE;
177     } else if (value instanceof BigDecimal) {
178       type = BIG_DECIMAL;
179     } else {
180       throw new UnsupportedTypeException(value.getClass());
181     }
182 
183     return type;
184   }

186   /**
187    * Converts a value of a lower domain numerical type to BigDecimal.
188    *
189    * @param value value
190    * @param type  type
191    * @return converted value
192    */
193   private static BigDecimal bigDecimalValue(Number value, int type) {
194     switch (type) {
195     case SHORT:
196       return BigDecimal.valueOf(value.shortValue());
197     case INT:
198       return BigDecimal.valueOf(value.intValue());
199     case LONG:
200       return BigDecimal.valueOf(value.longValue());
201     case FLOAT:
202       return BigDecimal.valueOf(value.floatValue());
203     default:
204       return BigDecimal.valueOf(value.doubleValue());
205     }
206   }
207 
208   /**
209    * Converts a value of a lower domain numerical type to Double.
210    *
211    * @param value value
212    * @param type  type
213    * @return converted value
214    */
215   private static double doubleValue(Number value, int type) {
216     switch (type) {
217     case SHORT:
218       return value.shortValue();
219     case INT:
220       return value.intValue();
221     case LONG:
222       return value.longValue();
223     default:
224       return value.floatValue();
225     }
226   }
227 
228   /**
229    * Converts a value of a lower domain numerical type to Long.
230    *
231    * @param value value
232    * @param type  type
233    * @return converted value
234    */
235   private static long longValue(Number value, int type) {
236     if (type == SHORT) {
237       return value.shortValue();
238     }
239     return value.intValue();
240   }
241 
242   /**
243    * Converts a value of a lower domain numerical type to Float.
244    *
245    * @param value value
246    * @param type  type
247    * @return converted value
248    */
249   private static float floatValue(Number value, int type) {
250     switch (type) {
251     case SHORT:
252       return value.shortValue();
253     case INT:
254       return value.intValue();
255     default:
256       return value.longValue();
257     }
258   }
