28  /**
29   * A single property in the EPGM model. A property consists of a property key
30   * and a property value, whereas a given property key does not enforce specific
31   * value type.
32   */
33  public class Property implements Value, Serializable, Comparable<Property> {
34  
35    /**
36     * Property key
37     */
38    private String key;
39  
40    /**
41     * Property value
42     */
43    private PropertyValue value;
44  
45    /**
46     * Creates a new property.
47     */
48    public Property() {
49    }
50  
51    /**
52     * Creates a new property from the given arguments.
53     *
54     * @param key   property key
55     * @param value property value
56     */
57    Property(String key, PropertyValue value) {
58      checkNotNull(key, "Property key was null");
59      checkNotNull(value, "Property value was null");
60      checkArgument(!key.isEmpty(), "Property key was empty");
61      this.key = key;
62      this.value = value;
63    }
64  
65    /**
66     * Creates a new property from the given arguments.
67     *
68     * @param key   property key
69     * @param value property value
70     * @return property
71     */
72    public static Property create(String key, PropertyValue value) {
73      return new Property(key, value);
74    }
75  
76    /**
77     * Creates a new property from the given arguments. Note that the given
78     * value type must be supported by {@link PropertyValue}.
79     *
80     * @param key   property key
81     * @param value property value
82     * @return property
83     */
84    public static Property create(String key, Object value) {
85      return new Property(key, PropertyValue.create(value));
86    }
87  
88    /**
89     * Returns the property key.
90     *
91     * @return property key
92     */
93    public String getKey() {
94      return key;
95    }
96  
97    /**
98     * Sets the property key.
99     *
100    * @param key property key (must not be {@code null})
101    */
102   public void setKey(String key) {
103     checkNotNull(key, "Property key was null");
104     checkArgument(!key.isEmpty(), "Property key was empty");
105     this.key = key;
106   }
107 
108   /**
109    * Returns the property value.
110    *
111    * @return property value
112    */
113   public PropertyValue getValue() {
114     return value;
115   }
116 
117   /**
118    * Sets the property value.
119    *
120    * @param value property value  (must not be {@code null})
121    */
122   public void setValue(PropertyValue value) {
123     this.value = checkNotNull(value, "Property value was null");
124   }
125 
126   @Override
127   public boolean equals(Object o) {
128     if (this == o) {
129       return true;
130     }
131     if (o == null || getClass() != o.getClass()) {
132       return false;
133     }
134 
135     Property property = (Property) o;
136 
137     return key.equals(property.key) && value.equals(property.value);
138   }
139 
140   @Override
141   public int hashCode() {
142     int result = key.hashCode();
143     result = 31 * result + value.hashCode();
144     return result;
145   }
146 
147   @Override
148   public int compareTo(Property o) {
149     return this.getKey().compareTo(o.getKey());
150   }
151 
152   @Override
153   public void write(DataOutputView outputView) throws IOException {
154     outputView.writeUTF(key);
155     value.write(outputView);
156   }
157 
158   @Override
159   public void read(DataInputView inputView) throws IOException {
160     key = inputView.readUTF();
161     value = new PropertyValue();
162     value.read(inputView);
163   }
164 
165   @Override
166   public String toString() {
167     return String.format("%s=%s:%s", key, value, value.getType() != null ?
168       value.getType().getSimpleName() : "null");
169   }
