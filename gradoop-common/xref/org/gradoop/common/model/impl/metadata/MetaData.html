72    /**

78    public static Class<?> getClassFromTypeString(String typeString) {

83    /**
84     * Create a map between type strings and their corresponding classes.
85     *
86     * @return map between string types and corresponding classes
87     */
88    private static Map<String, Class<?>> createStringToClassMap() {
89      Map<String, Class<?>> stringClassMap = new HashMap<>();
90      stringClassMap.put(TypeString.NULL.getTypeString(), null);
91      stringClassMap.put(TypeString.SHORT.getTypeString(), Short.class);
92      stringClassMap.put(TypeString.INTEGER.getTypeString(), Integer.class);
93      stringClassMap.put(TypeString.LONG.getTypeString(), Long.class);
94      stringClassMap.put(TypeString.FLOAT.getTypeString(), Float.class);
95      stringClassMap.put(TypeString.DOUBLE.getTypeString(), Double.class);
96      stringClassMap.put(TypeString.BOOLEAN.getTypeString(), Boolean.class);
97      stringClassMap.put(TypeString.STRING.getTypeString(), String.class);
98      stringClassMap.put(TypeString.BIGDECIMAL.getTypeString(), BigDecimal.class);
99      stringClassMap.put(TypeString.GRADOOPID.getTypeString(), GradoopId.class);
100     stringClassMap.put(TypeString.MAP.getTypeString(), Map.class);
101     stringClassMap.put(TypeString.LIST.getTypeString(), List.class);
102     stringClassMap.put(TypeString.LOCALDATE.getTypeString(), LocalDate.class);
103     stringClassMap.put(TypeString.LOCALTIME.getTypeString(), LocalTime.class);
104     stringClassMap.put(TypeString.LOCALDATETIME.getTypeString(), LocalDateTime.class);
105     stringClassMap.put(TypeString.SET.getTypeString(), Set.class);
106     return Collections.unmodifiableMap(stringClassMap);
107   }

167   /**
168    * Returns the type string for the specified property value.
169    *
170    * @param propertyValue property value
171    * @return property type string
172    */
173   public static String getTypeString(PropertyValue propertyValue) {
174     if (propertyValue.isNull()) {
175       return TypeString.NULL.getTypeString();
176     } else if (propertyValue.isShort()) {
177       return TypeString.SHORT.getTypeString();
178     } else if (propertyValue.isInt()) {
179       return TypeString.INTEGER.getTypeString();
180     } else if (propertyValue.isLong()) {
181       return TypeString.LONG.getTypeString();
182     } else if (propertyValue.isFloat()) {
183       return TypeString.FLOAT.getTypeString();
184     } else if (propertyValue.isDouble()) {
185       return TypeString.DOUBLE.getTypeString();
186     } else if (propertyValue.isBoolean()) {
187       return TypeString.BOOLEAN.getTypeString();
188     } else if (propertyValue.isString()) {
189       return TypeString.STRING.getTypeString();
190     } else if (propertyValue.isBigDecimal()) {
191       return TypeString.BIGDECIMAL.getTypeString();
192     } else if (propertyValue.isGradoopId()) {
193       return TypeString.GRADOOPID.getTypeString();
194     } else if (propertyValue.isMap()) {
195       // map type string is map:{keyType}:{valueType}
196       return TypeString.MAP.getTypeString() +
197         PROPERTY_TOKEN_DELIMITER +
198         getTypeString(propertyValue.getMap().keySet().iterator().next()) +
199         PROPERTY_TOKEN_DELIMITER +
200         getTypeString(propertyValue.getMap().values().iterator().next());
201     } else if (propertyValue.isList()) {
202       // list type string is list:{itemType}
203       return TypeString.LIST.getTypeString() +
204         PROPERTY_TOKEN_DELIMITER +
205         getTypeString(propertyValue.getList().get(0));
206     } else if (propertyValue.isDate()) {
207       return TypeString.LOCALDATE.getTypeString();
208     } else if (propertyValue.isTime()) {
209       return TypeString.LOCALTIME.getTypeString();
210     } else if (propertyValue.isDateTime()) {
211       return TypeString.LOCALDATETIME.getTypeString();
212     } else if (propertyValue.isSet()) {
213       // set type string is set:{itemType}
214       return TypeString.SET.getTypeString() +
215         PROPERTY_TOKEN_DELIMITER +
216         getTypeString(propertyValue.getSet().iterator().next());
217     } else {
218       throw new IllegalArgumentException("Type " + propertyValue.getType() + " is not supported");
219     }
220   }
