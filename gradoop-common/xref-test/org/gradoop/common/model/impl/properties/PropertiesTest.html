170   @Test
171   public void testEqualsAndHashCode() {
172     Properties properties1 = Properties.createFromMap(SUPPORTED_PROPERTIES);
173     Properties properties2 = Properties.createFromMap(SUPPORTED_PROPERTIES);
174     Properties properties3 = Properties.createFromMap(SUPPORTED_PROPERTIES);
175     // override property
176     properties3.set(KEY_1, INT_VAL_2);
177 
178     assertTrue("properties were not equal", properties1.equals(properties2));
179     assertFalse("properties were equal", properties1.equals(properties3));
180 
181     assertTrue("different hash code", properties1.hashCode() == properties2.hashCode());
182     assertTrue("same hash code", properties1.hashCode() != properties3.hashCode());
183 
184     properties1 = Properties.create();
185     properties1.set(KEY_1, BOOL_VAL_1);
186     properties1.set(KEY_2, INT_VAL_2);
187 
188     properties2 = Properties.create();
189     properties2.set(KEY_1, BOOL_VAL_1);
190 
191     assertFalse("properties were equal", properties1.equals(properties2));
192     assertTrue("same hash code", properties1.hashCode() != properties2.hashCode());
193   }
