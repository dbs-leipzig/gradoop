171   @Test
172   public void testEqualsAndHashCode() {
173     Properties properties1 = Properties.createFromMap(SUPPORTED_PROPERTIES);
174     Properties properties2 = Properties.createFromMap(SUPPORTED_PROPERTIES);
175     Properties properties3 = Properties.createFromMap(SUPPORTED_PROPERTIES);
176     // override property
177     properties3.set(KEY_1, INT_VAL_2);
178 
179     assertEquals("properties were not equal", properties2, properties1);
180     assertNotEquals(properties3, properties1, "properties were equal");
181 
182     assertEquals("different hash code", properties2.hashCode(), properties1.hashCode());
183     assertTrue("same hash code", properties1.hashCode() != properties3.hashCode());
184 
185     properties1 = Properties.create();
186     properties1.set(KEY_1, BOOL_VAL_1);
187     properties1.set(KEY_2, INT_VAL_2);
188 
189     properties2 = Properties.create();
190     properties2.set(KEY_1, BOOL_VAL_1);
191 
192     assertNotEquals(properties2, properties1, "properties were equal");
193     assertTrue("same hash code", properties1.hashCode() != properties2.hashCode());
194   }
