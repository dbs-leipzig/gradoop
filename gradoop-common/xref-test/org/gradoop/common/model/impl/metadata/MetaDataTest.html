81    /**
82     * Test the meta data maps for vertices, edges and graphs.
83     */
84    @Test
85    public void testMetaDataMaps() {
86  
87      Map<String, List<PropertyMetaData>> graphMetaData = new HashMap<>();
88  
89      List<PropertyMetaData> gPropMetaData1 = Arrays.asList(
90        new PropertyMetaData(
91          "gKey1", MetaData.getTypeString(propValues.get(5)), null), // Double
92        new PropertyMetaData(
93          "gKey2", MetaData.getTypeString(propValues.get(8)), null) // GradoopId
94      );
95      graphMetaData.put("g1", gPropMetaData1);
96  
97      List<PropertyMetaData> gPropMetaData2 = Arrays.asList(
98        new PropertyMetaData(
99          "gKey3", MetaData.getTypeString(propValues.get(10)), null), // List
100       new PropertyMetaData(
101         "gKey4", MetaData.getTypeString(propValues.get(2)), null) // Integer
102     );
103     graphMetaData.put("g2", gPropMetaData2);
104 
105     Map<String, List<PropertyMetaData>> vertexMetaData = new HashMap<>();
106 
107     List<PropertyMetaData> vPropMetaData1 = Arrays.asList(
108       new PropertyMetaData(
109         "vKey1", MetaData.getTypeString(propValues.get(11)), null), // LocalDate
110       new PropertyMetaData(
111         "vKey2", MetaData.getTypeString(propValues.get(15)), null) // Set
112     );
113     vertexMetaData.put("v1", vPropMetaData1);
114 
115     List<PropertyMetaData> vPropMetaData2 = Arrays.asList(
116       new PropertyMetaData(
117         "vKey3", MetaData.getTypeString(propValues.get(0)), null), // null
118       new PropertyMetaData(
119         "vKey4", MetaData.getTypeString(propValues.get(1)), null) // Boolean
120     );
121     vertexMetaData.put("v2", vPropMetaData2);
122 
123     Map<String, List<PropertyMetaData>> edgeMetaData = new HashMap<>();
124 
125     List<PropertyMetaData> ePropMetaData1 = Arrays.asList(
126       new PropertyMetaData(
127         "eKey1", MetaData.getTypeString(propValues.get(9)), null), // Map
128       new PropertyMetaData(
129         "eKey2", MetaData.getTypeString(propValues.get(7)), null) // BigDecimal
130     );
131     edgeMetaData.put("e1", ePropMetaData1);
132 
133     List<PropertyMetaData> ePropMetaData2 = Arrays.asList(
134       new PropertyMetaData(
135         "eKey3", MetaData.getTypeString(propValues.get(6)), null), // String
136       new PropertyMetaData(
137         "eKey4", MetaData.getTypeString(propValues.get(3)), null) // Long
138     );
139 
140     edgeMetaData.put("e2", ePropMetaData2);
141 
142     MetaData metaData = new MetaData(graphMetaData, vertexMetaData, edgeMetaData);
143 
144     assertTrue(metaData.getGraphLabels().contains("g1"));
145     assertTrue(metaData.getGraphLabels().contains("g2"));
146 
147     assertTrue(metaData.getVertexLabels().contains("v1"));
148     assertTrue(metaData.getVertexLabels().contains("v2"));
149 
150     assertTrue(metaData.getEdgeLabels().contains("e1"));
151     assertTrue(metaData.getEdgeLabels().contains("e2"));
152 
153     assertEquals(gPropMetaData1, metaData.getGraphPropertyMetaData("g1"));
154     assertEquals(gPropMetaData2, metaData.getGraphPropertyMetaData("g2"));
155 
156     assertEquals(vPropMetaData1, metaData.getVertexPropertyMetaData("v1"));
157     assertEquals(vPropMetaData2, metaData.getVertexPropertyMetaData("v2"));
158 
159     assertEquals(ePropMetaData1, metaData.getEdgePropertyMetaData("e1"));
160     assertEquals(ePropMetaData2, metaData.getEdgePropertyMetaData("e2"));
161   }
