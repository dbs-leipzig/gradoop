16  package org.gradoop.storage.common.predicate.query;

18  import org.gradoop.common.model.impl.id.GradoopId;
19  import org.gradoop.common.model.impl.id.GradoopIdSet;

22  import javax.annotation.Nonnull;
23  import javax.annotation.Nullable;
24  import java.io.Serializable;

28  /**
29   * Element Query Formula
30   * <p>
31   * A element query contains
32   * <ul>
33   * <li>id range set (which define the query id range of result element)</li>
34   * <li>element filter expression (which define should a result element be return from server)</li>
35   * </ul>
36   *
37   * @see Query#elements()
38   * @param <FilterImpl> filter implement type
39   */
40  public class ElementQuery<FilterImpl extends ElementFilter> implements Serializable {
41  
42    /**
43     * gradoop id range
44     */
45    private final GradoopIdSet ranges;
46  
47    /**
48     * filter predicate
49     */
50    private final FilterImpl filter;
51  
52    /**
53     * element query with range constructor
54     * @param range element query range
55     * @param filter filter definition
56     */
57    private ElementQuery(
58      @Nullable GradoopIdSet range,
59      @Nullable FilterImpl filter
60    ) {
61      this.ranges = range;
62      this.filter = filter;
63    }
64  
65    /**
66     * get filter predicate
67     *
68     * @return predicate implement
69     */
70    @Nullable
71    public FilterImpl getFilterPredicate() {
72      return this.filter;
73    }
74  
75    /**
76     * get query range by gradoop id
77     *
78     * @return gradoop id set
79     */
80    @Nullable
81    public GradoopIdSet getQueryRanges() {
82      return ranges;
83    }
84  
85    @Override
86    public String toString() {
87      return String.format("QUERY ELEMENT FROM %1$s %2$s",
88        ranges == null ? "ALL" : ranges,
89        filter == null ? "" : ("WHERE " + filter));
90    }
91  
92    /**
93     * {@code [Builder] - (set range) -> BuilderWithRange -(set filter) -> ElementQuery}
94     */
95    public static class Builder {
96  
97      /**
98       * query all range
99       *
100      * @return element query with all range
101      */
102     @Nonnull
103     public BuilderWithRange fromAll() {
104       return new BuilderWithRange(null);
105     }
106 
107     /**
108      * query element in a certain id range set
109      *
110      * @param gradoopIds query range
111      * @return element query with a certain range
112      */
113     @Nonnull
114     public BuilderWithRange fromSets(@Nonnull Collection<String> gradoopIds) {
115       return new BuilderWithRange(GradoopIdSet.fromExisting(gradoopIds.stream()
116         .map(GradoopId::fromString)
117         .collect(Collectors.toSet())));
118     }
119 
120     /**
121      * query element in a certain id range set
122      *
123      * @param gradoopIds query range
124      * @return element query with a certain range
125      */
126     @Nonnull
127     public BuilderWithRange fromSets(@Nonnull GradoopIdSet gradoopIds) {
128       return new BuilderWithRange(gradoopIds);
129     }
130 
131     /**
132      * query element in a certain id range set
133      *
134      * @param gradoopIds query range
135      * @return element query with a certain range
136      */
137     @Nonnull
138     public BuilderWithRange fromSets(@Nonnull GradoopId... gradoopIds) {
139       return new BuilderWithRange(GradoopIdSet.fromExisting(gradoopIds));
140     }
141 
142   }
143 
144   /**
145    * {@code Builder - (set range) -> [BuilderWithRange] -(set filter) -> ElementQuery}
146    */
147   public static class BuilderWithRange {
148 
149     /**
150      * gradoop id range
151      */
152     private final GradoopIdSet range;
153 
154     /**
155      * element query with range constructor
156      * @param range element query range
157      */
158     BuilderWithRange(@Nullable GradoopIdSet range) {
159       this.range = range;
160     }
161 
162     /**
163      * element query with range and without filter
164      *
165      * @param <FilterImpl> element filter type, a dummy definition
166      * @return element query with range and without filter
167      */
168     @Nonnull
169     public <FilterImpl extends ElementFilter> ElementQuery<FilterImpl> noFilter() {
170       return new ElementQuery<>(range, null);
171     }
172 
173     /**
174      * element query with range and filter
175      *
176      * @param filter element query with range and filter
177      * @param <FilterImpl> filter implement
178      * @return element query with range and filter
179      */
180     @Nonnull
181     public <FilterImpl extends ElementFilter> ElementQuery<FilterImpl> where(
182       @Nonnull FilterImpl filter
183     ) {
184       return new ElementQuery<>(range, filter);
185     }
186 
187   }
