21  /**
22   * Element Filter Formula
23   * A element filter predicate will be
24   *  - created by client,
25   *  - transform to query options and send to region servers
26   *  - decode query options by region servers,
27   *
28   * @param <FilterImpl> filter implement type
29   */
30  public interface ElementFilter<FilterImpl extends ElementFilter> extends Serializable {
31  
32    /**
33     * Conjunctive operator for element filter
34     *
35     * @param another another reduce filter
36     * @return conjunctive logic filter
37     */
38    @Nonnull
39    FilterImpl or(@Nonnull FilterImpl another);
40  
41    /**
42     * Disjunctive operator for element filter
43     *
44     * @param another another reduce filter
45     * @return disjunctive logic filter
46     */
47    @Nonnull
48    FilterImpl and(@Nonnull FilterImpl another);
49  
50    /**
51     * Negative operator for element filter
52     *
53     * @return negative logic for current filter
54     */
55    @Nonnull
56    FilterImpl negate();

58  }
