227   /**
228    * Enable/Disable the usage of pre-splitting regions at the moment of table creation.
229    * If the HBase table size grows, it should be created with pre-split regions in order to avoid
230    * region hotspots. If certain region servers are stressed by very intensive write/read
231    * operations, HBase may drop that region server because the Zookeeper connection will timeout.
232    *
233    * Note that this flag has no effect if the tables already exist.
234    *
235    * @param numberOfRegions the number of regions used for splitting
236    * @return this modified config
237    */
238   public GradoopHBaseConfig enablePreSplitRegions(final int numberOfRegions) {
239     RegionSplitter.getInstance().setNumberOfRegions(numberOfRegions);
240     this.vertexHandler.setPreSplitRegions(true);
241     this.edgeHandler.setPreSplitRegions(true);
242     this.graphHeadHandler.setPreSplitRegions(true);
243     return this;
244   }
245 
246   /**
247    * Enable/Disable the usage of a spreading byte as prefix of each HBase row key. This affects
248    * reading and writing from/to HBase.
249    *
250    * Records in HBase are sorted lexicographically by the row key. This allows fast access to
251    * an individual record by its key and fast fetching of a range of data given start and stop keys.
252    * But writing records with such naive keys will cause hotspotting because of how HBase writes
253    * data to its Regions. With this option you can disable this hotspotting.
254    *
255    * @param bucketCount the number of spreading bytes to use
256    * @return this modified config
257    */
258   public GradoopHBaseConfig useSpreadingByte(final int bucketCount) {
259     RowKeyDistributor.getInstance().setBucketCount((byte) bucketCount);
260     this.vertexHandler.setSpreadingByteUsage(true);
261     this.edgeHandler.setSpreadingByteUsage(true);
262     this.graphHeadHandler.setSpreadingByteUsage(true);
263     return this;
264   }
