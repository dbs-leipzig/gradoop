23  import org.gradoop.common.model.impl.id.GradoopId;
24  import org.gradoop.common.model.impl.properties.Property;

27  import javax.annotation.Nonnull;
28  import java.io.IOException;
29  import java.io.Serializable;
30  import java.util.List;

32  /**
33   * Handles writing and reading label and properties of an EPGM entity,
34   * i.e., vertex, edge and graph data.
35   */
36  public interface ElementHandler extends Serializable {
37    /**
38     * Creates a globally unique row key based on the given gradoop id. The
39     * created row key is used to persist the entity in the graph store.
40     *
41     * @param gradoopId the gradoop id used to create row key from
42     * @return persistent entity identifier
43     * @throws IOException on failure
44     */
45    byte[] getRowKey(@Nonnull final GradoopId gradoopId) throws IOException;
46  
47    /**
48     * Returns a list of all possible row keys for the given gradoop id according to
49     * the spreading byte.
50     *
51     * @param elementId the gradoop id of the element
52     * @return a list of all possible row keys
53     */
54    List<byte[]> getPossibleRowKeys(@Nonnull final GradoopId elementId);
55  
56    /**
57     * Adds the labels to the given {@link Put} and returns it.
58     *
59     * @param put    put to write label to
60     * @param entity entity to use the label from
61     * @return put with label
62     */
63    Put writeLabel(final Put put, final Element entity);
64  
65    /**
66     * Adds the given property to the {@link Put} and returns it.
67     *
68     * @param put   put to write property to
69     * @param property property
70     * @return put with property
71     */
72    Put writeProperty(final Put put, final Property property);
73  
74    /**
75     * Adds all properties of the given element to the given {@link Put} and
76     * returns it.
77     *
78     * @param put    put to write properties to
79     * @param entity entity to use properties from
80     * @return put with properties
81     */
82    Put writeProperties(final Put put, final Element entity);
83  
84    /**
85     * Creates an identifier from a given row {@link Result}.
86     *
87     * @param res row result
88     * @return entity identifier
89     * @throws IOException on failure
90     */
91    GradoopId readId(@Nonnull final Result res) throws IOException;
92  
93    /**
94     * Reads the label from the given row {@link Result}.
95     *
96     * @param res row result
97     * @return label contained in the row
98     */
99    String readLabel(final Result res);
100 
101   /**
102    * Reads all properties from the given row {@link Result}.
103    *
104    * @param res row result
105    * @return all properties contained in the row
106    */
107   Properties readProperties(final Result res);
108 
109   /**
110    * Creates table based on the given table descriptor.
111    *
112    * @param admin           HBase admin
113    * @param tableDescriptor description of the table used by that specific handler
114    * @throws IOException on failure
115    */
116   void createTable(final Admin admin, final HTableDescriptor tableDescriptor) throws IOException;
117 
118   /**
119    * Enable/Disable the usage of pre-splitting regions at the moment of table creation.
120    * If the HBase table size grows, it should be created with pre-split regions in order to avoid
121    * region hotspots. If certain region servers are stressed by very intensive write/read
122    * operations, HBase may drop that region server because the Zookeeper connection will timeout.
123    *
124    * Note that this flag has no effect if the tables already exist.
125    *
126    * @param usePreSplitRegions flag to decide if the pre-split of regions should be used
127    */
128   void setPreSplitRegions(boolean usePreSplitRegions);
129 
130   /**
131    * Enable/Disable the usage of a spreading byte as prefix of each HBase row key. This affects
132    * reading and writing from/to HBase.
133    *
134    * Records in HBase are sorted lexicographically by the row key. This allows fast access to
135    * an individual record by its key and fast fetching of a range of data given start and stop keys.
136    * But writing records with such naive keys will cause hotspotting because of how HBase writes
137    * data to its Regions. With this option you can disable this hotspotting.
138    *
139    * @param useSpreadingByte flag to decide if the spreading byte should be used
140    */
141   void setSpreadingByteUsage(boolean useSpreadingByte);
142 
143   /**
144    * Indicates whether the regions in HBase tables are pre-split or not.
145    *
146    * @return true, if a pre-split of regions is used
147    */
148   boolean isPreSplitRegions();
149 
150   /**
151    * Indicates whether the row key is salted with a spreading byte as prefix or not.
152    *
153    * @return true, if a spreading byte is used as row key prefix
154    */
155   boolean isSpreadingByteUsed();
156 }
