138   @Override
139   public void writeGraphHead(@Nonnull final GraphHead graphHead) throws IOException {
140     GraphHeadHandler graphHeadHandler = config.getGraphHeadHandler();
141     // graph id
142     Put put = new Put(graphHeadHandler.getRowKey(graphHead.getId()));
143     // write graph to Put
144     put = graphHeadHandler.writeGraphHead(put, graphHead);
145     // write to table
146     graphHeadTable.put(put);
147     if (autoFlush) {
148       admin.flush(graphHeadTable.getName());
149     }
150   }

180   @Override
181   public EPGMGraphHead readGraph(@Nonnull final GradoopId graphId) throws IOException {
182     EPGMGraphHead graphData = null;
183     GraphHeadHandler graphHeadHandler = config.getGraphHeadHandler();
184     List<Get> getList = new ArrayList<>();
185 
186     if (graphHeadHandler.isSpreadingByteUsed()) {
187       for (byte[] rowKey : graphHeadHandler.getPossibleRowKeys(graphId)) {
188         getList.add(new Get(rowKey));
189       }
190     } else {
191       getList.add(new Get(graphId.toByteArray()));
192     }
193     final Result[] results = graphHeadTable.get(getList);
194     for (Result res : results) {
195       if (!res.isEmpty()) {
196         graphData = graphHeadHandler.readGraphHead(res);
197         break;
198       }
199     }
200     return graphData;
201   }

258   ) throws IOException {
259     Scan scan = new Scan();

353   /**
354    * Attach a HBase filter represented by the given query to the given scan instance.
355    *
356    * @param query the query that represents a filter
357    * @param scan the HBase scan instance on which the filter will be applied
358    * @param isSpreadingByteUsed indicates whether a spreading byte is used as row key prefix or not
359    * @param <T> the type of the EPGM element
360    */
361   private <T extends Element> void attachFilter(
362     @Nonnull ElementQuery<HBaseElementFilter<T>> query,
363     @Nonnull Scan scan,
364     boolean isSpreadingByteUsed) {
365 
366     FilterList conjunctFilters = new FilterList(FilterList.Operator.MUST_PASS_ALL);
367 
368     if (query.getQueryRanges() != null && !query.getQueryRanges().isEmpty()) {
369       conjunctFilters.addFilter(HBaseFilterUtils.getIdFilter(query.getQueryRanges(),
370         isSpreadingByteUsed));
371     }
372 
373     if (query.getFilterPredicate() != null) {
374       conjunctFilters.addFilter(query.getFilterPredicate().toHBaseFilter(false));
375     }
376 
377     // if there are filters inside the root list, add it to the Scan object
378     if (!conjunctFilters.getFilters().isEmpty()) {
379       scan.setFilter(conjunctFilters);
380     }
381   }
