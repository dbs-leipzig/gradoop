39  /**
40   * Read logic graph or graph collection from accumulo store
41   */
42  public class AccumuloDataSource extends AccumuloBase implements FilterableDataSource<
43    ElementQuery<AccumuloElementFilter<EPGMGraphHead>>,
44    ElementQuery<AccumuloElementFilter<EPGMVertex>>,
45    ElementQuery<AccumuloElementFilter<EPGMEdge>>> {
46  
47    /**
48     * graph head filter
49     */
50    private final AccumuloQueryHolder<EPGMGraphHead> graphHeadQuery;
51  
52    /**
53     * vertex filter
54     */
55    private final AccumuloQueryHolder<EPGMVertex> vertexQuery;
56  
57    /**
58     * edge filter
59     */
60    private final AccumuloQueryHolder<EPGMEdge> edgeQuery;
61  
62    /**
63     * Creates a new Accumulo data source.
64     *
65     * @param store accumulo epgm store
66     * @param config gradoop flink configuration
67     */
68    public AccumuloDataSource(
69      @Nonnull AccumuloEPGMStore store,
70      @Nonnull GradoopFlinkConfig config
71    ) {
72      this(store, config, null, null, null);
73    }
74  
75    /**
76     * Creates a new Accumulo data source.
77     *
78     * @param store accumulo epgm store
79     * @param config gradoop flink configuration
80     * @param graphQuery graph head filter
81     * @param vertexQuery vertex filter
82     * @param edgeQuery edge filter
83     */
84    private AccumuloDataSource(
85      @Nonnull AccumuloEPGMStore store,
86      @Nonnull GradoopFlinkConfig config,
87      @Nullable AccumuloQueryHolder<EPGMGraphHead> graphQuery,
88      @Nullable AccumuloQueryHolder<EPGMVertex> vertexQuery,
89      @Nullable AccumuloQueryHolder<EPGMEdge> edgeQuery
90    ) {
91      super(store, config);
92      this.graphHeadQuery = graphQuery;
93      this.vertexQuery = vertexQuery;
94      this.edgeQuery = edgeQuery;
95    }
96  
97    @Override
98    public LogicalGraph getLogicalGraph() {
99      return getGraphCollection().reduce(new ReduceCombination<>());
100   }
101 
102   @Override
103   public GraphCollection getGraphCollection() {
104     GraphCollectionFactory factory = getFlinkConfig().getGraphCollectionFactory();
105     ExecutionEnvironment env = getFlinkConfig().getExecutionEnvironment();
106     return factory.fromDataSets(
107       /*graph head format*/
108       env.createInput(new GraphHeadInputFormat(
109         getStore().getConfig().getAccumuloProperties(),
110         graphHeadQuery)),
111       /*vertex input format*/
112       env.createInput(new VertexInputFormat(getStore().getConfig().getAccumuloProperties(),
113         vertexQuery)),
114       /*edge input format*/
115       env.createInput(new EdgeInputFormat(getStore().getConfig().getAccumuloProperties(),
116         edgeQuery)));
117   }
118 
119   @Nonnull
120   @Override
121   public AccumuloDataSource applyGraphPredicate(
122     @Nonnull ElementQuery<AccumuloElementFilter<EPGMGraphHead>> query
123   ) {
124     AccumuloQueryHolder<EPGMGraphHead> newGraphQuery = AccumuloQueryHolder.create(query);
125     return new AccumuloDataSource(
126       getStore(),
127       getFlinkConfig(),
128       newGraphQuery,
129       vertexQuery,
130       edgeQuery
131     );
132   }
133 
134   @Nonnull
135   @Override
136   public AccumuloDataSource applyVertexPredicate(
137     @Nonnull ElementQuery<AccumuloElementFilter<EPGMVertex>> query
138   ) {
139     AccumuloQueryHolder<EPGMVertex> newVertexQuery = AccumuloQueryHolder.create(query);
140     return new AccumuloDataSource(
141       getStore(),
142       getFlinkConfig(),
143       graphHeadQuery,
144       newVertexQuery,
145       edgeQuery
146     );
147   }
148 
149   @Nonnull
150   @Override
151   public AccumuloDataSource applyEdgePredicate(
152     @Nonnull ElementQuery<AccumuloElementFilter<EPGMEdge>> query
153   ) {
154     AccumuloQueryHolder<EPGMEdge> newEdgeQuery = AccumuloQueryHolder.create(query);
155     return new AccumuloDataSource(
156       getStore(),
157       getFlinkConfig(),
158       graphHeadQuery,
159       vertexQuery,
160       newEdgeQuery
161     );
162   }
163 
164   @Override
165   public boolean isFilterPushedDown() {
166     return this.graphHeadQuery != null ||
167       this.vertexQuery != null ||
168       this.edgeQuery != null;
169   }
