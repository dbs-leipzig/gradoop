37  /**
38   * Accumulo Tablet Server Iterator
39   * This Iterator will be created in accumulo tablet server runtime, when executing a partition
40   * range query. A Gradoop element iterator will decode query options as query filter, transform
41   * multi-rows into epgm element and check if this element should be return by predicate. Each
42   * element that fulfill the predicate will be serialized into one row.
43   *
44   * @param <E> gradoop epgm element
45   * @see <a href="https://accumulo.apache.org/1.9/accumulo_user_manual.html#_iterator_design">
46   *   accumulo iterator design</a>
47   */
48  public abstract class BaseElementIterator<E extends Element> implements
49    SortedKeyValueIterator<Key, Value> {
50  
51    /**
52     * Origin accumulo data source
53     */
54    private SortedKeyValueIterator<Key, Value> source;
55  
56    /**
57     * Mapped element iterator, which will handle "next fulfilled element" instrumentation
58     */
59    private InnerIterator seekIterator;
60  
61    /**
62     * Serialized top row
63     */
64    private Pair<Key, Value> top;
65  
66    /**
67     * EPGMElement filter predicate
68     */
69    private AccumuloElementFilter<E> filter;
70  
71    /**
72     * Deserialize from key-value pair
73     *
74     * @param pair k-v pair from accumulo row
75     * @return gradoop element instance
76     * @throws IOException on failure
77     */
78    @Nonnull
79    public abstract E fromRow(@Nonnull Map.Entry<Key, Value> pair) throws IOException;
80  
81    /**
82     * Serialize record to key-value pair
83     *
84     * @param record gradoop element instance
85     * @return k-v pair as accumulo row
86     * @throws IOException on failure
87     */
88    @Nonnull
89    public abstract Pair<Key, Value> toRow(@Nonnull E record) throws IOException;
90  
91    /**
92     * Read read next element definition from accumulo store.
93     * Return null if next element does not fulfill filter formula
94     *
95     * @param source origin store source
96     * @return decoded epgm element
97     * @throws IOException on failure
98     */
99    @Nullable
100   public abstract E readLine(
101     @Nonnull SortedKeyValueIterator<Key, Value> source
102   ) throws IOException;
103 
104   /**
105    * Get element filter predicate
106    *
107    * @return iterator element filter
108    */
109   protected AccumuloElementFilter<E> getFilter() {
110     return filter;
111   }
112 
113   @Override
114   public void init(
115     final SortedKeyValueIterator<Key, Value> source,
116     final Map<String, String> options,
117     final IteratorEnvironment env
118   ) {
119     this.source = source;
120     //read filter predicate
121     if (options != null && !options.isEmpty() &&
122       options.containsKey(AccumuloTables.KEY_PREDICATE)) {
123       this.filter = AccumuloElementFilter.decode(options.get(AccumuloTables.KEY_PREDICATE));
124     } else {
125       this.filter = (AccumuloElementFilter<E>) t -> true;
126     }
127   }
128 
129   @Override
130   public boolean hasTop() {
131     return top != null;
132   }
133 
134   @Override
135   public void next() throws IOException {
136     E topElement = seekIterator.hasNext() ? seekIterator.next() : null;
137     top = topElement == null ? null : toRow(topElement);
138   }
139 
140   @Override
141   public void seek(
142     final Range range,
143     final Collection<ByteSequence> columnFamilies,
144     final boolean inclusive
145   ) throws IOException {
146     //LOG.info("seek range {}", range);
147     source.seek(range, new ArrayList<>(), false);
148     seekIterator = new InnerIterator();
149     next();
150   }
151 
152   @Override
153   public Key getTopKey() {
154     return top == null ? null : top.getFirst();
155   }
156 
157   @Override
158   public Value getTopValue() {
159     return top == null ? null : top.getSecond();
160   }
161 
162   @Override
163   public SortedKeyValueIterator<Key, Value> deepCopy(IteratorEnvironment env) {
164     throw new UnsupportedOperationException("deep copy is not supported!");
165   }

167   /**
168    * Logical iterator for seeking fulfilled element
169    */
170   private class InnerIterator implements Iterator<E> {
171 
172     /**
173      * Next element head
174      */
175     private E head;
176 
177     /**
178      * Create a new inner iterator instance
179      */
180     private InnerIterator() {
181       this.head = readHead();
182     }
183 
184     /**
185      * Read next head element
186      *
187      * @return edge row head
188      */
189     private E readHead() {
190       E next;
191       do {
192         try {
193           next = readLine(source);
194           if (next != null && !getFilter().test(next)) {
195             next = null;
196           }
197         } catch (IOException err) {
198           throw new RuntimeException(err);
199         }
200       } while (source.hasTop() && next == null);
201       return next;
202     }
203 
204     @Override
205     public boolean hasNext() {
206       return head != null;
207     }
208 
209     @Override
210     public E next() {
211       E result = head;
212       head = readHead();
213       return result;
214     }
215   }
