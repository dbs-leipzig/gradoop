16  package org.gradoop.storage.accumulo.impl.handler;
17  
18  import org.apache.accumulo.core.data.Mutation;
19  import org.apache.accumulo.core.data.Value;
20  import org.gradoop.common.model.api.entities.Vertex;
21  import org.gradoop.common.model.api.entities.VertexFactory;
22  import org.gradoop.common.model.impl.id.GradoopIdSet;
23  import org.gradoop.common.model.impl.pojo.EPGMVertex;
24  import org.gradoop.storage.accumulo.impl.constants.AccumuloTables;
25  
26  /**
27   * Accumulo vertex handler for row's read/write operator
28   */
29  public class AccumuloVertexHandler implements AccumuloRowHandler<EPGMVertex, Vertex> {
30  
31    /**
32     * vertex factory
33     */
34    private final VertexFactory<EPGMVertex> factory;
35  
36    /**
37     * vertex handler factory constructor
38     *
39     * @param factory vertex factory
40     */
41    public AccumuloVertexHandler(VertexFactory<EPGMVertex> factory) {
42      this.factory = factory;
43    }
44  
45    @Override
46    public Mutation writeRow(
47      Mutation mutation,
48      Vertex record
49    ) {
50      mutation.put(AccumuloTables.KEY.LABEL, AccumuloTables.KEY.NONE, record.getLabel());
51  
52      //encode properties value bytes as base64 string
53      Iterable<String> keys = record.getPropertyKeys();
54      if (keys != null) {
55        keys.forEach(key -> mutation.put(
56          /*cf*/AccumuloTables.KEY.PROPERTY,
57          /*cq*/key,
58          /*value*/new Value(record.getPropertyValue(key).getRawBytes())));
59      }
60  
61      //write graph ids
62      GradoopIdSet ids = record.getGraphIds();
63      if (ids != null) {
64        ids.forEach(
65          id -> mutation.put(AccumuloTables.KEY.GRAPH, id.toString(), AccumuloTables.KEY.NONE));
66      }
67      return mutation;
68    }
69  
70    @Override
71    public EPGMVertex readRow(Vertex origin) {
72      return factory.initVertex(
73        origin.getId(),
74        origin.getLabel(),
75        origin.getProperties(),
76        origin.getGraphIds()
77      );
78    }
79  
80  }
