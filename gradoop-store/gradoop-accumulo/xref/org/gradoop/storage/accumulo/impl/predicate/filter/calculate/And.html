16  package org.gradoop.storage.accumulo.impl.predicate.filter.calculate;
17  
18  import org.gradoop.common.model.api.entities.Element;
19  import org.gradoop.storage.accumulo.impl.predicate.filter.api.AccumuloElementFilter;
20  
21  import java.util.ArrayList;
22  import java.util.Collections;
23  import java.util.List;
24  import java.util.StringJoiner;
25  
26  /**
27   * Conjunctive predicate filter
28   *
29   * @param <T> element type
30   */
31  public final class And<T extends Element> implements AccumuloElementFilter<T> {
32  
33    /**
34     * Predicate list
35     */
36    private final List<AccumuloElementFilter<T>> predicates = new ArrayList<>();
37  
38    /**
39     * Create a new conjunctive principles
40     *
41     * @param predicates predicates
42     */
43    private And(List<AccumuloElementFilter<T>> predicates) {
44      if (predicates.size() < 2) {
45        throw new IllegalArgumentException(String.format("predicates len(=%d) < 2",
46          predicates.size()));
47      }
48      this.predicates.addAll(predicates);
49    }
50  
51    /**
52     * Create a conjunctive formula
53     *
54     * @param predicates filter predicate
55     * @param <T> input type
56     * @return Conjunctive filter instance
57     */
58    @SafeVarargs
59    public static <T extends Element> And<T> create(AccumuloElementFilter<T>... predicates) {
60      List<AccumuloElementFilter<T>> formula = new ArrayList<>();
61      Collections.addAll(formula, predicates);
62      return new And<>(formula);
63    }
64  
65    @Override
66    public boolean test(T t) {
67      for (AccumuloElementFilter<T> predicate : predicates) {
68        if (!predicate.test(t)) {
69          return false;
70        }
71      }
72      return true;
73    }
74  
75    @Override
76    public String toString() {
77      StringJoiner joiner = new StringJoiner(" AND ");
78      for (AccumuloElementFilter<T> predicate : predicates) {
79        joiner.add("(" + predicate.toString() + ")");
80      }
81      return joiner.toString();
82    }
83  
84  }
