34  /**
35   * Accumulo predicate filter definition, this is a internal model, should not be used outside
36   *
37   * @param <T> element type
38   */
39  public class AccumuloQueryHolder<T extends Element> implements Serializable {
40  
41    /**
42     * Query ranges in accumulo table, should be serializable
43     */
44    private final byte[] queryRanges;
45  
46    /**
47     * Reduce filter for element
48     */
49    private final AccumuloElementFilter<T> reduceFilter;
50  
51    /**
52     * Accumulo predicate instance, low level api for store implement
53     *
54     * @param logicalRanges accumulo logical ranges for element table,
55     * @param reduceFilter query reduce filter
56     *                     only those in predicate should be return from tserver.
57     *                     if null, return all in range
58     */
59    private AccumuloQueryHolder(
60      @Nullable List<Range> logicalRanges,
61      @Nullable AccumuloElementFilter<T> reduceFilter
62    ) {
63      RangeWrapper wrapper = new RangeWrapper();
64      wrapper.ranges = logicalRanges;
65      this.queryRanges = wrapper.encrypt();
66      this.reduceFilter = reduceFilter;
67    }
68  
69    /**
70     * Create a predicate within a certain id ranges
71     *
72     * @param query element query
73     * @param <T>   element type
74     * @return accumulo predicate
75     */
76    public static <T extends Element> AccumuloQueryHolder<T> create(
77      @Nonnull ElementQuery<AccumuloElementFilter<T>> query
78    ) {
79      List<Range> ranges = Range.mergeOverlapping(Optional.ofNullable(query.getQueryRanges())
80        .orElse(GradoopIdSet.fromExisting())
81        .stream()
82        .map(GradoopId::toString)
83        .map(Range::exact)
84        .collect(Collectors.toList()));
85      return new AccumuloQueryHolder<>(
86        query.getQueryRanges() == null ? null : ranges,
87        query.getFilterPredicate());
88    }
89  
90    /**
91     * Create a predicate within a certain accumulo id ranges
92     *
93     * @param idRanges      gradoop row-id ranges for query element
94     * @param reduceFilter  reducer filter logic
95     * @param <T>           element type
96     * @return accumulo predicate
97     */
98    public static <T extends Element> AccumuloQueryHolder<T> create(
99      @Nonnull List<Range> idRanges,
100     @Nullable AccumuloElementFilter<T> reduceFilter) {
101     if (idRanges.isEmpty()) {
102       throw new IllegalArgumentException("id range is empty");
103     }
104     return new AccumuloQueryHolder<>(idRanges, reduceFilter);
105   }
106 
107   /**
108    * Get query ranges by anti-encrypt wrapper
109    *
110    * @return seek range
111    */
112   public List<Range> getQueryRanges() {
113     return queryRanges == null ? null : RangeWrapper.decrypt(queryRanges).ranges;
114   }
115 
116   /**
117    * Get reduce filter
118    *
119    * @return accumulo element filter
120    */
121   public AccumuloElementFilter<T> getReduceFilter() {
122     return reduceFilter;
123   }
124 
125   @Override
126   public String toString() {
127     List<String> ranges = getQueryRanges() == null ? null :
128       getQueryRanges().stream()
129         .map(it -> it == null ? null : String.format("%s:%s",
130           it.getStartKey().getRow(),
131           it.getEndKey().getRow()))
132         .collect(Collectors.toList());
133     return String.format("range=%1$s, filter=%2$s", ranges, getReduceFilter());
134   }
135 
136   /**
137    * Range wrapper definition, just for request transport
138    */
139   private static class RangeWrapper {
140 
141     /**
142      * Query ranges, may be null
143      */
144     private List<Range> ranges;
145 
146     /**
147      * Encrypt as byte array
148      *
149      * @return byte array result
150      */
151     private byte[] encrypt() {
152       try {
153         return KryoUtils.dumps(this);
154       } catch (IOException e) {
155         throw new RuntimeException(e);
156       }
157     }
158 
159     /**
160      * Decrypted from byte array
161      *
162      * @param data encrypted data
163      * @return range wrapper instance
164      */
165     private static RangeWrapper decrypt(byte[] data) {
166       try {
167         return KryoUtils.loads(data, RangeWrapper.class);
168       } catch (IOException e) {
169         throw new RuntimeException(e);
170       }
171     }
172   }
