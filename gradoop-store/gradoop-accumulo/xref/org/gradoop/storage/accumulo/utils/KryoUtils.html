16  package org.gradoop.storage.accumulo.utils;
17  
18  import com.esotericsoftware.kryo.Kryo;
19  import com.esotericsoftware.kryo.io.Input;
20  import com.esotericsoftware.kryo.io.Output;
21  import com.esotericsoftware.kryo.serializers.JavaSerializer;
22  
23  import java.io.ByteArrayInputStream;
24  import java.io.ByteArrayOutputStream;
25  import java.io.IOException;
26  import java.time.LocalDate;
27  import java.time.LocalDateTime;
28  import java.time.LocalTime;
29  import java.util.Collection;
30  import java.util.Map;
31  
32  /**
33   * kryo utils
34   */
35  public final class KryoUtils {
36  
37    /**
38     * kryo instance, should be thread safe
39     */
40    private static final ThreadLocal<Kryo> KRYO_POOLS;
41  
42    static {
43      KRYO_POOLS = ThreadLocal.withInitial(() -> {
44        Kryo kryo = new Kryo();
45        kryo.setReferences(false);
46        kryo.register(Collection.class);
47        kryo.register(Map.class);
48        kryo.register(LocalDateTime.class, new JavaSerializer());
49        kryo.register(LocalTime.class, new JavaSerializer());
50        kryo.register(LocalDate.class, new JavaSerializer());
51        return kryo;
52      });
53    }
54  
55    /**
56     * load object from byte array
57     *
58     * @param encoded encoded byte array
59     * @param type type template
60     * @param <T> type template
61     * @return object instance
62     * @throws IOException io error
63     */
64    public static <T> T loads(
65      byte[] encoded,
66      Class<T> type
67    ) throws IOException {
68      try (
69        ByteArrayInputStream content = new ByteArrayInputStream(encoded);
70        Input input = new Input(content)
71      ) {
72        return KRYO_POOLS.get().readObject(input, type);
73      }
74    }
75  
76    /**
77     * write object to byte array
78     *
79     * @param data data instance to be encode
80     * @return encrypted byte array
81     * @throws IOException io error
82     */
83    public static byte[] dumps(Object data) throws IOException {
84      try (
85        ByteArrayOutputStream content = new ByteArrayOutputStream();
86        Output output = new Output(content)
87      ) {
88        KRYO_POOLS.get().writeObject(output, data);
89        output.flush();
90        return content.toByteArray();
91      }
92    }
93  }
