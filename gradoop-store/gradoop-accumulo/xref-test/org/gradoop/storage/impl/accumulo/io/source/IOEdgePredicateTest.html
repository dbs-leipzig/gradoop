40    @Test
41    public void queryEdgeByProperty() throws Throwable {
42      doTest(TEST01, (loader, store, config) -> {
43        List<EPGMEdge> storeEdges = loader.getEdges().stream()
44          .filter(it -> {
45            assert it.getProperties() != null;
46            return it.getProperties().get("since") != null &&
47              Objects.equals(it.getProperties()
48                .get("since")
49                .getInt(), 2013);
50          })
51          .collect(Collectors.toList());
52  
53        AccumuloDataSource source = new AccumuloDataSource(
54          store,
55          GradoopFlinkConfig.createConfig(getExecutionEnvironment()));
56        List<EPGMEdge> query = source.applyEdgePredicate(
57          Query.elements()
58            .fromAll()
59            .where(AccumuloFilters.propEquals("since", 2013)))
60          .getGraphCollection()
61          .getEdges()
62          .collect();
63  
64        GradoopTestUtils.validateElementCollections(storeEdges, query);
65      });
66    }
67  
68    @Test
69    public void findEdgeByLabelRegex() throws Throwable {
70      doTest(TEST02, (loader, store, config) -> {
71        Pattern queryFormula = Pattern.compile("has.*+");
72  
73        //edge label query
74        List<EPGMEdge> storeEdges = loader.getEdges().stream()
75          .filter(it -> queryFormula.matcher(it.getLabel()).matches())
76          .collect(Collectors.toList());
77  
78        //edge label regex query
79        AccumuloDataSource source = new AccumuloDataSource(store, config);
80        List<EPGMEdge> query = source.applyEdgePredicate(
81          Query.elements()
82            .fromAll()
83            .where(AccumuloFilters.labelReg(queryFormula)))
84          .getGraphCollection()
85          .getEdges()
86          .collect();
87  
88        GradoopTestUtils.validateGraphElementCollections(storeEdges, query);
89      });
90    }
