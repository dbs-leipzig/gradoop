16  package org.gradoop.examples.sna;

18  import com.google.common.base.Preconditions;
19  import org.apache.flink.api.common.ProgramDescription;
20  import org.gradoop.examples.AbstractRunner;
21  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
22  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.EdgeCount;
23  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.VertexCount;
24  
25  import java.util.Arrays;
26  
27  /**
28   * The benchmark program executes the following workflow:
29   *
30   * 1) Extract subgraph with:
31   *    - vertex predicate: must be of type 'Person'
32   *    - edge predicate: must be of type 'knows'
33   * 2) Group the subgraph using the vertex attributes 'city' and 'gender' and
34   *    - count the number of vertices represented by each super vertex
35   *    - count the number of edges represented by each super edge
36   * 3) Aggregate the grouped graph:
37   *    - add the total vertex count as new graph property
38   *    - add the total edge count as new graph property
39   */
40  public class SNABenchmark1 extends AbstractRunner implements
41    ProgramDescription {
42  
43    /**
44     * Runs the example program.
45     *
46     * Need a (possibly HDFS) input directory that contains a EPGM graph
47     * in a given format (csv, indexed, json)
48     *
49     * Needs a (possibly HDFS) output directory to write the resulting graph to.
50     *
51     * @param args args[0] = input dir, args[1] input format, args[2] output dir
52     * @throws Exception on failure
53     */
54    @SuppressWarnings({
55      "unchecked",
56      "Duplicates"
57    })
58    public static void main(String[] args) throws Exception {
59      Preconditions.checkArgument(
60        args.length == 3, "input dir, input format and output dir required");
61      String inputDir  = args[0];
62      String inputFormat = args[1];
63      String outputDir = args[2];
64  
65      LogicalGraph epgmDatabase = readLogicalGraph(inputDir, inputFormat);
66  
67      LogicalGraph result = execute(epgmDatabase);
68  
69      writeLogicalGraph(result, outputDir);
70    }
71  
72    /**
73     * The actual computation.
74     *
75     * @param socialNetwork social network graph
76     * @return summarized, aggregated graph
77     */
78    private static LogicalGraph execute(LogicalGraph socialNetwork) {
79      return socialNetwork
80        .subgraph(
81          vertex -> vertex.getLabel().equals("Person"),
82          edge -> edge.getLabel().equals("knows"))
83        .groupBy(Arrays.asList("gender", "city"))
84        .aggregate(new VertexCount(), new EdgeCount());
85    }
86  
87    @Override
88    public String getDescription() {
89      return SNABenchmark1.class.getName();
90    }
91  }
