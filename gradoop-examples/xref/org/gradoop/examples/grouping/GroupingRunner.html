19  import org.apache.flink.api.common.ProgramDescription;
20  import org.gradoop.examples.AbstractRunner;

22  import org.gradoop.flink.model.impl.operators.aggregation.functions.count.Count;

26  /**
27   * A dedicated program for parametrized graph grouping.
28   */
29  public class GroupingRunner extends AbstractRunner implements ProgramDescription {
30  
31    /**
32     * Option to declare path to input graph
33     */
34    public static final String OPTION_INPUT_PATH = "i";
35    /**
36     * Option to declare input graph format (csv, indexed, json)
37     */
38    public static final String OPTION_INPUT_FORMAT = "f";
39    /**
40     * Option to declare path to output graph
41     */
42    public static final String OPTION_OUTPUT_PATH = "o";
43    /**
44     * Vertex grouping key option
45     */
46    public static final String OPTION_VERTEX_GROUPING_KEY = "vgk";
47    /**
48     * Edge grouping key option
49     */
50    public static final String OPTION_EDGE_GROUPING_KEY = "egk";
51    /**
52     * Use vertex label option
53     */
54    public static final String OPTION_USE_VERTEX_LABELS = "uvl";
55    /**
56     * Use edge label option
57     */
58    public static final String OPTION_USE_EDGE_LABELS = "uel";
59  
60    static {
61      OPTIONS.addOption(OPTION_INPUT_PATH, "vertex-input-path", true,
62        "Path to vertex file");
63      OPTIONS.addOption(OPTION_INPUT_FORMAT, "input-format", true,
64        "Format of the input [csv, indexed, json]. Default: " + DEFAULT_FORMAT);
65      OPTIONS.addOption(OPTION_OUTPUT_PATH, "output-path", true,
66        "Path to write output files to");
67      OPTIONS.addOption(OPTION_VERTEX_GROUPING_KEY, "vertex-grouping-key", true,
68        "EPGMProperty key to group vertices on.");
69      OPTIONS.addOption(OPTION_EDGE_GROUPING_KEY, "edge-grouping-key", true,
70        "EPGMProperty key to group edges on.");
71      OPTIONS.addOption(OPTION_USE_VERTEX_LABELS, "use-vertex-labels", false,
72        "Group on vertex labels");
73      OPTIONS.addOption(OPTION_USE_EDGE_LABELS, "use-edge-labels", false,
74        "Group on edge labels");
75    }
76  
77    /**
78     * Main program to run the example. Arguments are the available options.
79     *
80     * @param args program arguments
81     * @throws Exception on failure
82     */
83    @SuppressWarnings("unchecked")
84    public static void main(String[] args) throws Exception {
85      CommandLine cmd = parseArguments(args, GroupingRunner.class.getName());
86      if (cmd == null) {
87        return;
88      }
89      performSanityCheck(cmd);
90  
91      // read arguments from command line
92      final String inputPath = cmd.getOptionValue(OPTION_INPUT_PATH);
93      final String outputPath = cmd.getOptionValue(OPTION_OUTPUT_PATH);
94  
95      final String inputFormat = cmd.hasOption(OPTION_INPUT_FORMAT) ?
96        cmd.getOptionValue(OPTION_INPUT_FORMAT) : DEFAULT_FORMAT;
97  
98      boolean useVertexKey = cmd.hasOption(OPTION_VERTEX_GROUPING_KEY);
99      String vertexKey =
100       useVertexKey ? cmd.getOptionValue(OPTION_VERTEX_GROUPING_KEY) : null;
101     boolean useEdgeKey = cmd.hasOption(OPTION_EDGE_GROUPING_KEY);
102     String edgeKey =
103       useEdgeKey ? cmd.getOptionValue(OPTION_EDGE_GROUPING_KEY) : null;
104     boolean useVertexLabels = cmd.hasOption(OPTION_USE_VERTEX_LABELS);
105     boolean useEdgeLabels = cmd.hasOption(OPTION_USE_EDGE_LABELS);
106 
107     // initialize EPGM database
108     LogicalGraph graphDatabase = readLogicalGraph(inputPath, inputFormat);
109 
110     // initialize grouping method
111     Grouping grouping = getOperator(
112       vertexKey, edgeKey, useVertexLabels, useEdgeLabels);
113     // call grouping on whole database graph
114     LogicalGraph summarizedGraph = graphDatabase.callForGraph(grouping);
115 
116     if (summarizedGraph != null) {
117       writeLogicalGraph(summarizedGraph, outputPath);
118     } else {
119       System.err.println("wrong parameter constellation");
120     }
121   }
122 
123   /**
124    * Returns the grouping operator implementation based on the given strategy.
125    *
126    * @param vertexKey             vertex property key used for grouping
127    * @param edgeKey               edge property key used for grouping
128    * @param useVertexLabels       use vertex label for grouping, true/false
129    * @param useEdgeLabels         use edge label for grouping, true/false
130    * @return grouping operator implementation
131    */
132   private static Grouping getOperator(String vertexKey,
133     String edgeKey, boolean useVertexLabels, boolean useEdgeLabels) {
134     return new Grouping.GroupingBuilder()
135       .setStrategy(GroupingStrategy.GROUP_REDUCE)
136       .addVertexGroupingKey(vertexKey)
137       .addEdgeGroupingKey(edgeKey)
138       .useVertexLabel(useVertexLabels)
139       .useEdgeLabel(useEdgeLabels)
140       .addVertexAggregateFunction(new Count())
141       .addEdgeAggregateFunction(new Count())
142       .build();
143   }
144 
145   /**
146    * Checks if the minimum of arguments is provided
147    *
148    * @param cmd command line
149    */
150   private static void performSanityCheck(final CommandLine cmd) {
151     if (!cmd.hasOption(OPTION_INPUT_PATH)) {
152       throw new IllegalArgumentException("Define a graph input directory.");
153     }
154     if (!cmd.hasOption(OPTION_VERTEX_GROUPING_KEY) &&
155       !cmd.hasOption(OPTION_USE_VERTEX_LABELS)) {
156       throw new IllegalArgumentException(
157         "Chose at least a vertex grouping key or use vertex labels.");
158     }
159   }
160 
161   @Override
162   public String getDescription() {
163     return GroupingRunner.class.getName();
164   }
