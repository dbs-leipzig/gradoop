16  package org.gradoop.utils.statistics;
17  
18  import org.apache.flink.api.common.ProgramDescription;
19  import org.gradoop.examples.AbstractRunner;
20  import org.gradoop.flink.model.impl.operators.statistics.ConnectedComponentsDistribution;
21  import org.gradoop.flink.model.impl.epgm.LogicalGraph;
22  import org.gradoop.flink.model.impl.operators.sampling.common.SamplingEvaluationConstants;
23  import org.gradoop.flink.model.impl.operators.statistics.writer.StatisticWriter;
24  
25  /**
26   * Calls the computation of the weakly connected components for a logical graph. Uses the
27   * Gradoop-Wrapper {@link ConnectedComponentsDistribution} of Flinks ConnectedComponents-algorithm
28   * with a given or default maximum number of iterations. Writes the result to a csv-file named
29   * {@value SamplingEvaluationConstants#FILE_CONNECTED_COMPONENTS_DIST} in the output directory,
30   * containing a line for each connected component. A line contains the components id followed by
31   * the number of vertices and edges (if edges should be annotated) associated with this component.
32   * If edges are not annotated, the value for edges will be {@code -1}, e.g.:
33   * <pre>
34   * BOF
35   * 00000000000000000000000a, 10, 20
36   * 00000000000000000000000b, 5, 10
37   * EOF
38   * </pre>
39   */
40  public class ConnectedComponentsDistributionRunner extends AbstractRunner implements ProgramDescription {
41  
42    /**
43     * Default value for maximum number of iterations
44     */
45    private static final int DEFAULT_MAX_ITERATIONS = Integer.MAX_VALUE;
46  
47    /**
48     * Default value whether to annotate and count edges
49     */
50    private static final boolean DEFAULT_ANNOTATE_EDGES = true;
51  
52    /**
53     * Calls the computation of the weakly connected components for the graph.
54     *
55     * <pre>
56     * args[0] - path to graph
57     * args[1] - format of graph (csv, json, indexed)
58     * args[2] - output path
59     * args[3] - maximum number of iterations for the algorithm, e.g. 10
60     * args[4] - whether to annotate and count the edges (true) or not (false), e.g. true
61     * </pre>
62     *
63     * @param args command line arguments
64     * @throws Exception in case of read/write failure
65     */
66    public static void main(String[] args) throws Exception {
67  
68      LogicalGraph graph = readLogicalGraph(args[0], args[1]);
69  
70      int maxIteration = DEFAULT_MAX_ITERATIONS;
71      boolean annotateEdges = DEFAULT_ANNOTATE_EDGES;
72      String propertyKey = SamplingEvaluationConstants.PROPERTY_KEY_WCC_ID;
73  
74      if (args.length == 5) {
75        maxIteration = Integer.parseInt(args[3]);
76        annotateEdges = Boolean.parseBoolean(args[4]);
77      }
78  
79      StatisticWriter.writeCSV(new ConnectedComponentsDistribution(
80        propertyKey, maxIteration, annotateEdges).execute(graph), appendSeparator(args[2]) +
81          SamplingEvaluationConstants.FILE_CONNECTED_COMPONENTS_DIST);
82  
83      getExecutionEnvironment().execute("Sampling Statistics: Connected components distribution");
84    }
85  
86    @Override
87    public String getDescription() {
88      return ConnectedComponentsDistributionRunner.class.getName();
89    }
90  }
