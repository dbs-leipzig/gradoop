150   /**
151    * Convert a graph to a {@link TemporalGraph} and set temporal attributes using
152    * {@link TimeIntervalExtractor} functions.
153    *
154    * @param graph                  The graph.
155    * @param graphHeadTimeExtractor The function used to extract temporal attributes for graph heads.
156    * @param vertexTimeExtractor    The function used to extract temporal attributes for vertices.
157    * @param edgeTimeExtractor      The function used to extract temporal attributes for edges.
158    * @param <G> The graph head type.
159    * @param <V> The vertex type.
160    * @param <E> The edge type.
161    * @return A temporal graph with temporal attributes extracted from the original graph.
162    */
163   protected <G extends GraphHead, V extends Vertex, E extends Edge> TemporalGraph toTemporalGraph(
164     BaseGraph<G, V, E, ?, ?> graph,
165     TimeIntervalExtractor<G> graphHeadTimeExtractor,
166     TimeIntervalExtractor<V> vertexTimeExtractor,
167     TimeIntervalExtractor<E> edgeTimeExtractor) {
168     return getConfig().getTemporalGraphFactory().fromNonTemporalDataSets(
169       graph.getGraphHead(), graphHeadTimeExtractor, graph.getVertices(), vertexTimeExtractor,
170       graph.getEdges(), edgeTimeExtractor);
171   }

173   /**
174    * Convert a graph to a {@link TemporalGraph} with time extraction functions.
175    * This will use {@link TemporalGradoopTestUtils#extractTime(Element)} to extract temporal attributes.
176    *
177    * @param graph The graph.
178    * @return The temporal graph with extracted temporal information.
179    */
180   protected TemporalGraph toTemporalGraphWithDefaultExtractors(BaseGraph<?, ?, ?, ?, ?> graph) {
181     // We have to use lambda expressions instead of method references here, otherwise a
182     // ClassCastException will be thrown when those extractor functions are called.
183     // TODO: Find out why.
184     return toTemporalGraph(graph,
185       g -> TemporalGradoopTestUtils.extractTime(g),
186       v -> TemporalGradoopTestUtils.extractTime(v),
187       e -> TemporalGradoopTestUtils.extractTime(e));
188   }

271   /**
272    * Create a temporal edge with temporal attributes set.
273    *
274    * @param label     The label of the edge.
275    * @param source    The element used as a source for the edge.
276    * @param target    The element used as a target for the edge.
277    * @param txFrom    The start of the transaction time.
278    * @param txTo      The end of the transaction time.
279    * @param validFrom The start of the valid time.
280    * @param validTo   The end of the valid time.
281    * @return A temporal edge with those times set.
282    */
283   private TemporalEdge createEdge(String label, Identifiable source, Identifiable target, long txFrom,
284     long txTo, long validFrom, long validTo) {
285     TemporalEdge edge = getConfig().getTemporalGraphFactory().getEdgeFactory()
286       .createEdge(source.getId(), target.getId());
287     edge.setLabel(label);
288     edge.setTransactionTime(Tuple2.of(txFrom, txTo));
289     edge.setValidTime(Tuple2.of(validFrom, validTo));
290     return edge;
291   }

293   /**
294    * Create a temporal vertex with temporal attributes set.
295    *
296    * @param label     The label of the vertex.
297    * @param txFrom    The start of the transaction time.
298    * @param txTo      The end of the transaction time.
299    * @param validFrom The start of the valid time.
300    * @param validTo   The end of the valid time.
301    * @return A temporal vertex with those times set.
302    */
303   private TemporalVertex createVertex(String label, long txFrom, long txTo, long validFrom, long validTo) {
304     TemporalVertex vertex = getConfig().getTemporalGraphFactory().getVertexFactory().createVertex();
305     vertex.setLabel(label);
306     vertex.setTransactionTime(Tuple2.of(txFrom, txTo));
307     vertex.setValidTime(Tuple2.of(validFrom, validTo));
308     return vertex;
309   }

311   /**
312    * Convert a formatted date/time string to the time format (milliseconds since unix epoch) used by Gradoop.
313    * The date string is expected to be formatted according to {@link #DATE_FORMAT}.
314    *
315    * @param dateTimeString The date string.
316    * @return The time in milliseconds since unix epoch.
317    */
318   protected long asMillis(String dateTimeString) {
319     try {
320       return dateFormatter.parse(dateTimeString).getTime();
321     } catch (ParseException pe) {
322       throw new IllegalArgumentException("Failed to parse date.", pe);
323     }
