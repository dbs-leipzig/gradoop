38    public void testDirected() throws Exception {

41      loader.initDatabaseFromFile(getFilePath("/data/gdl/cam_test.gdl"));
42  
43      TemporalGraphCollection g = toTemporalGraphCollectionWithDefaultExtractors(loader.getGraphCollection());
44      g = g.apply(new ApplyTransformation<>(new RemoveProperties<>(PROPERTY_VALID_FROM, PROPERTY_VALID_TO),
45        new RemoveProperties<>(PROPERTY_VALID_FROM, PROPERTY_VALID_TO),
46        new RemoveProperties<>(PROPERTY_VALID_FROM, PROPERTY_VALID_TO)));
47  
48      CanonicalAdjacencyMatrixBuilder<TemporalGraphHead, TemporalVertex, TemporalEdge, TemporalGraph,
49        TemporalGraphCollection> cam =
50        new CanonicalAdjacencyMatrixBuilder<>(
51          new TemporalGraphHeadToDataString<>(),
52          new TemporalVertexToDataString<>(),
53          new TemporalEdgeToDataString<>(), true);
54  
55      String result = cam.execute(g).collect().get(0);
56  
57      String expectation = FileUtils.readFileToString(
58        FileUtils.getFile(getFilePath("/data/expected/cam_test_directed")));

60      assertEquals(expectation, result);
