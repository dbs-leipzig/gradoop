111   /**
112    * Test all {@link MinTime} and {@link MaxTime} related aggregate functions.
113    *
114    * @throws Exception when the execution in Flink fails.
115    */
116   @Test
117   public void testAggregationFunctions() throws Exception {
118     final String keyMaxEdge = "maxEdgeTime";
119     final String keyMinEdge = "minEdgeTime";
120     final String keyMaxVertex = "maxVertexTime";
121     final String keyMinVertex = "minVertexTime";
122     final String keyMax = "maxTime";
123     final String keyMin = "minTime";
124     TemporalGraph result = getTestGraphWithValues().aggregate(
125       new MaxEdgeTime(keyMaxEdge, temporalAttribute, field),
126       new MinEdgeTime(keyMinEdge, temporalAttribute, field),
127       new MaxVertexTime(keyMaxVertex, temporalAttribute, field),
128       new MinVertexTime(keyMinVertex, temporalAttribute, field),
129       new MinTime(keyMin, temporalAttribute, field),
130       new MaxTime(keyMax, temporalAttribute, field));
131     TemporalGraphHead head = result.getGraphHead().collect().get(0);
132     assertEquals(PropertyValue.create(expectedMaxEdge), head.getPropertyValue(keyMaxEdge));
133     assertEquals(PropertyValue.create(expectedMinEdge), head.getPropertyValue(keyMinEdge));
134     assertEquals(PropertyValue.create(expectedMaxVertex), head.getPropertyValue(keyMaxVertex));
135     assertEquals(PropertyValue.create(expectedMinVertex), head.getPropertyValue(keyMinVertex));
136     assertEquals(PropertyValue.create(expectedMax), head.getPropertyValue(keyMax));
137     assertEquals(PropertyValue.create(expectedMin), head.getPropertyValue(keyMin));
138   }

140   /**
141    * Test all {@link MinTime} and {@link MaxTime} related aggregate function where all
142    * temporal temporal attributes are set to default values.
143    * This will check if the aggregate values are null, when all of the values are set to the default value.
144    *
145    * @throws Exception when the execution in Flink fails.
146    */
147   @Test
148   public void testAggregationFunctionsWithAllDefaults() throws Exception {
149     final String keyMaxEdge = "maxEdgeTime";
150     final String keyMinEdge = "minEdgeTime";
151     final String keyMaxVertex = "maxVertexTime";
152     final String keyMinVertex = "minVertexTime";
153     final String keyMax = "maxTime";
154     final String keyMin = "minTime";
155     TemporalGraph result = getTestGraphWithAllDefaults().aggregate(
156       new MaxEdgeTime(keyMaxEdge, temporalAttribute, field),
157       new MinEdgeTime(keyMinEdge, temporalAttribute, field),
158       new MaxVertexTime(keyMaxVertex, temporalAttribute, field),
159       new MinVertexTime(keyMinVertex, temporalAttribute, field),
160       new MinTime(keyMin, temporalAttribute, field),
161       new MaxTime(keyMax, temporalAttribute, field));
162     TemporalGraphHead head = result.getGraphHead().collect().get(0);
163 
164     // The expected values for max and min aggregations. For valid times, they should be null.
165     PropertyValue defaultValue = PropertyValue.NULL_VALUE;
166 
167     // For transaction time, it depends on the chosen field. Min and max of FROM are the current
168     // time; min and max of TO are the MAX_VALUE.
169     if (temporalAttribute == TRANSACTION_TIME) {
170       if (field == FROM) {
171         defaultValue = PropertyValue.create(CURRENT_TIME);
172       } else {
173         defaultValue = PropertyValue.create(MAX_VALUE);
174       }
175     }
176     assertEquals(defaultValue, head.getPropertyValue(keyMaxEdge));
177     assertEquals(defaultValue, head.getPropertyValue(keyMinEdge));
178     assertEquals(defaultValue, head.getPropertyValue(keyMaxVertex));
179     assertEquals(defaultValue, head.getPropertyValue(keyMinVertex));
180     assertEquals(defaultValue, head.getPropertyValue(keyMax));
181     assertEquals(defaultValue, head.getPropertyValue(keyMin));
182   }

246   /**
247    * Create a temporal edge with temporal attributes set.
248    *
249    * @param source    The element used as a source for the edge.
250    * @param target    The element used as a target for the edge.
251    * @param txFrom    The start of the transaction time.
252    * @param txTo      The end of the transaction time.
253    * @param validFrom The start of the valid time.
254    * @param validTo   The end of the valid time.
255    * @return A temporal edge with those times set.
256    */
257   private TemporalEdge createEdge(Identifiable source, Identifiable target, long txFrom,
258     long txTo, long validFrom, long validTo) {
259     TemporalEdge edge = edgeFactory.createEdge(source.getId(), target.getId());
260     edge.setTransactionTime(Tuple2.of(txFrom, txTo));
261     edge.setValidTime(Tuple2.of(validFrom, validTo));
262     return edge;
263   }

267    *

269    * @param txTo      The end of the transaction time.

271    * @param validTo   The end of the valid time.

274   private TemporalVertex createVertex(long txFrom, long txTo, long validFrom, long validTo) {
275     TemporalVertex vertex = vertexFactory.createVertex();
276     vertex.setTransactionTime(Tuple2.of(txFrom, txTo));
277     vertex.setValidTime(Tuple2.of(validFrom, validTo));
278     return vertex;
