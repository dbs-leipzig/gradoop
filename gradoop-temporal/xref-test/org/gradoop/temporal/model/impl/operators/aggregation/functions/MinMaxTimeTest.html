96    /**
97     * Test all {@link MinTime} and {@link MaxTime} related aggregate function where all
98     * temporal temporal attributes are set to default values.
99     * This will check if the aggregate values are null, when all of the values are set to the default value.
100    *
101    * @throws Exception when the execution in Flink fails.
102    */
103   @Test(dataProvider = "defaultParameters")
104   public void testAggregationFunctionsWithAllDefaults(TimeDimension temporalAttribute, TimeDimension.Field field) throws Exception {
105     final String keyMaxEdge = "maxEdgeTime";
106     final String keyMinEdge = "minEdgeTime";
107     final String keyMaxVertex = "maxVertexTime";
108     final String keyMinVertex = "minVertexTime";
109     final String keyMax = "maxTime";
110     final String keyMin = "minTime";
111     TemporalGraph result = getTestGraphWithAllDefaults().aggregate(
112       new MaxEdgeTime(temporalAttribute, field, keyMaxEdge),
113       new MinEdgeTime(temporalAttribute, field, keyMinEdge),
114       new MaxVertexTime(temporalAttribute, field, keyMaxVertex),
115       new MinVertexTime(temporalAttribute, field, keyMinVertex),
116       new MinTime(temporalAttribute, field, keyMin),
117       new MaxTime(temporalAttribute, field, keyMax));
118     TemporalGraphHead head = result.getGraphHead().collect().get(0);
119 
120     // The expected values for max and min aggregations. For valid times, they should be null.
121     PropertyValue defaultValue = PropertyValue.NULL_VALUE;
122 
123     // For transaction time, it depends on the chosen field. Min and max of FROM are the current
124     // time; min and max of TO are the MAX_VALUE.
125     if (temporalAttribute == TRANSACTION_TIME) {
126       if (field == FROM) {
127         defaultValue = PropertyValue.create(CURRENT_TIME);
128       } else {
129         defaultValue = PropertyValue.create(MAX_VALUE);
130       }
131     }
132     assertEquals(defaultValue, head.getPropertyValue(keyMaxEdge));
133     assertEquals(defaultValue, head.getPropertyValue(keyMinEdge));
134     assertEquals(defaultValue, head.getPropertyValue(keyMaxVertex));
135     assertEquals(defaultValue, head.getPropertyValue(keyMinVertex));
136     assertEquals(defaultValue, head.getPropertyValue(keyMax));
137     assertEquals(defaultValue, head.getPropertyValue(keyMin));
138   }
