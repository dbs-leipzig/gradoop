51    /**
52     * Run the test. Calls the snapshot operator using a predicate and compares results with the expected
53     * result graph.
54     *
55     * @param predicate The {@link TemporalPredicate} to create the {@link Snapshot} operator with.
56     * @param dimension The {@link TimeDimension} to create the {@link Snapshot} operator with.
57     * @param expectedVertices An array of strings specifying the expected vertices.
58     * @param expectedEdges An array of string specifying the expected edges.
59     *
60     * @throws Exception when the Execution in Flink fails.
61     */
62    @Test(dataProvider = "parameters")
63    public void runTest(TemporalPredicate predicate, TimeDimension dimension, String[] expectedVertices,
64                        String[] expectedEdges) throws Exception {
65      TemporalGraph inputGraph = getTestGraphWithValues();
66  
67      Collection<TemporalVertex> resultVertices = new HashSet<>();
68      Collection<TemporalEdge> resultEdges = new HashSet<>();
69  
70      TemporalGraph resultGraph = inputGraph.callForGraph(new Snapshot(predicate, dimension));
71  
72      resultGraph.getVertices().output(new LocalCollectionOutputFormat<>(resultVertices));
73      resultGraph.getEdges().output(new LocalCollectionOutputFormat<>(resultEdges));
74  
75      getConfig().getExecutionEnvironment().execute();
76  
77      List<String> resultVertexLabels =
78        resultVertices.stream().map(EPGMElement::getLabel).collect(Collectors.toList());
79      List<String> resultEdgeLabels =
80        resultEdges.stream().map(EPGMElement::getLabel).collect(Collectors.toList());
81  
82      assertEquals(expectedVertices.length, resultVertices.size());
83      assertEquals(expectedEdges.length, resultEdges.size());
84  
85      Arrays.stream(expectedVertices).map(resultVertexLabels::contains).forEach(AssertJUnit::assertTrue);
86      Arrays.stream(expectedEdges).map(resultEdgeLabels::contains).forEach(AssertJUnit::assertTrue);
87    }
