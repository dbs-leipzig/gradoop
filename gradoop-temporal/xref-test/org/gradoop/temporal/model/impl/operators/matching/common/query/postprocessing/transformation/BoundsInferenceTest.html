48    @Test
49    public void preprocessingTest1() throws QueryContradictoryException {
50      // 1970-01-01 < c.val_to <= b.tx_from < a.val_from < 2020-05-25
51      // => lower(c.val_to) = 1L, upper(c.val_to) = ms(2020-05-25)-2
52      // => lower(b.tx_from) = 1L, upper(b.tx_from) = ms(2020-05-25)-2
53      // => lower(a.val_from) = 2L, upper(a.val_from) = ms(2020-05-25)-1
54  
55      CNF cnf = Util.cnfFromLists(
56        Collections.singletonList(new Comparison(bTxFrom, LT, aValFrom)),
57        Collections.singletonList(new Comparison(cValTo, LTE, bTxFrom)),
58        Collections.singletonList(new Comparison(literal1970, LT, cValTo)),
59        Collections.singletonList(new Comparison(aValFrom, LT, literal2020))
60      );
61      BoundsInference prep = new BoundsInference();
62      CNF processedCNF = prep.transformCNF(cnf);
63  
64      long l2020 = literal2020.getMilliseconds();
65  
66      CNF expectedCNF = Util.cnfFromLists(
67        Collections.singletonList(new Comparison(bTxFrom, LT, aValFrom)),
68        Collections.singletonList(new Comparison(cValTo, LTE, bTxFrom)),
69        Collections.singletonList(new Comparison(
70          new TimeLiteral(1), LTE, cValTo)),
71        Collections.singletonList(new Comparison(
72          cValTo, LTE, new TimeLiteral(l2020 - 1))),
73        Collections.singletonList(new Comparison(
74          new TimeLiteral(1), LTE, bTxFrom)),
75        Collections.singletonList(new Comparison(
76          bTxFrom, LTE, new TimeLiteral(l2020 - 1))),
77        Collections.singletonList(new Comparison(
78          new TimeLiteral(1), LTE, aValFrom)),
79        Collections.singletonList(new Comparison(
80          aValFrom, LTE, new TimeLiteral(l2020 - 1)))
81      );
82  
83      equalCNFs(expectedCNF, processedCNF);
84    }
