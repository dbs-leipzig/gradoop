216     testGraph.getVertices().map(v -> new Tuple2<>(v.getId(), v.getLabel()))
217       .returns(new TypeHint<Tuple2<GradoopId, String>>() {
218       }).output(new LocalCollectionOutputFormat<>(idLabelCollection));
219     getExecutionEnvironment().execute();
220     for (Tuple2<GradoopId, String> entity : idLabelCollection) {
221       idLabelMap.put(entity.f0, entity.f1);
222     }

225   /**
226    * Thest the temporal vertex degree operator with all 3 types of degree.
227    *
228    * @throws Exception in case of an error
229    */
230   @Test
231   public void testTemporalDegree() throws Exception {
232     Collection<Tuple4<GradoopId, Long, Long, Integer>> resultCollection = new HashSet<>();
233     HashMap<String, List<Tuple3<Long, Long, Integer>>> labelDegreeMap = new HashMap<>();
234 
235     TemporalVertexDegree operator = new TemporalVertexDegree(degreeType, TimeDimension.VALID_TIME);
236     operator.setIncludeVertexTime(enableVertexTime);
237     final DataSet<Tuple4<GradoopId, Long, Long, Integer>> resultDataSet = testGraph.callForValue(operator);
238 
239     resultDataSet.output(new LocalCollectionOutputFormat<>(resultCollection));
240     getExecutionEnvironment().execute();
241 
242     // Rearrange the result
243     for (Tuple4<GradoopId, Long, Long, Integer> resultEntity : resultCollection) {
244       if (!labelDegreeMap.containsKey(idLabelMap.get(resultEntity.f0))) {
245         labelDegreeMap.put(idLabelMap.get(resultEntity.f0), new ArrayList<>());
246       }
247       labelDegreeMap.get(idLabelMap.get(resultEntity.f0))
248         .add(new Tuple3<>(resultEntity.f1, resultEntity.f2, resultEntity.f3));
249     }
250 
251     labelDegreeMap.forEach((label, tuple3) -> {
252       assertTrue(tuple3.containsAll(expectedDegrees.get(label)));
253       assertTrue(expectedDegrees.get(label).containsAll(tuple3));
254     });
255   }
