56    @Test
57    public void testExecute() throws Exception {
58      GradoopId vertex1Id = GradoopId.get();
59      Map<String, Object> vertex1Props = new HashMap<>();
60      vertex1Props.put("foo", 23);
61      Long[] vertex1Time = new Long[] {123L, 1234L, 456L, 4567L};
62  
63      GradoopId vertex2Id = GradoopId.get();
64      Map<String, Object> vertex2Props = new HashMap<>();
65      vertex2Props.put("foo", 42);
66      Long[] vertex2Time = new Long[] {9876L, 98765L, 654L, 65432L};
67  
68      GradoopId vertex3Id = GradoopId.get();
69      Map<String, Object> vertex3Props = new HashMap<>();
70      vertex2Props.put("foo", 23);
71      Long[] vertex3Time = new Long[] {9876L, 98765L, 654L, 65432L};
72  
73      TemporalVertexFactory factory = new TemporalVertexFactory();
74  
75      TemporalVertex vertex1 = factory.createVertex("A", Properties.createFromMap(vertex1Props));
76      vertex1.setId(vertex1Id);
77      vertex1.setTransactionTime(new Tuple2<>(vertex1Time[0], vertex1Time[1]));
78      vertex1.setValidTime(new Tuple2<>(vertex1Time[2], vertex1Time[3]));
79  
80      TemporalVertex vertex2 = factory.createVertex("B", Properties.createFromMap(vertex2Props));
81      vertex2.setId(vertex2Id);
82      vertex2.setTransactionTime(new Tuple2<>(vertex2Time[0], vertex2Time[1]));
83      vertex2.setValidTime(new Tuple2<>(vertex2Time[2], vertex2Time[3]));
84  
85      TemporalVertex vertex3 = factory.createVertex("C", Properties.createFromMap(vertex3Props));
86      vertex3.setId(vertex3Id);
87      vertex3.setTransactionTime(new Tuple2<>(vertex3Time[0], vertex3Time[1]));
88      vertex3.setValidTime(new Tuple2<>(vertex3Time[2], vertex3Time[3]));
89  
90      DataSet<TemporalVertex> vertices = getExecutionEnvironment().fromElements(vertex1, vertex2, vertex3);
91  
92      String query = "MATCH (n) WHERE n.foo = 23 AND n.tx_from.before(Timestamp(1970-01-01T00:00:01)) " +
93        "AND n.tx_to.before(Timestamp(2020-01-01))";
94      TemporalQueryHandler queryHandler = new TemporalQueryHandler(query);
95      CNF filterPredicate = queryHandler.getPredicates().getSubCNF(Sets.newHashSet("n"));
96      Set<String> projectionKeys = queryHandler.getPredicates().getPropertyKeys("n");
97  
98      FilterAndProjectTemporalVerticesNode node = new FilterAndProjectTemporalVerticesNode(
99        vertices, "n", filterPredicate, projectionKeys);
100 
101     List<Embedding> filteredVertices = new ArrayList<>();
102     node.execute().output(new LocalCollectionOutputFormat<>(filteredVertices));
103     getExecutionEnvironment().execute();
104 
105     assertEquals(1, filteredVertices.size());
106     assertEquals(vertex1Id, filteredVertices.get(0).getId(0));
107 
108     EmbeddingMetaData metaData = node.getEmbeddingMetaData();
109     assertEquals(metaData.getEntryColumn("n"), 0);
110 
111     int fooColumn = metaData.getPropertyColumn("n", "foo");
112     int fromColumn = metaData.getPropertyColumn("n", TimeSelector.TimeField.TX_FROM.toString());
113     int toColumn = metaData.getPropertyColumn("n", TimeSelector.TimeField.TX_TO.toString());
114     assertTrue(fooColumn != fromColumn && fromColumn != toColumn && fooColumn != toColumn);
115   }
