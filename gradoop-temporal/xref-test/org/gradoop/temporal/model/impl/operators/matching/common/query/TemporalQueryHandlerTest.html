80    @Test
81    public void testSimplePredicates() throws Exception {
82      String testquery = "MATCH (a)-[e1:test]->(b) WHERE a.tx_from.before(b.tx_to)";
83      TemporalQueryHandler handler = new TemporalQueryHandler(testquery, new CNFPostProcessing(new ArrayList<>()));
84      Predicate and = new And(
85        new Comparison(new TimeSelector("a", "tx_from"),
86          LT,
87          new TimeSelector("b", "tx_to")),
88        new Comparison(new PropertySelector("e1", "__label__"),
89          Comparator.EQ,
90          new Literal("test"))
91      );
92      and = new And(and, getOverlapsPredicate());
93      assertPredicateEquals(and, handler.getPredicates());
94    }

96    @Test
97    public void testGlobalPredicates() throws QueryContradictoryException {
98      // no global predicates
99      String testquery = "MATCH (a)-[e1:test]->(b) WHERE a.tx_to.before(b.val_to)";
100     TimeSelector e1TxFrom = new TimeSelector("e1", TX_FROM);
101     TimeSelector aTxFrom = new TimeSelector("a", TX_FROM);
102     TimeSelector bTxFrom = new TimeSelector("b", TX_FROM);
103     TimeSelector aTxTo = new TimeSelector("a", TX_TO);
104     TimeSelector bTxTo = new TimeSelector("b", TX_TO);
105     TimeSelector e1TxTo = new TimeSelector("e1", TX_TO);
106     TimeSelector aValFrom = new TimeSelector("a", VAL_FROM);
107     TimeSelector bValFrom = new TimeSelector("b", VAL_FROM);
108     TimeSelector e1ValFrom = new TimeSelector("e1", VAL_FROM);
109     TimeSelector aValTo = new TimeSelector("a", VAL_TO);
110     TimeSelector bValTo = new TimeSelector("b", VAL_TO);
111     TimeSelector e1ValTo = new TimeSelector("e1", VAL_TO);
112     TemporalQueryHandler handler = new TemporalQueryHandler(
113       testquery, new CNFPostProcessing(new ArrayList<>()));
114     Predicate expectedPredicate = new And(
115       new Comparison(aTxTo, LT, bValTo),
116       new Comparison(
117         new PropertySelector("e1", "__label__"), Comparator.EQ, new Literal("test"))
118     );
119     expectedPredicate = new And(expectedPredicate, getOverlapsPredicate());
120     assertPredicateEquals(expectedPredicate, handler.getPredicates());
121 
122     // global and non-global predicates
123     testquery = "MATCH (a)-[e1:test]->(b) WHERE tx_to.before(b.val_to) AND a.val_to.after(b.val_to)";
124     handler = new TemporalQueryHandler(testquery, new CNFPostProcessing(new ArrayList<>()));
125     MinTimePoint globalTxTo = new MinTimePoint(e1TxTo, aTxTo, bTxTo);
126     MaxTimePoint globalTxFrom = new MaxTimePoint(e1TxFrom, aTxFrom, bTxFrom);
127     MaxTimePoint globalValFrom = new MaxTimePoint(e1ValFrom, aValFrom, bValFrom);
128 
129     expectedPredicate = new And(
130       new And(
131         new And(
132           new Comparison(globalTxTo, LT, bValTo),
133           new Comparison(globalTxFrom, LTE, globalTxTo)),
134         new Comparison(aValTo, GT, bValTo)
135       ),
136       new Comparison(new PropertySelector("e1", "__label__"),
137         Comparator.EQ,
138         new Literal("test"))
139     );
140 
141     expectedPredicate = new And(expectedPredicate, getOverlapsPredicate());
142     assertPredicateEquals(expectedPredicate, handler.getPredicates());
143 
144     // only global
145     testquery = "MATCH (a)-[e1]->(b) WHERE tx_to.before(b.val_to) AND a.val_to.after(val_to)";
146     handler = new TemporalQueryHandler(testquery, new CNFPostProcessing(new ArrayList<>()));
147     MinTimePoint globalValTo = new MinTimePoint(e1ValTo, aValTo, bValTo);
148 
149     expectedPredicate = new And(
150       new And(
151         new Comparison(globalTxTo, LT, bValTo),
152         new Comparison(globalTxFrom, LTE, globalTxTo)),
153       new And(
154         new Comparison(aValTo, GT, globalValTo),
155         new Comparison(globalValFrom, LTE, globalValTo))
156     );
157     expectedPredicate = new And(expectedPredicate, getOverlapsPredicate());
158     assertPredicateEquals(expectedPredicate, handler.getPredicates());
159   }

161   private void assertPredicateEquals(Predicate expectedGDL, CNF result) {
162     CNF expected = QueryPredicate.createFrom(expectedGDL, new ComparableTPGMFactory()).asCNF();
163     equalCNFs(expected, result);
164   }

167   // returns the overlaps predicate for the edge (a)-[e]->(b)
168   private Predicate getOverlapsPredicate() {
169     TimeSelector eFrom = new TimeSelector("e1", VAL_FROM);
170     TimeSelector eTo = new TimeSelector("e1", TimeSelector.TimeField.VAL_TO);
171 
172     TimeSelector sFrom = new TimeSelector("a", VAL_FROM);
173     TimeSelector sTo = new TimeSelector("a", TimeSelector.TimeField.VAL_TO);
174 
175     TimeSelector tFrom = new TimeSelector("b", VAL_FROM);
176     TimeSelector tTo = new TimeSelector("b", TimeSelector.TimeField.VAL_TO);
177 
178     return new Comparison(
179       new MaxTimePoint(eFrom, sFrom, tFrom), LTE, new MinTimePoint(eTo, sTo, tTo)
180     );
181   }
