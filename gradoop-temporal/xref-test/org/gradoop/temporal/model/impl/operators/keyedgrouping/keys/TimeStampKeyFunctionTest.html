39    /**
40     * Test if default values are handled properly without a {@link TemporalField} set.
41     */
42    @Test
43    public void testWithDefaultValueWithoutFieldExtraction() {
44      TemporalElement element = getConfig().getTemporalGraphFactory().getVertexFactory().createVertex();
45      element.setTxFrom(TemporalElement.DEFAULT_TIME_FROM);
46      element.setTxTo(TemporalElement.DEFAULT_TIME_TO);
47      KeyFunctionWithDefaultValue<TemporalElement, Long> function = TemporalGroupingKeys.timeStamp(
48        TimeDimension.VALID_TIME, TimeDimension.Field.FROM);
49      assertEquals(TemporalElement.DEFAULT_TIME_FROM, function.getKey(element));
50      assertEquals(TemporalElement.DEFAULT_TIME_FROM, function.getDefaultKey());
51      function = TemporalGroupingKeys.timeStamp(TimeDimension.VALID_TIME, TimeDimension.Field.TO);
52      assertEquals(TemporalElement.DEFAULT_TIME_TO, function.getKey(element));
53      assertEquals(TemporalElement.DEFAULT_TIME_TO, function.getDefaultKey());
54      function = TemporalGroupingKeys.timeStamp(TimeDimension.TRANSACTION_TIME, TimeDimension.Field.FROM);
55      assertEquals(TemporalElement.DEFAULT_TIME_FROM, function.getKey(element));
56      assertEquals(TemporalElement.DEFAULT_TIME_FROM, function.getDefaultKey());
57      function = TemporalGroupingKeys.timeStamp(TimeDimension.TRANSACTION_TIME, TimeDimension.Field.TO);
58      assertEquals(TemporalElement.DEFAULT_TIME_TO, function.getKey(element));
59      assertEquals(TemporalElement.DEFAULT_TIME_TO, function.getDefaultKey());
60    }

100       TimeDimension.TRANSACTION_TIME, TimeDimension.Field.FROM);

103     assertEquals((Long) 2L, function.getKey(element));
104     function = TemporalGroupingKeys.timeStamp(TimeDimension.VALID_TIME, TimeDimension.Field.FROM);
105     assertEquals((Long) 3L, function.getKey(element));

110   /**
111    * Test if keys are set properly on elements.
112    */
113   @Test
114   public void testAddKeyToElement() {
115     VertexFactory<TemporalVertex> vf = getConfig().getTemporalGraphFactory().getVertexFactory();
116     KeyFunctionWithDefaultValue<TemporalElement, Long> function = TemporalGroupingKeys.timeStamp(
117       TimeDimension.VALID_TIME, TimeDimension.Field.FROM);
118     TemporalElement element = vf.createVertex();
119     function.addKeyToElement(element, TemporalElement.DEFAULT_TIME_FROM);
120     assertEquals(PropertyValue.create(TemporalElement.DEFAULT_TIME_FROM),
121       element.getPropertyValue("time_VALID_TIME_FROM"));
122     function.addKeyToElement(element, -1L);
123     assertEquals(PropertyValue.create(-1L), element.getPropertyValue("time_VALID_TIME_FROM"));
124     element = vf.createVertex();
125     function = TemporalGroupingKeys.timeStamp(TimeDimension.VALID_TIME, TimeDimension.Field.FROM,
126       ChronoField.DAY_OF_MONTH);
127     function.addKeyToElement(element, -1L); // -1L is the default, since a TemporalField is set
128     assertNull(element.getPropertyValue("time_VALID_TIME_FROM_DayOfMonth"));
129     function.addKeyToElement(element, 2L);
130     assertEquals(PropertyValue.create(2L), element.getPropertyValue("time_VALID_TIME_FROM_DayOfMonth"));
131   }
