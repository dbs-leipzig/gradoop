54    public void simpleComparisonTest() throws Exception {
55      BinningTemporalGraphStatistics stats = getDummyStats();
56      //tx_from of v1 equally distributed from 100L to 200L
57      // => 175L is part of the 76th bin
58      // => 24 bins are greater => should yield 0.24
59      double estimation1 = stats.estimateTemporalProb(VERTEX,
60        Optional.of("v1"),
61        TimeSelector.TimeField.TX_FROM, Comparator.GT, 175L);
62      assertEquals(0.24, estimation1, 0.01);
63  
64      // half of the tx_to values are bounded => should yield 0.5
65      double estimation2 = stats.estimateTemporalProb(VERTEX,
66        Optional.of("v2"), TX_TO, LT, Long.MAX_VALUE);
67      assertEquals(0.5, estimation2, 0.01);
68  
69      // should yield 0.99 (is in the 100th bin, 99 are smaller)
70      double estimation3 = stats.estimateTemporalProb(VERTEX,
71        Optional.of("v1"), VAL_TO, LTE, Long.MAX_VALUE);
72      assertEquals(.99, estimation3, 0.01);
73  
74      double estimation4 = stats.estimateTemporalProb(VERTEX,
75        Optional.of("v2"), VAL_FROM, GTE, 3011L);
76      assertEquals(.96, estimation4, 0.01);
77  
78      double estimation5 = stats.estimateTemporalProb(VERTEX,
79        Optional.of("v1"), TX_FROM, GTE, Long.MIN_VALUE);
80      assertEquals(1., estimation5, 0.01);

471   @Test
472   public void complexTemporalTest() throws Exception {
473     // estimations are quite inaccurate
474     BinningTemporalGraphStatistics stats = getDummyStats();
475     // about 0.25, as differences between bound values are small, but half of v2.tx_to unbound
476     double estimation1 = stats.estimateTemporalProb(VERTEX, Optional.of("v1"), VAL_FROM,
477       GTE, VERTEX, Optional.of("v2"), TX_TO);
478     assertEquals(0.25, estimation1, 0.025);
479 
480     // P(both unbound) = 0.25
481     // P(both bound) = 0.25
482     // P(lhs bound, rhs unbound) = 0.25
483     // P(lhs unbound, rhs bound) = 0.25
484     // P(lhs <= rhs | both bound) = 0.5
485     // P(lhs <= rhs | both unbound) = 1.
486     // P(lhs <= rhs | lhs bound, rhs unbound) = 1.
487     // P(lhs <= rhs | lhs unbound, rhs bound) = 0
488     // => P(lhs <= rhs) = 0.625
489     double estimation2 = stats.estimateTemporalProb(VERTEX, Optional.of("v1"), VAL_TO,
490       LTE, VERTEX, Optional.of("v1"), VAL_TO);
491     assertEquals(0.625, estimation2, 0.01);
492 
493     // nearly same as above, but P(lhs< rhs | both unbound) = 0
494     // => P(lhs < rhs) = P(lhs <= rhs) - 0.25 = 0.375
495     double estimation3 = stats.estimateTemporalProb(VERTEX, Optional.of("v1"), VAL_TO,
496       LT, VERTEX, Optional.of("v1"), VAL_TO);
497     assertTrue(estimation3 <= estimation2);
498     assertEquals(0.375, estimation3, 0.01);
499 
500     // should be very similar to estimation 1 (where <= instead of < was used)
501     double estimation4 = stats.estimateTemporalProb(VERTEX, Optional.of("v1"), VAL_FROM,
502       GT, VERTEX, Optional.of("v2"), TX_TO);
503     assertTrue(estimation4 <= estimation1);

617    * = "f" (66 vertices)

619    * tx_to goes from 1500L to 2500L (step 20) for half of the vertices, other half is unbounded
620    * val_to goes from 3500L to 4500L for half of the vertices (step 20), other half is unbounded
621    * <p>
622    * Edges: identical tx and val times, their length equally distributed
623    * from 0 to 100L
624    * 10 different source vertices, 20 target vertices
625    *
626    * @return dummy statistics

631 
632     // first type of vertex has label1
633     String vLabel1 = "v1";
634     // tx_from goes from 100L to 200L, val_from from 150L to 250L (100 vertices)
635     // tx_to goes from 300L to 350L for half of the vertices, other half is unbounded
636     // val_to goes from 350L to 450L for half of the vertices, other half is unbounded
637     int numL1Vertices = 100;
638     for (int i = 0; i < numL1Vertices; i++) {
639       TemporalVertex vertex = new TemporalVertex();
640       vertex.setId(GradoopId.get());
641       vertex.setLabel(vLabel1);
642 
643       vertex.setTxFrom(100L + i);
644       long txTo = i % 2 == 0 ? 300L + i : Long.MAX_VALUE;
645       vertex.setTxTo(txTo);
646 
647       vertex.setValidFrom(150L + i);
648       long valTo = i % 2 == 0 ? 350L + i : Long.MAX_VALUE;
649       vertex.setValidTo(valTo);
650 
651       // 6 nodes with property value x
652       if (i % 10 == 0) {
653         vertex.setProperty("catProp1", PropertyValue.create("x"));
654       }
655       // 34 nodes with property value y
656       if (i % 3 == 0) {
657         vertex.setProperty("catProp1", PropertyValue.create("y"));
658       }
659 
660       // every second node has i as a property
661       if (i % 2 == 0) {
662         vertex.setProperty("numProp", PropertyValue.create(i));
663       }
664 
665       vertex.setProperty("numProp2", PropertyValue.create(3 * i));
666 
667       vertexList.add(vertex);
668     }
669 
670     // first type of vertex has label1
671     String vLabel2 = "v2";
672     // tx_from goes from 1000L to 2000L, val_from from 3000L to 4000L (100 vertices)
