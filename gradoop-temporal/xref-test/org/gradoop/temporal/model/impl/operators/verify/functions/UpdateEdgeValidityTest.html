33  /**
34   * Test for the {@link UpdateEdgeValidity} function used by the temporal verify operator.
35   */
36  public class UpdateEdgeValidityTest extends TemporalGradoopTestBase {
37  
38    /**
39     * A list storing the join results.
40     */
41    private List<TemporalEdge> result;
42  
43    /**
44     * The operator to test.
45     */
46    private UpdateEdgeValidity operator;
47  
48    /**
49     * The output for the operator.
50     */
51    private Collector<TemporalEdge> out;
52  
53    /**
54     * A test vertex.
55     */
56    private TemporalVertex vertex;
57  
58    /**
59     * Set up this test.
60     */
61    @BeforeMethod
62    public void setUp() {
63      result = new ArrayList<>();
64      out = new ListCollector<>(result);
65      operator = new UpdateEdgeValidity();
66      vertex = getVertexFactory().createVertex();
67      vertex.setValidFrom(asMillis("2020.02.02 12:00:00.000"));
68      vertex.setValidTo(asMillis("2020.02.03 12:00:00.000"));
69    }
70  
71    /**
72     * Test if non-overlapping edges will be removed properly.
73     */
74    @Test
75    public void testRemoveNonOverlapping() {
76      TemporalEdge edge = getEdgeFactory().createEdge(GradoopId.NULL_VALUE, GradoopId.NULL_VALUE);
77      // Edge is valid after the vertex.
78      edge.setValidFrom(asMillis("2020.02.03 12:00:00.000"));
79      edge.setValidTo(asMillis("2020.02.04 12:00:00.000"));
80      operator.join(edge, vertex, out);
81      assertTrue(result.isEmpty());
82      // Edge is valid before the vertex.
83      edge.setValidFrom(asMillis("2020.02.01 12:00:00.000"));
84      edge.setValidTo(asMillis("2020.02.02 11:59:59.999"));
85      operator.join(edge, vertex, out);
86      assertTrue(result.isEmpty());
87    }
88  
89    /**
90     * Test if the {@code validFrom} field is updated accordingly.
91     */
92    @Test
93    public void testUpdateValidFrom() {
94      TemporalEdge edge = getEdgeFactory().createEdge(GradoopId.NULL_VALUE, GradoopId.NULL_VALUE);
95      final long validTo = asMillis("2020.02.02 13:00:00.000");
96      edge.setValidFrom(asMillis("2019.12.31 00:00:00.000"));
97      edge.setValidTo(validTo);
98      operator.join(edge, vertex, out);
99      assertEquals(result.size(), 1);
100     TemporalEdge resultEdge = result.get(0);
101     assertEquals(resultEdge.getValidFrom(), vertex.getValidFrom());
102     assertEquals((long) resultEdge.getValidTo(), validTo);
103   }
104 
105   /**
106    * Test if the {@code validTo} field is updated accordingly.
107    */
108   @Test
109   public void testUpdateValidTo() {
110     TemporalEdge edge = getEdgeFactory().createEdge(GradoopId.NULL_VALUE, GradoopId.NULL_VALUE);
111     final long validFrom = asMillis("2020.02.02 13:00:00.000");
112     edge.setValidFrom(validFrom);
113     edge.setValidTo(asMillis("2020.03.01 12:00:00.000"));
114     operator.join(edge, vertex, out);
115     assertEquals(result.size(), 1);
116     TemporalEdge resultEdge = result.get(0);
117     assertEquals((long) resultEdge.getValidFrom(), validFrom);
118     assertEquals(resultEdge.getValidTo(), vertex.getValidTo());
119   }
120 
121   /**

123    */
124   @Test
125   public void testUpdateBoth() {

127     edge.setValidFrom(asMillis("2020.01.01 12:00:00.000"));

134   }
