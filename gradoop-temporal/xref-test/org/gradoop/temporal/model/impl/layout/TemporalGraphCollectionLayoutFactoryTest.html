61    @BeforeClass
62    public void setUp() throws Exception {
63      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
64  
65      graphHeads = loader.getGraphHeadsByVariables("g0").stream()
66        .map(getGraphHeadFactory()::fromNonTemporalGraphHead)
67        .collect(Collectors.toList());
68      vertices = loader.getVerticesByGraphVariables("g0").stream()
69        .map(getVertexFactory()::fromNonTemporalVertex)
70        .collect(Collectors.toList());
71      edges = loader.getEdgesByGraphVariables("g0").stream()
72        .map(getEdgeFactory()::fromNonTemporalEdge)
73        .collect(Collectors.toList());
74  
75      graphHeadDataSet = getExecutionEnvironment().fromCollection(graphHeads);
76      vertexDataSet = getExecutionEnvironment().fromCollection(vertices);
77      edgeDataSet = getExecutionEnvironment().fromCollection(edges);
78  
79      temporalGraph = getConfig().getTemporalGraphFactory()
80        .fromDataSets(graphHeadDataSet, vertexDataSet, edgeDataSet);
81  
82      factory = new TemporalGraphCollectionLayoutFactory();
83      factory.setGradoopFlinkConfig(getConfig());
84    }

86    /**
87     * Test the {@link TemporalGraphCollectionLayoutFactory#fromDataSets(DataSet, DataSet)} method.
88     *
89     * @throws Exception if the test fails
90     */
91    @Test
92    public void testFromDataSets() throws Exception {
93      final GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> layout =
94        factory.fromDataSets(graphHeadDataSet, vertexDataSet);
95  
96      Collection<TemporalGraphHead> loadedGraphHeads = new ArrayList<>();
97      Collection<TemporalVertex> loadedVertices = new ArrayList<>();
98      Collection<TemporalEdge> loadedEdges = new ArrayList<>();
99  
100     layout.getGraphHeads().output(new LocalCollectionOutputFormat<>(loadedGraphHeads));
101     layout.getVertices().output(new LocalCollectionOutputFormat<>(loadedVertices));
102     layout.getEdges().output(new LocalCollectionOutputFormat<>(loadedEdges));
103 
104     getExecutionEnvironment().execute();
105 
106     validateElementCollections(vertices, loadedVertices);
107     validateElementCollections(graphHeads, loadedGraphHeads);
108     assertEquals(0, loadedEdges.size());
109   }
