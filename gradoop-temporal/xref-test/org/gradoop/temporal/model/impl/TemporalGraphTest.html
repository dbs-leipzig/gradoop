170     temporalGraph.getEdges().output(new LocalCollectionOutputFormat<>(loadedEdges));
171 
172     Collection<EPGMGraphHead> epgmGraphHeads = new ArrayList<>();
173     Collection<EPGMVertex> epgmVertices = new ArrayList<>();
174     Collection<EPGMEdge> epgmEdges = new ArrayList<>();
175 
176     testLogicalGraph.getGraphHead().output(new LocalCollectionOutputFormat<>(epgmGraphHeads));
177     testLogicalGraph.getVertices().output(new LocalCollectionOutputFormat<>(epgmVertices));
178     testLogicalGraph.getEdges().output(new LocalCollectionOutputFormat<>(epgmEdges));
179 
180     getExecutionEnvironment().execute();
181 
182     assertFalse(loadedGraphHeads.isEmpty());
183     assertFalse(loadedVertices.isEmpty());
184     assertFalse(loadedEdges.isEmpty());
185 
186     validateElementCollections(epgmGraphHeads, loadedGraphHeads);
187     validateElementCollections(epgmVertices, loadedVertices);
188     validateElementCollections(epgmEdges, loadedEdges);
189     validateGraphElementCollections(epgmVertices, loadedVertices);
190     validateGraphElementCollections(epgmEdges, loadedEdges);
191 
192     loadedGraphHeads.forEach(this::checkDefaultTemporalElement);
193     loadedVertices.forEach(this::checkDefaultTemporalElement);
194     loadedEdges.forEach(this::checkDefaultTemporalElement);
