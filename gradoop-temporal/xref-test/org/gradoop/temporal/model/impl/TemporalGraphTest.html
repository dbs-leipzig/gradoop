164    *
165    * @throws Exception if loading the graph fails
166    */
167   @Test
168   public void testFromGraph() throws Exception {
169     TemporalGraph temporalGraph = TemporalGraph.fromGraph(testLogicalGraph);
170 
171     Collection<TemporalGraphHead> loadedGraphHeads = new ArrayList<>();
172     Collection<TemporalVertex> loadedVertices = new ArrayList<>();
173     Collection<TemporalEdge> loadedEdges = new ArrayList<>();
174 
175     temporalGraph.getGraphHead().output(new LocalCollectionOutputFormat<>(loadedGraphHeads));
176     temporalGraph.getVertices().output(new LocalCollectionOutputFormat<>(loadedVertices));
177     temporalGraph.getEdges().output(new LocalCollectionOutputFormat<>(loadedEdges));
178 
179     Collection<EPGMGraphHead> epgmGraphHeads = new ArrayList<>();
180     Collection<EPGMVertex> epgmVertices = new ArrayList<>();
181     Collection<EPGMEdge> epgmEdges = new ArrayList<>();
182 
183     testLogicalGraph.getGraphHead().output(new LocalCollectionOutputFormat<>(epgmGraphHeads));
184     testLogicalGraph.getVertices().output(new LocalCollectionOutputFormat<>(epgmVertices));
185     testLogicalGraph.getEdges().output(new LocalCollectionOutputFormat<>(epgmEdges));
186 
187     getExecutionEnvironment().execute();
188 
189     assertFalse(loadedGraphHeads.isEmpty());
190     assertFalse(loadedVertices.isEmpty());
191     assertFalse(loadedEdges.isEmpty());
192 
193     validateElementCollections(epgmGraphHeads, loadedGraphHeads);
194     validateElementCollections(epgmVertices, loadedVertices);
195     validateElementCollections(epgmEdges, loadedEdges);
196     validateGraphElementCollections(epgmVertices, loadedVertices);
197     validateGraphElementCollections(epgmEdges, loadedEdges);
198 
199     loadedGraphHeads.forEach(this::checkDefaultTemporalElement);
200     loadedVertices.forEach(this::checkDefaultTemporalElement);
201     loadedEdges.forEach(this::checkDefaultTemporalElement);
202   }
