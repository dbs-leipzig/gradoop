56    /**
57     * Creates a test temporal graph from the social network loader.
58     *
59     * @throws Exception if loading the graph fails
60     */
61    @Before
62    public void setUp() throws Exception {
63      testGraph = toTemporalGraph(getSocialNetworkLoader().getLogicalGraph())
64        .transformGraphHead((current, trans) -> {
65          current.setProperty("testGraphHeadProperty", PropertyValue.create(1L));
66          return current;
67        });
68  
69      testGraphCollection = toTemporalGraphCollection(getSocialNetworkLoader().getGraphCollection());
70    }
71  
72    /**
73     * Tests the function {@link TemporalIndexedCSVDataSource#getTemporalGraph()}.
74     *
75     * @throws Exception if loading the graph fails
76     */
77    @Test
78    public void testGetTemporalGraph() throws Exception {
79      new TemporalIndexedCSVDataSink(testFolder.getRoot().getPath(), getConfig()).write(testGraph);
80      getExecutionEnvironment().execute();
81      TemporalIndexedCSVDataSource source = new TemporalIndexedCSVDataSource(testFolder.getRoot().getPath(),
82        getConfig());
83      collectAndAssertTrue(source.getTemporalGraph().equalsByData(testGraph));
84    }
85  
86    /**
87     * Tests the function {@link TemporalIndexedCSVDataSource#getTemporalGraphCollection()}.
88     *
89     * @throws Exception if loading the collection fails
90     */
91    @Test
92    public void testGetTemporalGraphCollection() throws Exception {
93      new TemporalIndexedCSVDataSink(testFolder.getRoot().getPath(), getConfig()).write(testGraphCollection);
94      getExecutionEnvironment().execute();
95      TemporalIndexedCSVDataSource source = new TemporalIndexedCSVDataSource(testFolder.getRoot().getPath(),
96        getConfig());
97      collectAndAssertTrue(source.getTemporalGraphCollection().equalsByGraphData(testGraphCollection));
98    }
