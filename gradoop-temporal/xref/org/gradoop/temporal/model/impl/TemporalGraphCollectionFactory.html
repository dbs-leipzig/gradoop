131   @Override
132   public TemporalGraphCollection fromGraphs(
133     LogicalGraphLayout<TemporalGraphHead, TemporalVertex, TemporalEdge>... logicalGraphLayout) {
134     if (logicalGraphLayout.length == 0) {
135       return createEmptyCollection();
136     } else if (logicalGraphLayout.length == 1) {
137       return fromGraph(logicalGraphLayout[0]);
138     } else {
139       return fromDataSets(
140         Arrays.stream(logicalGraphLayout).map(LogicalGraphLayout::getGraphHead).reduce(DataSet::union)
141           .get().distinct(new Id<>()),
142         Arrays.stream(logicalGraphLayout).map(LogicalGraphLayout::getVertices).reduce(DataSet::union)
143           .get().distinct(new Id<>()),
144         Arrays.stream(logicalGraphLayout).map(LogicalGraphLayout::getEdges).reduce(DataSet::union)
145           .get().distinct(new Id<>()));
146     }
147   }

181   /**
182    * Creates a {@link TemporalGraphCollection} instance by the given graph head, vertex and edge datasets.
183    *
184    * @param graphHead graph head DataSet
185    * @param vertices vertex DataSet
186    * @param edges edge DataSet
187    * @return a temporal graph collection
188    */
189   public TemporalGraphCollection fromNonTemporalDataSets(
190     DataSet<? extends GraphHead> graphHead,
191     DataSet<? extends Vertex> vertices,
192     DataSet<? extends Edge> edges) {
193     return new TemporalGraphCollection(this.layoutFactory.fromDataSets(
194       graphHead.map(new GraphHeadToTemporalGraphHead<>(getGraphHeadFactory())),
195       vertices.map(new VertexToTemporalVertex<>(getVertexFactory())),
196       edges.map(new EdgeToTemporalEdge<>(getEdgeFactory()))), config);
197   }

199   /**
200    * Creates a {@link TemporalGraphCollection} instance. By the provided timestamp extractors, it is
201    * possible to extract temporal information from the data to define a timestamp or time interval that
202    * represents the beginning and end of the element's validity (valid time).
203    *
204    * @param graphHead graph head DataSet
205    * @param graphHeadTimeIntervalExtractor extractor to pick the time interval from graph heads
206    * @param vertices vertex DataSet
207    * @param vertexTimeIntervalExtractor extractor to pick the time interval from vertices
208    * @param edges edge DataSet
209    * @param edgeTimeIntervalExtractor extractor to pick the time interval from edges
210    * @return the logical graph represented as temporal graph with defined valid times
211    * @param <G> The graph head type.
212    * @param <V> The vertex type.
213    * @param <E> The edge type.
214    */
215   public <
216     G extends GraphHead,
217     V extends Vertex,
218     E extends Edge> TemporalGraphCollection fromNonTemporalDataSets(
219     DataSet<G> graphHead,
220     TimeIntervalExtractor<G> graphHeadTimeIntervalExtractor,
221     DataSet<V> vertices,
222     TimeIntervalExtractor<V> vertexTimeIntervalExtractor,
223     DataSet<E> edges,
224     TimeIntervalExtractor<E> edgeTimeIntervalExtractor) {
225 
226     return new TemporalGraphCollection(this.layoutFactory.fromDataSets(
227       graphHead.map(new GraphHeadToTemporalGraphHead<>(getGraphHeadFactory(),
228         graphHeadTimeIntervalExtractor)),
229       vertices.map(new VertexToTemporalVertex<>(getVertexFactory(),
230         vertexTimeIntervalExtractor)),
231       edges.map(new EdgeToTemporalEdge<>(getEdgeFactory(),
232         edgeTimeIntervalExtractor))), config);
233   }

235   /**
236    * Creates a {@link TemporalGraphCollection} from a (non-temporal) base graph collection.
237    *
238    * This method calls {@link #fromNonTemporalDataSets(DataSet, DataSet, DataSet)} on the graph collections
239    * element data sets.
240    *
241    * @param collection Some base graph collection.
242    * @return The resulting temporal graph collection.
243    */
244   public TemporalGraphCollection fromNonTemporalGraphCollection(
245     BaseGraphCollection<?, ?, ?, ?, ?> collection) {
246     return fromNonTemporalDataSets(
247       collection.getGraphHeads(), collection.getVertices(), collection.getEdges());
248   }
