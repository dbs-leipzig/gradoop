33  /**
34   * Factory responsible for creating temporal GVE graph collection layouts.
35   */
36  public class TemporalGraphCollectionLayoutFactory extends TemporalBaseLayoutFactory
37    implements GraphCollectionLayoutFactory<TemporalGraphHead, TemporalVertex, TemporalEdge> {
38  
39    @Override
40    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromDataSets(
41      DataSet<TemporalGraphHead> graphHeads, DataSet<TemporalVertex> vertices) {
42      return fromDataSets(graphHeads, vertices, createEdgeDataSet(Collections.emptyList()));
43    }
44  
45    @Override
46    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromDataSets(
47      DataSet<TemporalGraphHead> graphHeads, DataSet<TemporalVertex> vertices,
48      DataSet<TemporalEdge> edges) {
49      return new TemporalGVELayout(
50        Objects.requireNonNull(graphHeads),
51        Objects.requireNonNull(vertices),
52        Objects.requireNonNull(edges));
53    }
54  
55    @Override
56    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromIndexedDataSets(
57      Map<String, DataSet<TemporalGraphHead>> graphHeads,
58      Map<String, DataSet<TemporalVertex>> vertices, Map<String, DataSet<TemporalEdge>> edges) {
59      return new TemporalIndexedLayout(graphHeads, vertices, edges);
60  
61    }
62  
63    @Override
64    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromCollections(
65      Collection<TemporalGraphHead> graphHeads,
66      Collection<TemporalVertex> vertices,
67      Collection<TemporalEdge> edges) {
68      return fromDataSets(
69        createGraphHeadDataSet(Objects.requireNonNull(graphHeads)),
70        createVertexDataSet(Objects.requireNonNull(vertices)),
71        createEdgeDataSet(Objects.requireNonNull(edges)));
72    }
73  
74    @Override
75    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromGraphLayout(
76      LogicalGraphLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> logicalGraphLayout) {
77      return fromDataSets(
78        logicalGraphLayout.getGraphHead(),
79        logicalGraphLayout.getVertices(),
80        logicalGraphLayout.getEdges());
81    }
82  
83    /**
84     * {@inheritDoc}
85     *
86     * Creating a temporal graph layout from graph transactions is not supported yet.
87     */
88    @Override
89    public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromTransactions(
90      DataSet<GraphTransaction> transactions) {
91      throw new UnsupportedOperationException(
92        "Creating a temporal graph layout from graph transactions is not supported yet.");
93    }
94  
95    /**
96     * {@inheritDoc}
97     *
98     * Creating a temporal graph layout from graph transactions is not supported yet.
99     */
100   @Override
101   public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> fromTransactions(
102     DataSet<GraphTransaction> transactions,
103     GroupReduceFunction<TemporalVertex, TemporalVertex> vertexMergeReducer,
104     GroupReduceFunction<TemporalEdge, TemporalEdge> edgeMergeReducer) {
105     throw new UnsupportedOperationException(
106       "Creating a temporal graph layout from graph transactions is not supported yet.");
107   }
108 
109   @Override
110   public GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> createEmptyCollection() {
111     return fromCollections(Collections.emptyList(), Collections.emptyList(), Collections.emptyList());
112   }
113 }
