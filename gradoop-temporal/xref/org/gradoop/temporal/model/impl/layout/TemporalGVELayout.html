16  package org.gradoop.temporal.model.impl.layout;
17  
18  import org.apache.flink.api.java.DataSet;
19  import org.gradoop.flink.model.api.layouts.GraphCollectionLayout;
20  import org.gradoop.flink.model.api.layouts.LogicalGraphLayout;
21  import org.gradoop.flink.model.impl.functions.epgm.ByLabel;
22  import org.gradoop.flink.model.impl.layouts.transactional.tuples.GraphTransaction;
23  import org.gradoop.temporal.model.impl.pojo.TemporalEdge;
24  import org.gradoop.temporal.model.impl.pojo.TemporalGraphHead;
25  import org.gradoop.temporal.model.impl.pojo.TemporalVertex;
26  
27  /**
28   * Represents a temporal graph or graph collection using three separate datasets:
29   * <ol>
30   *   <li>a dataset containing the temporal graph heads which are the metadata of logical graphs</li>
31   *   <li>a dataset containing the temporal vertices</li>
32   *   <li>a dataset containing the temporal edges</li>
33   * </ol>
34   */
35  public class TemporalGVELayout implements LogicalGraphLayout<TemporalGraphHead, TemporalVertex, TemporalEdge>,
36    GraphCollectionLayout<TemporalGraphHead, TemporalVertex, TemporalEdge> {
37  
38    /**
39     * Graph data associated with the logical graphs in that collection.
40     */
41    private final DataSet<TemporalGraphHead> temporalGraphHeads;
42    /**
43     * DataSet containing temporal vertices associated with that graph.
44     */
45    private final DataSet<TemporalVertex> temporalVertices;
46    /**
47     * DataSet containing temporal edges associated with that graph.
48     */
49    private final DataSet<TemporalEdge> temporalEdges;
50  
51    /**
52     * Creates a new temporal layout holding the graph elements.
53     *
54     * @param temporalGraphHeads graph head dataset
55     * @param temporalVertices vertex dataset
56     * @param temporalEdges edge dataset
57     */
58    TemporalGVELayout(
59      DataSet<TemporalGraphHead> temporalGraphHeads,
60      DataSet<TemporalVertex> temporalVertices,
61      DataSet<TemporalEdge> temporalEdges) {
62  
63      this.temporalGraphHeads = temporalGraphHeads;
64      this.temporalVertices = temporalVertices;
65      this.temporalEdges = temporalEdges;
66    }
67  
68    @Override
69    public DataSet<TemporalVertex> getVertices() {
70      return this.temporalVertices;
71    }
72  
73    @Override
74    public DataSet<TemporalVertex> getVerticesByLabel(String label) {
75      return this.temporalVertices.filter(new ByLabel<>(label));
76    }
77  
78    @Override
79    public DataSet<TemporalEdge> getEdges() {
80      return this.temporalEdges;
81    }
82  
83    @Override
84    public DataSet<TemporalEdge> getEdgesByLabel(String label) {
85      return this.temporalEdges.filter(new ByLabel<>(label));
86    }
87  
88    @Override
89    public boolean isGVELayout() {
90      return true;
91    }
92  
93    @Override
94    public boolean isIndexedGVELayout() {
95      return false;
96    }
97  
98    @Override
99    public boolean isTransactionalLayout() {
100     return false;
101   }
102 
103   @Override
104   public DataSet<TemporalGraphHead> getGraphHead() {
105     return this.temporalGraphHeads;
106   }
107 
108   @Override
109   public DataSet<TemporalGraphHead> getGraphHeads() {
110     return this.temporalGraphHeads;
111   }
112 
113   @Override
114   public DataSet<TemporalGraphHead> getGraphHeadsByLabel(String label) {
115     return this.temporalGraphHeads.filter(new ByLabel<>(label));
116   }
117 
118   @Override
119   public DataSet<GraphTransaction> getGraphTransactions() {
120     throw new UnsupportedOperationException(
121       "Converting a temporal graph to graph transactions is not supported yet.");
122   }
123 }
