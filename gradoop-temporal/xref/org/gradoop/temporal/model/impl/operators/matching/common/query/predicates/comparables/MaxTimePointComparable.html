32  /**
33   * Wraps a {@link org.gradoop.gdl.model.comparables.time.MaxTimePoint}
34   */
35  public class MaxTimePointComparable extends TemporalComparable {
36  
37    /**
38     * The wrapped MaxTimePoint
39     */
40    private final MaxTimePoint maxTimePoint;
41  
42    /**
43     * Wrappers for the arguments
44     */
45    private final ArrayList<QueryComparable> args;
46  
47    /**
48     * Creates a new wrapper.
49     *
50     * @param maxTimePoint the wrapped MaxTimePoint.
51     */
52    public MaxTimePointComparable(MaxTimePoint maxTimePoint) {
53      this.maxTimePoint = maxTimePoint;
54      args = new ArrayList<>();
55      for (TimePoint arg : maxTimePoint.getArgs()) {
56        args.add(ComparableFactory.createComparableFrom(arg));
57      }
58    }
59  
60    @Override
61    public PropertyValue evaluate(Embedding embedding, EmbeddingMetaData metaData) {
62      long max = TemporalElement.DEFAULT_TIME_FROM;
63      for (QueryComparable arg : args) {
64        long argValue = arg.evaluate(embedding, metaData).getLong();
65        if (argValue > max) {
66          max = argValue;
67        }
68      }
69      return PropertyValue.create(max);
70    }
71  
72    @Override
73    public PropertyValue evaluate(GraphElement element) {
74      if (maxTimePoint.getVariables().size() > 1) {
75        throw new UnsupportedOperationException("can not evaluate an expression with >1 variable on" +
76          " a single GraphElement!");
77      }
78      long max = TemporalElement.DEFAULT_TIME_FROM;
79      for (QueryComparable arg : args) {
80        long argValue = arg.evaluate(element).getLong();
81        if (argValue > max) {
82          max = argValue;
83        }
84      }
85      return PropertyValue.create(max);
86    }
87  
88    @Override
89    public Set<String> getPropertyKeys(String variable) {
90      HashSet<String> keys = new HashSet<>();
91      for (QueryComparable comp: getArgs()) {
92        keys.addAll(comp.getPropertyKeys(variable));
93      }
94      return keys;
95    }
96  
97    @Override
98    public boolean equals(Object o) {
99      if (this == o) {
100       return true;
101     }
102     if (o == null || getClass() != o.getClass()) {
103       return false;
104     }
105 
106     MaxTimePointComparable that = (MaxTimePointComparable) o;
107     if (that.args.size() != args.size()) {
108       return false;
109     }
110 
111     for (QueryComparable arg : args) {
112       boolean foundMatch = false;
113       for (QueryComparable candidate : that.args) {
114         if (arg.equals(candidate)) {
115           foundMatch = true;
116           break;
117         }
118       }
119       if (!foundMatch) {
120         return false;
121       }
122     }
123     return true;
124   }
125 
126   @Override
127   public int hashCode() {
128     return maxTimePoint != null ? maxTimePoint.hashCode() : 0;
129   }
130 
131   @Override
132   public TimePoint getWrappedComparable() {
133     return maxTimePoint;
134   }
135 
136   public MaxTimePoint getMaxTimePoint() {
137     return maxTimePoint;
138   }
139 
140   public ArrayList<QueryComparable> getArgs() {
141     return args;
142   }
143 
144 }
