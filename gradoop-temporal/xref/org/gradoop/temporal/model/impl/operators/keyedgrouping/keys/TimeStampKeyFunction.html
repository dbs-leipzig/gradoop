16  package org.gradoop.temporal.model.impl.operators.keyedgrouping.keys;

18  import org.apache.flink.api.common.typeinfo.BasicTypeInfo;
19  import org.apache.flink.api.common.typeinfo.TypeInformation;

24  import org.gradoop.temporal.model.impl.pojo.TemporalElement;

26  import java.time.Instant;
27  import java.time.LocalDateTime;
28  import java.time.temporal.ChronoField;
29  import java.time.temporal.TemporalField;
30  import java.util.Objects;
31  
32  import static java.time.ZoneOffset.UTC;

34  /**
35   * A key function extracting a time stamp of a {@link TemporalElement}.<p>
36   * An optional {@link TemporalField} parameter can be given to this key function. In this case the time stamp
37   * will be converted to a date (with a time zone set to {@link java.time.ZoneOffset#UTC UTC}) and the field
38   * (e. g. {@link ChronoField#MONTH_OF_YEAR month}) will be extracted from that date.<p>
39   * When no {@link TemporalField} is given, the time stamp will be used as is, in milliseconds since unix
40   * epoch.<p>
41   * The final grouping key will be stored on the super element as a property with key
42   * {@code time_INTERVAL_FIELD_CALCULATEDFIELD} where {@code INTERVAL} is the {@link TimeDimension},
43   * {@code FIELD} the {@link TimeDimension.Field} and {@code CALCULATEDFIELD} the {@link TemporalField}
44   * extracted from the element. When no {@link TemporalField} is given, the property will just be called
45   * {@code time_INTERVAL_FIELD}.<p>
46   * When the extracted {@link TimeDimension.Field field} of the {@link TimeDimension} is set to a default
47   * value and a {@link TemporalField} was set, a default value ({@code -1}) will be returned instead.
48   * In this case no property will be set.
49   *
50   * @param <T> The type of the elements to group.
51   */
52  public class TimeStampKeyFunction<T extends TemporalElement> implements KeyFunctionWithDefaultValue<T, Long> {
53  
54    /**
55     * The time dimension of the temporal element to extract.
56     */
57    private final TimeDimension timeDimension;
58  
59    /**
60     * The field of that dimension to extract.
61     */
62    private final TimeDimension.Field timeDimensionField;
63  
64    /**
65     * The time field to calculate.
66     */
67    private final TemporalField fieldOfTimeStamp;
68  
69    /**
70     * The property key used to store the grouping key on the super-element.
71     */
72    private final String targetPropertyKey;
73  
74    /**
75     * Create a new instance of this grouping key function.
76     *
77     * @param timeDimension      The time dimension of the temporal element to consider.
78     * @param timeDimensionField The field of that time dimension to consider.
79     * @param fieldOfTimeStamp   The time field of that field to calculate. May be {@code null},
80     *                           in that case nothing will be calculated, the time stamp will be
81     *                           returned as is.
82     */
83    public TimeStampKeyFunction(TimeDimension timeDimension,
84      TimeDimension.Field timeDimensionField, TemporalField fieldOfTimeStamp) {
85      this.timeDimension = Objects.requireNonNull(timeDimension);
86      this.timeDimensionField = Objects.requireNonNull(timeDimensionField);
87      this.fieldOfTimeStamp = fieldOfTimeStamp;
88      this.targetPropertyKey = "time_" + timeDimension + "_" + timeDimensionField +
89        (fieldOfTimeStamp != null ? "_" + fieldOfTimeStamp : "");
90    }
91  
92    @Override
93    public Long getKey(T element) {
94      final Tuple2<Long, Long> interval;
95      switch (timeDimension) {
96      case TRANSACTION_TIME:
97        interval = element.getTransactionTime();
98        break;
99      case VALID_TIME:
100       interval = element.getValidTime();
101       break;
102     default:
103       throw new UnsupportedOperationException(
104         "Time interval not supported by this element: " + timeDimension);
105     }
106     final Long fieldValue;
107     switch (timeDimensionField) {
108     case FROM:
109       fieldValue = interval.f0;
110       if (fieldValue.equals(TemporalElement.DEFAULT_TIME_FROM) && (fieldOfTimeStamp != null)) {
111         return getDefaultKey();
112       }
113       break;
114     case TO:
115       fieldValue = interval.f1;
116       if (fieldValue.equals(TemporalElement.DEFAULT_TIME_TO) && (fieldOfTimeStamp != null)) {
117         return getDefaultKey();
118       }
119       break;
120     default:
121       throw new UnsupportedOperationException("Field is not supported: " + timeDimensionField);
122     }
123     if (fieldOfTimeStamp == null) {
124       return fieldValue;
125     }
126     final LocalDateTime date = LocalDateTime.ofInstant(Instant.ofEpochMilli(fieldValue), UTC);
127     return fieldOfTimeStamp.getFrom(date);
128   }
129 
130   @Override
131   public void addKeyToElement(T element, Object key) {
132     if (!(key instanceof Long)) {
133       throw new IllegalArgumentException("Invalid type for key: " + key.getClass().getSimpleName());
134     }
135     // Do not set the key if field extraction is enabled and the key is -1
136     if (fieldOfTimeStamp == null || !getDefaultKey().equals(key)) {
137       element.setProperty(targetPropertyKey, PropertyValue.create(key));
138     }
139   }
140 
141   @Override
142   public TypeInformation<Long> getType() {
143     return BasicTypeInfo.LONG_TYPE_INFO;
144   }
145 
146   @Override
147   public Long getDefaultKey() {
148     if (fieldOfTimeStamp == null) {
149       switch (timeDimensionField) {
150       case FROM:
151         return TemporalElement.DEFAULT_TIME_FROM;
152       case TO:
153         return TemporalElement.DEFAULT_TIME_TO;
154       default:
155         throw new UnsupportedOperationException("Field not supported: " + timeDimensionField);
156       }
157     }
158     return -1L;
159   }
