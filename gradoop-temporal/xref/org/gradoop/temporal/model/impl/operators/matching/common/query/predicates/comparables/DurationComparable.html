31  /**
32   * Wraps a {@link org.gradoop.gdl.model.comparables.time.Duration}
33   */
34  public class DurationComparable extends TemporalComparable {
35  
36    /**
37     * The wrapped duration
38     */
39    private final Duration duration;
40    /**
41     * the from value of the interval
42     */
43    private final TemporalComparable from;
44    /**
45     * the to value of the interval
46     */
47    private final TemporalComparable to;
48    /**
49     * Long describing the system time of the query
50     */
51    private final Long now;
52  
53    /**
54     * Creates a new wrapper
55     *
56     * @param duration the Duration to be wrapped
57     */
58    public DurationComparable(Duration duration) {
59      this(duration, new TimeLiteral("now"));
60    }
61  
62    /**
63     * Creates a new wrapper
64     *
65     * @param duration the duration to be wrapped
66     * @param now system time of the query
67     */
68    public DurationComparable(Duration duration, TimeLiteral now) {
69      this.duration = duration;
70      this.from = (TemporalComparable) ComparableFactory.createComparableFrom(duration.getFrom());
71      this.to = (TemporalComparable) ComparableFactory.createComparableFrom(duration.getTo());
72      this.now = now.getMilliseconds();
73    }
74  
75    public Duration getDuration() {
76      return duration;
77    }
78  
79    public TemporalComparable getFrom() {
80      return from;
81    }
82  
83    public TemporalComparable getTo() {
84      return to;
85    }
86  
87    public Long getNow() {
88      return now;
89    }
90  
91    @Override
92    public PropertyValue evaluate(Embedding embedding, EmbeddingMetaData metaData) {
93      long toLong = to.evaluate(embedding, metaData).getLong() == TemporalElement.DEFAULT_TIME_TO ?
94        now : to.evaluate(embedding, metaData).getLong();
95      return PropertyValue.create(toLong -
96        from.evaluate(embedding, metaData).getLong());
97    }
98  
99    @Override
100   public PropertyValue evaluate(GraphElement element) {
101     long toLong = to.evaluate(element).getLong() == TemporalElement.DEFAULT_TIME_TO ?
102       now : to.evaluate(element).getLong();
103     return PropertyValue.create(toLong -
104       from.evaluate(element).getLong());
105   }
106 
107   @Override
108   public Set<String> getPropertyKeys(String variable) {
109     HashSet<String> keys = new HashSet<>();
110     keys.addAll(from.getPropertyKeys(variable));
111     keys.addAll(to.getPropertyKeys(variable));
112     return keys;
113   }
114 
115 
116   @Override
117   public TimePoint getWrappedComparable() {
118     return duration;
119   }
120 
121   @Override
122   public boolean equals(Object o) {
123     if (this == o) {
124       return true;
125     }
126     if (o == null || getClass() != o.getClass()) {
127       return false;
128     }
129 
130     DurationComparable that = (DurationComparable) o;
131     return that.from.equals(from) && that.to.equals(to);
132   }
133 
134   @Override
135   public int hashCode() {
136     return duration.hashCode();
137   }
138 }
