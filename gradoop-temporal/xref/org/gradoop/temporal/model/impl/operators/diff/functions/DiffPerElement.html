16  package org.gradoop.temporal.model.impl.operators.diff.functions;
17  
18  import org.apache.flink.api.common.functions.FlatMapFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.api.java.tuple.Tuple2;
21  import org.apache.flink.util.Collector;
22  import org.gradoop.common.model.impl.properties.PropertyValue;
23  import org.gradoop.temporal.model.api.TimeDimension;
24  import org.gradoop.temporal.model.api.functions.TemporalPredicate;
25  import org.gradoop.temporal.model.impl.operators.diff.Diff;
26  import org.gradoop.temporal.model.impl.pojo.TemporalElement;
27  
28  import java.util.Objects;
29  
30  /**
31   * Evaluates two temporal predicates per element of a data set and sets the {@value Diff#PROPERTY_KEY}
32   * accordingly. Values not matching any of the two predicates will be discarded.
33   *
34   * @param <E> The element type.
35   */
36  @FunctionAnnotation.NonForwardedFields("properties")
37  public class DiffPerElement<E extends TemporalElement> implements FlatMapFunction<E, E> {
38  
39    /**
40     * The predicate used to determine the first snapshot.
41     */
42    private final TemporalPredicate first;
43  
44    /**
45     * The predicate used to determine the second snapshot.
46     */
47    private final TemporalPredicate second;
48  
49    /**
50     * Specifies the time dimension that will be considered by the operator.
51     */
52    private TimeDimension dimension;
53  
54    /**
55     * Create an instance of this function, setting the two temporal predicates used to determine the snapshots.
56     *
57     * @param first     The predicate used for the first snapshot.
58     * @param second    The predicate used for the second snapshot.
59     * @param dimension The time dimension that will be used.
60     */
61    public DiffPerElement(TemporalPredicate first, TemporalPredicate second, TimeDimension dimension) {
62      this.first = Objects.requireNonNull(first);
63      this.second = Objects.requireNonNull(second);
64      this.dimension = Objects.requireNonNull(dimension);
65    }
66  
67    @Override
68    public void flatMap(E value, Collector<E> out) {
69      Tuple2<Long, Long> timeValues;
70  
71      switch (dimension) {
72      case VALID_TIME:
73        timeValues = value.getValidTime();
74        break;
75      case TRANSACTION_TIME:
76        timeValues = value.getTransactionTime();
77        break;
78      default:
79        throw new IllegalArgumentException("Unknown dimension [" + dimension + "].");
80      }
81  
82      boolean inFirst = first.test(timeValues.f0, timeValues.f1);
83      boolean inSecond = second.test(timeValues.f0, timeValues.f1);
84      PropertyValue result;
85      if (inFirst && inSecond) {
86        result = Diff.VALUE_EQUAL;
87      } else if (inFirst) {
88        result = Diff.VALUE_REMOVED;
89      } else if (inSecond) {
90        result = Diff.VALUE_ADDED;
91      } else {
92        return;
93      }
94      value.setProperty(Diff.PROPERTY_KEY, result);
95      out.collect(value);
96    }
97  }
