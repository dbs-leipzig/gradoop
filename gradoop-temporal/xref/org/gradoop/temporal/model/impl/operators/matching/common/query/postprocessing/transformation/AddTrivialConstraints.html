43  /**
44   * Adds constraints a.tx_from<=a.tx_to and a.val_from<=a.val_to
45   * for every variable a.
46   * A constraint a.tx_from<=a.tx_to is only added, if a selector
47   * a.tx_from or a.tx_to is found within the singleton clauses of the CNF (analogously for val)
48   */
49  public class AddTrivialConstraints implements QueryTransformation {
50  
51    @Override
52    public CNF transformCNF(CNF cnf) throws QueryContradictoryException {
53      // determine all variables of interest for tx and val intervals
54      List<Set<String>> variableSets = getNecessaryFields(cnf);
55      Set<String> txVars = variableSets.get(0);
56      Set<String> valVars = variableSets.get(1);
57  
58      // add trivial constraints a.tx_from <= a.tx_to, if necessary
59      for (String variable : txVars) {
60        cnf = addSingletonClause(cnf, new Comparison(
61          new TimeSelector(variable, TimeSelector.TimeField.TX_FROM),
62          LTE, new TimeSelector(variable, TimeSelector.TimeField.TX_TO)
63        ));
64      }
65  
66      // add trivial constraints a.val_from <= a.val_to, if necessary
67      for (String variable : valVars) {
68        cnf = addSingletonClause(cnf, new Comparison(
69          new TimeSelector(variable, TimeSelector.TimeField.VAL_FROM),
70          LTE, new TimeSelector(variable, TimeSelector.TimeField.VAL_TO)
71        ));
72      }
73  
74      // add all "<" relations between literals that occur in singleton clauses
75      ArrayList<TimeLiteral> literals = new ArrayList<>(getNecessaryLiterals(cnf));
76      literals.sort(new Comparator<TimeLiteral>() {
77        @Override
78        public int compare(TimeLiteral timeLiteral, TimeLiteral t1) {
79          return Long.compare(timeLiteral.getMilliseconds(), t1.getMilliseconds());
80        }
81      });
82      for (int i = 0; i < literals.size(); i++) {
83        for (int j = i + 1; j < literals.size(); j++) {
84          if (literals.get(i).getMilliseconds() == literals.get(j).getMilliseconds()) {
85            continue;
86          }
87          cnf = addSingletonClause(cnf, new Comparison(
88            literals.get(i), LT, literals.get(j)
89          ));
90        }
91      }
92      return cnf;
93    }
94  
95    /**
96     * Adds a singleton clause to an existing CNF
97     *
98     * @param cnf        CNF
99     * @param comparison comparison that makes up the single clause to add
100    * @return CNF with singleton clause appended
101    */
102   private CNF addSingletonClause(CNF cnf, Comparison comparison) {
103     return cnf.and(new CNF(Collections.singletonList(
104       new CNFElement(Collections.singletonList(
105         new ComparisonExpression(comparison, new ComparableTPGMFactory())))
106     )));
107   }

109   /**
110    * Generates a set of variables for transaction times and valid times.
111    * A variable is included in the set for transaction times, iff
112    * a selector a.tx_(from/to) occurs in a singleton CNF clause.
113    * (analogously for valid times)
114    *
115    * @param cnf CNF
116    * @return list containing set of tx variables at index 0, set of
117    * val variables at index 1
118    */
119   private List<Set<String>> getNecessaryFields(CNF cnf) {
120     HashSet<String> txSet = new HashSet<>();
121     HashSet<String> valSet = new HashSet<>();
122     for (CNFElement clause : cnf.getPredicates()) {
123 
124       // only temporal singleton clauses are relevant
125       if (clause.size() != 1 || !ComparisonUtil.isTemporal(clause.getPredicates().get(0))) {
126         continue;
127       } else {
128 
129         // unwrap the comparison...
130         QueryComparable[] comparables = new QueryComparable[]{
131           clause.getPredicates().get(0).getLhs(),
132           clause.getPredicates().get(0).getRhs()
133         };
134         for (QueryComparable comparable : comparables) {
135           // ...and check it for time selectors
136           if (comparable instanceof TimeSelectorComparable) {
137             TimeSelector selector = (TimeSelector)
138               comparable.getWrappedComparable();
139             String variable = selector.getVariable();
140             // add the variable to the corresponding set(s)
141             TimeSelector.TimeField field = selector.getTimeProp();
142             if (field == TimeSelector.TimeField.TX_FROM ||
143               field == TimeSelector.TimeField.TX_TO) {
144               txSet.add(variable);
145             } else {
146               valSet.add(variable);
147             }
148           }
149         }
150       }
151     }
152     return new ArrayList<>(Arrays.asList(txSet, valSet));
153   }

155   /**
156    * Finds all time literals that are included in necessary comparisons, i.e.
157    * comparisons that make up a single cnf clause
158    *
159    * @param cnf CNF to search for necessary time literals
160    * @return list of time literals that are included in necessary comparisons
161    */
162   private Set<TimeLiteral> getNecessaryLiterals(CNF cnf) {
163     HashSet<TimeLiteral> literals = new HashSet<>();
164     for (CNFElement clause : cnf.getPredicates()) {
165       // only temporal singleton clauses are relevant
166       if (clause.size() != 1 || !(ComparisonUtil.isTemporal(clause.getPredicates().get(0)))) {
167         continue;
168       } else {
169         // check the clause for literals and add them to the set
170         ComparisonExpression comparison = clause.getPredicates().get(0);
171         if (comparison.getLhs() instanceof TimeLiteralComparable) {
172           literals.add((TimeLiteral) (comparison.getLhs()).getWrappedComparable());
173         }
174         if (comparison.getRhs() instanceof TimeLiteralComparable) {
175           literals.add((TimeLiteral) (comparison.getRhs()).getWrappedComparable());
176         }
177       }
178     }
179     return literals;
180   }
