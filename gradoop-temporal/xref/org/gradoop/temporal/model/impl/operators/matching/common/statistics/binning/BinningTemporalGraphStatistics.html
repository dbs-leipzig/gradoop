50  /**
51   * Wraps for a graph a set of {@link TemporalElementStats}, one for each combination
52   * (vertex / edge) x (label).
53   * Provides methods to estimate the probability that comparisons hold, based on
54   * the wrapped statistics.
55   */
56  @SuppressWarnings("CanBeFinal")
57  public class BinningTemporalGraphStatistics extends TemporalGraphStatistics implements Serializable {

59    /**
60     * A value to return for a probability close to 0
61     */
62    static final double VERY_LOW_PROB = 0.001;

64    /**
65     * A value to return for a probability close to 1
66     */
67    static final double VERY_HIGH_PROB = 0.999;

107   /**
108    * Creates new graph statistics
109    *
110    * @param vertexStats        list of vertex statistics (one element for each label)
111    * @param edgeStats          list of edge statistics (one element for each label)
112    * @param relevantProperties properties to consider
113    */
114   protected BinningTemporalGraphStatistics(List<TemporalElementStats> vertexStats,
115                                            List<TemporalElementStats> edgeStats,
116                                            Set<String> relevantProperties) {
117     this.vertexStats = new HashMap<>();
118     this.edgeStats = new HashMap<>();
119     this.vertexCount = 0L;
120     this.edgeCount = 0L;
121     for (TemporalElementStats edgeStat : edgeStats) {
122       this.edgeStats.put(edgeStat.getLabel(), edgeStat);
123       this.edgeCount += edgeStat.getElementCount();
124     }
125     for (TemporalElementStats vertexStat : vertexStats) {
126       this.vertexStats.put(vertexStat.getLabel(), vertexStat);
127       this.vertexCount += vertexStat.getElementCount();
128     }
129     this.relevantProperties = relevantProperties;
130     initDistinctVertices();
131   }

133   /**
134    * Initializes the statistics for distinct vertices
135    */
136   private void initDistinctVertices() {
137     this.distinctSourceCount = new HashMap<>();
138     this.distinctTargetCount = new HashMap<>();
139 
140     // for every type of edge...
141     for (Map.Entry<String, TemporalElementStats> entries : edgeStats.entrySet()) {
142       TemporalElementStats stats = edgeStats.get(entries.getKey());
143       List<TemporalElement> sample = stats.getSample();
144       int sampleSize = sample.size();
145       // ...extract source and target vertex IDs of the edges...
146       HashSet<GradoopId> sourceIds = new HashSet<>();
147       HashSet<GradoopId> targetIds = new HashSet<>();
148       for (TemporalElement edge : sample) {
149         sourceIds.add(((TemporalEdge) edge).getSourceId());
150         targetIds.add(((TemporalEdge) edge).getTargetId());
151       }
152       // ... and estimate distinct source and target vertices naively
153       long sourceEstimation = sourceIds.size() *
154         (stats.getElementCount() / sampleSize);
155       long targetEstimation = targetIds.size() *
156         (stats.getElementCount() / sampleSize);
157       distinctSourceCount.put(entries.getKey(), sourceEstimation);
158       distinctTargetCount.put(entries.getKey(), targetEstimation);
159     }
160   }

232   @Override
233   public double estimateTemporalProb(ElementType type1, Optional<String> label1,
234                                      TimeSelector.TimeField field1, Comparator comp, Long value) {
235     // no further estimations for = and !=
236     if (comp == EQ) {
237       return VERY_LOW_PROB;
238     } else if (comp == Comparator.NEQ) {
239       return VERY_HIGH_PROB;
240     }
241     // <, <=, >=, >
242     if (!(label1.isPresent())) {
243       return estimateTemporalProb(type1, field1, comp, value);
244     } else {
245       TemporalElementStats elementStats = (type1 == ElementType.VERTEX) ?
246         vertexStats.getOrDefault(label1.get(), null) :
247         edgeStats.getOrDefault(label1.get(), null);
248       // label not in keys
249       if (elementStats == null) {
250         return VERY_LOW_PROB;
251       }
252       Binning bins = getBins(elementStats, field1);
253       return estimateFromBins(bins, comp, value);
254     }
255   }

257   @Override
258   public double estimateTemporalProb(ElementType type1, Optional<String> label1,
259                                      TimeSelector.TimeField field1, Comparator comp, ElementType type2,
260                                      Optional<String> label2, TimeSelector.TimeField field2) {
261     // no further estimations for = and !=
262     if (comp == EQ) {
263       return VERY_LOW_PROB;
264     } else if (comp == Comparator.NEQ) {
265       return VERY_HIGH_PROB;
266     }
267     // stats for lhs (can be more than one, if no label is specified)
268     List<TemporalElementStats> statsLhs;
269     // same for rhs
270     List<TemporalElementStats> statsRhs;
271 
272     // determine relevant statistics for rhs and lhs
273     if (!(label1.isPresent())) {
274       statsLhs = type1 == ElementType.VERTEX ?
275         new ArrayList<>(vertexStats.values()) :
276         new ArrayList<>(edgeStats.values());
277     } else {
278       statsLhs = type1 == ElementType.VERTEX ?
279         new ArrayList<>(
280           Collections.singletonList(vertexStats.getOrDefault(label1.get(), null))) :
281         new ArrayList<>(
282           Collections.singletonList(edgeStats.getOrDefault(label1.get(), null)));
283       // label not in the DB
284       if (statsLhs.get(0) == null) {
285         return 0.;
286       }
287     }
288     // same for rhs
289     if (!(label2.isPresent())) {
290       statsRhs = type2 == ElementType.VERTEX ?
291         new ArrayList<>(vertexStats.values()) :
292         new ArrayList<>(edgeStats.values());
293     } else {
294       statsRhs = type2 == ElementType.VERTEX ?
295         new ArrayList<>(
296           Collections.singletonList(vertexStats.getOrDefault(label2.get(), null))) :
297         new ArrayList<>(
298           Collections.singletonList(edgeStats.getOrDefault(label2.get(), null)));
299       // label not in the DB
300       if (statsRhs.get(0) == null) {
301         return 0.;
302       }
303     }
304     return estimateTimeSelectorComparison(statsLhs, field1, comp, statsRhs, field2);
305   }

307   /**
308    * Estimates the probability that a comparison between two time selectors holds
309    * Both selectors are described by the statistics that are relevant for them
310    * (one statistic if label was specified, more than one if not).
311    *
312    * @param statsLhs   list of statistics for the lhs
313    * @param fieldLhs   lhs time field (tx_from, tx_to, val_from, val_to)
314    * @param comparator comparator of the comparison
315    * @param statsRhs   list of statistics for the rhs
316    * @param fieldRhs   rhs time field
317    * @return estimation of the probability that the comparison holds
318    */
319   private double estimateTimeSelectorComparison(List<TemporalElementStats> statsLhs,
320                                                 TimeSelector.TimeField fieldLhs,
321                                                 Comparator comparator, List<TemporalElementStats> statsRhs,
322                                                 TimeSelector.TimeField fieldRhs) {
323     // count overall number of elements for both sides
324     long lhsOverallCount = statsLhs.stream()
325       .map(TemporalElementStats::getElementCount)
326       .reduce(Long::sum)
327       .orElse(0L);
328     long rhsOverallCount = statsRhs.stream()
329       .map(TemporalElementStats::getElementCount)
330       .reduce(Long::sum)
331       .orElse(0L);
332     double sum = 0.;
333 
334     for (TemporalElementStats lhs : statsLhs) {
335       double lhsWeight = (double) lhs.getElementCount() / lhsOverallCount;
336       for (TemporalElementStats rhs : statsRhs) {
337         double rhsWeight = (double) rhs.getElementCount() / rhsOverallCount;
338         sum += lhsWeight * rhsWeight *
339           estimateTimeSelectorComparison(lhs, fieldLhs, comparator, rhs, fieldRhs);
340       }
341     }
342     return sum;
343   }

345   /**
346    * Estimates the probability that a comparison between two time selectors holds
347    * Both selectors are described by the statistics that are relevant for them
348    * (one statistic for both lhs and rhs).
349    *
350    * @param statsLhs   statistics for the lhs
351    * @param fieldLhs   lhs time field (tx_from, tx_to, val_from, val_to)
352    * @param comparator comparator of the comparison
353    * @param statsRhs   statistics for the rhs
354    * @param fieldRhs   rhs time field
355    * @return estimation of the probability that the comparison holds
356    */
357   private double estimateTimeSelectorComparison(TemporalElementStats statsLhs,
358                                                 TimeSelector.TimeField fieldLhs,
359                                                 Comparator comparator, TemporalElementStats statsRhs,
360                                                 TimeSelector.TimeField fieldRhs) {
361 
362     double sum = 0.;
363 
364     // first, compare without considering Long.MIN values, TemporalElement.DEFAULT_TIME_TO values
365 
366     double[] lhs = statsLhs.getTemporalPropertyStats(fieldLhs);
367     double[] rhs = statsRhs.getTemporalPropertyStats(fieldRhs);
368     double lhsMean = lhs[0];
369     double lhsVariance = lhs[1];
370     double lhsExtreme = lhs[2];
371     double rhsMean = rhs[0];
372     double rhsVariance = rhs[1];
373     double rhsExtreme = rhs[2];
374     // probability that both lhs and rhs not TemporalElement.DEFAULT_TIME_FROM or
375     // TemporalElement.DEFAULT_TIME_TO
376     double probBothNotExtreme = (1 - lhsExtreme) * (1 - rhsExtreme);
377     // distribution for difference between lhs and rhs values
378     NormalDistribution diffDist = new NormalDistribution(lhsMean - rhsMean,
379       Math.max(lhsVariance + rhsVariance, VERY_LOW_PROB));
380 
381     // P(lhs < rhs)
382     double probNotExtremeLTE = diffDist.cumulativeProbability(0.);
383     // P(lhs = rhs)
384     double probNotExtremeEQ = diffDist.probability(0.);
385     // P(lhs comp rhs)
386     double probCompNotExtreme = 0.;
387 
388     if (comparator == LTE) {
389       probCompNotExtreme = probNotExtremeLTE;
390     } else if (comparator == LT) {
391       probCompNotExtreme = probNotExtremeLTE - probNotExtremeEQ;
392     } else if (comparator == EQ) {
393       probCompNotExtreme = probNotExtremeEQ;
394     } else if (comparator == NEQ) {
395       probCompNotExtreme = 1. - probNotExtremeEQ;
396     } else if (comparator == GTE) {
397       probCompNotExtreme = 1. - (probNotExtremeLTE - probNotExtremeEQ);
398     } else if (comparator == GT) {
399       probCompNotExtreme = 1. - probNotExtremeLTE;
400     }
401 
402     sum += probBothNotExtreme * probCompNotExtreme;
403 
404     // also consider cases in which one or both sides have extreme values
405     double probCompExtreme = 0.;
406     // lhs from
407     if (fieldLhs == TX_FROM || fieldLhs == VAL_FROM) {
408       // rhs from
409       if (fieldRhs == TX_FROM || fieldRhs == VAL_FROM) {
410         if (comparator == EQ) {
411           probCompExtreme = lhsExtreme * rhsExtreme;
412         } else if (comparator == NEQ) {
413           probCompExtreme = (1 - lhsExtreme) * rhsExtreme + lhsExtreme * (1 - rhsExtreme);
414         } else if (comparator == LTE) {
415           probCompExtreme = lhsExtreme;
416         } else if (comparator == LT) {
417           probCompExtreme = lhsExtreme * (1 - rhsExtreme);
418         } else if (comparator == GTE) {
419           probCompExtreme = rhsExtreme;
420         } else if (comparator == GT) {
421           probCompExtreme = (1 - lhsExtreme) * rhsExtreme;
422         }
423       } else {
424         // rhs to
425         if (comparator == NEQ || comparator == LT || comparator == LTE) {
426           probCompExtreme = lhsExtreme * rhsExtreme + (1 - lhsExtreme) * rhsExtreme +
427             lhsExtreme * (1 - rhsExtreme);
428         } //for all other comparators 0, if lhs and/or rhs extreme
429       }
430     } else { // lhs to
431       // rhs from
432       if (fieldRhs == TX_FROM || fieldRhs == VAL_FROM) {
433         if (comparator == NEQ || comparator == GT || comparator == GTE) {
434           probCompExtreme = lhsExtreme * rhsExtreme + (1 - lhsExtreme) * rhsExtreme +
435             lhsExtreme * (1 - rhsExtreme);
436         } // for all other comparators 0, if lhs and/or rhs extreme
437       } else {
438         // rhs to
439         if (comparator == EQ) {
440           probCompExtreme = lhsExtreme * rhsExtreme;
441         } else if (comparator == NEQ) {
442           probCompExtreme = (1 - lhsExtreme) * rhsExtreme + lhsExtreme * (1 - rhsExtreme);
443         } else if (comparator == LTE) {
444           probCompExtreme = rhsExtreme;
445         } else if (comparator == LT) {
446           probCompExtreme = (1 - lhsExtreme) * rhsExtreme;
447         } else if (comparator == GTE) {
448           probCompExtreme = lhsExtreme;
449         } else if (comparator == GT) {
450           probCompExtreme = lhsExtreme * (1 - rhsExtreme);
451         }
452       }
453     }
454 
455     sum += probCompExtreme;
456 
457     return sum;
458   }

461   @Override
462   public double estimateDurationProb(ElementType type, Optional<String> label, Comparator comp,
463                                      boolean transaction, Long value) {
464     Map<String, TemporalElementStats> statsMap = (type == ElementType.VERTEX) ? vertexStats : edgeStats;
465     List<TemporalElementStats> relevantStats = label.isPresent() ?
466       new ArrayList<>(Collections.singletonList(statsMap.get(label.get()))) :
467       new ArrayList<>(statsMap.values());
468 
469     double sum = 0.;
470     long numElements = 0;
471     for (TemporalElementStats stat : relevantStats) {
472       numElements += stat.getElementCount();
473     }
474 
475     for (TemporalElementStats stat : relevantStats) {
476 
477       double[] durationStats = transaction ?
478         stat.getTxDurationStats() : stat.getValDurationStats();
479       NormalDistribution durationDist = new NormalDistribution(
480         durationStats[0], Math.max(Math.sqrt(durationStats[1]), VERY_LOW_PROB));
481       double estimation = 0.;
482 
483       if (comp == EQ) {
484         estimation = durationDist.density(value);
485       } else if (comp == NEQ) {
486         estimation = 1. - durationDist.density(value);
487       } else if (comp == LTE) {
488         estimation = durationDist.cumulativeProbability(value);
489       } else if (comp == LT) {
490         estimation = durationDist.cumulativeProbability(value) -
491           durationDist.density(value);
492       } else if (comp == GT) {
493         estimation = 1 - durationDist.cumulativeProbability(value);
494       } else if (comp == GTE) {
495         estimation = 1 - (durationDist.cumulativeProbability(value)) +
496           durationDist.density(value);
497       }
498       sum += estimation * ((double) stat.getElementCount() / numElements);
499     }
500     return sum;
501   }

503   @Override
504   public double estimateDurationProb(ElementType type1, Optional<String> label1, boolean transaction1,
505                                      Comparator comp, ElementType type2, Optional<String> label2,
506                                      boolean transaction2) {
507     Map<String, TemporalElementStats> statsMapLhs = (type1 == ElementType.VERTEX) ? vertexStats : edgeStats;
508     List<TemporalElementStats> relevantStatsLhs = label1.isPresent() ?
509       new ArrayList<>(Collections.singletonList(statsMapLhs.get(label1.get()))) :
510       new ArrayList<>(statsMapLhs.values());
511 
512     Map<String, TemporalElementStats> statsMapRhs = (type2 == ElementType.VERTEX) ? vertexStats : edgeStats;
513     List<TemporalElementStats> relevantStatsRhs = label2.isPresent() ?
514       new ArrayList<>(Collections.singletonList(statsMapRhs.get(label2.get()))) :
515       new ArrayList<>(statsMapRhs.values());
516 
517     double sum = 0.;
518     long numElements = 0;
519     for (TemporalElementStats statL : relevantStatsLhs) {
520       for (TemporalElementStats statR : relevantStatsRhs) {
521         numElements += statL.getElementCount() * statR.getElementCount();
522       }
523     }
524 
525     for (TemporalElementStats statLhs : relevantStatsLhs) {
526       long countLhs = statLhs.getElementCount();
527 
528       for (TemporalElementStats statRhs : relevantStatsRhs) {
529         long countRhs = statRhs.getElementCount();
530 
531         double[] durationStatsLhs = transaction1 ?
532           statLhs.getTxDurationStats() : statLhs.getValDurationStats();
533         double[] durationStatsRhs = transaction2 ?
534           statRhs.getTxDurationStats() : statRhs.getValDurationStats();
535 
536         // distribution of differences (LHS - RHS)
537         NormalDistribution diffDist = new NormalDistribution(
538           durationStatsLhs[0] - durationStatsRhs[0],
539           Math.max(Math.sqrt(durationStatsLhs[1] + durationStatsRhs[1]), VERY_LOW_PROB));
540 
541         double estimation = 0.;
542 
543         if (comp == EQ) {
544           estimation = diffDist.density(0);
545         } else if (comp == NEQ) {
546           estimation = 1. - diffDist.density(0);
547         } else if (comp == LTE) {
548           estimation = diffDist.cumulativeProbability(0);
549         } else if (comp == LT) {
550           estimation = diffDist.cumulativeProbability(0) -
551             diffDist.density(0);
552         } else if (comp == GT) {
553           estimation = 1 - diffDist.cumulativeProbability(0);
554         } else if (comp == GTE) {
555           estimation = 1 - (diffDist.cumulativeProbability(0)) +
556             diffDist.density(0);
557         }
558 
559         sum += estimation * ((double) countLhs * (double) countRhs) / (double) numElements;
560       }
561     }
562     return sum;
563   }

566   public double estimatePropertyProb(ElementType type, Optional<String> label, String property,

568     Map<String, TemporalElementStats> statsMap = type == ElementType.VERTEX ?
569       vertexStats : edgeStats;

571     List<TemporalElementStats> relevantStats;
572     if (label.isPresent()) {
573       if (statsMap.containsKey(label.get())) {
574         relevantStats = new ArrayList<>(Collections.singletonList(statsMap.get(label.get())));
575       } else {
576         return VERY_LOW_PROB;
577       }
578     } else {
579       relevantStats = new ArrayList<>(statsMap.values());
580     }
581 
582 
583     if (value.isNumber()) {
584       return estimateNumericalPropertyProb(relevantStats, property, comp, value);
585     } else {
586       return estimateCategoricalPropertyProb(relevantStats, property, comp, value);
587     }

591   @Override
592   public double estimatePropertyProb(ElementType type1, Optional<String> label1, String property1,
593                                      Comparator comp,
594                                      ElementType type2, Optional<String> label2, String property2) {
595     Map<String, TemporalElementStats> statsMap1 = type1 == ElementType.VERTEX ? vertexStats : edgeStats;
596     List<TemporalElementStats> relevantStats1 = label1.isPresent() ?
597       new ArrayList<>(Collections.singletonList(statsMap1.get(label1.get()))) :
598       new ArrayList<>(statsMap1.values());
599     Map<String, TemporalElementStats> statsMap2 = type2 == ElementType.VERTEX ? vertexStats : edgeStats;
600     List<TemporalElementStats> relevantStats2 = label2.isPresent() ?
601       new ArrayList<>(Collections.singletonList(statsMap2.get(label2.get()))) :
602       new ArrayList<>(statsMap1.values());
603 
604     // check if numerical or categorical
605     boolean numerical1 = false;
606     boolean numerical2 = false;
607     for (TemporalElementStats s : relevantStats1) {
608       if (s.getNumericalPropertyStatsEstimation().containsKey(property1)) {
609         numerical1 = true;
610       }
611     }
612     for (TemporalElementStats s : relevantStats2) {
613       if (s.getNumericalPropertyStatsEstimation().containsKey(property2)) {
614         numerical2 = true;
615       }
616     }
617 
618     if (isPropertyRelevant(property1) != isPropertyRelevant(property2)) {
619       return 0.5;
620     }
621 
622     if (numerical1 != numerical2) {
623       // numerical properties can not be compared to categorical.
624       return 0.;
625     } else if (!numerical1) {
626       return estimateCategoricalPropertyProb(relevantStats1, property1, comp,
627         relevantStats2, property2);
628     } else {
629       return estimateNumericalPropertyProb(relevantStats1, property1, comp,
630         relevantStats2, property2);
631     }
632   }

635    * Estimates the probability that a comparison on numerical property data holds.
636    * RHS and LHS elements are described by a list of element statistics
637    * Note that the method assumes lhs and rhs to be independent, i.e. a call
638    * does not yield a reasonable value for comparing the same property of the
639    * exact same element.
640    *
641    * @param relevantStats1 statistics for lhs elements
642    * @param property1      lhs property to compare
643    * @param comp           comparator
644    * @param relevantStats2 statistics for the rhs elements
645    * @param property2      rhs property to compare
646    * @return estimation of probability that the comparison holds
647    */
648   private double estimateNumericalPropertyProb(List<TemporalElementStats> relevantStats1, String property1,
649                                                Comparator comp, List<TemporalElementStats> relevantStats2,
650                                                String property2) {
651     long count1 = relevantStats1.stream()
652       .map(TemporalElementStats::getElementCount)
653       .reduce(0L, Long::sum);
654     long count2 = relevantStats2.stream()
655       .map(TemporalElementStats::getElementCount)
656       .reduce(0L, Long::sum);
657     long totalCount = count1 * count2;
658 
659     double outerSum = 0.;
660     for (TemporalElementStats s1 : relevantStats1) {
661       double innerSum = 0.;
662       for (TemporalElementStats s2 : relevantStats2) {
663         double prob = estimateNumericalPropertyProb(s1, property1, comp, s2, property2);
664         double weight = ((double) s1.getElementCount() * s2.getElementCount()) /
665           totalCount;
666         innerSum += prob * weight;
667       }
668       outerSum += innerSum;
669     }
670     return outerSum;

673   /**
674    * Estimates the probability that a comparison between numerical properties holds.
675    * LHS and RHS are described by one statistics element. They are assumed to be
676    * independent, otherwise the estimation may not be reasonable.
677    *
678    * @param stats1    lhs statistics
679    * @param property1 lhs property to compare
680    * @param comp      comparator
681    * @param stats2    rhs statistics
682    * @param property2 rhs property to compare
683    * @return estimation of probability that the comparison holds
684    */
685   private double estimateNumericalPropertyProb(TemporalElementStats stats1, String property1,
686                                                Comparator comp,
687                                                TemporalElementStats stats2, String property2) {
688     Map<String, Double[]> map1 = stats1.getNumericalPropertyStatsEstimation();
689     Map<String, Double[]> map2 = stats2.getNumericalPropertyStatsEstimation();
690     Double[] propStats1 = map1.getOrDefault(property1, null);
691     Double[] propStats2 = map2.getOrDefault(property2, null);
692     if (propStats1 == null || propStats2 == null) {
693       return 0.0001;
694     }
695     double occurrence = stats1.getNumericalOccurrenceEstimation()
696       .getOrDefault(property1, 0.) *
697       stats2.getNumericalOccurrenceEstimation().getOrDefault(property2, 0.);
698     // assuming both properties are normally distributed,
699     // their difference is also normally distributed
700     NormalDistribution differenceDist = new NormalDistribution(
701       propStats1[0] - propStats2[0],
702       Math.max(Math.sqrt(propStats1[1] + propStats2[1]), VERY_LOW_PROB));
703 
704     if (comp == EQ) {
705       return occurrence * differenceDist.density(0.);
706     } else if (comp == NEQ) {
707       return occurrence * (1. - differenceDist.density(0.));
708     } else if (comp == LTE) {
709       return occurrence * (differenceDist.cumulativeProbability(0.));
710     } else if (comp == LT) {
711       return occurrence * (differenceDist.cumulativeProbability(0.) -
712         differenceDist.density(0.));
713     } else if (comp == GTE) {
714       return occurrence * (1. - differenceDist.cumulativeProbability(0.) +
715         differenceDist.density(0.));
716     } else {
717       //GT
718       return occurrence * (1. - differenceDist.cumulativeProbability(0.));
719     }
720   }

761    * Estimates the probability that a comparison between categorical properties holds.
762    * LHS and RHS are both described by a statistics element.
763    *
764    * @param stats1    lhs statistics
765    * @param property1 lhs property to compare
766    * @param comp      comparator
767    * @param stats2    rhs statistics
768    * @param property2 rhs property to compare
769    * @return estimation of the probability that the comparison holds
770    */
771   private double estimateCategoricalPropertyProb(TemporalElementStats stats1, String property1,
772                                                  Comparator comp,
773                                                  TemporalElementStats stats2, String property2) {
774     Map<String, Map<PropertyValue, Double>> map1 =
775       stats1.getCategoricalSelectivityEstimation();
776     Map<String, Map<PropertyValue, Double>> map2 =
777       stats2.getCategoricalSelectivityEstimation();
778     Map<PropertyValue, Double> propStats1 = map1.getOrDefault(property1, null);
779     Map<PropertyValue, Double> propStats2 = map2.getOrDefault(property2, null);
780     // property not sampled or not considered
781     if (propStats1 == null || propStats2 == null) {
782       // property not considered => return 0.5
783       if (!isPropertyRelevant(property1) && !isPropertyRelevant(property2)) {
784         return 0.5;
785       } else {
786         // property not sampled => very rare => return very small value
787         return 0.0001;
788       }
789     }
790     if (comp == EQ || comp == NEQ) {
791       double sum = 0.;
792       for (Map.Entry<PropertyValue, Double> entry1 : propStats1.entrySet()) {
793         double val1Selectivity = entry1.getValue();
794         for (Map.Entry<PropertyValue, Double> entry2 : propStats2.entrySet()) {
795           if (entry1.getKey().equals(entry2.getKey())) {
796             double val2Selectivity = entry2.getValue();
797             sum += val1Selectivity * val2Selectivity;
798           }
799         }
800       }
801       return comp == EQ ? sum : 1. - sum;
802     } else {
803       // shouldn't happen, categorical variables can only be compared with EQ or NEQ
804       return 0.;
805     }
806   }

809    * Estimates the probability that a comparison between a categorical property value
810    * and a constant holds
811    *
812    * @param relevantStats statistics for the lhs (the elements that can have the
813    *                      property in question)
814    * @param property      the property
815    * @param comp          comparator of the comparison
816    * @param value         rhs of the comparison
817    * @return estimation of the probability that the comparison holds
818    */
819   private double estimateCategoricalPropertyProb(List<TemporalElementStats> relevantStats, String property,
820                                                  Comparator comp, PropertyValue value) {
821     if (comp != EQ && comp != NEQ) {
822       return 0.;
823     }
824     long overallCount = 0L;
825     boolean found = false;
826     for (TemporalElementStats stat : relevantStats) {
827       overallCount += stat.getElementCount();
828       if (stat.getCategoricalSelectivityEstimation().containsKey(property)) {
829         found = true;
830       }
831     }
832     // property not sampled or not considered
833     if (!found) {
834       // not relevant/considered => simply return 0.5
835       if (!isPropertyRelevant(property)) {
836         return 0.5;
837       } else {
838         // not sampled => very rare or not even there => return very small value
839         return 0.0001;
840       }
841     }
842 
843     double prob = 0.;
844     for (TemporalElementStats stat : relevantStats) {
845       if (!stat.getCategoricalSelectivityEstimation().containsKey(property)) {
846         // not sampled ? ....
847         double factor = 0.0001;
848         // ...or just excluded/not relevant?
849         if (!isPropertyRelevant(property)) {
850           factor = 0.5;
851         }
852         prob += factor * ((double) stat.getElementCount() / overallCount);
853         continue;
854       }
855       prob += stat.getCategoricalSelectivityEstimation()
856         .get(property)
857         .getOrDefault(value, 0.0001) *
858         ((double) stat.getElementCount() / overallCount);
859     }
860     if (comp == EQ) {
861       return prob;
862     } else {
863       //NEQ
864       return 1 - prob;
865     }

899    * @param comp          comparator of the comparison

905     // property not sampled or not considered
906     if (!stat.getNumericalPropertyStatsEstimation().containsKey(property)) {
907       // if property not excluded, return very small value
908       // if excluded, it is irrelevant, return 0.5
909       return isPropertyRelevant(property) ? 0.0001 : 0.5;
910     }
911     Double[] propertyStats = stat.getNumericalPropertyStatsEstimation().get(property);

913       Math.max(Math.sqrt(propertyStats[1]), VERY_LOW_PROB));

915     double occurenceProb = stat.getNumericalOccurrenceEstimation().get(property);

918     } else if (comp == NEQ) {

958   /**
959    * Estimates the probability that a comparison holds using bin statistics
960    *
961    * @param bins  the bins that describe the lhs of the comparison
962    * @param comp  comparator, not EQ or NEQ
963    * @param value rhs of the comparison
964    * @return estimation of probability that the comparison holds
965    */
966   private double estimateFromBins(Binning bins, Comparator comp, Long value) {
967     int[] valueBins = findBins(bins, value);
968     int numBins = bins.getBins().length;
969 
970     if (comp == Comparator.LT) {
971       return ((double) valueBins[0]) / ((double) numBins);
972     } else if (comp == LTE) {
973       return ((double) valueBins[1]) / ((double) numBins);
974     } else if (comp == Comparator.GT) {
975       // 1 - LTE estimation
976       return 1 - ((double) valueBins[1]) / ((double) numBins);
977     } else {
978       // 1- LT estimation for GTE
979       return 1 - ((double) valueBins[0]) / ((double) numBins);
980     }
981   }

995   /**
996    * Finds the bins in which a value falls
997    *
998    * @param bins  bins to search for the value
999    * @param value value to search
1000    * @return integer array containing the lowest suitable bin index and the highest.
1001    * A value can fall into more than one bin in a row, as the bins are equal width.
1002    * E.g., binning a sequence {@code 1,2,2,3} into 4 bins would yield 2 suitable bins
1003    * for the value {@code 2}
1004    */
1005   private int[] findBins(Binning bins, Long value) {
1006     Long[] binReps = bins.getBins();
1007     int bin = Arrays.binarySearch(binReps, value);
1008     if (bin < 0) {
1009       bin = -bin;
1010     }
1011     bin = Math.min(bin, binReps.length - 1);
1012     // bins are equal size, so the same bin could occur more than once
1013     // the lowest bin in which value fits
1014     int minBin = bin;
1015     // the "highest" bin in which value fits
1016     int maxBin = bin;
1017     while (minBin >= 0 && matchesBin(binReps, value, minBin)) {
1018       minBin--;
1019     }
1020     minBin++;
1021     while (maxBin < binReps.length && matchesBin(binReps, value, maxBin)) {
1022       maxBin++;
1023     }
1024     maxBin--;
1025     return new int[]{minBin, maxBin};
1026   }

1028   /**
1029    * Checks whether a suitable bin was found for a value.
1030    *
1031    * @param arr   the array of bins
1032    * @param value the value to search
1033    * @param index index of the potentially suitable bin
1034    * @return true iff index denotes a suitable bin for value
1035    */
1036   private boolean matchesBin(Long[] arr, Long value, int index) {
1037     if (arr.length == 1) {
1038       return true;
1039     }
1040     if (index == arr.length - 1) {
1041       return value >= arr[index];
1042     } else if (index == 0) {
1043       return value < arr[index + 1];
1044     } else {
1045       boolean lowerCond = value >= arr[index];
1046       boolean upperCond = value.equals(TemporalElement.DEFAULT_TIME_TO) ? value <= arr[index + 1] :
1047         value < arr[index + 1];
1048       return lowerCond && upperCond;
1049     }
1050   }
