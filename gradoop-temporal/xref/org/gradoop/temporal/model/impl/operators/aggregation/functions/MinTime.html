18  import org.gradoop.common.model.impl.properties.PropertyValue;
19  import org.gradoop.common.model.impl.properties.PropertyValueUtils;

23  /**
24   * Base class for calculating the minimum of a field of a time-interval for temporal
25   * elements. If a valid time attribute is used for aggregation, this function ignores the default
26   * values ({@link TemporalElement#DEFAULT_TIME_FROM} and {@link TemporalElement#DEFAULT_TIME_TO})
27   * and handles it the same way as {@code null}.
28   */
29  public class MinTime extends AbstractTimeAggregateFunction {
30  
31    /**
32     * Sets attributes used to initialize this aggregate function.
33     *
34     * @param aggregatePropertyKey The aggregate property key.
35     * @param dimension            The time dimension to consider.
36     * @param field                The field of the time-interval to consider.
37     */
38    public MinTime(String aggregatePropertyKey, TimeDimension dimension, TimeDimension.Field field) {
39      super(aggregatePropertyKey, dimension, field);
40    }
41  
42    @Override
43    public PropertyValue aggregate(PropertyValue aggregate, PropertyValue increment) {
44      return applyAggregateWithDefaults(aggregate, increment, PropertyValueUtils.Numeric::min);
45    }
46  }
