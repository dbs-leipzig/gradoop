26  import org.gradoop.temporal.model.impl.pojo.TemporalVertex;

28  /**
29   * Verifies the edge set of a graph. This removes dangling edges (in the same way as
30   * {@link org.gradoop.flink.model.impl.operators.verify.Verify} would) and updates the remaining edges so that
31   * the edge is only valid when both its source- and target-vertex are valid.
32   */
33  public class VerifyAndUpdateEdgeValidity implements UnaryBaseGraphToBaseGraphOperator<TemporalGraph> {
34  
35    @Override
36    public TemporalGraph execute(TemporalGraph graph) {
37      final DataSet<TemporalVertex> vertices = graph.getVertices();
38      final DataSet<TemporalEdge> verifiedEdges = graph.getEdges()
39        .join(vertices)
40        .where(new SourceId<>())
41        .equalTo(new Id<>())
42        .with(new UpdateEdgeValidity())
43        .name("Verify and update edges (1/2)")
44        .join(vertices)
45        .where(new TargetId<>())
46        .equalTo(new Id<>())
47        .with(new UpdateEdgeValidity())
48        .name("Verify and update edges (2/2)");
49      return graph.getFactory().fromDataSets(graph.getGraphHead(), vertices, verifiedEdges);
50    }
51  }
