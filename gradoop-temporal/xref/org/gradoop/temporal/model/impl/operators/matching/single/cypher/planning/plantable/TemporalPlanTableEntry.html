16  package org.gradoop.temporal.model.impl.operators.matching.single.cypher.planning.plantable;

18  import org.apache.commons.lang3.tuple.Pair;
19  import org.gradoop.flink.model.impl.operators.matching.common.query.predicates.CNF;
20  import org.gradoop.flink.model.impl.operators.matching.single.cypher.planning.plantable.PlanTableEntry;
21  import org.gradoop.flink.model.impl.operators.matching.single.cypher.planning.queryplan.QueryPlan;
22  import org.gradoop.temporal.model.impl.operators.matching.single.cypher.planning.estimation.TemporalQueryPlanEstimator;
23  
24  import java.util.HashSet;
25  import java.util.Set;
26  import java.util.stream.Collectors;

28  /**
29   * Represents a query plan and additional meta data. Plan table entries are managed in a
30   * {@link org.gradoop.flink.model.impl.operators.matching.single.cypher.planning.plantable.PlanTable}.
31   */
32  public class TemporalPlanTableEntry extends PlanTableEntry {
33    /**
34     * The estimator containing the query plan.
35     */
36    private final TemporalQueryPlanEstimator estimator;
37    /**
38     * Estimated cardinality of the query result as computed by the estimator.
39     */
40    private long estimatedCardinality = -1;
41  
42    /**
43     * Creates a new plan table entry.
44     *
45     * @param type          the partial match type this entry represents
46     * @param processedVars processed query variables
47     * @param predicates    predicates not covered by this plan
48     * @param estimator     cardinality estimator
49     */
50    public TemporalPlanTableEntry(Type type, Set<String> processedVars, CNF predicates,
51                                  TemporalQueryPlanEstimator estimator) {
52      super(type, processedVars, predicates, null);
53      this.estimator = estimator;
54    }
55  
56    @Override
57    public Set<String> getAllVariables() {
58      return new HashSet<>(estimator.getQueryPlan().getRoot().getEmbeddingMetaData()
59        .getVariables());
60    }
61  
62    @Override
63    public Set<String> getAttributedVariables() {
64      return new HashSet<>(estimator.getQueryPlan().getRoot().getEmbeddingMetaData()
65        .getVariablesWithProperties());
66    }
67  
68    /**
69     * Returns all (variable,key) pairs needed for further query processing.
70     *
71     * @return (variable, key) pairs needed for further processing
72     */
73    public Set<Pair<String, String>> getProjectionPairs() {
74      return getPredicates().getVariables().stream()
75        .flatMap(var -> getPredicates().getPropertyKeys(var).stream().map(key -> Pair.of(var, key)))
76        .collect(Collectors.toSet());
77    }
78  
79    @Override
80    public long getEstimatedCardinality() {
81      if (estimatedCardinality == -1) {
82        estimatedCardinality = estimator.getCardinality();
83      }
84      return estimatedCardinality;
85    }
86  
87    @Override
88    public QueryPlan getQueryPlan() {
89      return estimator.getQueryPlan();
90    }
91  }
