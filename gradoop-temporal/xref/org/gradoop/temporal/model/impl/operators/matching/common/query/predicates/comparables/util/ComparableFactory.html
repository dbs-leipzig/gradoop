39  /**
40   * Class for creating a {@link QueryComparable} wrapper for a GDL {@link ComparableExpression}
41   */
42  public class ComparableFactory {
43  
44    /**
45     * Create a wrapper for a GDL comparable
46     *
47     * @param expression the GDL element to wrap
48     * @return wrapper for expression
49     * @throws IllegalArgumentException if expression is no GDL ComparableExpression
50     */
51    public static QueryComparable createComparableFrom(ComparableExpression expression) {
52      if (expression.getClass() == Literal.class) {
53        return new LiteralComparable((Literal) expression);
54      } else if (expression.getClass() == PropertySelector.class) {
55        return new PropertySelectorComparable((PropertySelector) expression);
56      } else if (expression.getClass() == ElementSelector.class) {
57        return new ElementSelectorComparable((ElementSelector) expression);
58      } else if (expression.getClass() == TimeLiteral.class) {
59        return new TimeLiteralComparable((TimeLiteral) expression);
60      } else if (expression.getClass() == TimeSelector.class) {
61        return new TimeSelectorComparable((TimeSelector) expression);
62      } else if (expression.getClass() == MinTimePoint.class) {
63        return new MinTimePointComparable((MinTimePoint) expression);
64      } else if (expression.getClass() == MaxTimePoint.class) {
65        return new MaxTimePointComparable((MaxTimePoint) expression);
66      } else if (expression.getClass() == TimeConstant.class) {
67        return new TimeConstantComparable((TimeConstant) expression);
68      } else if (expression.getClass() == Duration.class) {
69        return new DurationComparable((Duration) expression);
70      } else {
71        throw new IllegalArgumentException(
72          expression.getClass() + " is not a GDL ComparableExpression"
73        );
74      }
75    }
76  }
