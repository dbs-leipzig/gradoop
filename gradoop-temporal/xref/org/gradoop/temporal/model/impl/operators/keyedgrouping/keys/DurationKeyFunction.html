26  import java.time.Instant;
27  import java.time.LocalDateTime;
28  import java.time.temporal.ChronoUnit;
29  import java.time.temporal.TemporalUnit;
30  import java.util.Objects;
31  
32  import static java.time.ZoneOffset.UTC;
33  
34  /**
35   * A key function extracting the duration of a temporal attribute.<p>
36   * This key function extracts a time interval from an element and calculates the duration of that element
37   * in a certain {@link TemporalUnit} (e.g. {@link ChronoUnit#SECONDS seconds}).
38   * Calculations will assume the time zone to be {@link java.time.ZoneOffset#UTC UTC}.<p>
39   * The final grouping key will be stored on the super element as a property with key
40   * {@code duration_INTERVAL_UNIT} where {@code INTERVAL} is the {@link TimeDimension} extracted and
41   * {@code UNIT} the {@link TemporalUnit} to be calculated.
42   *
43   * @param <T> The type of the temporal elements.
44   */
45  public class DurationKeyFunction<T extends TemporalElement> implements KeyFunctionWithDefaultValue<T, Long> {
46  
47    /**
48     * A key function used to extract the interval from the element.
49     */
50    private final TimeIntervalKeyFunction<T> intervalExtractor;
51  
52    /**
53     * The unit to get the duration as.
54     */
55    private final TemporalUnit timeUnit;
56  
57    /**
58     * Should the duration be calculated from dates instead of {@link Instant}s? This will support
59     * units larger than {@link ChronoUnit#DAYS}.
60     */
61    private final boolean calculateAsDate;
62  
63    /**
64     * Create a new instance of this grouping key function.
65     *
66     * @param timeDimension The time dimension to get the duration of.
67     * @param unit          The unit to get the duration as.
68     */
69    public DurationKeyFunction(TimeDimension timeDimension, TemporalUnit unit) {
70      this.intervalExtractor = new TimeIntervalKeyFunction<>(timeDimension);
71      this.timeUnit = Objects.requireNonNull(unit);
72      if (unit instanceof ChronoUnit) {
73        calculateAsDate = ((ChronoUnit) unit).compareTo(ChronoUnit.DAYS) >= 0;
74      } else {
75        calculateAsDate = false;
76      }
77    }
78  
79    @Override
80    public Long getKey(T element) {
81      final Tuple2<Long, Long> interval = intervalExtractor.getKey(element);
82      final Instant start = Instant.ofEpochMilli(interval.f0);
83      final Instant end = Instant.ofEpochMilli(interval.f1);
84      if (calculateAsDate) {
85        return LocalDateTime.ofInstant(start, UTC).until(LocalDateTime.ofInstant(end, UTC), timeUnit);
86      } else {
87        return start.until(end, timeUnit);
88      }
89    }
90  
91    @Override
92    public Long getDefaultKey() {
93      return -1L;
94    }
95  
96    @Override
97    public TypeInformation<Long> getType() {
98      return BasicTypeInfo.LONG_TYPE_INFO;
99    }
100 
101   @Override
102   public void addKeyToElement(T element, Object key) {
103     if (!(key instanceof Long)) {
104       throw new IllegalArgumentException("Invalid type for key: " + key.getClass().getSimpleName());
105     }
106     if (getDefaultKey().equals(key)) {
107       return;
108     }
109     element.setProperty("duration_" + intervalExtractor + "_" + timeUnit, PropertyValue.create(key));
110   }
111 }
