16  package org.gradoop.temporal.model.impl.pojo;

18  import org.apache.flink.api.java.tuple.Tuple2;
19  import org.gradoop.common.model.api.entities.Element;
20  import org.gradoop.common.model.impl.id.GradoopId;
21  import org.gradoop.common.model.impl.pojo.EPGMElement;
22  import org.gradoop.common.model.impl.properties.Properties;
23  import org.gradoop.temporal.model.api.TimeDimension;

25  import java.util.Objects;

27  /**
28   * Base class for all temporal elements. Contains interval definitions for transaction time and valid time.
29   */
30  public abstract class TemporalElement extends EPGMElement implements Element {
31  
32    /**
33     * The default value for unset times (validFrom and txFrom).
34     */
35    public static final Long DEFAULT_TIME_FROM = Long.MIN_VALUE;
36    /**
37     * The default value for unset times (validTo and txTo).
38     */
39    public static final Long DEFAULT_TIME_TO = Long.MAX_VALUE;
40    /**
41     * Transaction time interval containing the beginning and end of the elements transaction time.
42     * Its values are unix timestamps in milliseconds.
43     */
44    private Tuple2<Long, Long> transactionTime;
45    /**
46     * Valid time interval containing the beginning and end of the elements validity.
47     * Its values are unix timestamps in milliseconds.
48     */
49    private Tuple2<Long, Long> validTime;
50  
51    /**
52     * Default constructor. Needed because of Flink's POJO rules and serialization.
53     */
54    public TemporalElement() {
55      transactionTime = new Tuple2<>();
56      validTime = new Tuple2<>();
57    }
58  
59    /**
60     * Creates a new temporal graph element.
61     *
62     * @param id the element identifier
63     * @param label the element label
64     * @param properties the element properties
65     * @param validFrom the beginning of the elements validity as unix timestamp in milliseconds
66     * @param validTo the end of the elements validity as unix timestamp in milliseconds
67     */
68    TemporalElement(GradoopId id, String label, Properties properties, Long validFrom, Long validTo) {
69      super(id, label, properties);
70      // Set transaction time beginning to the current system time
71      transactionTime = new Tuple2<>(System.currentTimeMillis(), DEFAULT_TIME_TO);
72  
73      Tuple2<Long, Long> newValidTime = new Tuple2<>();
74      newValidTime.f0 = validFrom == null ? DEFAULT_TIME_FROM : validFrom;
75      newValidTime.f1 = validTo == null ? DEFAULT_TIME_TO : validTo;
76      setValidTime(newValidTime);
77    }
78  
79    /**
80     * Get the transaction time tuple (tx-from, tx-to). Needed because of Flink's POJO rules.
81     *
82     * @return a {@link Tuple2} representing the transaction time interval
83     */
84    public Tuple2<Long, Long> getTransactionTime() {
85      return transactionTime;
86    }
87  
88    /**
89     * Set the transaction time tuple (tx-from, tx-to). Needed because of Flink's POJO rules.
90     *
91     * @param transactionTime a {@link Tuple2} representing the transaction time interval
92     */
93    public void setTransactionTime(Tuple2<Long, Long> transactionTime) {
94      Objects.requireNonNull(transactionTime);
95      Objects.requireNonNull(transactionTime.f0);
96      Objects.requireNonNull(transactionTime.f1);
97      if (transactionTime.f0 > transactionTime.f1) {
98        throw new IllegalArgumentException("tx-from time can not be after tx-to time");
99      }
100     this.transactionTime = transactionTime;
101   }
102 
103   /**
104    * Get the valid time tuple (valid-from, valid-to). Needed because of Flink's POJO rules.
105    *
106    * @return a {@link Tuple2} representing the valid time interval
107    */
108   public Tuple2<Long, Long> getValidTime() {
109     return validTime;
110   }
111 
112   /**
113    * Set the valid time tuple (valid-from, valid-to). Needed because of Flink's POJO rules.
114    *
115    * @param validTime a {@link Tuple2} representing the valid time interval
116    */
117   public void setValidTime(Tuple2<Long, Long> validTime) {
118     Objects.requireNonNull(validTime);
119     Objects.requireNonNull(validTime.f0);
120     Objects.requireNonNull(validTime.f1);
121     if (validTime.f0 > validTime.f1) {
122       throw new IllegalArgumentException("valid-from time can not be after valid-to time");
123     }
124     this.validTime = validTime;
125   }
126 
127   /**
128    * Set value time by field.
129    *
130    * @param validTime time value
131    * @param field time field
132    */
133   public void setValidTime(long validTime, TimeDimension.Field field) {
134     switch (Objects.requireNonNull(field)) {
135     case FROM:
136       setValidFrom(validTime);
137       break;
138     case TO:
139       setValidTo(validTime);
140       break;
141     default:
142       throw new IllegalArgumentException("Unknown field [" + field + "].");
143     }
144   }
145 
146   /**
147    * Get the time tuple (from, to) regarding to the given {@link TimeDimension}.
148    *
149    * @param dimension the time dimension of the returned values
150    * @return a tuple 2 representing the time interval of the given dimension
151    */
152   public Tuple2<Long, Long> getTimeByDimension(TimeDimension dimension) {
153     switch (Objects.requireNonNull(dimension)) {
154     case VALID_TIME:
155       return this.validTime;
156     case TRANSACTION_TIME:
157       return this.transactionTime;
158     default:
159       throw new IllegalArgumentException("Unknown dimension [" + dimension + "].");
160     }
161   }
162 
163   /**
164    * Get the beginning of the elements validity as unix timestamp in milliseconds.
165    *
166    * @return the beginning of the elements validity as unix timestamp in milliseconds

169     return this.validTime.f0;

179   }
180 
181   /**
182    * Get the end of the elements validity as unix timestamp in milliseconds.
183    *
184    * @return the end of the elements validity as unix timestamp in milliseconds
185    */
186   public Long getValidTo() {
187     return this.validTime.f1;
188   }
189 
190   /**
191    * Sets the end of the elements validity as unix timestamp in milliseconds.
192    *
193    * @param validTo the end of the elements validity as unix timestamp in milliseconds
194    */
195   public void setValidTo(long validTo) {
196     this.validTime.f1 = validTo;
197   }

199   /**
200    * Get the beginning of the elements transaction interval as unix timestamp in milliseconds.
201    *
202    * @return the beginning of the elements transaction interval as unix timestamp in milliseconds
203    */
204   public Long getTxFrom() {
205     return this.transactionTime.f0;
206   }

208   /**
209    * Set the beginning of the elements transaction interval as unix timestamp in milliseconds.
210    *
211    * @param txFrom the beginning of the elements transaction interval as unit timestamp in milliseconds
212    */
213   public void setTxFrom(long txFrom) {
214     this.transactionTime.f0 = txFrom;
215   }
216 
217   /**
218    * Get the end of the elements transaction interval as unix timestamp in milliseconds.
219    *
220    * @return the end of the elements transaction interval as unix timestamp in milliseconds
221    */
222   public Long getTxTo() {
223     return this.transactionTime.f1;
224   }
225 
226   /**
227    * Set the end of the elements transaction interval as unix timestamp in milliseconds.
228    *
229    * @param txTo the end of the elements transaction interval as unix timestamp in milliseconds
230    */
231   public void setTxTo(long txTo) {
232     this.transactionTime.f1 = txTo;
233   }

235   @Override
236   public String toString() {
237     return String.format("%s%s%s[tx%s,val%s]{%s}",
238       id,
239       label == null || label.equals("") ? "" : ":",
