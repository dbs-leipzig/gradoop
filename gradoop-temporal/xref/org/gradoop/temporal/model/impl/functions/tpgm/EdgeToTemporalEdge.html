16  package org.gradoop.temporal.model.impl.functions.tpgm;
17  
18  import org.apache.flink.api.common.functions.MapFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.gradoop.common.model.api.entities.Edge;
21  import org.gradoop.common.model.api.entities.EdgeFactory;
22  import org.gradoop.common.model.impl.id.GradoopId;
23  import org.gradoop.temporal.model.api.functions.TimeIntervalExtractor;
24  import org.gradoop.temporal.model.impl.pojo.TemporalEdge;
25  
26  /**
27   * Initializes a {@link TemporalEdge} from a {@link Edge} instance by setting either
28   * default temporal information or, if a timeIntervalExtractor is given, by the extracted time information.
29   *
30   * @param <E> The (non-temporal) edge type.
31   */
32  @FunctionAnnotation.ForwardedFields("id;sourceId;targetId;label;properties;graphIds")
33  public class EdgeToTemporalEdge<E extends Edge> implements MapFunction<E, TemporalEdge> {
34    /**
35     * The user defined timestamp extractor.
36     */
37    private TimeIntervalExtractor<E> timeIntervalExtractor;
38    /**
39     * Reuse this instance to reduce instantiations.
40     */
41    private TemporalEdge reuse;
42  
43    /**
44     * Creates an instance of the TemporalEdgeFromNonTemporal map function. The temporal instance
45     * will have default temporal information.
46     *
47     * @param elementFactory factory that is responsible for creating a temporal edge instance
48     */
49    public EdgeToTemporalEdge(EdgeFactory<TemporalEdge> elementFactory) {
50      this.reuse = elementFactory.createEdge(GradoopId.NULL_VALUE, GradoopId.NULL_VALUE);
51    }
52  
53    /**
54     * Creates an instance of the TemporalEdgeFromNonTemporal map function. The temporal instance
55     * will have valid times extracted from the non-temporal instance by the given timeIntervalExtractor.
56     *
57     * @param elementFactory factory that is responsible for creating a temporal edge instance
58     * @param timeIntervalExtractor the extractor instance fetches the validFrom and validTo values
59     */
60    public EdgeToTemporalEdge(
61      EdgeFactory<TemporalEdge> elementFactory,
62      TimeIntervalExtractor<E> timeIntervalExtractor) {
63      this(elementFactory);
64      this.timeIntervalExtractor = timeIntervalExtractor;
65    }
66  
67    @Override
68    public TemporalEdge map(E value) throws Exception {
69      reuse.setId(value.getId());
70      reuse.setLabel(value.getLabel());
71      reuse.setSourceId(value.getSourceId());
72      reuse.setTargetId(value.getTargetId());
73      reuse.setProperties(value.getProperties());
74      reuse.setGraphIds(value.getGraphIds());
75      if (timeIntervalExtractor != null) {
76        reuse.setValidTime(timeIntervalExtractor.map(value));
77      }
78      return reuse;
79    }
80  }
