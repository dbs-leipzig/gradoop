16  package org.gradoop.temporal.model.api.functions;
17  
18  import org.gradoop.common.model.api.entities.Element;
19  import org.gradoop.common.model.impl.properties.PropertyValue;
20  import org.gradoop.flink.model.api.functions.AggregateFunction;
21  import org.gradoop.flink.model.impl.operators.aggregation.Aggregation;
22  import org.gradoop.temporal.model.impl.pojo.TemporalElement;
23  
24  /**
25   * An aggregate function used by the {@link Aggregation} operator that handles {@link TemporalElement}s
26   * separately.
27   */
28  public interface TemporalAggregateFunction extends AggregateFunction {
29  
30    /**
31     * Get the increment to be added to the aggregate from an element.
32     * The default implementation of this method will check if the element is a temporal element,
33     * in that case the increment will be determined by {@link #getIncrement(TemporalElement)}.
34     * Otherwise the value of {@link #getNonTemporalIncrement(Element)} will be used.
35     * <br>
36     * <b>Note: </b> Implementations of the {@link TemporalAggregateFunction} interface should
37     * implement {@link #getIncrement(TemporalElement)} (and optionally
38     * {@link #getNonTemporalIncrement(Element)}) instead of this method.
39     *
40     * @param element element used to get the increment
41     * @return The increment, possibly {@code null}.
42     * @see #getIncrement(TemporalElement) method to be implemented instead of this method.
43     */
44    @Override
45    default PropertyValue getIncrement(Element element) {
46      if (element instanceof TemporalElement) {
47        return getIncrement((TemporalElement) element);
48      } else {
49        return getNonTemporalIncrement(element);
50      }
51    }
52  
53    /**
54     * Get the default value used for non-temporal elements or throw an exception.
55     * By default, an {@link UnsupportedOperationException} is thrown.
56     *
57     * @param element The non-temporal element.
58     * @return The default aggregate value.
59     */
60    default PropertyValue getNonTemporalIncrement(Element element) {
61      throw new UnsupportedOperationException("This aggregate function only supports temporal elements.");
62    }
63  
64    /**
65     * Get the increment of a temporal element to be added to the aggregate value.
66     *
67     * @param element The temporal element.
68     * @return The increment, possibly {@code null}.
69     * @see AggregateFunction#getIncrement(Element) about this method.
70     */
71    PropertyValue getIncrement(TemporalElement element);
72  }
