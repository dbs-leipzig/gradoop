169   /**
170    * Comparison according to gSpan lexicographic order.
171    *
172    * @param that other extension
173    * @return comparison result
174    */
175   @Override
176   public int compareTo(Traversal<C> that) {
177 
178     int comparison;
179 
180     // no difference is times
181     if (this.fromTime == that.fromTime && this.toTime == that.toTime) {
182 
183       // compare from values
184       comparison = this.fromValue.compareTo(that.fromValue);
185 
186       if (comparison == 0) {
187 
188         // compare direction
189         boolean thisIsOutgoing = this.isOutgoing();
190         boolean thatIsOutgoing = that.isOutgoing();
191 
192         if (thisIsOutgoing && !thatIsOutgoing) {
193           comparison = -1;
194         } else if (thatIsOutgoing && !thisIsOutgoing) {
195           comparison = 1;
196         } else {
197 
198           // compare edge values
199           comparison = this.edgeValue.compareTo(that.edgeValue);
200 
201           if (comparison == 0) {
202 
203             // compare to values
204             comparison = this.toValue.compareTo(that.toValue);
205           }
206         }
207       }
208 
209       // time differences
210     } else {
211 
212       // compare backtracking
213       boolean thisIsBacktrack = this.isBackwards();
214       boolean thatIsBacktrack = that.isBackwards();
215 
216       if (thisIsBacktrack && !thatIsBacktrack) {
217         comparison = -1;
218       } else if (thatIsBacktrack && !thisIsBacktrack) {
219         comparison = 1;
220       } else {
221 
222         // both forwards
223         if (thatIsBacktrack) {
224 
225           // back to earlier vertex is smaller
226           comparison = this.toTime - that.toTime;
227 
228           // both backwards
229         } else {
230 
231           // forwards from later vertex is smaller
232           comparison = that.fromTime - this.fromTime;
233         }
234       }
235     }
236 
237     return comparison;
238   }
