72  /**
73   * A graph collection graph is one of the base concepts of the Extended Property Graph Model. From
74   * a model perspective, the collection represents a set of logical graphs. From a data perspective
75   * this is reflected by providing three concepts:
76   *
77   * - a set of graph heads assigned to the graphs in that collection
78   * - a set of vertices which is the union of all vertex sets of the represented graphs
79   * - a set of edges which is the union of all edge sets of the represented graphs
80   *
81   * Furthermore, a graph collection provides operations that are performed on the underlying data.
82   * These operations result in either another graph collection or in a {@link LogicalGraph}.
83   *
84   * A graph collection is wrapping a {@link GraphCollectionLayout} which defines, how the collection
85   * is represented in Apache Flink. Note that the GraphCollection also implements that interface and
86   * just forward the calls to the layout. This is just for convenience and API synchronicity.
87   */
88  public class GraphCollection implements
89    BaseGraphCollection<GraphHead, Vertex, Edge, LogicalGraph, GraphCollection>,
90    GraphCollectionOperators {

92     * Layout for that graph collection
93     */

170   @Override
171   public LogicalGraph getGraph(final GradoopId graphID) {
172     // filter vertices and edges based on given graph id
173     DataSet<GraphHead> graphHead = getGraphHeads()
174       .filter(new BySameId<>(graphID));
175     DataSet<Vertex> vertices = getVertices()
176       .filter(new InGraph<>(graphID));
177     DataSet<Edge> edges = getEdges()
178       .filter(new InGraph<>(graphID));
179 
180     return new LogicalGraph(
181       config.getLogicalGraphFactory().fromDataSets(graphHead, vertices, edges),
182       getConfig());
183   }
184 
185   @Override
186   public GraphCollection getGraphs(final GradoopId... identifiers) {
187 
188     GradoopIdSet graphIds = new GradoopIdSet();
189 
190     graphIds.addAll(Arrays.asList(identifiers));
191 
192     return getGraphs(graphIds);
193   }
194 
195   @Override
196   public GraphCollection getGraphs(final GradoopIdSet identifiers) {
197 
198     DataSet<GraphHead> newGraphHeads = this.getGraphHeads()
199       .filter(new FilterFunction<GraphHead>() {
200         @Override
201         public boolean filter(GraphHead graphHead) {
202           return identifiers.contains(graphHead.getId());
203         }
204       });
205 
206     // build new vertex set
207     DataSet<Vertex> vertices = getVertices()
208       .filter(new InAnyGraph<>(identifiers));
209 
210     // build new edge set
211     DataSet<Edge> edges = getEdges()
212       .filter(new InAnyGraph<>(identifiers));
213 
214     return new GraphCollection(getFactory().fromDataSets(newGraphHeads, vertices, edges),
215       getConfig());
216   }
