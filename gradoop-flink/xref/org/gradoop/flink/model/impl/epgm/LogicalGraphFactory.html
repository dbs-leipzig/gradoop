33  /**
34   * Responsible for creating instances of {@link LogicalGraph} based on a specific
35   * {@link org.gradoop.flink.model.api.layouts.LogicalGraphLayout}.
36   */
37  public class LogicalGraphFactory
38    implements BaseGraphFactory<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection> {
39    /**
40     * Creates the layout from given data.
41     */
42    private LogicalGraphLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> layoutFactory;
43    /**
44     * The Gradoop Flink configuration.
45     */
46    private final GradoopFlinkConfig config;
47  
48    /**
49     * Creates a new factory.
50     *
51     * @param config the Gradoop Flink configuration
52     */
53    public LogicalGraphFactory(GradoopFlinkConfig config) {
54      this.config = config;
55    }
56  
57    @Override
58    public LogicalGraphLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> getLayoutFactory() {
59      return layoutFactory;
60    }
61  
62    @Override
63    public void setLayoutFactory(LogicalGraphLayoutFactory<EPGMGraphHead, EPGMVertex, EPGMEdge> layoutFactory) {
64      Objects.requireNonNull(layoutFactory);
65      this.layoutFactory = layoutFactory;
66      this.layoutFactory.setGradoopFlinkConfig(config);
67    }
68  
69    @Override
70    public GraphHeadFactory<EPGMGraphHead> getGraphHeadFactory() {
71      return layoutFactory.getGraphHeadFactory();
72    }
73  
74    @Override
75    public VertexFactory<EPGMVertex> getVertexFactory() {
76      return layoutFactory.getVertexFactory();
77    }
78  
79    @Override
80    public EdgeFactory<EPGMEdge> getEdgeFactory() {
81      return layoutFactory.getEdgeFactory();
82    }
83  
84    @Override
85    public LogicalGraph fromDataSets(DataSet<EPGMVertex> vertices) {
86      return new LogicalGraph(layoutFactory.fromDataSets(vertices), config);
87    }
88  
89    @Override
90    public LogicalGraph fromDataSets(DataSet<EPGMVertex> vertices, DataSet<EPGMEdge> edges) {
91      return new LogicalGraph(layoutFactory.fromDataSets(vertices, edges), config);
92    }
93  
94    @Override
95    public LogicalGraph fromDataSets(DataSet<EPGMGraphHead> graphHead, DataSet<EPGMVertex> vertices,
96      DataSet<EPGMEdge> edges) {
97      return new LogicalGraph(layoutFactory.fromDataSets(graphHead, vertices, edges), config);
98    }
99  
100   @Override
101   public LogicalGraph fromIndexedDataSets(Map<String, DataSet<EPGMGraphHead>> graphHead,
102     Map<String, DataSet<EPGMVertex>> vertices, Map<String, DataSet<EPGMEdge>> edges) {
103     return new LogicalGraph(layoutFactory.fromIndexedDataSets(graphHead, vertices, edges), config);
104   }
105 
106   @Override
107   public LogicalGraph fromCollections(EPGMGraphHead graphHead, Collection<EPGMVertex> vertices,
108     Collection<EPGMEdge> edges) {
109     return new LogicalGraph(layoutFactory.fromCollections(graphHead, vertices, edges), config);
110   }
111 
112   @Override
113   public LogicalGraph fromCollections(Collection<EPGMVertex> vertices, Collection<EPGMEdge> edges) {
114     return new LogicalGraph(layoutFactory.fromCollections(vertices, edges), config);
115   }
116 
117   @Override
118   public LogicalGraph createEmptyGraph() {
119     return new LogicalGraph(layoutFactory.createEmptyGraph(), config);
120   }
121 }
