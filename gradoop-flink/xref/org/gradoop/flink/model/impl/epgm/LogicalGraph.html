61  /**
62   * A logical graph is one of the base concepts of the Extended Property Graph Model. A logical graph
63   * encapsulates three concepts:
64   * <ul>
65   * <li>a graph head, that stores information about the graph (i.e. label and properties)</li>
66   * <li>a set of vertices assigned to the graph</li>
67   * <li>a set of directed, possibly parallel edges assigned to the graph</li>
68   * </ul>
69   * Furthermore, a logical graph provides operations that are performed on the underlying data. These
70   * operations result in either another logical graph or in a {@link GraphCollection}.
71   * <p>
72   * A logical graph is wrapping a {@link LogicalGraphLayout} which defines, how the graph is
73   * represented in Apache Flink. Note that the LogicalGraph also implements that interface and
74   * just forward the calls to the layout. This is just for convenience and API synchronicity.
75   */
76  public class LogicalGraph implements
77    BaseGraph<EPGMGraphHead, EPGMVertex, EPGMEdge, LogicalGraph, GraphCollection>, LogicalGraphOperators {
78    /**
79     * Layout for that logical graph.
80     */
81    private final LogicalGraphLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> layout;
82    /**
83     * Configuration
84     */
85    private final GradoopFlinkConfig config;
86  
87    /**
88     * Creates a new logical graph based on the given parameters.
89     *
90     * @param layout representation of the logical graph
91     * @param config the Gradoop Flink configuration
92     */
93    LogicalGraph(LogicalGraphLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> layout,
94      GradoopFlinkConfig config) {
95      Objects.requireNonNull(layout);
96      Objects.requireNonNull(config);
97      this.layout = layout;
98      this.config = config;
99    }

218   @Override

220     List<String> vertexGroupingKeys, List<AggregateFunction> vertexAggregateFunctions,

300   @Override
301   public DataSet<Boolean> isEmpty() {
302     return getVertices()
303       .map(new True<>())
304       .distinct()
305       .union(getConfig().getExecutionEnvironment().fromElements(false))
306       .reduce(new Or())
307       .map(new Not());
308   }
