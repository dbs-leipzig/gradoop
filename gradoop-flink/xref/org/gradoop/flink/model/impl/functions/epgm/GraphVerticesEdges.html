33  /**
34   * combiner: {@code (graphId, [vertex|edges] -> (graphId, {vertex,...}, {edge,...})}<br>
35   * reducer: {@code (graphId, {vertex,...}, {edge,...}) -> (graphId, {vertex,...}, {edge,...})}
36   * <p>
37   * Forwarded fields:
38   * <br>
39   * {@code f0}: {@code graphId}
40   */
41  @FunctionAnnotation.ForwardedFields("f0")
42  public class GraphVerticesEdges implements
43    GroupCombineFunction<
44      Tuple2<GradoopId, EPGMGraphElement>,
45      Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>>,
46    GroupReduceFunction<
47      Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>,
48      Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> {
49  
50    /**
51     * Creates vertex and edge sets for each transaction.
52     *
53     * @param values vertices and edges associated to the same graph
54     * @param out collector
55     * @throws Exception The function may throw Exceptions, which will cause the program to cancel,
56     *                   and may trigger the recovery logic.
57     */
58    @Override
59    public void combine(Iterable<Tuple2<GradoopId, EPGMGraphElement>> values,
60      Collector<Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> out) throws Exception {
61  
62      Iterator<Tuple2<GradoopId, EPGMGraphElement>> iterator = values.iterator();
63  
64      GradoopId graphId    = null;
65      Set<EPGMVertex> vertices = new HashSet<>();
66      Set<EPGMEdge> edges      = new HashSet<>();
67  
68      while (iterator.hasNext()) {
69        Tuple2<GradoopId, EPGMGraphElement> next = iterator.next();
70        graphId = next.f0;
71  
72        EPGMGraphElement element = next.f1;
73        if (element instanceof EPGMVertex) {
74          vertices.add((EPGMVertex) element);
75        } else {
76          edges.add((EPGMEdge) element);
77        }
78      }
79  
80      out.collect(Tuple3.of(graphId, vertices, edges));
81    }
82  
83    /**
84     * Merges sets created by the combiners.
85     *
86     * @param values partial vertex and edge sets for one transaction
87     * @param out collector
88     * @throws Exception The function may throw Exceptions, which will cause the program to cancel,
89     *                   and may trigger the recovery logic.
90     */
91    @Override
92    public void reduce(Iterable<Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> values,
93      Collector<Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> out) throws Exception {
94  
95      Iterator<Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>>> iterator = values.iterator();
96      Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>> first = iterator.next();
97  
98      GradoopId graphId    = first.f0;
99      Set<EPGMVertex> vertices = first.f1;
100     Set<EPGMEdge> edges      = first.f2;
101 
102     while (iterator.hasNext()) {
103       Tuple3<GradoopId, Set<EPGMVertex>, Set<EPGMEdge>> next = iterator.next();
104       vertices.addAll(next.f1);
105       edges.addAll(next.f2);
106     }
107 
108     out.collect(Tuple3.of(graphId, vertices, edges));
109   }
