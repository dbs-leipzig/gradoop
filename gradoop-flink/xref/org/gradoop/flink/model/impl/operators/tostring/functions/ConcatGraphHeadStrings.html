16  package org.gradoop.flink.model.impl.operators.tostring.functions;
17  
18  import org.apache.commons.lang3.StringUtils;
19  import org.apache.flink.api.common.functions.GroupReduceFunction;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.flink.model.impl.operators.tostring.tuples.GraphHeadString;
22  
23  import java.util.ArrayList;
24  import java.util.Collections;
25  import java.util.List;
26  
27  import static org.gradoop.flink.model.impl.operators.tostring.CanonicalAdjacencyMatrixBuilder.LINE_SEPARATOR;
28  
29  /**
30   * concatenates the sorted string representations of graph heads to represent a
31   * collection
32   */
33  public class ConcatGraphHeadStrings
34    implements GroupReduceFunction<GraphHeadString, String> {
35  
36    @Override
37    public void reduce(Iterable<GraphHeadString> graphHeadLabels,
38                       Collector<String> collector) throws Exception {
39  
40      List<String> graphLabels = new ArrayList<>();
41  
42      for (GraphHeadString graphHeadString : graphHeadLabels) {
43        String graphLabel = graphHeadString.getLabel();
44        if (!graphLabel.equals("")) {
45          graphLabels.add(graphLabel);
46        }
47      }
48  
49      Collections.sort(graphLabels);
50  
51      collector.collect(StringUtils.join(graphLabels, LINE_SEPARATOR));
52    }
53  }
