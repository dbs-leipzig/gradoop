16  package org.gradoop.flink.model.impl.operators.matching.single.cypher.operators.expand.tuples;

18  import org.apache.commons.lang3.ArrayUtils;
19  import org.apache.flink.api.java.tuple.Tuple3;
20  import org.gradoop.common.model.impl.id.GradoopId;
21  import org.gradoop.flink.model.impl.operators.matching.single.cypher.pojos.Embedding;
22  
23  /**
24   * Represents an intermediate result for the expand operator.
25   *
26   * The base part (f0) is the immutable input generated by previous operations.
27   * The path (f1) is the path we grow while expanding.
28   * The end node (f2) is the last element in expanded path (a node).
29   * It is stored separately to enable using it as a join column.
30   */
31  public class ExpandEmbedding extends Tuple3<Embedding, GradoopId[], GradoopId> {
32  
33    /**
34     * Create a new ExpandIntermediateResult.
35     */
36    public ExpandEmbedding() {
37      super();
38    }
39  
40    /**
41     * Create a new expand intermediate result
42     *
43     * @param base the base part
44     * @param path the path along we expanded
45     */
46    public ExpandEmbedding(Embedding base, GradoopId... path) {
47      super(base, ArrayUtils.subarray(path, 0, path.length - 1), path[path.length - 1]);
48  
49    }
50  
51    /**
52     * Returns the base part
53     * @return the base part
54     */
55    public Embedding getBase() {
56      return f0;
57    }
58  
59    /**
60     * Returns the path
61     * @return the path
62     */
63    public GradoopId[] getPath() {
64      return f1;
65    }
66  
67    /**
68     * Returns the end element
69     * @return the end element
70     */
71    public GradoopId getEnd() {
72      return f2;
73    }
74  
75    /**
76     * Expands a previous intermediate result by the given edge
77     * {@code (base,(a,b,c),d) x (d,e,f) -> (base, (a,b,c,d,e), f)}
78     *
79     * @param edge the edge along which we expand
80     * @return new expanded intermediate result
81     */
82    public ExpandEmbedding grow(EdgeWithTiePoint edge) {
83      return new ExpandEmbedding(
84        f0,
85        ArrayUtils.addAll(f1, f2, edge.f1, edge.f2)
86      );
87    }
88  
89    /**
90     * Size of the path
91     * @return path size
92     */
93    public int pathSize() {
94      return f1.length;
95    }
96  
97    /**
98     * Turns the intermediate result into an embedding
99     * {@code (base,(a,b,c),d) -> (base, IdListEntry(a,b,c), IdEntry(d))}
100    *
101    * @return embedding representation of the expand intermediate result
102    */
103   public Embedding toEmbedding() {
104     Embedding embedding = getBase();
105     embedding.add(f1);
106     embedding.add(f2);
107 
108     return getBase();
109   }
