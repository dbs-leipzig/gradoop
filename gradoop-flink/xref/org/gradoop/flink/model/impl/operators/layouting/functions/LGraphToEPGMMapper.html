32  /**
33   * Converts an LGraph to an EPGM Graph
34   */
35  public class LGraphToEPGMMapper {
36  
37    /**
38     * Build a new EPGM-Graph from the given LGraph, by converting LVertices to vertices 1:1.
39     *
40     * @param input    The original EPGMGraph, to use it's factory
41     * @param layouted The LGraph to convert
42     * @return The converted LGraph
43     */
44    public LogicalGraph buildSimplifiedGraph(LogicalGraph input, LGraph layouted) {
45      DataSet<EPGMVertex> vertices = layouted.getVertices().map(LGraphToEPGMMapper::mapTpEPGMVertex);
46      DataSet<EPGMEdge> edges = layouted.getEdges().map(LGraphToEPGMMapper::mapToEPGMEdge);
47      return input.getFactory().fromDataSets(vertices, edges);
48    }
49  
50    /**
51     * Map LVertex to EPGMVertex.
52     *
53     * @param lv Input LVertex
54     * @return Output EPGMVertex
55     */
56    protected static EPGMVertex mapTpEPGMVertex(LVertex lv) {
57      EPGMVertex v = new EPGMVertex(lv.getId(), "vertex", Properties.create(), null);
58      lv.getPosition().setVertexPosition(v);
59      v.setProperty(FusingFRLayouter.VERTEX_SIZE_PROPERTY, lv.getCount());
60      v.setProperty(FusingFRLayouter.SUB_ELEMENTS_PROPERTY,
61        getSubelementListValue(lv.getSubVertices()));
62      return v;
63    }
64  
65    /**
66     * Map LEdge to EPGMEdge.
67     *
68     * @param le Input LEdge
69     * @return Output EPGMEdge
70     */
71    protected static EPGMEdge mapToEPGMEdge(LEdge le) {
72      EPGMEdge e = new EPGMEdge(le.getId(), "edge", le.getSourceId(), le.getTargetId(), Properties.create(),
73          null);
74      e.setProperty(FusingFRLayouter.VERTEX_SIZE_PROPERTY, le.getCount());
75      e.setProperty(FusingFRLayouter.SUB_ELEMENTS_PROPERTY, getSubelementListValue(le.getSubEdges()));
76      return e;
77    }
78  
79    /**
80     * Helper function to convert the List of sub-elements into a comma seperated string
81     * Gradoop (especially the CSVDataSink) seems to have trouble with lists of PropertyValues, so
82     * this is the easies workaround
83     *
84     * @param ids List of GradoopIds
85     * @return A comma seperated string of ids
86     */
87    protected static String getSubelementListValue(List<GradoopId> ids) {
88      return ids.stream().map(GradoopId::toString).collect(Collectors.joining(","));
89    }
90  }
