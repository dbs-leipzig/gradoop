16  package org.gradoop.flink.model.impl.operators.grouping.functions;
17  
18  import org.apache.flink.api.common.functions.GroupReduceFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.common.model.impl.id.GradoopId;
22  import org.gradoop.flink.model.impl.operators.grouping.tuples.VertexGroupItem;
23  import org.gradoop.common.model.impl.properties.PropertyValueList;
24  
25  /**
26   * Reduces a group of {@link VertexGroupItem} instances.
27   */
28  @FunctionAnnotation.ForwardedFields(
29      "f0;" + // vertex id
30      "f2;" + // label
31      "f3;" + // properties
32      "f4;" + // aggregates
33      "f6"    // label group
34  )
35  public class ReduceVertexGroupItems
36    extends ReduceVertexGroupItemBase
37    implements GroupReduceFunction<VertexGroupItem, VertexGroupItem> {
38  
39    /**
40     * Creates group reduce function.
41     *
42     * @param useLabel true, iff labels are used for grouping
43     */
44    public ReduceVertexGroupItems(boolean useLabel) {
45      super(useLabel);
46    }
47  
48    @Override
49    public void reduce(Iterable<VertexGroupItem> vertexGroupItems,
50      Collector<VertexGroupItem> collector) throws Exception {
51  
52      GradoopId superVertexId                         = null;
53      String groupLabel                               = null;
54      PropertyValueList groupPropertyValues           = null;
55  
56      VertexGroupItem reuseTuple = getReuseVertexGroupItem();
57  
58      boolean isFirst = true;
59  
60      for (VertexGroupItem groupItem : vertexGroupItems) {
61        if (isFirst) {
62          superVertexId       = GradoopId.get();
63          groupLabel          = groupItem.getGroupLabel();
64          groupPropertyValues = groupItem.getGroupingValues();
65  
66          reuseTuple.setGroupLabel(groupLabel);
67          reuseTuple.setGroupingValues(groupPropertyValues);
68          reuseTuple.setSuperVertexId(superVertexId);
69          reuseTuple.setAggregateValues(groupItem.getAggregateValues());
70          reuseTuple.setSuperVertex(groupItem.isSuperVertex());
71          reuseTuple.setLabelGroup(groupItem.getLabelGroup());
72  
73          isFirst = false;
74        }
75        reuseTuple.setVertexId(groupItem.getVertexId());
76  
77        // collect updated vertex item
78        collector.collect(reuseTuple);
79  
80        reuseTuple.getLabelGroup().aggregate(groupItem.getAggregateValues());
81      }
82  
83      VertexGroupItem superVertex = createSuperVertexTuple(
84        superVertexId,
85        groupLabel,
86        groupPropertyValues,
87        reuseTuple.getLabelGroup().getAggregateValueList());
88      superVertex.getLabelGroup().resetAggregateValues();
89      collector.collect(superVertex);
90    }
91  }
