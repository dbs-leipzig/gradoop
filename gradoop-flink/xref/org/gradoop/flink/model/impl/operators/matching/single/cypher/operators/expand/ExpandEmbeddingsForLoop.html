28  /**
29   * Expands an vertex along the edges. The number of hops can be specified via upper and lower bound
30   * The input embedding is appended by 2 Entries, the first one represents the path (edge, vertex,
31   * edge, vertex, ..., edge), the second one the end vertex
32   *
33   * Iteration is done with {@code LoopUnrolling}
34   */

46     * @param distinctVertexColumns indices of distinct input vertex columns

94    public ExpandEmbeddingsForLoop(DataSet<Embedding> input, DataSet<Embedding> candidateEdges,

96      List<Integer> distinctVertexColumns, List<Integer> distinctEdgeColumns, int closingColumn) {
97  
98      this(input, candidateEdges, expandColumn, lowerBound, Integer.MAX_VALUE, direction,
99        distinctVertexColumns, distinctEdgeColumns, closingColumn,
100       JoinOperatorBase.JoinHint.OPTIMIZER_CHOOSES);
101   }

103   @Override
104   protected DataSet<ExpandEmbedding> iterate(DataSet<ExpandEmbedding> initialWorkingSet) {
105     List<DataSet<ExpandEmbedding>> intermediateResults = Lists.newArrayList(initialWorkingSet);
106 
107     for (int i = 1; i < upperBound; i++) {
108       DataSet<ExpandEmbedding> nextResult =
109         intermediateResults.get(i - 1)
110           .join(candidateEdgeTuples, joinHint)
111             .where(2).equalTo(0)
112             .with(new MergeExpandEmbeddings(distinctVertexColumns, distinctEdgeColumns,
113               closingColumn))
114             .name(getName() + " - Expansion " + i);
115 
116       intermediateResults.add(nextResult);
117     }
118 
119     int startIndex = Math.max(lowerBound - 1, 0);
120     DataSet<ExpandEmbedding> result = intermediateResults.get(startIndex);
121 
122     for (int i = startIndex + 1; i < upperBound; i++) {
123       result = result.union(intermediateResults.get(i));
124     }
125 
126     return result;
127   }
