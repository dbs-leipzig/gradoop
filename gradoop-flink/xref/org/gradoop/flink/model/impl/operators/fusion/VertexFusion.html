44  /**
45   * Fusion is a binary operator taking two graphs: a search graph (first parameter) and a
46   * pattern graph (second parameter) [This means that this is not a symmetric operator
47   * (F(a,b) != F(b,a))]. The general idea of this operator is that everything that appears
48   * in the pattern graph is fused into a single vertex into the search graph. A new logical graph
49   * is returned.
50   *
51   * 1) If any search graph is empty, an empty result will always be returned
52   * 2) If a pattern graph is empty, then the search graph will always be returned
53   * 3) The vertices of the pattern graph appearing in the first parameter are replaced by a
54   *    new vertex. The old edges of the search graph are updated in the final result so that
55   *    each vertex, either pointing to a to-be-fused vertex or starting from a to-be-fused vertex,
56   *    are respectively updated as either pointing or starting from the fused vertex.
57   * 4) Pattern graph's edges are also taken into account: any edge between to-be-fused vertices
58   *    appearing in the search graph that are not expressed in the pattern graph are
59   *    rendered as hooks over the fused vertex
60   */
61  public class VertexFusion implements BinaryBaseGraphToBaseGraphOperator<LogicalGraph> {
62  
63    @Override
64    public LogicalGraph execute(LogicalGraph searchGraph, LogicalGraph graphPatterns) {
65      return execute(searchGraph,
66          graphPatterns.getCollectionFactory()
67          .fromDataSets(
68              graphPatterns.getGraphHead(),
69              graphPatterns.getVertices(),
70              graphPatterns.getEdges()));
71    }

74    /**
75     * Fusing the already-combined sources.
76     *
77     * @param searchGraph   Logical Graph defining the data lake
78     * @param graphPatterns Collection of elements representing which vertices will be merged into
79     *                      a vertex
80     * @return              A single merged graph
81     */
82    public LogicalGraph execute(LogicalGraph searchGraph, GraphCollection graphPatterns) {
83  
84      // Missing in the theoric definition: creating a new header
85      GradoopId newGraphid = GradoopId.get();
86  
87      DataSet<EPGMGraphHead> gh = searchGraph.getGraphHead()
88        .map(new MapGraphHeadForNewGraph(newGraphid));
89  
90      DataSet<GradoopId> patternVertexIds = graphPatterns.getVertices()
91              .map(new Id<>());
92      DataSet<GradoopId> patternEdgeIds = graphPatterns.getEdges()
93              .map(new Id<>());
94  
95      // PHASE 1: Induced Subgraphs
96      // Associate each vertex to its graph id
97      DataSet<Tuple2<EPGMVertex, GradoopId>> patternVerticesWithGraphIDs =
98          graphPatterns.getVertices()
99          .coGroup(searchGraph.getVertices())
100         .where(new Id<>()).equalTo(new Id<>())
101         .with(new LeftSide<>())
102         .flatMap(new MapVertexToPairWithGraphId());
103 
104     // Associate each gid in hypervertices.H to the merged vertices
105     DataSet<Tuple2<EPGMVertex, GradoopId>> mergedVertices =
106         graphPatterns.getGraphHeads()
107         .map(new CoGroupGraphHeadToVertex());
108 
109     // PHASE 2: Recreating the vertices
110     DataSet<EPGMVertex> vi = searchGraph.getVertices()
111       .filter(new IdNotInBroadcast<>())
112       .withBroadcastSet(patternVertexIds, IdNotInBroadcast.IDS);
113 
114     DataSet<Tuple2<EPGMVertex, GradoopId>> idJoin = patternVerticesWithGraphIDs
115       .coGroup(mergedVertices)
116       .where(new Value1Of2<>()).equalTo(new Value1Of2<>())
117       .with(new CoGroupAssociateOldVerticesWithNewIds())
118       .union(vi.map(new MapVerticesAsTuplesWithNullId()));
119 
120     DataSet<EPGMVertex> vToRet = mergedVertices
121       .coGroup(patternVerticesWithGraphIDs)
122       .where(new Value1Of2<>()).equalTo(new Value1Of2<>())
123       .with(new LeftSide<>())
124       .map(new Value0Of2<>())
125       .union(vi)
126       .map(new MapFunctionAddGraphElementToGraph2<>(newGraphid));
127 
128     // PHASE 3: Recreating the edges
129     DataSet<EPGMEdge> edges = searchGraph.getEdges()
130       .filter(new IdNotInBroadcast<>())
131       .withBroadcastSet(patternEdgeIds, IdNotInBroadcast.IDS)
132       .leftOuterJoin(idJoin)
133       .where(new SourceId<>()).equalTo(new LeftElementId<>())
134       .with(new FlatJoinSourceEdgeReference(true))
135       .leftOuterJoin(idJoin)
136       .where(new TargetId<>()).equalTo(new LeftElementId<>())
137       .with(new FlatJoinSourceEdgeReference(false))
138       .groupBy(new Id<>())
139       .reduceGroup(new AddNewIdToDuplicatedEdge())
140       .map(new MapFunctionAddGraphElementToGraph2<>(newGraphid));
141 
142     return searchGraph.getFactory().fromDataSets(gh, vToRet, edges);
143   }
