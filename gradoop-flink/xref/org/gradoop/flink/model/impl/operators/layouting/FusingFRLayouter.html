38  /**
39   * A special variant of the FRLayouter that combines similar vertices during the layouting,
40   * creating a simplified version of the graph.
41   */
42  public class FusingFRLayouter extends FRLayouter {
43  
44    /**
45     * Specifies the different available Output-Formats for the layouter
46     */
47    public enum OutputFormat {
48      /**
49       * Output the simplified graph. The output-graph will loose all information except for the
50       * GradoopIds. Vertices/Edges will have "SUBELEMENTS"-Property containing a comma
51       * separated string listing all element-ids that were
52       * combined into the super-element. Edges and Vertices will have a "SIZE"-Property containing
53       * the number of sub-elements contained in this super-element.
54       */
55      SIMPLIFIED,
56      /**
57       * Grouped vertices will be resolved into the original vertices and will be placed randomly
58       * close to another.
59       */
60      EXTRACTED,
61      /**
62       * Like EXTRACTED, but vertices of a supernode will be placed at exactly the same position.
63       */
64      RAWEXTRACTED,
65      /**
66       * Like EXTRACTED, but performs some more layouting-iterations after the extraction. At least 11
67       * iterations are needed to used POSTLAYOUT.
68       */
69      POSTLAYOUT
70    }

72    /**
73     * Name of the property that will contain the number of sub-vertices or sub-edges for a vertex or
74     * edge
75     */
76    public static final String VERTEX_SIZE_PROPERTY = "SIZE";
77    /**
78     * The name of the property where the ids of the sub-vertices (or sub-edges) of a

81    public static final String SUB_ELEMENTS_PROPERTY = "SUBELEMENTS";

85    private static final int POST_ITERATIONS = 10;
86    /**
87     * Only vertices with a similarity of at least threshold are combined
88     */
89    protected double threshold;

91     * Compare function to use. Null means use default.

93    protected VertexCompareFunction compareFunction = null;

95     * The output format chosen by the user

97    protected OutputFormat outputFormat;

103   /**
104    * Create new FusingFRLayouter
105    *
106    * @param iterations   Number of iterations to perform
107    * @param vertexCount  Number of vertices in the input-graph (used to compute default-values)
108    * @param threshold    Only vertices with a similarity of at least threshold are combined. Lower
109    *                     values will lead to a more simplified output-graph. Valid values are {@code
110    *                     >= 0 }
111    *                     and {@code <= 1 }.
112    * @param outputFormat Chosen OutputFormat. See {@link OutputFormat}
113    */
114   public FusingFRLayouter(int iterations, int vertexCount, double threshold,
115     OutputFormat outputFormat) {
116     super(iterations, vertexCount);
117 
118     if (threshold < 0 || threshold > 1) {
119       throw new IllegalArgumentException("Threshold must be between 0 and 1");
120     }
121 
122     if (outputFormat == OutputFormat.POSTLAYOUT && iterations <= POST_ITERATIONS) {
123       throw new IllegalArgumentException("When using OutputFormat.POSTLAYOUT, at least 11 " +
124         "iterations are needed");
125     }
126 
127     this.threshold = threshold;
128     this.outputFormat = outputFormat;
129   }

152   @Override
153   public LogicalGraph execute(LogicalGraph g) {
154 
155     applicator = new FRForceApplicator(getWidth(), getHeight(), getK(),
156       (outputFormat != OutputFormat.POSTLAYOUT) ? iterations : iterations - POST_ITERATIONS);
157 
158     g = createInitialLayout(g);
159 
160     DataSet<EPGMVertex> gradoopVertices = g.getVertices();
161     DataSet<EPGMEdge> gradoopEdges = g.getEdges();
162 
163     // Flink can only iterate over a single dataset. Therefore vertices and edges have to be
164     // temporarily combined into a single dataset.
165     // Also the Grapdoop datatypes are converted to internal datatypes
166     DataSet<SimpleGraphElement> tmpvertices = gradoopVertices.map((v) -> new LVertex(v));
167     DataSet<SimpleGraphElement> tmpedges = gradoopEdges.map((e) -> new LEdge(e));
168     DataSet<SimpleGraphElement> graphElements = tmpvertices.union(tmpedges);
169 
170     IterativeDataSet<SimpleGraphElement> loop = graphElements.iterate(
171       (outputFormat != OutputFormat.POSTLAYOUT) ? iterations : iterations - POST_ITERATIONS);
172 
173     // split the combined dataset to work with the edges and vertices
174     LGraph graph = new LGraph(loop);
175 
176     // perform the layouting
177     layout(graph);
178 
179     // Use the VertexFusor to create a simplified version of the graph
180     graph = new VertexFusor(getCompareFunction(), threshold).execute(graph);
181 
182     // again, combine vertices and edges into a single dataset to perform iterations
183     graphElements = graph.getGraphElements();
184     graphElements = loop.closeWith(graphElements);
185 
186     // again, split the combined dataset  (after all iterations have been completed)
187     graph = new LGraph(graphElements);
188 
189 
190     switch (outputFormat) {
191     case SIMPLIFIED:
192       return buildSimplifiedGraph(g, graph);
193     case EXTRACTED:
194       return buildExtractedGraph(g, graph, true);
195     case RAWEXTRACTED:
196       return buildExtractedGraph(g, graph, false);
197     case POSTLAYOUT:
198       return buildPostLayoutGraph(g, graph);
199     default:
200       throw new IllegalArgumentException("Unsupported output-format");
201     }
202 
203   }

205   /**
206    * Extract all subverties/subedges from the super-vertices/super-edges and place them at the
207    * location of the super-vertex (and add some random jitter to the positions).
208    * Then some more layouting-iteraions are performed.
209    *
210    * @param input Original input graph
211    * @param graph Result of the layouting
212    * @return The final graph, containing all vertices and edges from the original graph.
213    */
214   protected LogicalGraph buildPostLayoutGraph(LogicalGraph input, LGraph graph) {
215     DataSet<LVertex> vertices = graph.getVertices().flatMap(new LVertexFlattener(true, getK()));
216 
217     DataSet<LEdge> edges = input.getEdges().map(LEdge::new);
218     graph.setEdges(edges);
219 
220 
221     // use a new applicator for all following layouting iterations. The new applicator will
222     // behave as if iteration x of n is actually iterations+x of n+POST_ITERATIONS
223     applicator =
224       new FRForceApplicator(getWidth(), getHeight(), getK(), iterations + POST_ITERATIONS);
225     applicator.setPreviousIterations(iterations);
226 
227     // do some more layouting iterations
228     IterativeDataSet<LVertex> loop = vertices.iterate(POST_ITERATIONS);
229     graph.setVertices(loop);
230     layout(graph);
231     vertices = loop.closeWith(graph.getVertices());
232 
233 
234     DataSet<EPGMVertex> gradoopVertices =
235       vertices.join(input.getVertices()).where(LVertex.ID_POSITION).equalTo("id")
236         .with(new LVertexEPGMVertexJoinFunction());
237 
238     return input.getFactory().fromDataSets(gradoopVertices, input.getEdges());
239   }

241   /**
242    * Simply translate the internal representations back to {@link LogicalGraph}.
243    *
244    * @param input    Original input graph
245    * @param layouted Result of the layouting
246    * @return The layouted graph in the Gradoop-format
247    */
248   protected LogicalGraph buildSimplifiedGraph(LogicalGraph input, LGraph layouted) {
249     return new LGraphToEPGMMapper().buildSimplifiedGraph(input, layouted);
250   }
251 
252   /**
253    * Extract all subverties/subedges from the super-vertices/super-edges and place them at the
254    * location of the super-vertex (and add some random jitter to the positions)
255    *
256    * @param input    Original input graph
257    * @param layouted Result of the layouting
258    * @param jitter   Enable/disable adding jitter to subvertex positions
259    * @return The final graph, containing all vertices and edges from the original graph.
260    */
261   protected LogicalGraph buildExtractedGraph(LogicalGraph input, LGraph layouted,
262     final boolean jitter) {
263 
264     DataSet<EPGMVertex> vertices =
265       layouted.getVertices().flatMap(new LVertexFlattener(jitter, getK())).join(input.getVertices())
266         .where(LVertex.ID_POSITION).equalTo(new Id<>()).with(new LVertexEPGMVertexJoinFunction());
267     return input.getFactory().fromDataSets(vertices, input.getEdges());
268   }
269 
270   @Override
271   protected DataSet<LVertex> applyForces(DataSet<LVertex> vertices, DataSet<Force> forces, int iterations) {
272     return vertices.join(forces).where(LVertex.ID_POSITION).equalTo(Force.ID_POSITION).with(applicator);
273   }
