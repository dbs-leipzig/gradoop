16  package org.gradoop.flink.model.impl.operators.kmeans.util;
17  
18  import java.io.Serializable;
19  import java.util.Objects;
20  
21  /**
22   * Point class that can be created by defining its coordinates.
23   * Extends the existing code snippet from GitHub repository
24   * https://github.com/TU-Berlin-DIMA/flink-tutorial.git
25   */
26  public class Point implements Serializable {
27  
28    /**
29     * Latitude coordinate of the point
30     */
31    private double lat;
32  
33    /**
34     * Longitude coordinate of the point
35     */
36    private double lon;
37  
38    /**
39     * Empty constructor of Point class
40     */
41    public Point() {
42    }
43  
44    /**
45     * Creates a point with the coordinates assigned to it.
46     *
47     * @param lat First spatial property of the point
48     * @param lon Second spatial property of thw point
49     */
50    public Point(double lat, double lon) {
51      this.lat = lat;
52      this.lon = lon;
53    }
54  
55    /**
56     * Defines how to sum up two points. Done by summing up the latitudes and longitudes of each point.
57     *
58     * @param other Point that is added
59     * @return Returns a point with the summed up longitudes and latitudes
60     */
61    public Point add(Point other) {
62      this.lat += other.getLat();
63      this.lon += other.getLon();
64      return this;
65    }
66  
67    /**
68     * Computes the euclidean distance between two points.
69     *
70     * @param other Point to which the distance is computed
71     * @return Returns the euclidean distance between the two points
72     */
73    public double euclideanDistance(Point other) {
74      return Math.sqrt(
75        (lat - other.getLat()) * (lat - other.getLat()) + (lon - other.getLon()) * (lon - other.getLon()));
76    }
77  
78    /**
79     * Defines how a point is divided by a value. Done by dividing the latitude and longitude by the value.
80     *
81     * @param val Value the points is divided by
82     * @return Returns the point with its divided latitude and longitude
83     */
84    public Point div(long val) {
85      lat /= val;
86      lon /= val;
87      return this;
88    }
89  
90    /**
91     * Gets the latitude.
92     *
93     * @return Returns the value of the latitude
94     */
95    public Double getLat() {
96      return this.lat;
97    }
98  
99    /**
100    * Gets Longitude
101    *
102    * @return Returns the value of the longitude
103    */
104   public double getLon() {
105     return this.lon;
106   }
107 
108   /**
109    * Sets Latitude
110    *
111    * @param lat New value for lat
112    */
113   public void setLat(double lat) {
114     this.lat = lat;
115   }
116 
117   /**
118    * Sets Longitude
119    *
120    * @param lon New value for lon
121    */
122   public void setLon(double lon) {
123     this.lon = lon;
124   }
125 
126 
127   /**
128    * Equals method implemented to compare two points.
129    *
130    * @param o Other point this point is compared to
131    * @return Returns if the points are equal
132    */
133   @Override
134   public boolean equals(Object o) {
135     if (this == o) {
136       return true;
137     }
138     if (o == null || getClass() != o.getClass()) {
139       return false;
140     }
141     Point point = (Point) o;
142     return Double.compare(point.getLat(), lat) == 0 && Double.compare(point.getLon(), lon) == 0;
143   }
144 
145   /**
146    * HashCode method implemented to compare two points.
147    *
148    * @return Returns a unique hash for this instance of point.
149    */
150   @Override
151   public int hashCode() {
152     return Objects.hash(lat, lon);
153   }
154 }
