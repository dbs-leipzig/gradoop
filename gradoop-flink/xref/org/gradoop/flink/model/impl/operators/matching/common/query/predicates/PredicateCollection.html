19  import org.gradoop.flink.model.impl.operators.matching.single.cypher.pojos.Embedding;

22  import java.io.Serializable;
23  import java.util.Iterator;
24  import java.util.List;
25  import java.util.Objects;
26  import java.util.Set;
27  
28  import static java.util.stream.Collectors.joining;
29  
30  /**
31   * represents a collection of predicates which are connected via a certain operator
32   *
33   * @param <P> the stored predicate type
34   */
35  public abstract class PredicateCollection<P> implements Iterable<P>, Serializable {
36    /**
37     * Holds the predicate collection
38     */
39    protected List<P> predicates;
40  
41    /**
42     * Returns the stored predicates
43     *
44     * @return predicates
45     */
46    public List<P> getPredicates() {
47      return this.predicates;
48    }
49  
50    /**
51     * Sets the predicates to the given list
52     *
53     * @param predicates new predicates
54     */
55    public void setPredicates(List<P> predicates) {
56      this.predicates = predicates;
57    }
58  
59    /**
60     * Add a single predicate to the collection
61     *
62     * @param predicate the predicate to be added
63     */
64    public void addPredicate(P predicate) {
65      this.predicates.add(predicate);
66    }
67  
68    /**
69     * Add a list of predicates
70     *
71     * @param predicateList the predicaes to be added
72     */
73    public void addPredicates(List<P> predicateList) {
74      this.predicates.addAll(predicateList);
75    }
76  
77    /**
78     * Returns the number of predicates contained in this predicate collection.
79     *
80     * @return number of predicates
81     */
82    public int size() {
83      return predicates.size();
84    }
85  
86    /**
87     * Evaluates the predicate collection with respect to the given Embedding
88     *
89     * @param embedding the embedding record holding the data
90     * @param metaData the embedding meta data
91     * @return evaluation result
92     */
93    public abstract boolean evaluate(Embedding embedding, EmbeddingMetaData metaData);
94  
95    /**
96     * Evaluates the predicate collection with respect to the given EPGMGraphElement
97     *
98     * @param element GraphElement under which the predicate will be evaluated
99     * @return evaluation result
100    */
101   public abstract boolean evaluate(GraphElement element);
102 
103   /**
104    * Retrieves a set of all variables included in the predicate collection
105    * @return set of variables
106    */
107   public abstract Set<String> getVariables();
108 
109   /**
110    * Retrieves a set of all property keys referenced by the predicate for a given variable
111    * @param variable the variable
112    * @return set of referenced properties
113    */
114   public abstract Set<String> getPropertyKeys(String variable);
115 
116   /**
117    * Stores the name of the operator predicates are connected with
118    *
119    * @return operator name
120    */
121   public abstract String operatorName();
122 
123   @Override
124   public Iterator<P> iterator() {
125     return predicates.iterator();
126   }
127 
128   @Override
129   public String toString() {
130     return "(" + predicates.stream()
131             .map(P::toString)
132             .collect(joining(" " + operatorName() + " ")) + ")";
133   }
134 
135   @Override
136   public boolean equals(Object o) {
137     if (this == o) {
138       return true;
139     }
140     if (o == null || getClass() != o.getClass()) {
141       return false;
142     }
143 
144     PredicateCollection predicateCollection = (PredicateCollection) o;
145 
146     return Objects.equals(predicates, predicateCollection.predicates);
147   }
148 
149   @Override
150   public int hashCode() {
151     return predicates != null ? predicates.hashCode() : 0;
152   }
153 }
