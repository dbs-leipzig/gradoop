16  package org.gradoop.flink.model.impl.operators.layouting.util;
17  
18  import org.apache.flink.api.java.DataSet;
19  
20  /**
21   * Lightweight/Layouting-Graph. This way we do not need to drag around
22   * a full {@link org.gradoop.flink.model.impl.epgm.LogicalGraph} through every operation.
23   */
24  public class LGraph {
25    /**
26     * Vertices of the graph
27     */
28    private DataSet<LVertex> vertices;
29    /**
30     * Edges of the graph
31     */
32    private DataSet<LEdge> edges;
33  
34    /**
35     * Create new graph
36     * @param vertices The vertices for the graph
37     * @param edges The edges for the graph
38     */
39    public LGraph(DataSet<LVertex> vertices, DataSet<LEdge> edges) {
40      this.vertices = vertices;
41      this.edges = edges;
42    }
43  
44    /**
45     * Create new graph
46     * @param g The GraphElements (vertices and edges) for the graph
47     */
48    public LGraph(DataSet<SimpleGraphElement> g) {
49      vertices = g.filter(e -> e instanceof LVertex).map(e -> (LVertex) e);
50      edges = g.filter(e -> e instanceof LEdge).map(e -> (LEdge) e);
51    }
52  
53    /**
54     * Get a DataSet containing all elements of the graph
55     * @return The DataSet
56     */
57    public DataSet<SimpleGraphElement> getGraphElements() {
58      return vertices.map(x -> (SimpleGraphElement) x).union(edges.map(x -> (SimpleGraphElement) x));
59    }
60  
61    /**
62     * Gets vertices
63     *
64     * @return value of vertices
65     */
66    public DataSet<LVertex> getVertices() {
67      return vertices;
68    }
69  
70    /**
71     * Sets vertices
72     *
73     * @param vertices the new value
74     */
75    public void setVertices(DataSet<LVertex> vertices) {
76      this.vertices = vertices;
77    }
78  
79    /**
80     * Gets edges
81     *
82     * @return value of edges
83     */
84    public DataSet<LEdge> getEdges() {
85      return edges;
86    }
87  
88    /**
89     * Sets edges
90     *
91     * @param edges the new value
92     */
93    public void setEdges(DataSet<LEdge> edges) {
94      this.edges = edges;
95    }
96  }
