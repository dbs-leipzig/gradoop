22  /**
23   * Represents a single node in a {@link QueryPlan}
24   */
25  public abstract class PlanNode {

27     * Describes the output of that node.
28     */
29    private EmbeddingMetaData embeddingMetaData;

32     * Recursively executes this node and returns the resulting {@link Embedding} data set.
33     *
34     * @return embeddings
35     */
36    public abstract DataSet<Embedding> execute();
37  
38    /**
39     * Returns the meta data describing the embeddings produced by this node.
40     *
41     * @return meta data describing the output
42     */
43    public EmbeddingMetaData getEmbeddingMetaData() {
44      if (this.embeddingMetaData == null) {
45        this.embeddingMetaData = computeEmbeddingMetaData();
46      }
47      return embeddingMetaData;
48    }
49  
50    /**
51     * Computes the meta data returned by the specific node.
52     *
53     * @return meta data
54     */
55    protected abstract EmbeddingMetaData computeEmbeddingMetaData();
