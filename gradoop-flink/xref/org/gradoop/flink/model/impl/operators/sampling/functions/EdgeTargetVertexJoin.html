16  package org.gradoop.flink.model.impl.operators.sampling.functions;

18  import org.apache.flink.api.common.functions.JoinFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.api.java.tuple.Tuple3;
21  import org.gradoop.common.model.api.entities.Edge;
22  import org.gradoop.common.model.api.entities.Vertex;
23  import org.gradoop.common.model.impl.id.GradoopId;
24  
25  /**
26   * Joins to get the edge target:<br>
27   * {@code (edge,edge.targetId,bool-source),(target) -> (edge,bool-source,(bool)target[propertyKey])}
28   *
29   * @param <V> The vertex type.
30   * @param <E> The edge type.
31   */
32  @FunctionAnnotation.ForwardedFieldsFirst({"f0->f0", "f2->f1"})
33  @FunctionAnnotation.ReadFieldsSecond("properties")
34  public class EdgeTargetVertexJoin<V extends Vertex, E extends Edge>
35    implements JoinFunction<Tuple3<E, GradoopId, Boolean>, V, Tuple3<E, Boolean, Boolean>> {
36  
37    /**
38     *  Reduce object instantiations
39     */
40    private Tuple3<E, Boolean, Boolean> reuse;
41  
42    /**
43     * Property key of vertex value
44     */
45    private final String propertyKey;
46  
47    /**
48     * Creates an instance of this join function
49     *
50     * @param propertyKey property key of marked vertex value
51     */
52    public EdgeTargetVertexJoin(String propertyKey) {
53      this.reuse = new Tuple3<>();
54      this.propertyKey = propertyKey;
55    }
56  
57    @Override
58    public Tuple3<E, Boolean, Boolean> join(Tuple3<E, GradoopId, Boolean> interim, V vertex) {
59      reuse.f0 = interim.f0;
60      reuse.f1 = interim.f2;
61      reuse.f2 = vertex.getPropertyValue(propertyKey).getBoolean();
62      return reuse;
63    }
64  }
