16  package org.gradoop.flink.model.impl.operators.neighborhood;
17  
18  import org.gradoop.flink.model.api.epgm.BaseGraph;
19  import org.gradoop.flink.model.api.functions.AggregateFunction;
20  import org.gradoop.flink.model.api.operators.UnaryBaseGraphToBaseGraphOperator;
21  
22  /**
23   * Super class for all neighborhood operators.
24   *
25   * @param <LG> graph type
26   */
27  public abstract class Neighborhood<LG extends BaseGraph>
28    implements UnaryBaseGraphToBaseGraphOperator<LG> {
29  
30    /**
31     * Defines which edge direction shall be considered for aggregation. Incoming edges, outgoing
32     * edges or both.
33     */
34    public enum EdgeDirection {
35      /**
36       * Incoming edge.
37       */
38      IN,
39      /**
40       * Outgoing edge.
41       */
42      OUT,
43      /**
44       * Incoming and outgoing edges.
45       */
46      BOTH
47    }
48  
49    /**
50     * EPGMEdge aggregate function.
51     */
52    private AggregateFunction function;
53  
54    /**
55     * Considered edge direction for aggregation.
56     */
57    private EdgeDirection direction;
58  
59    /**
60     * Valued constructor.
61     *
62     * @param function aggregate function
63     * @param direction considered edge direction
64     */
65    public Neighborhood(AggregateFunction function, EdgeDirection direction) {
66      this.function = function;
67      this.direction = direction;
68    }
69  
70    /**
71     * Returns the aggregate function.
72     *
73     * @return aggregate function
74     */
75    public AggregateFunction getFunction() {
76      return function;
77    }
78  
79    /**
80     * Returns the considered edge direction.
81     *
82     * @return edge direction
83     */
84    public EdgeDirection getDirection() {
85      return direction;
86    }
87  }
