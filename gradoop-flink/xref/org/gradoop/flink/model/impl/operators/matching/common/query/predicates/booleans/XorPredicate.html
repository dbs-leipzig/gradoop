19  import org.gradoop.flink.model.impl.operators.matching.common.query.predicates.QueryComparableFactory;

21  import org.gradoop.gdl.model.predicates.Predicate;

27  import java.util.Objects;

29  /**
30   * Wraps a {@link Xor} predicate
31   */
32  public class XorPredicate extends QueryPredicate {
33    /**
34     * Holdes the wrapped predicate
35     */
36    private final Xor xor;
37  
38    /**
39     * Optional factory for creating QueryComparables
40     */
41    private final QueryComparableFactory comparableFactory;
42  
43    /**
44     * Creates a new Wrapper
45     *
46     * @param xor The wrapped xor predicate
47     */
48    public XorPredicate(Xor xor) {
49      this(xor, null);
50    }
51  
52    /**
53     * Creates a new Wrapper
54     *
55     * @param xor The wrapped xor predicate
56     * @param comparableFactory factory for comparables
57     */
58    public XorPredicate(Xor xor, QueryComparableFactory comparableFactory) {
59      this.xor = xor;
60      this.comparableFactory = comparableFactory;
61    }
62  
63    /**
64     * Converts the predicate into conjunctive normal form
65     *
66     * @return predicate in cnf
67     */
68    public CNF asCNF() {
69      Predicate lhs = xor.getArguments()[0];
70      Predicate rhs = xor.getArguments()[1];
71  
72      QueryPredicate wrapper = QueryPredicate.createFrom(
73        new Or(new And(lhs, new Not(rhs)), new And(new Not(lhs), rhs)),
74        comparableFactory
75      );
76  
77      return wrapper.asCNF();
78    }
79  
80    /**
81     * Returns the wrapped left hand side predicate
82     *
83     * @return wrapped left hand side predicate
84     */
85    public QueryPredicate getLhs() {
86      return QueryPredicate.createFrom(xor.getArguments()[0], comparableFactory);
87    }
88  
89    /**
90     * Returns the wrapped right hand side predicate
91     *
92     * @return wrapped right hand side predicate
93     */
94    public QueryPredicate getRhs() {
95      return QueryPredicate.createFrom(xor.getArguments()[1], comparableFactory);
96    }
97  
98    @Override
99    public boolean equals(Object o) {
100     if (this == o) {
101       return true;
102     }
103     if (o == null || getClass() != o.getClass()) {
104       return false;
105     }
106 
107     XorPredicate that = (XorPredicate) o;
108 
109     return Objects.equals(xor, that.xor);
110   }
111 
112   @Override
113   public int hashCode() {
114     return xor != null ? xor.hashCode() : 0;
115   }
116 }
