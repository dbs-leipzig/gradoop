31  /**
32   * Converts an EPGM vertex to a Tuple2 with its graph ids in field 0 and a
33   * {@link IdWithCandidates} in field 1.
34   * <p>
35   * {@code (vId,props,graphs) -> {(graphId, (vid,candidates)}}
36   *
37   * @param <V> vertex type
38   */
39  public class BuildIdWithCandidatesAndGraphs<V extends Vertex>
40    extends AbstractBuilder<V, Tuple2<GradoopIdSet, IdWithCandidates<GradoopId>>> {
41    /**
42     * serial version uid
43     */
44    private static final long serialVersionUID = 42L;
45    /**
46     * Query vertices to match against.
47     */
48    private transient Collection<org.gradoop.gdl.model.Vertex> queryVertices;
49    /**
50     * Number of vertices in the query graph
51     */
52    private int vertexCount;
53    /**
54     * Reduce instantiations
55     */
56    private final Tuple2<GradoopIdSet, IdWithCandidates<GradoopId>> reuseTuple;
57  
58    /**
59     * Constructor
60     *
61     * @param query GDL query
62     */
63    public BuildIdWithCandidatesAndGraphs(String query) {
64      super(query);
65      reuseTuple = new Tuple2<>();
66      reuseTuple.f1 = new IdWithCandidates<>();
67    }
68  
69    @Override
70    public void open(Configuration parameters) throws Exception {
71      super.open(parameters);
72      queryVertices = getQueryHandler().getVertices();
73      vertexCount   = queryVertices.size();
74    }
75  
76    @Override
77    public Tuple2<GradoopIdSet, IdWithCandidates<GradoopId>> map(V v)
78      throws Exception {
79      reuseTuple.f0 = v.getGraphIds();
80      reuseTuple.f1.setId(v.getId());
81      reuseTuple.f1.setCandidates(getCandidates(vertexCount,
82        ElementMatcher.getMatches(v, queryVertices, DEFAULT_VERTEX_LABEL)));
83      return reuseTuple;
84    }
85  }
