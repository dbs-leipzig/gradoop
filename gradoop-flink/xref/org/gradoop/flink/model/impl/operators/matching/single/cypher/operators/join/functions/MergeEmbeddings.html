33  /**
34   * Given two input embeddings, the function merges them according to the given parameters and
35   * constraints.
36   *
37   * The constraints for merging are defined at {@link JoinEmbeddings}.
38   */
39  public class MergeEmbeddings implements
40    FlatJoinFunction<Embedding, Embedding, Embedding>,
41    FlatMapFunction<Tuple2<Embedding, Embedding>, Embedding> {
42  
43    /**
44     * Reduce object instantiations.
45     */
46    protected final Embedding reuseEmbedding;
47    /**
48     * Non-Join columns from the right side.
49     */
50    private final int[] nonJoinColumnsRight;
51    /**
52     * Number of join columns of the right side.
53     */
54    private final int joinColumnsRightSize;
55    /**
56     * Vertex columns of the left embedding that need to have distinct id values.
57     */
58    private final int[] distinctVertexColumnsLeft;
59    /**
60     * Vertex columns of the right embedding that need to have distinct id values.
61     */
62    private final int[] distinctVertexColumnsRight;
63    /**
64     * Edge columns of the left embedding that need to have distinct id values.
65     */
66    private final int[] distinctEdgeColumnsLeft;
67    /**
68     * Edge columns of the right embedding that need to have distinct id values.
69     */
70    private final int[] distinctEdgeColumnsRight;
71    /**
72     * Flag, if vertex distinctiveness needs to be checked.
73     */
74    private final boolean checkDistinctVertices;
75    /**
76     * Flag, if vertex distinctiveness needs to be checked.
77     */
78    private final boolean checkDistinctEdges;
79  
80    /**
81     * Creates a new UDF instance.
82     *
83     * @param rightColumns number of columns in the right embedding
84     * @param joinColumnsRight join columns of the right side
85     * @param distinctVertexColumnsLeft distinct vertex columns of the left embedding
86     * @param distinctVertexColumnsRight distinct vertex columns of the right embedding
87     * @param distinctEdgeColumnsLeft distinct edge columns of the left embedding
88     * @param distinctEdgeColumnsRight distinct edge columns of the right embedding
89     */
90    public MergeEmbeddings(int rightColumns,
91      List<Integer> joinColumnsRight,
92      List<Integer> distinctVertexColumnsLeft,
93      List<Integer> distinctVertexColumnsRight,
94      List<Integer> distinctEdgeColumnsLeft,
95      List<Integer> distinctEdgeColumnsRight) {
96  
97      this.nonJoinColumnsRight = IntStream.range(0, rightColumns)
98        .filter(col -> !joinColumnsRight.contains(col))
99        .toArray();
100     this.joinColumnsRightSize = joinColumnsRight.size();
101 
102     ToIntFunction<Integer> f = i -> i;
103     this.distinctVertexColumnsLeft = distinctVertexColumnsLeft.stream().mapToInt(f).toArray();
104     this.distinctVertexColumnsRight = distinctVertexColumnsRight.stream().mapToInt(f).toArray();
105     this.distinctEdgeColumnsLeft = distinctEdgeColumnsLeft.stream().mapToInt(f).toArray();
106     this.distinctEdgeColumnsRight = distinctEdgeColumnsRight.stream().mapToInt(f).toArray();
107 
108     this.checkDistinctVertices = distinctVertexColumnsLeft.size() > 0 ||
109       distinctVertexColumnsRight.size() > 0;
110     this.checkDistinctEdges = distinctEdgeColumnsLeft.size() > 0 ||
111       distinctEdgeColumnsRight.size() > 0;
112     this.reuseEmbedding = new Embedding();
113   }
114 
115   @Override
116   public void join(Embedding left, Embedding right, Collector<Embedding> out) throws Exception {
117     if (isValid(left, right)) {
118       buildEmbedding(left, right);
119       out.collect(reuseEmbedding);
120     }
121   }
122 
123   @Override
124   public void flatMap(Tuple2<Embedding, Embedding> value, Collector<Embedding> out)
125       throws Exception {
126     join(value.f0, value.f1, out);
127   }

129   /**
130    * Checks if the merged embedding would hold under morphism setting
131    * @param left left embedding
132    * @param right right embedding
133    * @return true if the morphism condition holds
134    */
135   protected boolean isValid(Embedding left, Embedding right) {
136     boolean collect = false;
137 
138     // Vertex-Homomorphism + Edge-Homomorphism
139     if (!checkDistinctVertices && !checkDistinctEdges) {
140       collect = true;
141       // Vertex-Homomorphism + Edge-Isomorphism
142     } else if (!checkDistinctVertices) {
143       if (isDistinct(distinctEdgeColumnsLeft, distinctEdgeColumnsRight, left, right)) {
144         collect = true;
145       }
146       // Vertex-Isomorphism + Edge-Isomorphism
147     } else {
148       if (isDistinct(distinctVertexColumnsLeft, distinctVertexColumnsRight, left, right) &&
149         isDistinct(distinctEdgeColumnsLeft, distinctEdgeColumnsRight, left, right)) {
150         collect = true;
151       }
152     }
153 
154     return collect;
155   }
156 
157   /**
158    * Merges left and right embeddings into {@link MergeEmbeddings#reuseEmbedding}
159    * @param left left embedding
160    * @param right right embedding
161    */
162   protected void buildEmbedding(Embedding left, Embedding right) {
163     reuseEmbedding.setIdData(mergeIdData(left, right));
164     reuseEmbedding.setPropertyData(mergePropertyData(left, right));
165     reuseEmbedding.setIdListData(mergeIdListData(left, right));
166   }

168   /**
169    * Checks if both given embeddings contain distinct id values at all specified columns.
170    *
171    * @param columnsLeft left columns to include in check for uniqueness
172    * @param columnsRight right columns to include in check for uniqueness
173    * @param left left embedding
174    * @param right right embedding
175    * @return true, if both embeddings contain unique id values for all specified columns
176    */
177   private boolean isDistinct(int[] columnsLeft, int[] columnsRight,
178     Embedding left, Embedding right) {
179 
180     Set<GradoopId> ids = new HashSet<>(left.size() + right.size());
181     return isDistinct(ids, columnsLeft, left) && isDistinct(ids, columnsRight, right);
182   }
183 
184   /**
185    * Checks if the specified embeddings contains distinct ids at the specified columns.
186    *
187    * @param ids used to track uniqueness
188    * @param columns columns to check for uniqueness
189    * @param embedding embedding to check
190    * @return true, if the embedding contains distinct Ids at the specified columns
191    */
192   private boolean isDistinct(Set<GradoopId> ids, int[] columns, Embedding embedding) {
193     boolean isDistinct = true;
194     for (int column : columns) {
195       isDistinct = ids.addAll(embedding.getIdAsList(column));
196       if (!isDistinct) {
197         break;
198       }
199     }
200     return isDistinct;
201   }

203   /**
204    * Merges the idData columns of left and right
205    * All entries of left are kept as well as all right entries which aren't join columns
206    *
207    * @param left the left hand side embedding
208    * @param right the right hand side embedding
209    * @return the merged data represented as byte array
210    */
211   private byte[] mergeIdData(Embedding left, Embedding right) {
212     byte[] newIdData = new byte[
213       left.getIdData().length +
214       right.getIdData().length -
215       (joinColumnsRightSize * (Embedding.ID_ENTRY_SIZE))
216     ];
217 
218     int offset = left.getIdData().length;
219     System.arraycopy(left.getIdData(), 0, newIdData, 0, offset);
220 
221     for (int i : nonJoinColumnsRight) {
222       System.arraycopy(right.getRawIdEntry(i), 0, newIdData, offset, Embedding.ID_ENTRY_SIZE);
223       offset += Embedding.ID_ENTRY_SIZE;
224     }
225 
226     return newIdData;
227   }
228 
229   /**
230    * Merges the propertyData columns of the left and right embeddings
231    * All entries from both sides are kept.
232    *
233    * @param left the left hand side embedding
234    * @param right the right hand side embedding
235    * @return the merged data represented as byte array
236    */
237   private byte[] mergePropertyData(Embedding left, Embedding right) {
238     return ArrayUtils.addAll(left.getPropertyData(), right.getPropertyData());
239   }
240 
241   /**
242    * Merges the idListData columns of the left and right embeddings
243    * All entries from both sides are kept.
244    *
245    * @param left the left hand side embedding
246    * @param right the right hand side embedding
247    * @return the merged data represented as byte array
248    */
249   private byte[] mergeIdListData(Embedding left, Embedding right) {
250     return ArrayUtils.addAll(left.getIdListData(), right.getIdListData());
251   }
