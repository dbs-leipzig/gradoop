16  package org.gradoop.flink.model.impl.operators.grouping.functions;
17  
18  import org.apache.flink.api.common.functions.FlatMapFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.util.Collector;
21  import org.gradoop.common.model.api.entities.Edge;
22  import org.gradoop.flink.model.impl.operators.grouping.tuples.EdgeGroupItem;
23  import org.gradoop.flink.model.impl.operators.grouping.tuples.LabelGroup;
24  
25  import java.util.List;
26  
27  /**
28   * Takes an EPGM edge as input and creates a {@link EdgeGroupItem} which
29   * contains only necessary information for further processing.
30   *
31   * @param <E> The edge type.
32   */
33  @FunctionAnnotation.ForwardedFields("sourceId->f0;targetId->f1;")
34  @FunctionAnnotation.ReadFields("label;properties")
35  public class BuildEdgeGroupItem<E extends Edge>
36    extends BuildGroupItemBase
37    implements FlatMapFunction<E, EdgeGroupItem> {
38  
39    /**
40     * Avoid object initialization in each call.
41     */
42    private final EdgeGroupItem reuseEdgeGroupItem;
43  
44    /**
45     * Creates map function.
46     *
47     * @param useLabel true, if vertex label shall be used
48     * @param edgeLabelGroups stores grouping properties for edge labels
49     */
50    public BuildEdgeGroupItem(boolean useLabel, List<LabelGroup> edgeLabelGroups) {
51      super(useLabel, edgeLabelGroups);
52      this.reuseEdgeGroupItem = new EdgeGroupItem();
53    }
54  
55    @Override
56    public void flatMap(E edge, Collector<EdgeGroupItem> collector) throws Exception {
57      boolean usedEdgeLabelGroup = false;
58      reuseEdgeGroupItem.setSourceId(edge.getSourceId());
59      reuseEdgeGroupItem.setTargetId(edge.getTargetId());
60  
61      // check if edge shall be grouped by a special set of keys
62      for (LabelGroup edgeLabelGroup : getLabelGroups()) {
63        if (edgeLabelGroup.getGroupingLabel().equals(edge.getLabel())) {
64          usedEdgeLabelGroup = true;
65          setGroupItem(reuseEdgeGroupItem, edge, edgeLabelGroup);
66          collector.collect(reuseEdgeGroupItem);
67        }
68      }
69      // standard grouping case
70      if (!usedEdgeLabelGroup) {
71        setGroupItem(reuseEdgeGroupItem, edge, getDefaultLabelGroup());
72        collector.collect(reuseEdgeGroupItem);
73      }
74    }
75  }
