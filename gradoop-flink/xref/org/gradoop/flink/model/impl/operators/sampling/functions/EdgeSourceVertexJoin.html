16  package org.gradoop.flink.model.impl.operators.sampling.functions;
17  
18  import org.apache.flink.api.common.functions.JoinFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.api.java.tuple.Tuple3;
21  import org.gradoop.common.model.api.entities.Edge;
22  import org.gradoop.common.model.api.entities.Vertex;
23  import org.gradoop.common.model.impl.id.GradoopId;
24  
25  /**
26   * Joins to get the edge source:<br>
27   * {@code (edge),(vertex) -> (edge,edge.targetId,(bool)vertex[propertyKey])}
28   *
29   * @param <V> The vertex type.
30   * @param <E> The edge type.
31   */
32  @FunctionAnnotation.ForwardedFieldsFirst({"*->f0", "id->f1"})
33  @FunctionAnnotation.ReadFieldsSecond("properties")
34  public class EdgeSourceVertexJoin<V extends Vertex, E extends Edge>
35    implements JoinFunction<E, V, Tuple3<E, GradoopId, Boolean>> {
36    /**
37     * Reduce object instantiations
38     */
39    private Tuple3<E, GradoopId, Boolean> reuse;
40  
41    /**
42     * Property key of marked value
43     */
44    private String propertyKey;
45  
46    /**
47     * Creates an instance of this join function
48     *
49     * @param propertyKey vertex property key
50     */
51    public EdgeSourceVertexJoin(String propertyKey) {
52      this.reuse = new Tuple3<>();
53      this.propertyKey = propertyKey;
54    }
55  
56    @Override
57    public Tuple3<E, GradoopId, Boolean> join(E edge, V vertex) throws Exception {
58      reuse.f0 = edge;
59      reuse.f1 = edge.getTargetId();
60      reuse.f2 = vertex.getPropertyValue(propertyKey).getBoolean();
61      return reuse;
62    }
63  }
