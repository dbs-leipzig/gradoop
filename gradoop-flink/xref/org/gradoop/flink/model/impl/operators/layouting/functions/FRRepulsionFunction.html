16  package org.gradoop.flink.model.impl.operators.layouting.functions;

18  import org.apache.flink.api.common.functions.CrossFunction;
19  import org.apache.flink.api.common.functions.FlatJoinFunction;
20  import org.apache.flink.api.common.functions.JoinFunction;
21  import org.apache.flink.util.Collector;
22  import org.gradoop.flink.model.impl.operators.layouting.util.Force;
23  import org.gradoop.flink.model.impl.operators.layouting.util.LVertex;
24  import org.gradoop.flink.model.impl.operators.layouting.util.Vector;
25  
26  import java.util.Random;
27  
28  /**
29   * A Join/Cross/FlatJoin-Function that computes the repulsion-forces between two given vertices.
30   * Implements both Join and FlatJoin to compute repulsions for a single vertex or for both
31   * vertices at once.
32   */
33  public class FRRepulsionFunction implements JoinFunction<LVertex, LVertex, Force>,
34    CrossFunction<LVertex, LVertex, Force>, FlatJoinFunction<LVertex, LVertex, Force> {
35    /**
36     * Rng. Used to get random directions for vertices at the same position
37     */
38    private Random rng;
39    /**
40     * Parameter for the FR-Algorithm
41     */
42    private double k;
43    /**
44     * Maximum distance between two vertices that still produces a repulsion
45     */
46    private double maxDistance;
47  
48    /**
49     * Object reuse for output
50     */
51    private Force firstForce = new Force();
52    /**
53     * Object reuse for output
54     */
55    private Force secondForce = new Force();
56    /**
57     * Object reuse
58     */
59    private Vector calculatedForce = new Vector();
60    /**
61     * Object reuse
62     */
63    private Vector calculatedForce2 = new Vector();
64  
65    /**
66     * Create new RepulsionFunction
67     *
68     * @param k A parameter of the FR-Algorithm
69     */
70    public FRRepulsionFunction(double k) {
71      this(k, Float.MAX_VALUE);
72    }
73  
74    /**
75     * Create new RepulsionFunction
76     *
77     * @param k           A parameter of the FR-Algorithm
78     * @param maxDistance Maximum distance between two vertices that still produces a repulsion
79     */
80    public FRRepulsionFunction(double k, double maxDistance) {
81      rng = new Random();
82      this.k = k;
83      this.maxDistance = maxDistance;
84    }
85  
86    /**
87     * Computes the repulsion force between two vertices ONLY FOR THE FIRST vertex
88     *
89     * @param first  First Vertex
90     * @param second Second Certex
91     * @return A force-tuple representing the repulsion-force for the first vertex
92     */
93    @Override
94    public Force join(LVertex first, LVertex second) {
95      Vector force = calculateForce(first, second);
96      firstForce.set(first.getId(), force);
97      return firstForce;
98    }
99  
100   /**
101    * Alias for join() to fullfill the CrossFunction-Interface.
102    *
103    * @param vertex  First Vertex
104    * @param vertex2 Second Certex
105    * @return A force-tuple representing the repulsion-force for the first vertex
106    */
107   @Override
108   public Force cross(LVertex vertex, LVertex vertex2) {
109     return join(vertex, vertex2);
110   }
111 
112 
113   /**
114    * Computes the repulsion force between two vertices ONLY FOR THE FIRST vertex
115    * ATTENTION! Subsequent calls will modify previously returned results!
116    *
117    * @param first  First Vertex
118    * @param second Second Certex
119    * @return A force-tuple representing the repulsion-force for the first vertex
120    */
121   protected Vector calculateForce(LVertex first, LVertex second) {
122     Vector pos1 = first.getPosition();
123     Vector pos2 = second.getPosition();
124     double distance = pos1.distance(pos2);
125     Vector direction = pos2.mSub(pos1);
126 
127     if (first.getId().equals(second.getId())) {
128       calculatedForce.reset();
129       return calculatedForce;
130     }
131 
132     if (distance > maxDistance) {
133       calculatedForce.reset();
134       return calculatedForce;
135     }
136 
137     if (distance == 0) {
138       // generate a pseudo-random (but deterministic) direction
139       int hash = first.getId().hashCode() + second.getId().hashCode();
140       int x = hash >> 16;
141       int y = hash & 0xFFFF;
142       distance = 0.1;
143       direction.setX(x);
144       direction.setY(y);
145       if (first.getId().compareTo(second.getId()) > 0) {
146         direction.mMul(-1);
147       }
148     }
149 
150     calculatedForce.set(direction.mNormalized()
151       .mMul(-Math.pow(k, 2) / distance).mMul(first.getCount() * second.getCount()));
152     return calculatedForce;
153   }
154 
155   /**
156    * Implement FlatJoin and produce the force-tuples for both vertices at once.
157    * (Forces of magnitude 0 will be ignored)
158    *
159    * @param first     The first vertex
160    * @param second    The second vertex
161    * @param collector Contains up to two force-tuples representing repulsion-forces between both
162    *                  vertices.
163    */
164   @Override
165   public void join(LVertex first, LVertex second, Collector<Force> collector) {
166 
167     Vector force = calculateForce(first, second);
168 
169     if (force.magnitude() == 0) {
170       return;
171     }
172 
173     firstForce.set(first.getId(), force);
174 
175     calculatedForce2.set(force).mMul(-1);
176     secondForce.set(second.getId(), calculatedForce2);
177 
178     collector.collect(firstForce);
179     collector.collect(secondForce);
180   }
