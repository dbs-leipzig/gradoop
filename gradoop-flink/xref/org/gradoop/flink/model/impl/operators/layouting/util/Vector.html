24  /**
25   * Simple helper-class for some vector-math.
26   * All math-operations will return a new Vector (instead of modifying the existing vector). This
27   * prevents strange side-effects when performing complex computations.
28   */
29  public class Vector implements Serializable {
30  
31    /**
32     * X-coordinate of vector
33     */
34    private double x;
35    /**
36     * Y-coordinate of vector
37     */
38    private double y;

40    /**
41     * Construct a vector from x and y coordinates
42     *
43     * @param x X-Coordinate of the new vector
44     * @param y Y-Coordinate of the new vector
45     */
46    public Vector(double x, double y) {
47      this.x = x;
48      this.y = y;
49      check();
50    }
51  
52    /**
53     * Construct new zero-Vector
54     */
55    public Vector() {
56      x = 0d;
57      y = 0d;
58    }

60    /**
61     * Ensure that this vector does not contain any NaN-values.
62     * Throws IllegalStateException if x or y is NaN.
63     */
64    private void check() {
65      if (Double.isNaN(x) || Double.isNaN(y)) {
66        throw new IllegalStateException(
67          "A vector should never be NaN. There is probably a bug in " + "your code");
68      }
69    }

71    /**
72     * Create a vector from the coordinate-properties of a Vertex
73     *
74     * @param v The vertex to extract position coordinates from properties (X,Y)
75     * @return A matching vector
76     */
77    public static Vector fromVertexPosition(EPGMVertex v) {
78      double x = v.getPropertyValue(LayoutingAlgorithm.X_COORDINATE_PROPERTY).getInt();
79      double y = v.getPropertyValue(LayoutingAlgorithm.Y_COORDINATE_PROPERTY).getInt();
80      return new Vector(x, y);
81    }

83    /**
84     * Set the coordinate-properties of a vertex to the values of this vector
85     *
86     * @param v The vertex that will receive the values of this vector as coordinates
87     */
88    public void setVertexPosition(EPGMVertex v) {
89      v.setProperty(LayoutingAlgorithm.X_COORDINATE_PROPERTY, (int) x);
90      v.setProperty(LayoutingAlgorithm.Y_COORDINATE_PROPERTY, (int) y);
91    }

133   /**
134    * Calculate the euclidean distance between this vector and another vector
135    *
136    * @param other The other vector
137    * @return Math.sqrt(Math.pow ( x - other.x, 2) + Math.pow(y - other.y, 2))
138    */
139   public double distance(Vector other) {
140     return Math.sqrt(Math.pow(x - other.x, 2) + Math.pow(y - other.y, 2));
141   }

143   /**
144    * Calculate the scalar-product of this vector
145    *
146    * @param other The other vector
147    * @return Skalar-product of this and other
148    */
149   public double scalar(Vector other) {
150     return x * other.x + y * other.y;
151   }

153   /**
154    * Clamp this vector to a given length.
155    * The returned vector will have the same orientation as this one, but will have at most a
156    * length of maxLen.
157    * If maxLen is smaller the the lenght of this Vector this vector (a copy of it) will be returned.
158    *
159    * @param maxLen maximum lenght of vector
160    * @return This vector but constrained to the given length
161    */
162   public Vector clamped(double maxLen) {
163     double len = magnitude();
164     if (len == 0) {
165       return new Vector(0, 0);
166     }
167     double newx = (x / len) * Math.min(len, maxLen);
168     double newy = (y / len) * Math.min(len, maxLen);
169     return new Vector(newx, newy);
170   }

172   /**
173    * Normalize this vector.
174    *
175    * @return a vector with the same orientation as this one and a length of 1. If this vector is
176    * (0,0) then (0,0) will be returned instead.
177    */
178   public Vector normalized() {
179     double len = magnitude();
180     if (len == 0) {
181       return new Vector(0, 0);
182     }
183     double newx = x / len;
184     double newy = y / len;
185     return new Vector(newx, newy);
186   }

188   /**
189    * Get the lenght of this vector
190    *
191    * @return euclidean length of this vector
192    */
193   public double magnitude() {
194     return Math.sqrt(Math.pow(x, 2) + Math.pow(y, 2));
195   }

197   /**
198    * Confine this point to the given bounding-box.
199    *
200    * @param minX Bounding-box
201    * @param maxX Bounding-box
202    * @param minY Bounding-box
203    * @param maxY Bounding-box
204    * @return A vector that does not violate the given bounding box.
205    */
206   public Vector confined(double minX, double maxX, double minY, double maxY) {
207     double newx = Math.min(Math.max(x, minX), maxX);
208     double newy = Math.min(Math.max(y, minY), maxY);
209     return new Vector(newx, newy);
210   }

212   /**
213    * Calculates the (unsigned) angle between this vector and the given vecotr
214    *
215    * @param other The other vector
216    * @return The angle
217    */
218   public double angle(Vector other) {
219     return Math.acos(scalar(other) / (magnitude() * other.magnitude())) / Math.PI * 180;
220   }

222   /**
223    * Rotate this vector anti-clockwise by the given angle (in degrees)
224    *
225    * @param angle The angle o rotate this vector by
226    * @return The rotated vector
227    */
228   public Vector rotate(double angle) {
229     angle = angle / 180 * Math.PI;
230     double newx = x * Math.cos(angle) - y * Math.sin(angle);
231     double newy = x * Math.sin(angle) + y * Math.cos(angle);
232     return new Vector(newx, newy);
233   }
