35  /**
36   * Updates the state of a {@link FatVertex} according to the message it
37   * receives.
38   *
39   * Forwarded Fields First:
40   *
41   * f0: vertex id
42   */
43  @FunctionAnnotation.ForwardedFieldsFirst("f0")
44  public class UpdateVertexState
45    extends RichJoinFunction<FatVertex, Message, FatVertex> {
46  
47    /**
48     * serial version uid
49     */
50    private static final long serialVersionUID = 42L;
51  
52    /**
53     * GDL query
54     */
55    private final String query;
56  
57    /**
58     * Query handler
59     */
60    private transient QueryHandler queryHandler;
61  
62    /**
63     * Constructor
64     *
65     * @param query GDL query
66     */
67    public UpdateVertexState(String query) {
68      this.query = query;
69    }
70  
71    @Override
72    public void open(Configuration parameters) throws Exception {
73      super.open(parameters);
74      queryHandler = new QueryHandler(query);
75    }
76  
77    @Override
78    public FatVertex join(FatVertex fatVertex, Message message) throws Exception {
79      if (message != null) {
80        // TODO: message to SELF should be processed first
81        for (int i = 0; i < message.getSenderIds().size(); i++) {
82          processDeletion(fatVertex, message.getSenderIds().get(i),
83            message.getDeletions().get(i), message.getMessageTypes().get(i));
84        }
85        fatVertex.setUpdated(true);
86      } else {
87        fatVertex.setUpdated(false);
88      }
89  
90      return fatVertex;
91    }
92  
93    /**
94     * Processes a deletion on the current vertex.
95     *
96     * @param fatVertex   fat vertex
97     * @param senderId    sender vertexId
98     * @param deletion    sender vertex candidate deletion id
99     * @param messageType message type
100    */
101   private void processDeletion(FatVertex fatVertex, GradoopId senderId,
102     Long deletion, MessageType messageType) {
103     switch (messageType) {
104     case FROM_SELF:
105       updateCandidates(fatVertex, deletion);
106       break;
107     case FROM_CHILD:
108       updateOutgoingEdges(fatVertex,
109         queryHandler.getEdgeIdsByTargetVertexId(deletion), senderId);
110       break;
111     case FROM_PARENT:
112       updateIncomingEdges(fatVertex,
113         queryHandler.getEdgeIdsBySourceVertexId(deletion));
114       break;
115     case FROM_CHILD_REMOVE:
116       updateOutgoingEdges(fatVertex, senderId);
117       break;
118     case FROM_PARENT_REMOVE:
119       updateIncomingEdges(fatVertex,
120         queryHandler.getEdgeIdsBySourceVertexId(deletion));
121       updateParentIds(fatVertex, senderId);
122       break;
123     default:
124       throw new IllegalArgumentException("Unsupported type: " + messageType);
125     }
126   }
127 
128   /**
129    * Removes the sender vertex id from the parents of the current vertex.
130    *
131    * @param fatVertex fat vertex
132    * @param senderId  sender vertex id
133    */
134   private void updateParentIds(FatVertex fatVertex, GradoopId senderId) {
135     fatVertex.getParentIds().remove(senderId);
136   }
137 
138   /**
139    * Updates the candidates of the given vertex. If the vertex has still
140    * candidates left, the outgoing edges will be updated accordingly.
141    *
142    * @param fatVertex fat vertex
143    * @param deletion  vertex candidate to be removed
144    */
145   private void updateCandidates(FatVertex fatVertex, Long deletion) {
146     fatVertex.getCandidates().remove(deletion);
147     if (!fatVertex.getCandidates().isEmpty()) {
148       updateOutgoingEdges(fatVertex,
149         queryHandler.getEdgeIdsBySourceVertexId(deletion));
150     }
151   }
152 
153   /**
154    * Updates incoming edge of the given vertex. The method decrements the
155    * counter for each query edge candidate.
156    *
157    * @param fatVertex   fat vertex
158    * @param queryEdges  edge candidates to be removed
159    */
160   private void updateIncomingEdges(FatVertex fatVertex, Collection<Long>
161     queryEdges) {
162     for (Long eQ : queryEdges) {
163       if (fatVertex.getIncomingCandidateCounts()[eQ.intValue()] > 0) {
164         fatVertex.getIncomingCandidateCounts()[eQ.intValue()]--;
165       }
166     }
167   }
168 
169   /**
170    * Deletes all outgoing edges which point to the given target id.
171    *
172    * @param vertex    fat vertex
173    * @param targetId  target vertex
174    */
175   private void updateOutgoingEdges(FatVertex vertex, GradoopId targetId) {
176     Iterator<IdPair> iterator = vertex.getEdgeCandidates().keySet().iterator();
177     while (iterator.hasNext()) {
178       if (iterator.next().getTargetId().equals(targetId)) {
179         iterator.remove();
180       }
181     }
182   }
183 
184   /**
185    * Updates the outgoing edges of the given vertex according to a collection
186    * of edge candidates which will be deleted.
187    *
188    * @param fatVertex   fat vertex
189    * @param queryEdges  edge candidates to be removed
190    */
191   private void updateOutgoingEdges(FatVertex fatVertex,
192     Collection<Long> queryEdges) {
193     updateOutgoingEdges(fatVertex, queryEdges, null);
194   }
195 
196   /**
197    * Updates the outgoing edges of the given vertex according to a collection of
198    * edge candidates which will be deleted. Updates only those outgoing edges
199    * which point to a specific target vertex.
200    *
201    * @param fatVertex     fat vertex
202    * @param queryEdges    edge candidates to be removed
203    * @param targetVertex  target vertex to consider
204    */
205   private void updateOutgoingEdges(FatVertex fatVertex,
206     Collection<Long> queryEdges, GradoopId targetVertex) {
207     if (queryEdges != null) {
208       Iterator<Map.Entry<IdPair, boolean[]>> edgeIterator = fatVertex
209         .getEdgeCandidates().entrySet().iterator();
210       while (edgeIterator.hasNext()) {
211         Map.Entry<IdPair, boolean[]> e = edgeIterator.next();
212         if (targetVertex == null ||
213           e.getKey().getTargetId().equals(targetVertex)) {
214           // remove edge candidates for that edge
215           for (Long eQ : queryEdges) {
216             e.getValue()[eQ.intValue()] = false;
217           }
218           // remove edge if there are no candidates left
219           if (allFalse(e.getValue())) {
220             edgeIterator.remove();
221           }
222         }
223       }
224     }
225   }
226 
227   /**
228    * Checks if the given array contains only false values.
229    *
230    * @param a array
231    * @return true, iff all values are false
232    */
233   private boolean allFalse(boolean[] a) {
234     for (boolean b : a) {
235       if (b) {
236         return false;
237       }
238     }
239     return true;
240   }
