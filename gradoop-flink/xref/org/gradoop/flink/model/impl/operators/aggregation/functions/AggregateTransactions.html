16  package org.gradoop.flink.model.impl.operators.aggregation.functions;
17  
18  import org.apache.flink.api.common.functions.MapFunction;
19  import org.gradoop.common.model.impl.pojo.EPGMEdge;
20  import org.gradoop.common.model.impl.pojo.EPGMVertex;
21  import org.gradoop.common.model.impl.properties.PropertyValue;
22  import org.gradoop.flink.model.api.functions.AggregateFunction;
23  import org.gradoop.flink.model.impl.layouts.transactional.tuples.GraphTransaction;
24  
25  import java.util.HashMap;
26  import java.util.Map;
27  import java.util.Set;
28  import java.util.stream.Collectors;
29  
30  /**
31   * Applies an {@link AggregateFunction} to the vertex or edge set of a graph transaction.
32   */
33  public class AggregateTransactions implements MapFunction<GraphTransaction, GraphTransaction> {
34  
35    /**
36     * Set of all aggregate functions.
37     */
38    private final Set<AggregateFunction> aggregateFunctions;
39    /**
40     * Set of aggregate vertex functions.
41     */
42    private final Set<AggregateFunction> vertexAggregateFunctions;
43    /**
44     * Set of aggregate edge functions.
45     */
46    private final Set<AggregateFunction> edgeAggregateFunctions;
47  
48    /**
49     * Creates a new instance of a AggregateTransactions map function.
50     *
51     * @param aggregateFunctions vertex or edge aggregate functions with possible default value
52     */
53    public AggregateTransactions(Set<AggregateFunction> aggregateFunctions) {
54      // initialization logic to avoid instanceOf checking during execution
55      this.aggregateFunctions = aggregateFunctions;
56  
57      vertexAggregateFunctions = aggregateFunctions.stream()
58        .filter(AggregateFunction::isVertexAggregation)
59        .collect(Collectors.toSet());
60  
61      edgeAggregateFunctions = aggregateFunctions.stream()
62        .filter(AggregateFunction::isEdgeAggregation)
63        .collect(Collectors.toSet());
64    }
65  
66    @Override
67    public GraphTransaction map(GraphTransaction graphTransaction) throws Exception {
68      Map<String, PropertyValue> aggregate = new HashMap<>();
69      aggregate = aggregateVertices(aggregate, graphTransaction);
70      aggregate = aggregateEdges(aggregate, graphTransaction);
71  
72      for (AggregateFunction function : aggregateFunctions) {
73        aggregate.computeIfPresent(function.getAggregatePropertyKey(),
74          (k, v) -> function.postAggregate(v));
75        function.applyResult(graphTransaction.getGraphHead(), aggregate
76          .getOrDefault(function.getAggregatePropertyKey(), AggregateUtil.getDefaultAggregate(function)));
77      }
78      return graphTransaction;
79    }
80  
81    /**
82     * Applies the aggregate functions on the vertices of the given graph transaction.
83     *
84     * @param aggregate map to hold the aggregate values
85     * @param graphTransaction graph transaction
86     * @return final vertex aggregate value
87     */
88    private Map<String, PropertyValue> aggregateVertices(Map<String, PropertyValue> aggregate,
89                                                         GraphTransaction graphTransaction) {
90      for (EPGMVertex vertex : graphTransaction.getVertices()) {
91        aggregate = AggregateUtil.increment(aggregate, vertex, vertexAggregateFunctions);
92      }
93      return aggregate;
94    }
95  
96    /**
97     * Applies the aggregate functions on the edges of the given graph transaction.
98     *
99     * @param aggregate map to hold the aggregate values
100    * @param graphTransaction graph transaction
101    * @return final edge aggregate value
102    */
103   private Map<String, PropertyValue> aggregateEdges(Map<String, PropertyValue> aggregate,
104                                                     GraphTransaction graphTransaction) {
105     for (EPGMEdge edge : graphTransaction.getEdges()) {
106       aggregate = AggregateUtil.increment(aggregate, edge, edgeAggregateFunctions);
107     }
108     return aggregate;
109   }
110 }
