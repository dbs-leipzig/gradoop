38  /**
39   * Extracts a subgraph from a base graph using the given filter functions.
40   * The graph head stays unchanged for the subgraph.
41   * <p>
42   * The operator is able to:
43   * <ol>
44   *   <li>extract vertex-induced subgraph</li>
45   *   <li>extract edge-induced subgraph via join + union strategy</li>
46   *   <li>extract edge-induced subgraph via project + union + join strategy</li>
47   *   <li>extract subgraph based on vertex and edge filter function without verification
48   *   (no joins, use {@link Verify} to verify the subgraph)</li>
49   * </ol>
50   *
51   * @param <G>  The graph head type.
52   * @param <V>  The vertex type.
53   * @param <E>  The edge type.
54   * @param <LG> The type of the graph.
55   * @param <GC> The type of the graph collection.
56   */
57  public class Subgraph<
58    G extends GraphHead,
59    V extends Vertex,
60    E extends Edge,
61    LG extends BaseGraph<G, V, E, LG, GC>,
62    GC extends BaseGraphCollection<G, V, E, LG, GC>>
63    implements UnaryBaseGraphToBaseGraphOperator<LG> {
64  
65    /**
66     * Used to filter vertices from the graph.
67     */
68    protected final FilterFunction<V> vertexFilterFunction;
69  
70    /**
71     * Used to filter edges from the graph.
72     */
73    protected final FilterFunction<E> edgeFilterFunction;
74  
75    /**
76     * Execution strategy for the operator.
77     */
78    protected final Strategy strategy;
79  
80    /**
81     * Available execution strategies.
82     */
83    public enum Strategy {
84      /**
85       * Applies both filter functions on the input vertex and edge data set.
86       */
87      BOTH,
88      /**
89       * Only applies the vertex filter function and adds the incident edges connecting those
90       * vertices via a join.
91       */
92      VERTEX_INDUCED,
93      /**
94       * Only applies the edge filter function and computes the resulting vertices via:<br>
95       * {@code (E |><| V ON e.source = v.id) U (E |><| V on e.target = v.id)}
96       */
97      EDGE_INDUCED,
98      /**
99       * Only applies the edge filter function and computes the resulting vertices via:<br>
100      * {@code DISTINCT((π_source(E) U π_target(E))) |><| V}
101      */
102     EDGE_INDUCED_PROJECT_FIRST
103   }
104 
105   /**
106    * Creates a new sub graph operator instance.
107    * <p>
108    * If both parameters are not {@code null}, the operator returns the subgraph
109    * defined by filtered vertices and edges.
110    * <p>
111    * If the {@code edgeFilterFunction} is {@code null}, the operator returns the vertex-induced subgraph.
112    * <p>
113    * If the {@code vertexFilterFunction} is {@code null}, the operator returns the edge-induced subgraph.
114    *
115    * @param vertexFilterFunction vertex filter function
116    * @param edgeFilterFunction   edge filter function
117    * @param strategy             sets the execution strategy for the operator
118    */
119   public Subgraph(FilterFunction<V> vertexFilterFunction,
120     FilterFunction<E> edgeFilterFunction, Strategy strategy) {
121 
122     if (strategy == BOTH &&
123       (vertexFilterFunction == null || edgeFilterFunction == null)) {
124       throw new IllegalArgumentException("No vertex or no edge filter function was given.");
125     }
126 
127     if (strategy == VERTEX_INDUCED && vertexFilterFunction == null) {
128       throw new IllegalArgumentException("No vertex filter functions was given.");
129     }
130 
131     if ((strategy == EDGE_INDUCED || strategy == EDGE_INDUCED_PROJECT_FIRST) &&
132       edgeFilterFunction == null) {
133       throw new IllegalArgumentException("No vertex edge functions was given.");
134     }
135 
136     this.strategy = strategy;
137 
138     this.vertexFilterFunction = vertexFilterFunction;
139     this.edgeFilterFunction = edgeFilterFunction;
140   }
141 
142   @Override
143   public LG execute(LG superGraph) {
144 
145     LG result;
146     switch (strategy) {
147     case BOTH:
148       result = subgraph(superGraph);
149       break;
150     case VERTEX_INDUCED:
151       result = vertexInducedSubgraph(superGraph);
152       break;
153     case EDGE_INDUCED:
154       result = edgeInducedSubgraph(superGraph);
155       break;
156     case EDGE_INDUCED_PROJECT_FIRST:
157       result = edgeInducedSubgraphProjectFirst(superGraph);
158       break;
159     default:
160       throw new IllegalArgumentException("Strategy " + strategy + " is not supported");
161     }
162 
163     return result;
164   }
165 
166   /**
167    * Returns the subgraph of the given supergraph that is induced by the
168    * vertices that fulfil the given filter function.
169    *
170    * @param superGraph supergraph
171    * @return vertex-induced subgraph
172    */
173   private LG vertexInducedSubgraph(LG superGraph) {
174     DataSet<V> filteredVertices = superGraph.getVertices().filter(vertexFilterFunction);
175 
176     return superGraph.getFactory()
177       .fromDataSets(superGraph.getGraphHead(), filteredVertices, superGraph.getEdges())
178       .verify();
179   }
180 
181   /**
182    * Returns the subgraph of the given supergraph that is induced by the
183    * edges that fulfil the given filter function.
184    *
185    * @param superGraph supergraph
186    * @return edge-induced subgraph
187    */
188   private LG edgeInducedSubgraph(LG superGraph) {
189     DataSet<E> filteredEdges = superGraph.getEdges().filter(edgeFilterFunction);
190     DataSet<V> inducedVertices = filteredEdges
191       .join(superGraph.getVertices())
192       .where(new SourceId<>()).equalTo(new Id<>())
193       .with(new RightSide<>())
194       .union(filteredEdges
195         .join(superGraph.getVertices())
196         .where(new TargetId<>()).equalTo(new Id<>())
197         .with(new RightSide<>()))
198       .distinct(new Id<>());
199 
200     return superGraph.getFactory()
201       .fromDataSets(superGraph.getGraphHead(), inducedVertices, filteredEdges);
202   }
203 
204   /**
205    * Returns the subgraph of the given supergraph that is induced by the
206    * edges that fulfil the given filter function.
207    *
208    * @param superGraph supergraph
209    * @return edge-induced subgraph
210    */
211   private LG edgeInducedSubgraphProjectFirst(LG superGraph) {
212     DataSet<E> filteredEdges = superGraph.getEdges().filter(edgeFilterFunction);
213     DataSet<V> inducedVertices = filteredEdges
214       .flatMap(new EdgeToSourceAndTargetId<>())
215       .distinct()
216       .join(superGraph.getVertices())
217       .where("*").equalTo(new Id<>())
218       .with(new RightSide<>());
219 
220     return superGraph.getFactory()
221       .fromDataSets(superGraph.getGraphHead(), inducedVertices, filteredEdges);
222   }
223 
224   /**
225    * Returns the subgraph of the given supergraph that is defined by the
226    * vertices that fulfil the vertex filter function and edges that fulfill the edge filter function.
227    * <p>
228    * <b>Note:</b> The operator does not verify the consistency of the resulting graph.
229    *
230    * @param superGraph supergraph
231    * @return subgraph
232    */
233   private LG subgraph(LG superGraph) {
234     return superGraph.getFactory().fromDataSets(
235       superGraph.getGraphHead(),
236       superGraph.getVertices().filter(vertexFilterFunction),
237       superGraph.getEdges().filter(edgeFilterFunction));
238   }
