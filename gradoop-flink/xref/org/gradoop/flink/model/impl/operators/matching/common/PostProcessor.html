49    /**
50     * Extracts a {@link GraphCollection} from a set of {@link Element}.
51     *
52     * @param elements   elements
53     * @param factory    element factory
54     * @param mayOverlap elements may be contained in multiple graphs
55     * @param <G> The graph head type.
56     * @param <V> The vertex type.
57     * @param <E> The edge type.
58     * @param <LG> The graph type.
59     * @param <GC> The graph collection type.
60     * @return Graph collection
61     */
62    public static <G extends GraphHead,
63      V extends Vertex,
64      E extends Edge,
65      LG extends BaseGraph<G, V, E, LG, GC>,
66      GC extends BaseGraphCollection<G, V, E, LG, GC>> GC extractGraphCollection(
67      DataSet<Element> elements, BaseGraphCollectionFactory<G, V, E, LG, GC> factory, boolean mayOverlap) {
68  
69      Class<G> graphHeadType = factory.getGraphHeadFactory().getType();
70      Class<V> vertexType = factory.getVertexFactory().getType();
71      Class<E> edgeType = factory.getEdgeFactory().getType();
72      return factory.fromDataSets(
73        extractGraphHeads(elements, graphHeadType),
74        extractVertices(elements, vertexType, mayOverlap),
75        extractEdges(elements, edgeType, mayOverlap)
76      );
77    }

79    /**
80     * Extracts a {@link GraphCollection} from a set of {@link Element} and
81     * attaches the original data from the input {@link LogicalGraph}.
82     *
83     * @param elements      elements
84     * @param inputGraph    original input graph
85     * @param mayOverlap    true, if elements may be contained in multiple graphs
86     * @param <G> The graph head type.
87     * @param <V> The vertex type.
88     * @param <E> The edge type.
89     * @param <LG> The graph type.
90     * @param <GC> The graph collection type.
91     * @return Graph collection
92     */
93    public static <G extends GraphHead,
94      V extends Vertex,
95      E extends Edge,
96      LG extends BaseGraph<G, V, E, LG, GC>,
97      GC extends BaseGraphCollection<G, V, E, LG, GC>> GC extractGraphCollectionWithData(
98      DataSet<Element> elements, LG inputGraph, boolean mayOverlap) {
99  
100     BaseGraphCollectionFactory<G, V, E, LG, GC> factory = inputGraph.getCollectionFactory();
101 
102     // get result collection without data
103     GC collection = extractGraphCollection(elements, factory, mayOverlap);
104 
105     // attach data by joining first and merging the graph head ids
106     DataSet<V> newVertices = inputGraph.getVertices()
107       .rightOuterJoin(collection.getVertices())
108       .where(new Id<>()).equalTo(new Id<>())
109       .with(new MergedGraphIds<>())
110       .withForwardedFieldsFirst("id;label;properties;");
111 
112 
113     DataSet<E> newEdges = inputGraph.getEdges()
114       .rightOuterJoin(collection.getEdges())
115       .where(new Id<>()).equalTo(new Id<>())
116       .with(new MergedGraphIds<>())
117       .withForwardedFieldsFirst("id;label;properties");
118 
119     return factory.fromDataSets(collection.getGraphHeads(), newVertices, newEdges);
120   }
121   /**
122    * Extracts vertex ids from the given pattern matching result.
123    *
124    * @param result pattern matching result
125    * @return dataset with (vertexId) tuples
126    */
127   private static DataSet<Tuple1<GradoopId>> extractVertexIds(
128     DataSet<FatVertex> result) {
129     return result.project(0);
130   }
131 
132   /**
133    * Extracts edge-source-target id triples from the given pattern matching
134    * result.
135    *
136    * @param result pattern matching result
137    * @return dataset with (edgeId, sourceId, targetId) tuples
138    */
139   private static DataSet<Tuple3<GradoopId, GradoopId, GradoopId>> extractEdgeIds(
140     DataSet<FatVertex> result) {
141     return result.flatMap(new EdgeTriple());
142   }

144   /**
145    * Filters and casts graph heads from a given set of {@link Element}
146    *
147    * @param elements      elements
148    * @param graphHeadType graph head type
149    * @param <G> The graph head type
150    * @return graph heads
151    */
152   private static <G extends GraphHead> DataSet<G> extractGraphHeads(DataSet<Element> elements,
153     Class<G> graphHeadType) {
154     return elements
155       .filter(new IsInstance<>(graphHeadType))
156       .map(new Cast<>(graphHeadType))
157       .returns(TypeExtractor.createTypeInfo(graphHeadType));
158   }
159 
160   /**
161    * Initializes vertices from the given pattern matching result.
162    *
163    * @param result        pattern matching result
164    * @param vertexFactory vertex factory
165    * @param <V> The produced vertex type.
166    * @return vertices
167    */
168   public static <V extends Vertex> DataSet<V> extractVertices(DataSet<FatVertex> result,
169     VertexFactory<V> vertexFactory) {
170     return extractVertexIds(result).map(new VertexFromId<>(vertexFactory));
171   }
172 
173   /**
174    * Filters and casts vertices from a given set of {@link Element}
175    *
176    * @param elements      elements
177    * @param vertexType    vertex type
178    * @param mayOverlap    vertices may be contained in multiple graphs
179    * @param <V> The produced vertex type.
180    * @return vertices
181    */
182   private static <V extends Vertex> DataSet<V> extractVertices(DataSet<Element> elements,
183     Class<V> vertexType, boolean mayOverlap) {
184     DataSet<V> result = elements
185       .filter(new IsInstance<>(vertexType))
186       .map(new Cast<>(vertexType))
187       .returns(TypeExtractor.createTypeInfo(vertexType));
188     // TODO: Replace two group-by statements with a combinable reduce function.
189     return mayOverlap ? result
190       .groupBy(new Id<>())
191       .combineGroup(new MergedGraphIds<>())
192       .groupBy(new Id<>())
193       .reduceGroup(new MergedGraphIds<>()) : result;
194   }
195 
196   /**
197    * Initializes edges from the given pattern matching result.
198    *
199    * @param result      pattern matching result
200    * @param edgeFactory edge factory
201    * @param <E> The produced edge type.
202    * @return edges
203    */
204   public static <E extends Edge> DataSet<E> extractEdges(DataSet<FatVertex> result,
205     EdgeFactory<E> edgeFactory) {
206     return extractEdgeIds(result).map(new EdgeFromIds<>(edgeFactory));
207   }
208 
209   /**
210    * Filters and casts edges from a given set of {@link Element}
211    *
212    * @param elements      elements
213    * @param edgeType      edge type
214    * @param mayOverlap    edges may be contained in multiple graphs
215    * @param <E> The produced edge type.
216    * @return edges
217    */
218   private static <E extends Edge> DataSet<E> extractEdges(DataSet<Element> elements,
219     Class<E> edgeType, boolean mayOverlap) {
220     DataSet<E> result = elements
221       .filter(new IsInstance<>(edgeType))
222       .map(new Cast<>(edgeType))
223       .returns(TypeExtractor.createTypeInfo(edgeType));
224 
225     return mayOverlap ? result
226       .groupBy(new Id<>())
227       .combineGroup(new MergedGraphIds<>()).groupBy(new Id<>())
228       .reduceGroup(new MergedGraphIds<>()) : result;
229   }

231   /**
232    * Initializes vertices including their original data from the given
233    * pattern matching result.
234    *
235    * @param result        pattern matching result
236    * @param inputVertices original data graph vertices
237    * @param <V> The vertex type.
238    * @return vertices including data
239    */
240   public static <V extends Vertex> DataSet<V> extractVerticesWithData(
241     DataSet<FatVertex> result, DataSet<V> inputVertices) {
242     return extractVertexIds(result)
243       .join(inputVertices)
244       .where(0).equalTo(new Id<>())
245       .with(new RightSide<>());
246   }
247 
248   /**
249    * Initializes edges including their original data from the given pattern
250    * matching result.
251    *
252    * @param result      pattern matching result
253    * @param inputEdges  original data graph edges
254    * @param <E> The edge type.
255    * @return edges including data
256    */
257   public static <E extends Edge> DataSet<E> extractEdgesWithData(DataSet<FatVertex> result,
258     DataSet<E> inputEdges) {
259     return extractEdgeIds(result)
260       .join(inputEdges)
261       .where(0).equalTo(new Id<>())
262       .with(new RightSide<>());
263   }
