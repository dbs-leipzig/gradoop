32  /**
33   * Binary node that wraps a {@link CartesianProduct} operator.
34   */
35  public class CartesianProductNode extends BinaryNode implements JoinNode {
36    /**
37     * Morphism type for vertices
38     */
39    private final MatchStrategy vertexStrategy;
40    /**
41     * Morphism type for edges
42     */
43    private final MatchStrategy edgeStrategy;
44    /**
45     * Cross hint for Flink optimizer
46     */
47    private final CrossOperatorBase.CrossHint crossHint;
48  
49    /**
50     * Creates  a new node.
51     *
52     * @param leftChild left input plan node
53     * @param rightChild right right input plan node
54     * @param vertexStrategy morphism setting for vertices
55     * @param edgeStrategy morphism setting for edges
56     */
57    public CartesianProductNode(PlanNode leftChild, PlanNode rightChild,
58      MatchStrategy vertexStrategy, MatchStrategy edgeStrategy) {
59      this(leftChild, rightChild,
60        vertexStrategy, edgeStrategy, CrossOperatorBase.CrossHint.OPTIMIZER_CHOOSES);
61    }
62  
63    /**
64     * Creates a new node.
65     *
66     * @param leftChild left input plan node
67     * @param rightChild right input plan node
68     * @param vertexStrategy morphism setting for vertices
69     * @param edgeStrategy morphism setting for edges
70     * @param crossHint Join hint for the Flink optimizer
71     */
72    public CartesianProductNode(PlanNode leftChild, PlanNode rightChild,
73      MatchStrategy vertexStrategy, MatchStrategy edgeStrategy,
74      CrossOperatorBase.CrossHint crossHint) {
75      super(leftChild, rightChild);
76      this.vertexStrategy = vertexStrategy;
77      this.edgeStrategy = edgeStrategy;
78      this.crossHint = crossHint;
79    }
80  
81    @Override
82    public DataSet<Embedding> execute() {
83      CartesianProduct op = new CartesianProduct(getLeftChild().execute(), getRightChild().execute(),
84        getRightChild().getEmbeddingMetaData().getEntryCount(),
85        getDistinctVertexColumnsLeft(), getDistinctVertexColumnsRight(),
86        getDistinctEdgeColumnsLeft(), getDistinctEdgeColumnsRight(), crossHint);
87      op.setName(this.toString());
88      return op.evaluate();
89    }
90  
91    @Override
92    protected EmbeddingMetaData computeEmbeddingMetaData() {
93      EmbeddingMetaData leftInputMetaData = getLeftChild().getEmbeddingMetaData();
94      EmbeddingMetaData rightInputMetaData = getRightChild().getEmbeddingMetaData();
95      EmbeddingMetaData embeddingMetaData = new EmbeddingMetaData(leftInputMetaData);
96  
97      int entryCount = leftInputMetaData.getEntryCount();
98  
99      // append the non-join entry mappings from the right to the left side
100     for (String var : rightInputMetaData.getVariables()) {
101       embeddingMetaData.setEntryColumn(var, rightInputMetaData.getEntryType(var), entryCount++);
102     }
103 
104     // append all property mappings from the right to the left side
105     int propertyCount = leftInputMetaData.getPropertyCount();
106     for (String var : rightInputMetaData.getVariables()) {
107       for (String key : rightInputMetaData.getPropertyKeys(var)) {
108         embeddingMetaData.setPropertyColumn(var, key, propertyCount++);
109       }
110     }
111     return embeddingMetaData;
112   }
113 
114   /**
115    * According to the specified {@link CartesianProductNode#vertexStrategy}, the method returns
116    * the columns that need to contain distinct entries in the left embedding.
117    *
118    * @return distinct vertex columns of the left embedding
119    */
120   private List<Integer> getDistinctVertexColumnsLeft() {
121     return getDistinctVertexColumns(getLeftChild().getEmbeddingMetaData());
122   }
123 
124   /**
125    * According to the specified {@link CartesianProductNode#vertexStrategy}, the method returns
126    * the columns that need to contain distinct entries in the right embedding.
127    *
128    * @return distinct vertex columns of the right embedding
129    */
130   private List<Integer> getDistinctVertexColumnsRight() {
131     return getDistinctVertexColumns(getRightChild().getEmbeddingMetaData());
132   }
133 
134   /**
135    * According to the specified {@link CartesianProductNode#vertexStrategy} and the specified
136    * {@link EmbeddingMetaData}, the method returns the columns that need to contain distinct
137    * entries.
138    *
139    * @param metaData meta data for the embedding
140    * @return distinct vertex columns
141    */
142   private List<Integer> getDistinctVertexColumns(final EmbeddingMetaData metaData) {
143     return vertexStrategy == MatchStrategy.ISOMORPHISM ?
144       metaData.getVertexVariables().stream()
145         .map(metaData::getEntryColumn)
146         .collect(Collectors.toList()) : Collections.emptyList();
147   }
148 
149   /**
150    * According to the specified {@link CartesianProductNode#edgeStrategy}, the method returns
151    * the columns that need to contain distinct entries in the left embedding.
152    *
153    * @return distinct edge columns of the left embedding
154    */
155   private List<Integer> getDistinctEdgeColumnsLeft() {
156     return getDistinctEdgeColumns(getLeftChild().getEmbeddingMetaData());
157   }
158 
159   /**
160    * According to the specified {@link CartesianProductNode#edgeStrategy}, the method returns
161    * the columns that need to contain distinct entries in the right embedding.
162    *
163    * @return distinct edge columns of the right embedding
164    */
165   private List<Integer> getDistinctEdgeColumnsRight() {
166     return getDistinctEdgeColumns(getRightChild().getEmbeddingMetaData());
167   }
168 
169   /**
170    * According to the specified {@link CartesianProductNode#edgeStrategy} and the specified
171    * {@link EmbeddingMetaData}, the method returns the columns that need to contain distinct
172    * entries.
173    *
174    * @param metaData meta data for the embedding
175    * @return distinct edge columns
176    */
177   private List<Integer> getDistinctEdgeColumns(EmbeddingMetaData metaData) {
178     return edgeStrategy == MatchStrategy.ISOMORPHISM ?
179       metaData.getEdgeVariables().stream()
180         .map(metaData::getEntryColumn)
181         .collect(Collectors.toList()) : Collections.emptyList();
182   }
183 
184   @Override
185   public String toString() {
186     return String.format("CartesianProductNode{" +
187       "vertexMorphismType=%s, " +
188       "edgeMorphismType=%s}",
189       vertexStrategy, edgeStrategy);
190   }
