16  package org.gradoop.flink.model.api.functions;
17  
18  import org.gradoop.common.model.api.entities.Element;
19  import org.gradoop.common.model.impl.properties.PropertyValue;
20  import org.gradoop.flink.model.impl.operators.aggregation.Aggregation;
21  
22  import java.io.Serializable;
23  
24  /**
25   * Describes an aggregate function as input for the {@link Aggregation} operator.
26   */
27  public interface AggregateFunction extends Serializable {
28  
29    /**
30     * Describes the aggregation logic.
31     *
32     * @param aggregate previously aggregated value
33     * @param increment value that is added to the aggregate
34     *
35     * @return new aggregate
36     */
37    PropertyValue aggregate(PropertyValue aggregate, PropertyValue increment);
38  
39    /**
40     * Returns the property key used to store the aggregate value.
41     *
42     * @return aggregate property key
43     */
44    String getAggregatePropertyKey();
45  
46    /**
47     * Describes the increment of an element that should be added to the aggregate.
48     *
49     * @param element element used to get the increment
50     * @return increment, may be NULL, which is handled in the operator
51     */
52    PropertyValue getIncrement(Element element);
53  
54    /**
55     * Returns whether this function aggregates vertices.
56     *
57     * @return true, if it aggregates vertices
58     */
59    default boolean isVertexAggregation() {
60      return true;
61    }
62    /**
63     * Returns whether this function aggregates edges.
64     *
65     * @return true, if it aggregates edges
66     */
67    default boolean isEdgeAggregation() {
68      return true;
69    }
70  
71    /**
72     * Compute the final result after the aggregation step was completed. This method may be
73     * implemented if the aggregate function uses some internal aggregate value not equal to the
74     * result.<br>
75     * The default implementation of this method does nothing, it returns the result unchanged.<br>
76     * This method will not be called when the aggregate value is {@code null}.
77     *
78     * @param result The result of the aggregation step.
79     * @return The final result, computed from the previous result in a post-processing step.
80     */
81    default PropertyValue postAggregate(PropertyValue result) {
82      return result;
83    }
84  }
