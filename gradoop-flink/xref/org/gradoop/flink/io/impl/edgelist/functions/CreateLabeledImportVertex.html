16  package org.gradoop.flink.io.impl.edgelist.functions;
17  
18  import org.apache.flink.api.common.functions.MapFunction;
19  import org.apache.flink.api.java.functions.FunctionAnnotation;
20  import org.apache.flink.api.java.tuple.Tuple2;
21  import org.gradoop.common.model.impl.properties.Properties;
22  import org.gradoop.flink.io.impl.graph.tuples.ImportVertex;
23  import org.gradoop.common.util.GradoopConstants;
24  
25  /**
26   * {@code (vertexId, label) => ImportVertex}
27   * <p>
28   * Forwarded fields:
29   * <br>
30   * {@code f0}: {@code vertexId}
31   *
32   * @param <K> id type
33   */
34  @FunctionAnnotation.ForwardedFields("f0")
35  public class CreateLabeledImportVertex<K extends Comparable<K>>
36    implements MapFunction<Tuple2<K, String>, ImportVertex<K>> {
37    /**
38     * reused ImportVertex
39     */
40    private ImportVertex<K> reuseVertex;
41    /**
42     * PropertyKey of property value
43     */
44    private String propertyKey;
45  
46    /**
47     * Constructor
48     *
49     * @param propertyKey used PropertyKey
50     */
51    public CreateLabeledImportVertex(String propertyKey) {
52      this.propertyKey = propertyKey;
53      this.reuseVertex = new ImportVertex<>();
54      reuseVertex.setLabel(GradoopConstants.DEFAULT_VERTEX_LABEL);
55      reuseVertex.setProperties(Properties.createWithCapacity(1));
56    }
57  
58    @Override
59    public ImportVertex<K> map(Tuple2<K, String> inputTuple) throws Exception {
60      reuseVertex.setId(inputTuple.f0);
61      reuseVertex.getProperties().set(propertyKey, inputTuple.f1);
62      return reuseVertex;
63    }
64  }
