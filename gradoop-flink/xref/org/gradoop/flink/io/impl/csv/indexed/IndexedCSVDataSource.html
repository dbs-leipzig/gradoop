43  /**
44   * A graph data source for CSV files indexed by label.
45   * <p>
46   * The datasource expects files separated by label, e.g. in the following directory structure:
47   * <p>
48   * csvRoot
49   * |- Person.csv     # contains all vertices with label 'Person'
50   * |- University.csv # contains all vertices with label 'University'
51   * |- knows.csv      # contains all edges with label 'knows'
52   * |- studyAt.csv    # contains all edges with label 'studyAt'
53   * |- metadata.csv   # Meta data for all data contained in the graph
54   */
55  public class IndexedCSVDataSource extends CSVBase implements DataSource {
56    /**
57     * HDFS Configuration.
58     */
59    private final Configuration hdfsConfig;
60  
61    /**
62     * Creates a new data source. The constructor creates a default HDFS configuration.
63     *
64     * @param csvPath root path of csv files
65     * @param config  gradoop configuration
66     */
67    public IndexedCSVDataSource(String csvPath, GradoopFlinkConfig config) {
68      this(csvPath, config, new Configuration());
69    }
70  
71    /**
72     * Creates a new data source.
73     *
74     * @param csvPath  root path of csv files
75     * @param conf     gradoop configuration
76     * @param hdfsConf HDFS configuration
77     */
78    public IndexedCSVDataSource(String csvPath, GradoopFlinkConfig conf, Configuration hdfsConf) {
79      super(csvPath, conf);
80      this.hdfsConfig = Objects.requireNonNull(hdfsConf);
81    }
82  
83    /**
84     * Will use a single graph head of the collection as final graph head for the graph.
85     * Issue #1217 (https://github.com/dbs-leipzig/gradoop/issues/1217) will optimize further.
86     *
87     * {@inheritDoc}
88     */
89    @Override
90    public LogicalGraph getLogicalGraph() throws IOException {
91      GraphCollection collection = getGraphCollection();
92      return collection.getGraphFactory()
93        .fromDataSets(collection.getGraphHeads().first(1), collection.getVertices(), collection.getEdges());
94    }
95  
96    @Override
97    public GraphCollection getGraphCollection() throws IOException {
98      CSVMetaDataSource source = new CSVMetaDataSource();
99      CSVMetaData metaData = source.readLocal(getMetaDataPath(), hdfsConfig);
100     DataSet<Tuple3<String, String, String>> metaDataBroadcast =
101       source.readDistributed(getMetaDataPath(), getConfig());
102 
103     ExecutionEnvironment env = getConfig().getExecutionEnvironment();
104     GraphCollectionFactory factory = getConfig().getGraphCollectionFactory();
105 
106     Map<String, DataSet<EPGMGraphHead>> graphHeads = metaData.getGraphLabels().stream()
107       .map(label -> Tuple2.of(label, env.readTextFile(getGraphHeadCSVPath(label))
108         .map(new CSVLineToGraphHead(factory.getGraphHeadFactory()))
109         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
110         .filter(graphHead -> graphHead.getLabel().equals(label))))
111       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
112 
113     Map<String, DataSet<EPGMVertex>> vertices = metaData.getVertexLabels().stream()
114       .map(label -> Tuple2.of(label, env.readTextFile(getVertexCSVPath(label))
115         .map(new CSVLineToVertex(factory.getVertexFactory()))
116         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
117         .filter(vertex -> vertex.getLabel().equals(label))))
118       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
119 
120     Map<String, DataSet<EPGMEdge>> edges = metaData.getEdgeLabels().stream()
121       .map(label -> Tuple2.of(label, env.readTextFile(getEdgeCSVPath(label))
122         .map(new CSVLineToEdge(factory.getEdgeFactory()))
123         .withBroadcastSet(metaDataBroadcast, BC_METADATA)
124         .filter(edge -> edge.getLabel().equals(label))))
125       .collect(Collectors.toMap(t -> t.f0, t -> t.f1));
126 
127     return factory.fromIndexedDataSets(graphHeads, vertices, edges);
128   }
