129   /**
130    * Creates a GDL formatted string from the graph heads, vertices and edges.
131    *
132    * @return GDL formatted string
133    */
134   public String getGDLString() {
135     Map<GradoopId, String> idToGraphHeadName = getGraphHeadNameMapping(graphHeads);
136     Map<GradoopId, String> idToVertexName = getVertexNameMapping(vertices);
137     Map<GradoopId, String> idToEdgeName = getEdgeNameMapping(edges);
138 
139     Set<GradoopId> usedVertexIds = new HashSet<>();
140     Set<GradoopId> usedEdgeIds = new HashSet<>();
141 
142     StringBuilder result = new StringBuilder();
143 
144     for (G graphHead : graphHeads) {
145       StringBuilder verticesString = new StringBuilder();
146       StringBuilder edgesString = new StringBuilder();
147 
148       for (V vertex : vertices) {
149         boolean containedInGraph = vertex.getGraphIds().contains(graphHead.getId());
150         boolean firstOccurrence = !usedVertexIds.contains(vertex.getId());
151 
152         if (containedInGraph) {
153           String vertexString = vertexToGDLString(vertex, idToVertexName, firstOccurrence);
154           usedVertexIds.add(vertex.getId());
155           verticesString.append(vertexString).append(System.lineSeparator());
156         }
157       }
158 
159       for (E edge : edges) {
160         if (edge.getGraphIds().contains(graphHead.getId())) {
161           boolean firstOccurrence = !usedEdgeIds.contains(edge.getId());
162           String edgeString = edgeToGDLString(edge, idToVertexName, idToEdgeName, firstOccurrence);
163           usedEdgeIds.add(edge.getId());
164           edgesString.append(edgeString).append(System.lineSeparator());
165         }
166       }
167 
168       result
169         .append(graphHeadToGDLString(graphHead, idToGraphHeadName))
170         .append(GRAPH_ELEMENTS_DEFINITION_START).append(System.lineSeparator())
171         .append(verticesString)
172         .append(edgesString.length() > 0 ? System.lineSeparator() : "")
173         .append(edgesString)
174         .append(GRAPH_ELEMENTS_DEFINITION_END)
175         .append(System.lineSeparator()).append(System.lineSeparator());
176     }
177     return result.toString();
178   }

180   /**
181    * Returns a mapping between the graph heads gradoop ids and the GDL variable names.
182    *
183    * @param graphHeads The graph heads.
184    * @return Mapping between graph head and GDL variable name.
185    */
186   private Map<GradoopId, String> getGraphHeadNameMapping(List<G> graphHeads) {
187     Map<GradoopId, String> idToGraphHeadName = new HashMap<>(graphHeads.size());
188     for (int i = 0; i < graphHeads.size(); i++) {
189       G graphHead = graphHeads.get(i);
190       String gName = String.format("%s%s", GRAPH_VARIABLE_PREFIX, i);
191       idToGraphHeadName.put(graphHead.getId(), gName);
192     }
193     return idToGraphHeadName;
194   }

196   /**
197    * Returns a mapping between the vertex GradoopID and the GDL variable name.
198    *
199    * @param vertices The graph vertices.
200    * @return Mapping between vertex and GDL variable name.
201    */
202   private Map<GradoopId, String> getVertexNameMapping(List<V> vertices) {
203     Map<GradoopId, String> idToVertexName = new HashMap<>(vertices.size());
204     for (int i = 0; i < vertices.size(); i++) {
205       V vertex = vertices.get(i);
206       String vName = String.format("%s_%s_%s", VERTEX_VARIABLE_PREFIX, vertex.getLabel(), i);
207       idToVertexName.put(vertex.getId(), vName);
208     }
209     return idToVertexName;
210   }

212   /**
213    * Returns a mapping between the edge GradoopId and the GDL variable name.
214    *
215    * @param edges The graph edges.
216    * @return Mapping between edge and GDL variable name.
217    */
218   private Map<GradoopId, String> getEdgeNameMapping(List<E> edges) {
219     Map<GradoopId, String> idToEdgeName = new HashMap<>(edges.size());
220     for (int i = 0; i < edges.size(); i++) {
221       E edge = edges.get(i);
222       String eName = String.format("%s_%s_%s", EDGE_VARIABLE_PREFIX, edge.getLabel(), i);
223       idToEdgeName.put(edge.getId(), eName);
224     }
225     return idToEdgeName;
226   }

228   /**
229    * Returns a GDL formatted graph head string.
230    *
231    * @param graphhead graph head
232    * @param idToGraphHeadName mapping from graph head id to its GDL variable name
233    * @return GDL formatted string
234    */
235   private String graphHeadToGDLString(G graphhead, Map<GradoopId, String> idToGraphHeadName) {
236     return String.format("%s:%s %s",
237       idToGraphHeadName.get(graphhead.getId()),
238       graphhead.getLabel(),
239       propertiesToGDLString(graphhead.getProperties()));
240   }

242   /**
243    * Returns the gdl formatted vertex including the properties and the label on first occurrence
244    * or otherwise just the variable name.
245    *
246    * @param vertex The vertex that should be formatted.
247    * @param idToVertexName Maps GradoopId of a vertex to a string that represents the gdl
248    *                       variable name
249    * @param firstOccurrence Is it the first occurrence of the vertex in all graphs?
250    * @return A GDL formatted vertex string.
251    */
252   private String vertexToGDLString(V vertex, Map<GradoopId, String> idToVertexName,
253     boolean firstOccurrence) {
254     if (firstOccurrence) {
255       return String.format("(%s:%s %s)",
256         idToVertexName.get(vertex.getId()),
257         vertex.getLabel(),
258         propertiesToGDLString(vertex.getProperties()));
259     } else {
260       return String.format("(%s)", idToVertexName.get(vertex.getId()));
261     }
262   }
263 
264   /**
265    * Returns the GDL formatted edge, including the properties and the label on first occurrence
266    * or otherwise just the variable name.
267    *
268    * @param edge The edge to be formatted.
269    * @param idToVertexName Maps GradoopId of a vertex to a string that represents the GDL
270    *                       variable name
271    * @param idToEdgeName Maps GradoopId of an edge to a string that represents the GDL variable
272    *                     name.
273    * @param firstOccurrence Is it the first occurrence of the edge in all graphs?
274    * @return A GDL formatted edge string.
275    */
276   private String edgeToGDLString(E edge, Map<GradoopId, String> idToVertexName,
277     Map<GradoopId, String> idToEdgeName, boolean firstOccurrence) {
278     String result;
279     if (firstOccurrence) {
280       result =  String.format("(%s)-[%s:%s%s]->(%s)",
281         idToVertexName.get(edge.getSourceId()),
282         idToEdgeName.get(edge.getId()),
283         edge.getLabel(),
284         propertiesToGDLString(edge.getProperties()),
285         idToVertexName.get(edge.getTargetId()));
286     } else {
287       result = String.format("(%s)-[%s]->(%s)",
288         idToVertexName.get(edge.getSourceId()),
289         idToEdgeName.get(edge.getId()),
290         idToVertexName.get(edge.getTargetId()));
291     }
292     return result;
293   }

295   /**
296    * Returns the properties as a GDL formatted String.
297    *
298    * @param properties The properties to be formatted.
299    * @return A GDL formatted string that represents the properties.
300    */
301   private String propertiesToGDLString(Properties properties) {
302     if (properties == null || properties.isEmpty()) {
303       return "";
304     } else {
305       return properties.toList().stream()
306         .map(this::propertyToGDLString)
307         .collect(Collectors.joining(PROPERTIES_SEPARATOR, PROPERTIES_PREFIX, PROPERTIES_SUFFIX));
308     }
309   }

311   /**
312    * Returns this property as a GDL formatted String.
313    *
314    * @param property The property.
315    * @return A GDL formatted string that represents the property.
316    */
317   private String propertyToGDLString(Property property) {
318     StringBuilder result = new StringBuilder()
319       .append(property.getKey())
320       .append(KEY_VALUE_SEPARATOR);
321 
322     PropertyValue value = property.getValue();
323 
324     if (value.isString()) {
325       result.append(STRING_PREFIX).append(value.toString()).append(STRING_SUFFIX);
326     } else if (value.isNull()) {
327       result.append(NULL_STRING);
328     } else if (value.isDouble()) {
329       result.append(value.toString()).append(DOUBLE_SUFFIX);
330     } else if (value.isFloat()) {
331       result.append(value.toString()).append(FLOAT_SUFFIX);
332     } else if (value.isLong()) {
333       result.append(value.toString()).append(LONG_SUFFIX);
334     } else {
335       result.append(value.toString());
336     }
337 
338     return result.toString();
339   }
