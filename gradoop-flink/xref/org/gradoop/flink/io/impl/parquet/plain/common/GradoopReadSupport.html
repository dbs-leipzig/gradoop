16  package org.gradoop.flink.io.impl.parquet.plain.common;
17  
18  import org.apache.hadoop.conf.Configuration;
19  import org.apache.hadoop.mapreduce.Job;
20  import org.apache.hadoop.mapreduce.JobContext;
21  import org.apache.parquet.hadoop.api.InitContext;
22  import org.apache.parquet.hadoop.api.ReadSupport;
23  import org.apache.parquet.hadoop.util.ContextUtil;
24  import org.apache.parquet.io.api.RecordMaterializer;
25  import org.apache.parquet.schema.MessageType;
26  
27  import java.lang.reflect.Constructor;
28  import java.util.Map;
29  
30  /**
31   * Simple parquet read support for a {@link GradoopRootConverter<R>}
32   *
33   * @param <R> the record type
34   */
35  public class GradoopReadSupport<R> extends ReadSupport<R> {
36  
37    /**
38     * The configuration key for the {@link GradoopRootConverter}
39     */
40    public static final String GRADOOP_ROOT_CONVERTER_CLASS = "gradoop.parquet.rootConverterClass";
41  
42    /**
43     * The constructor of the root converter
44     */
45    private Constructor<? extends GradoopRootConverter> rootConverterConstructor;
46  
47    /**
48     * Sets the root converter class of a {@link JobContext}.
49     *
50     * @param job the job's context
51     * @param rootConverter the root converter class
52     */
53    public static void setRootConverter(Job job, Class<? extends GradoopRootConverter<?>> rootConverter) {
54      ContextUtil.getConfiguration(job)
55        .setClass(GRADOOP_ROOT_CONVERTER_CLASS, rootConverter, GradoopRootConverter.class);
56    }
57  
58    /**
59     * Gets the root converter class of a job's {@link Configuration}.
60     *
61     * @param configuration the job's configuration
62     * @return the specified root converter class or defaults to null
63     */
64    public static Class<? extends GradoopRootConverter> getRootConverter(Configuration configuration) {
65      return configuration.getClass(GRADOOP_ROOT_CONVERTER_CLASS, null, GradoopRootConverter.class);
66    }
67  
68    @Override
69    public ReadContext init(InitContext context) {
70      try {
71        Class<? extends GradoopRootConverter> rootConverter = getRootConverter(context.getConfiguration());
72        this.rootConverterConstructor = rootConverter.getConstructor(MessageType.class);
73      } catch (ReflectiveOperationException | SecurityException e) {
74        throw new RuntimeException("can't initialize ReadSupport", e);
75      }
76  
77      return new ReadContext(context.getFileSchema());
78    }
79  
80    @Override
81    public RecordMaterializer<R> prepareForRead(Configuration configuration,
82      Map<String, String> keyValueMetaData,
83      MessageType fileSchema, ReadContext readContext) {
84      try {
85        GradoopRootConverter rootConverter = this.rootConverterConstructor
86          .newInstance(readContext.getRequestedSchema());
87        return new GradoopRecordMaterializer<>(rootConverter);
88      } catch (ReflectiveOperationException e) {
89        throw new RuntimeException("can't initialize GradoopRecordMaterializer", e);
90      }
91    }
92  }
