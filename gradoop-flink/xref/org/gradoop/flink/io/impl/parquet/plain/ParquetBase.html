35  import java.io.File;
36  import java.io.IOException;
37  import java.util.Objects;

39  /**
40   * Base class for parquet data source and data sink.
41   */
42  abstract class ParquetBase {
43  
44    /**
45     * File ending for parquet-protobuf files.
46     */
47    private static final String PARQUET_FILE_SUFFIX = ".parquet";
48    /**
49     * parquet-protobuf file for vertices.
50     */
51    private static final String VERTEX_FILE = "vertices" + PARQUET_FILE_SUFFIX;
52    /**
53     * parquet-protobuf file for graph heads.
54     */
55    private static final String GRAPH_HEAD_FILE = "graphs" + PARQUET_FILE_SUFFIX;
56    /**
57     * parquet-protobuf file for edges.
58     */
59    private static final String EDGE_FILE = "edges" + PARQUET_FILE_SUFFIX;
60  
61    /**
62     * Root directory containing the CSV and metadata files.
63     */
64    private final String basePath;
65    /**
66     * Gradoop Flink configuration
67     */
68    private final GradoopFlinkConfig config;
69  
70    /**
71     * Constructor.
72     *
73     * @param basePath directory to the parquet files
74     * @param config   Gradoop Flink configuration
75     */
76    protected ParquetBase(String basePath, GradoopFlinkConfig config) {
77      Objects.requireNonNull(basePath);
78      Objects.requireNonNull(config);
79  
80      this.basePath = basePath.endsWith(File.separator) ? basePath : basePath + File.separator;
81      this.config = config;
82    }
83  
84    /**
85     * Returns the path to the graph head file.
86     *
87     * @return graph head file path
88     */
89    protected String getGraphHeadPath() {
90      return basePath + GRAPH_HEAD_FILE;
91    }
92  
93    /**
94     * Returns the path to the vertex file.
95     *
96     * @return vertex file path
97     */
98    protected String getVertexPath() {
99      return basePath + VERTEX_FILE;
100   }
101 
102   /**
103    * Returns the path to the edge file.
104    *
105    * @return edge file path
106    */
107   protected String getEdgePath() {
108     return basePath + EDGE_FILE;
109   }
110 
111   /**
112    * Get the gradoop configuration.
113    *
114    * @return the gradoop configuration instance
115    */
116   protected GradoopFlinkConfig getConfig() {
117     return config;
118   }
119 
120   /**
121    * Writes a {@link DataSet} as parquet file(s) to the specified location.
122    *
123    * @param data the {@link DataSet}
124    * @param outputPath the path of the file
125    * @param writeSupport the write support for the datasets type
126    * @param overwrite the behavior regarding existing files
127    * @param <T> the datasets object type
128    * @throws IOException if an I/O error occurs
129    */
130   protected <T> void write(DataSet<T> data, String outputPath,
131     Class<? extends WriteSupport<T>> writeSupport, boolean overwrite) throws IOException {
132     Job job = Job.getInstance();
133 
134     FileOutputFormat.setOutputPath(job, new Path(outputPath));
135     ParquetOutputFormat.setCompression(job, CompressionCodecName.SNAPPY);
136     ParquetOutputFormat.setEnableDictionary(job, false);
137     ParquetOutputFormat.setWriteSupportClass(job, writeSupport);
138     ParquetOutputFormatWithMode.setFileCreationMode(job,
139       overwrite ? ParquetFileWriter.Mode.OVERWRITE : ParquetFileWriter.Mode.CREATE);
140 
141     data.output(new HadoopValueOutputFormat<T>(new ParquetOutputFormatWithMode<>(), job));
142   }
143 
144   /**
145    * Creates a {@link DataSet} that represents the elements produced by reading the given parquet file.
146    *
147    * @param type the element type
148    * @param inputPath the path of the file
149    * @param rootConverter the root converter for the element type
150    * @return A {@link DataSet} that represents the data read from the given file.
151    * @param <T> element type
152    * @throws IOException if an I/O error occurs
153    */
154   protected <T> DataSet<T> read(Class<T> type, String inputPath,
155     Class<? extends GradoopRootConverter<T>> rootConverter) throws IOException {
156     Job job = Job.getInstance();
157 
158     FileInputFormat.addInputPath(job, new Path(inputPath));
159     GradoopReadSupport.setRootConverter(job, rootConverter);
160     ParquetInputFormat.setReadSupportClass(job, GradoopReadSupport.class);
161 
162     return config.getExecutionEnvironment().createInput(
163       new HadoopValueInputFormat<T>(new ParquetInputFormat<>(GradoopReadSupport.class), type, job));
164   }
165 }
