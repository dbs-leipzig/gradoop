34  /**
35   * A gradoop operator wrapping {@link org.apache.flink.graph.library.linkanalysis.PageRank}.
36   *
37   * @param <G>  Gradoop graph head type.
38   * @param <V>  Gradoop vertex type.
39   * @param <E>  Gradoop edge type.
40   * @param <LG> Gradoop type of the graph.
41   * @param <GC> Gradoop type of the graph collection.
42   */
43  public class PageRank<
44    G extends GraphHead,
45    V extends Vertex,
46    E extends Edge,
47    LG extends BaseGraph<G, V, E, LG, GC>,
48    GC extends BaseGraphCollection<G, V, E, LG, GC>>
49    extends GradoopGellyAlgorithm<G, V, E, LG, GC, NullValue, NullValue> {
50  
51    /**
52     * Property key to store the page rank in.
53     */
54    private final String propertyKey;
55  
56    /**
57     * Damping factor.
58     */
59    private final double dampingFactor;
60  
61    /**
62     * Number of iterations.
63     */
64    private final int iterations;
65  
66    /**
67     * Whether to include "zero-degree" vertices in the PageRank computation and result. These
68     * vertices only affect the scores of other vertices indirectly through influencing the initial
69     * proportional score of {@code (1 - damping factor) / number of vertices}.
70     * If set to {@code false}, these vertices will NOT be part of the computation and the returned
71     * result graph.
72     */
73    private final boolean includeZeroDegrees;
74  
75    /**
76     * Constructor for Page Rank with fixed number of iterations and {@link #includeZeroDegrees}
77     * set to {@code false}.
78     *
79     * @param propertyKey Property key to store the page rank in.
80     * @param dampingFactor Damping factor.
81     * @param iterations    Number of iterations.
82     */
83    public PageRank(String propertyKey, double dampingFactor, int iterations) {
84      this(propertyKey, dampingFactor, iterations, false);
85    }
86  
87    /**
88     * Constructor for Page Rank with fixed number of iterations.
89     *
90     * @param propertyKey Property key to store the page rank in.
91     * @param dampingFactor Damping factor.
92     * @param iterations    Number of iterations.
93     * @param includeZeroDegrees Whether to include "zero-degree" vertices in the PageRank
94     *                                  computation and result.
95     */
96    public PageRank(String propertyKey, double dampingFactor, int iterations, boolean includeZeroDegrees) {
97      super(new VertexToGellyVertexWithNullValue<>(), new EdgeToGellyEdgeWithNullValue<>());
98      this.propertyKey = propertyKey;
99      this.dampingFactor = dampingFactor;
100     this.iterations = iterations;
101     this.includeZeroDegrees = includeZeroDegrees;
102   }
103 
104   @Override
105   public LG executeInGelly(Graph<GradoopId, NullValue, NullValue> gellyGraph) throws Exception {
106     DataSet<V> newVertices =
107       new org.apache.flink.graph.library.linkanalysis.PageRank<GradoopId, NullValue, NullValue>(
108         dampingFactor, iterations).setIncludeZeroDegreeVertices(includeZeroDegrees).run(gellyGraph)
109       .join(currentGraph.getVertices())
110       .where(new PageRankResultKey()).equalTo(new Id<>())
111       .with(new PageRankToAttribute<>(propertyKey));
112     return currentGraph.getFactory().fromDataSets(
113       currentGraph.getGraphHead(), newVertices, currentGraph.getEdges());
114   }
