55  /**
56   * Performs the RandomJump using Gellys VertexCentricIteration (VCI).
57   * Uniformly at random picks {@link #numberOfStartVertices} starting vertices and then simulates a RandomWalk
58   * for each starting vertex on the graph, where once visited edges are not used again. For each walker,
59   * with a given {@link #jumpProbability}, or if the walk ends in a sink, or if all outgoing edges
60   * of the current vertex were visited, randomly jumps to any vertex in the graph and starts a new
61   * RandomWalk from there.
62   * Unlike the RandomWalk algorithm, RandomJump does not have problems of getting stuck or not being
63   * able to visit enough vertices. The algorithm converges when the maximum number of iterations
64   * has been reached, or enough vertices have been visited (with the percentage of vertices to
65   * visit at least given in {@link #percentageToVisit}).
66   * Returns the initial graph with vertices and edges annotated by a boolean property named
67   * "sampled", which is set to {@code true} if visited, or {@code false} if not.
68   *
69   * @param <G>  Gradoop graph head type.
70   * @param <V>  Gradoop vertex type.
71   * @param <E>  Gradoop edge type.
72   * @param <LG> Gradoop type of the graph.
73   * @param <GC> Gradoop type of the graph collection.
74   */
75  public class KRandomJumpGellyVCI<
76    G extends GraphHead,
77    V extends Vertex,
78    E extends Edge,
79    LG extends BaseGraph<G, V, E, LG, GC>,
80    GC extends BaseGraphCollection<G, V, E, LG, GC>>
81    extends BaseGellyAlgorithm<G, V, E, LG, GC, Long, VCIVertexValue, Long, LG>
82    implements UnaryBaseGraphToBaseGraphOperator<LG> {

86    protected LG currentGraph;

118   /**
119    * Creates an instance of KRandomJumpGellyVCI.
120    *
121    * @param numberOfStartVertices Number of starting vertices.
122    * @param maxIterations Value for maximum number of iterations for the algorithm.
123    * @param jumpProbability Probability for jumping to random vertex instead of walking to random
124    *                       neighbor.
125    * @param percentageToVisit Relative amount of vertices to visit at least.
126    */
127   public KRandomJumpGellyVCI(int numberOfStartVertices, int maxIterations, double jumpProbability,
128     double percentageToVisit) {
129     checkArgument(numberOfStartVertices >= 1,
130       "at least 1 starting vertex is needed, numberOfStartVertices must be equal or greater 1");
131     checkArgument(maxIterations > 0,
132       "maxIterations must be greater than 0");
133     checkArgument(jumpProbability >= 0d && jumpProbability <= 1d,
134       "jumpProbability must be equal/greater than 0.0 and smaller/equal 1.0");
135     checkArgument(percentageToVisit > 0d && percentageToVisit <= 1d,
136       "percentageToVisit must be greater than 0.0 and smaller/equal 1.0");
137     this.numberOfStartVertices = numberOfStartVertices;
138     this.maxIterations = maxIterations;
139     this.jumpProbability = jumpProbability;
140     this.percentageToVisit = percentageToVisit;
141   }

143   @Override
144   public Graph<Long, VCIVertexValue, Long> transformToGelly(LG graph) {
145     this.currentGraph = graph;
146 
147     indexToVertexIdMap = DataSetUtils.zipWithIndex(graph.getVertices().map(new Id<>()));
148     indexToEdgeIdMap = DataSetUtils.zipWithIndex(graph.getEdges().map(new Id<>()));
149 
150     DataSet<org.apache.flink.graph.Vertex<Long, VCIVertexValue>> vertices = indexToVertexIdMap
151       .map(new LongIdToGellyVertexWithVCIValueMap());
152 
153     DataSet<org.apache.flink.graph.Edge<Long, Long>> edges = graph.getEdges()
154       .join(indexToVertexIdMap)
155       .where(new SourceId<>()).equalTo(1)
156       .with(new LongIdWithEdgeToTupleJoin<>())
157       .join(indexToVertexIdMap)
158       .where(1).equalTo(1)
159       .with(new ReplaceTargetWithLongIdJoin())
160       .join(indexToEdgeIdMap)
161       .where(2).equalTo(1)
162       .with(new LongIdTupleToGellyEdgeWithLongValueJoin());
163 
164     return Graph.fromDataSet(vertices, edges, graph.getConfig().getExecutionEnvironment());
165   }

167   @Override
168   public LG executeInGelly(Graph<Long, VCIVertexValue, Long> gellyGraph) throws Exception {
169 
170     long vertexCount = gellyGraph.numberOfVertices();
171 
172     //--------------------------------------------------------------------------
173     // pre compute
174     //--------------------------------------------------------------------------
175 
176     // define start vertices
177     Set<Long> randomStartIndices = new HashSet<>();
178     while (randomStartIndices.size() < numberOfStartVertices) {
179       long randomLongInBounds = (long) (Math.random() * (vertexCount - 1L));
180       randomStartIndices.add(randomLongInBounds);
181     }
182     DataSet<Long> startIndices = currentGraph.getConfig().getExecutionEnvironment()
183       .fromCollection(randomStartIndices);
184 
185     // define how many vertices to visit
186     long verticesToVisit = (long) Math.ceil((double) vertexCount * percentageToVisit);
187 
188     // set compute parameters
189     VertexCentricConfiguration parameters = new VertexCentricConfiguration();
190     parameters.addBroadcastSet(VCIComputeFunction.START_INDICES_BROADCAST_SET, startIndices);
191     parameters.addBroadcastSet(VCIComputeFunction.VERTEX_INDICES_BROADCAST_SET,
192       indexToVertexIdMap.map(new Value0Of2<>()));
193     parameters.registerAggregator(VCIComputeFunction.VISITED_VERTICES_AGGREGATOR_NAME,
194       new LongSumAggregator());
195 
196     // run gelly
197     Graph<Long, VCIVertexValue, Long> resultGraph = gellyGraph.runVertexCentricIteration(
198       new VCIComputeFunction(jumpProbability, verticesToVisit),
199         null, maxIterations, parameters);
200 
201     //--------------------------------------------------------------------------
202     // post compute
203     //--------------------------------------------------------------------------
204 
205     DataSet<GradoopId> visitedGellyEdgeIds = resultGraph.getVertices()
206       .flatMap(new GetVisitedGellyEdgeLongIdsFlatMap())
207       .join(indexToEdgeIdMap)
208       .where("*").equalTo(0)
209       .with(new VisitedGellyEdgesWithLongIdToGradoopIdJoin());
210 
211     // compute new visited edges
212     DataSet<E> visitedEdges = currentGraph.getEdges()
213       .leftOuterJoin(visitedGellyEdgeIds)
214       .where(new Id<>()).equalTo("*")
215       .with(new EdgeWithGellyEdgeIdJoin<>(SamplingConstants.PROPERTY_KEY_SAMPLED));
216 
217     DataSet<GradoopId> visitedSourceTargetIds = visitedEdges
218       .flatMap(new GetVisitedSourceTargetIdsFlatMap<>(SamplingConstants.PROPERTY_KEY_SAMPLED))
219       .distinct();
220 
221     // compute new visited vertices
222     DataSet<V> visitedVertices = resultGraph.getVertices()
223       .join(indexToVertexIdMap)
224       .where(0).equalTo(0)
225       .with(new GellyVertexWithLongIdToGradoopIdJoin())
226       .join(currentGraph.getVertices())
227       .where(0).equalTo(new Id<>())
228       .with(new GellyVertexWithVertexJoin<>(SamplingConstants.PROPERTY_KEY_SAMPLED));
229 
230     visitedVertices = visitedVertices.leftOuterJoin(visitedSourceTargetIds)
231       .where(new Id<>()).equalTo("*")
232       .with(new VertexWithVisitedSourceTargetIdJoin<>(SamplingConstants.PROPERTY_KEY_SAMPLED));
233 
234     // return graph
235     return currentGraph.getFactory().fromDataSets(
236       currentGraph.getGraphHead(), visitedVertices, visitedEdges);
237   }
