23  /**

36    /**

41     * @param frequentPatterns k-edge frequent patterns
42     * @param rightmostPaths k-edge rightmost paths
43     * @param uncompressEmbeddings flag, to enable embedding decompression (true=enabled)
44     * @param compressedPatterns compressed k-edge frequent patterns for map lookup
45     *
46     * @return map of grown supported patterns
47     */
48    PatternEmbeddingsMap growPatterns(
49      int[] graph,
50      PatternEmbeddingsMap parentMap,
51      List<int[]> frequentPatterns,
52      List<int[]> rightmostPaths,
53      boolean uncompressEmbeddings,
54      List<int[]> compressedPatterns
55    );
56  
57    /**
58     * Verifies if a pattern in DFS-code model is minimal according to gSpan lexicographic order.
59     *
60     * @param pattern pattern
61     * @return true, if minimal
62     */
63    boolean isMinimal(int[] pattern);
64  
65    /**
66     * Calculates the rightmost path of a given pattern.
67     *
68     * @param pattern input pattern
69     * @return rightmost path (vertex times)
70     */
71    int[] getRightmostPathTimes(int[] pattern);
72  
73    /**
74     * Turns a pattern in DFS code model into a graph.
75     *
76     * @param pattern DFS code
77     * @return graph
78     */
79    int[] getGraph(int[] pattern);
