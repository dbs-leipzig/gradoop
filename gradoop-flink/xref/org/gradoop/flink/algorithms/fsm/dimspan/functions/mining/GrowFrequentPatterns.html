33  /**
34   * {@code (graph, k-edge pattern -> embeddings) => (graph, k+1-edge pattern -> embeddings)}
35   */
36  public class GrowFrequentPatterns
37    extends RichMapFunction<GraphWithPatternEmbeddingsMap, GraphWithPatternEmbeddingsMap> {
38  
39    /**
40     * compressed k-edge frequent patterns for fast embedding map lookup
41     */
42    private List<int[]> compressedFrequentPatterns;
43  
44    /**
45     * uncompressed k-edge frequent patterns for pattern growth
46     */
47    private List<int[]> frequentPatterns;
48  
49    /**
50     * patterns with frequency for collector
51     */
52    private List<WithCount<int[]>> patternFrequencies;
53  
54    /**
55     * list of rightmost paths, index relates to frequent patterns
56     */
57    private List<int[]> rightmostPaths;
58  
59    /**
60     * pattern growth logic (directed or undirected mode)
61     */
62    private final GSpanLogic gSpan;
63  
64    /**
65     * flag to enable graph compression (true=enabled)
66     */
67    private final boolean compressGraphs;
68  
69    /**
70     * flag to enable pattern compression (true=enabled)
71     */
72    private final boolean compressPatterns;
73  
74    /**
75     * flag to enable pattern decompression (true=enabled)
76     */
77    private final boolean uncompressFrequentPatterns;
78  
79    /**
80     * flag to enable embedding compression (true=enabled)
81     */
82    private final boolean compressEmbeddings;
83  
84    /**
85     * flag to enable pattern verification before counting (true=enabled)
86     */
87    private final boolean validatePatterns;
88  
89    /**
90     * Constructor.
91     *
92     * @param gSpan pattern growth logic
93     * @param fsmConfig FSM Configuration
94     */
95    public GrowFrequentPatterns(GSpanLogic gSpan, DIMSpanConfig fsmConfig) {
96  
97      // set pattern growth logic for directed or undirected mode
98      this.gSpan = gSpan;
99  
100     // cache compression flags
101     compressGraphs = fsmConfig.isGraphCompressionEnabled();
102     compressEmbeddings = fsmConfig.isEmbeddingCompressionEnabled();
103     compressPatterns = fsmConfig.getPatternCompressionInStep() == DataflowStep.MAP;
104     uncompressFrequentPatterns = fsmConfig.getPatternCompressionInStep() != DataflowStep.WITHOUT;
105 
106     // cache validation flag
107     validatePatterns = fsmConfig.getPatternVerificationInStep() == DataflowStep.MAP;
108   }
109 
110   @Override
111   public void open(Configuration parameters) throws Exception {
112     super.open(parameters);
113 
114     // broadcast reception
115 
116     patternFrequencies =
117       getRuntimeContext().getBroadcastVariable(DIMSpanConstants.FREQUENT_PATTERNS);
118 
119     int patternCount = patternFrequencies.size();
120 
121     this.frequentPatterns = Lists.newArrayListWithExpectedSize(patternCount);
122 
123     for (WithCount<int[]> patternWithCount : patternFrequencies) {
124       int[] pattern = patternWithCount.getObject();
125 
126       // uncompress
127       if (uncompressFrequentPatterns) {
128         pattern = Simple16Compressor.uncompress(pattern);
129       }
130       frequentPatterns.add(pattern);
131     }
132 
133     // sort
134     frequentPatterns.sort(new DFSCodeComparator());
135 
136     // calculate rightmost paths
137     this.rightmostPaths = Lists.newArrayListWithExpectedSize(patternCount);
138     this.compressedFrequentPatterns = Lists.newArrayListWithExpectedSize(patternCount);
139 
140     for (int[] pattern : frequentPatterns) {
141       rightmostPaths.add(gSpan.getRightmostPathTimes(pattern));
142 
143       // TODO: directly store compressed patterns at reception
144       compressedFrequentPatterns
145         .add(compressPatterns ? Simple16Compressor.compress(pattern) : pattern);
146     }
147   }
148 
149   @Override
150   public GraphWithPatternEmbeddingsMap map(GraphWithPatternEmbeddingsMap pair) throws Exception {
151 
152     // union k-1 edge frequent patterns with k-edge ones
153     if (pair.isFrequentPatternCollector()) {
154       for (WithCount<int[]> patternWithFrequency : patternFrequencies) {
155         pair.getMap().collect(patternWithFrequency);
156       }
157     } else {
158       int[] graph = pair.getGraph();
159 
160       // uncompress graph
161       if (compressGraphs) {
162         graph = Simple16Compressor.uncompress(graph);
163       }
164 
165       // execute pattern growth for all supported frequent patterns
166       PatternEmbeddingsMap childMap = gSpan.growPatterns(graph, pair.getMap(),
167         frequentPatterns, rightmostPaths, compressEmbeddings, compressedFrequentPatterns);
168 
169       // drop non-minimal patterns if configured to be executed here
170       if (validatePatterns) {
171         PatternEmbeddingsMap validatedMap = PatternEmbeddingsMap.getEmptyOne();
172 
173         for (int i = 0; i < childMap.getPatternCount(); i++) {
174           int[] pattern = childMap.getPattern(i);
175 
176           if (gSpan.isMinimal(pattern)) {
177             int[] embeddingData = childMap.getValues()[i];
178             validatedMap.put(pattern, embeddingData);
179           }
180         }
181 
182         childMap = validatedMap;
183       }
184 
185       // update pattern-embedding map
186       pair.setPatternEmbeddings(childMap);
187 
188       // compress patterns and embedding, if configured
189       // NOTE: graphs will remain compressed
190       if (compressPatterns) {
191         Simple16Compressor.compressPatterns(pair.getMap());
192       }
193 
194       if (compressEmbeddings) {
195         Simple16Compressor.compressEmbeddings(pair.getMap());
196       }
197     }
198 
199     return pair;
200   }
