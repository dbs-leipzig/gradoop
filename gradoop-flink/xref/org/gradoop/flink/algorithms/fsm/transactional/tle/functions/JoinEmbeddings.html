37  /**
38   * Superclass of classes joining subgraph embeddings.
39   *
40   * @param <G> graph type
41   * @param <SE> subgraph type
42   */
43  public class JoinEmbeddings<G extends FSMGraph, SE extends SubgraphEmbeddings>
44    implements FlatJoinFunction<SE, G, SE>, FlatMapFunction<SE, SE> {
45  
46    /**
47     * Labeler used to generate canonical labels.
48     */
49    private final CanonicalLabeler canonicalLabeler;
50  
51    /**
52     * Constructor.
53     *
54     * @param fsmConfig FSM configuration.
55     */
56    public JoinEmbeddings(FSMConfig fsmConfig) {
57      this.canonicalLabeler = new CanonicalLabeler(fsmConfig.isDirected());
58    }
59  
60    @Override
61    public void join(SE embeddings, G graph, Collector<SE> out) throws
62      Exception {
63  
64      Set<TreeSet<Integer>> edgeSets = Sets.newHashSet();
65      Map<String, List<Embedding>> subgraphEmbeddings = Maps.newHashMap();
66  
67      for (Embedding parent : embeddings.getEmbeddings()) {
68        for (Map.Entry<Integer, FSMEdge> entry : graph.getEdges().entrySet()) {
69  
70          int edgeId = entry.getKey();
71  
72          if (! parent.getEdges().containsKey(edgeId)) {
73            FSMEdge edge = entry.getValue();
74  
75            int sourceId = edge.getSourceId();
76            boolean containsSourceId =
77              parent.getVertices().containsKey(sourceId);
78  
79            int targetId = edge.getTargetId();
80            boolean containsTargetId =
81              parent.getVertices().containsKey(targetId);
82  
83            if (containsSourceId ||  containsTargetId) {
84              TreeSet<Integer> edgeSet = Sets.newTreeSet(parent.getEdgeIds());
85              edgeSet.add(edgeId);
86  
87              if (! edgeSets.contains(edgeSet)) {
88                edgeSets.add(edgeSet);
89  
90                Embedding child = parent.deepCopy();
91                child.getEdges().put(edgeId, edge);
92  
93                if (! containsSourceId) {
94                  child.getVertices()
95                    .put(sourceId, graph.getVertices().get(sourceId));
96                }
97                if (! containsTargetId) {
98                  child.getVertices()
99                    .put(targetId, graph.getVertices().get(targetId));
100               }
101 
102               String canonicalLabel = canonicalLabeler.label(child);
103 
104               List<Embedding> siblings =
105                 subgraphEmbeddings.get(canonicalLabel);
106 
107               if (siblings == null) {
108                 siblings = Lists.newArrayList(child);
109                 subgraphEmbeddings.put(canonicalLabel, siblings);
110               } else {
111                 siblings.add(child);
112               }
113             }
114           }
115         }
116       }
117     }
118 
119     collect(embeddings, out, subgraphEmbeddings);
120   }
121 
122   @Override
123   public void flatMap(SE embeddings, Collector<SE> collector) throws Exception {
124 
125     Map<String, List<Embedding>> subgraphEmbeddings;
126 
127     if (embeddings.getSize() == 1) {
128       subgraphEmbeddings = joinOnVertex(embeddings);
129     } else {
130       subgraphEmbeddings = joinOnEdgeOverlap(embeddings);
131     }
132 
133     collect(embeddings, collector, subgraphEmbeddings);
134   }

136   /**
137    * Cross-joins k-edge (k>1) embeddings on k-1-edge overlap.
138    *
139    * @param embeddings k-edge embeddings
140    * @return Map of k+1-edge subgraphs and embeddings
141    */
142   private Map<String, List<Embedding>> joinOnEdgeOverlap(SE embeddings) {
143     Map<String, List<Embedding>> subgraphEmbeddings;
144     subgraphEmbeddings = Maps.newHashMap();
145     List<Embedding> parents = embeddings.getEmbeddings();
146 
147     Set<TreeSet<Integer>> edgeSets = Sets.newHashSet();
148 
149     // index parents
150 
151     Map<TreeSet<Integer>, List<Integer>> subsetParents =  Maps.newHashMap();
152 
153     int parentIndex = 0;
154     for (Embedding parent : parents) {
155 
156       for (Integer edgeId : parent.getEdgeIds()) {
157         TreeSet<Integer> subset = Sets.newTreeSet(parent.getEdgeIds());
158         subset.remove(edgeId);
159 
160         Collection<Integer> siblings = subsetParents.get(subset);
161 
162         if (siblings == null) {
163           subsetParents.put(subset, Lists.newArrayList(parentIndex));
164         } else {
165           siblings.add(parentIndex);
166         }
167       }
168 
169       parentIndex++;
170     }
171 
172     // join overlapping parents
173 
174     for (List<Integer> overlappingParents : subsetParents.values()) {
175       int size = overlappingParents.size();
176 
177       if (size > 1) {
178 
179         for (int i = 0; i < size - 1; i++) {
180           for (int j = i + 1; j < size; j++) {
181 
182             Embedding left = parents.get(overlappingParents.get(i));
183             Embedding right = parents.get(overlappingParents.get(j));
184 
185             TreeSet<Integer> edgeSet = Sets.newTreeSet(left.getEdgeIds());
186             edgeSet.addAll(right.getEdgeIds());
187 
188             if (!edgeSets.contains(edgeSet)) {
189               edgeSets.add(edgeSet);
190 
191               Embedding child = left.deepCopy();
192               child.getVertices().putAll(right.getVertices());
193               child.getEdges().putAll(right.getEdges());
194 
195               String canonicalLabel = canonicalLabeler.label(child);
196 
197               List<Embedding> siblings =
198                 subgraphEmbeddings.get(canonicalLabel);
199 
200               if (siblings == null) {
201                 siblings = Lists.newArrayList(child);
202                 subgraphEmbeddings.put(canonicalLabel, siblings);
203               } else {
204                 siblings.add(child);
205               }
206             }
207           }
208         }
209       }
210     }
211     return subgraphEmbeddings;
212   }

214   /**
215    * Cross-joins 1-edge embeddings on source and target vertices.
216    *
217    * @param embeddings 1-edge embeddings
218    * @return Map of 2-edge subgraphs and embeddings
219    */
220   private Map<String, List<Embedding>> joinOnVertex(SE embeddings) {
221     Map<String, List<Embedding>> subgraphEmbeddings;
222     subgraphEmbeddings = Maps.newHashMap();
223     List<Embedding> parents = embeddings.getEmbeddings();
224 
225     int size = parents.size();
226 
227     for (int i = 0; i < size - 1; i++) {
228       for (int j = i + 1; j < size; j++) {
229 
230         Embedding left = parents.get(i);
231         Embedding right = parents.get(j);
232 
233         FSMEdge leftEdge = left.getEdges().values().iterator().next();
234         FSMEdge rightEdge = right.getEdges().values().iterator().next();
235 
236         if (leftEdge.getSourceId() == rightEdge.getSourceId() ||
237           leftEdge.getSourceId() == rightEdge.getTargetId() ||
238           leftEdge.getTargetId() == rightEdge.getSourceId() ||
239           leftEdge.getTargetId() == rightEdge.getTargetId()) {
240 
241           Embedding child = left.deepCopy();
242           child.getVertices().putAll(right.getVertices());
243           child.getEdges().putAll(right.getEdges());
244 
245           String canonicalLabel = canonicalLabeler.label(child);
246 
247           List<Embedding> siblings =
248             subgraphEmbeddings.get(canonicalLabel);
249 
250           if (siblings == null) {
251             siblings = Lists.newArrayList(child);
252             subgraphEmbeddings.put(canonicalLabel, siblings);
253           } else {
254             siblings.add(child);
255           }
256         }
257       }
258     }
259     return subgraphEmbeddings;
260   }

262   /**
263    * Triggers output of subgraphs and embeddings.
264    *
265    * @param reuseTuple reuse tuple to avoid instantiations
266    * @param out Flink output collector
267    * @param subgraphEmbeddings subgraphs and embeddings.
268    */
269   private void collect(SE reuseTuple, Collector<SE> out,
270     Map<String, List<Embedding>> subgraphEmbeddings) {
271 
272     reuseTuple.setSize(reuseTuple.getSize() + 1);
273 
274     for (Map.Entry<String, List<Embedding>> entry :
275       subgraphEmbeddings.entrySet()) {
276 
277       reuseTuple.setCanonicalLabel(entry.getKey());
278       reuseTuple.setEmbeddings(entry.getValue());
279 
280       out.collect(reuseTuple);
281     }
282   }
