33  public class CombinableFilterTest extends GradoopFlinkTestBase {
34  
35    private final CombinableFilter<Object> alwaysTrue = e -> true;
36  
37    private final CombinableFilter<Object> alwaysFalse = e -> false;
38  
39    private final Object testObject = new Object();
40  
41    @Test
42    public void testAnd() throws Exception {
43      assertTrue(alwaysTrue.and(alwaysTrue).filter(testObject));
44      assertFalse(alwaysFalse.and(alwaysTrue).filter(testObject));
45      assertFalse(alwaysTrue.and(alwaysFalse).filter(testObject));
46      assertFalse(alwaysFalse.and(alwaysFalse).filter(testObject));
47  
48      assertTrue(new And<>(alwaysTrue, alwaysTrue).filter(testObject));
49      assertFalse(new And<>(alwaysFalse, alwaysTrue).filter(testObject));
50      assertFalse(new And<>(alwaysTrue, alwaysFalse).filter(testObject));
51      assertFalse(new And<>(alwaysFalse, alwaysFalse).filter(testObject));
52    }
53  
54    @Test
55    public void testOr() throws Exception {
56      assertTrue(alwaysTrue.or(alwaysTrue).filter(testObject));
57      assertTrue(alwaysFalse.or(alwaysTrue).filter(testObject));
58      assertTrue(alwaysTrue.or(alwaysFalse).filter(testObject));
59      assertFalse(alwaysFalse.or(alwaysFalse).filter(testObject));
60  
61      assertTrue(new Or<>(alwaysTrue, alwaysTrue).filter(testObject));
62      assertTrue(new Or<>(alwaysFalse, alwaysTrue).filter(testObject));
63      assertTrue(new Or<>(alwaysTrue, alwaysFalse).filter(testObject));
64      assertFalse(new Or<>(alwaysFalse, alwaysFalse).filter(testObject));
65    }
66  
67    @Test
68    public void testNegate() throws Exception {
69      assertTrue(alwaysFalse.negate().filter(testObject));
70      assertFalse(alwaysTrue.negate().filter(testObject));
71  
72      assertTrue(new Not<>(alwaysFalse).filter(testObject));
73      assertFalse(new Not<>(alwaysTrue).filter(testObject));
74    }
75  
76    @Test
77    public void testWithGraph() throws Exception {
78      FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
79      loader.appendToDatabaseFromString("expected[" +
80        // gender = 'f' and city = 'leipzig'
81        "(alice)" +
82        // label = 'Tag'
83        "(databases)(graphs)(hadoop)" +
84        "]");
85      LogicalGraph databaseGraph = loader.getLogicalGraph();
86      // Filter vertices where
87      // (gender = 'f' AND city = 'Leipzig') OR NOT(label = 'Person' OR label = 'Forum')
88      FilterFunction<EPGMVertex> vertexFilterFunction =
89        new ByProperty<EPGMVertex>("gender", PropertyValue.create("f"))
90        .and(new ByProperty<>("city", PropertyValue.create("Leipzig"))).or(
91          new ByLabel<>("Person").or(new ByLabel<>("Forum")).negate());
92      // Filter edges where
93      // label <> 'hasInterest'
94      LogicalGraph subgraph = databaseGraph.subgraph(vertexFilterFunction,
95        new ByLabel<EPGMEdge>("hasInterest").negate(), Subgraph.Strategy.BOTH);
96      collectAndAssertTrue(subgraph.equalsByElementIds(loader.getLogicalGraphByVariable("expected")));
97    }
