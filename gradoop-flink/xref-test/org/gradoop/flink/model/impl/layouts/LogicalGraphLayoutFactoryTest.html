184   /**
185    * Check if the {@link LogicalGraphLayoutFactory#fromCollections(Collection, Collection)}
186    * method returns the same graph as
187    * {@link LogicalGraphLayoutFactory#fromDataSets(DataSet, DataSet)} using datasets
188    * created from the same collections.
189    *
190    * @throws Exception on failure.
191    */
192   @Test
193   public void testFromCollectionsWithoutGraphHead() throws Exception {
194     FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
195 
196     Collection<EPGMVertex> vertices = loader.getVertices();
197     Collection<EPGMEdge> edges = loader.getEdges();
198     LogicalGraphLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromCollections = getFactory()
199       .fromCollections(vertices, edges);
200 
201     DataSet<EPGMVertex> vertexDataSet = getExecutionEnvironment().fromCollection(vertices);
202     DataSet<EPGMEdge> edgeDataSet = getExecutionEnvironment().fromCollection(edges);
203     LogicalGraphLayout<EPGMGraphHead, EPGMVertex, EPGMEdge> fromDataSets = getFactory()
204       .fromDataSets(vertexDataSet, edgeDataSet);
205 
206     LogicalGraph graphFromCollections = getConfig().getLogicalGraphFactory()
207       .fromDataSets(fromCollections.getGraphHead(), fromCollections.getVertices(),
208         fromCollections.getEdges());
209     LogicalGraph graphFromDataSets = getConfig().getLogicalGraphFactory()
210       .fromDataSets(fromDataSets.getGraphHead(), fromDataSets.getVertices(),
211         fromDataSets.getEdges());
212     collectAndAssertTrue(graphFromCollections.equalsByData(graphFromDataSets));
213   }
