60        "(pL)-[:knows {since : 2014, count : 2L}]->(pL)" +
61        "(pL)-[:knows {since : 2013, count : 1L}]->(pD)" +
62        "(pB)-[:knows {since : 2015, count : 2L}]->(pD)" +

66        Arrays.asList(Grouping.LABEL_SYMBOL, "city"),
