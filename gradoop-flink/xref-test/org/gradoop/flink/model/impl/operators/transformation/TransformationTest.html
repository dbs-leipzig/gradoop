36    final String testGraphString = "" +
37      "g0:A  { a : 1 } [(:A { a : 1, b : 2 })-[:a { a : 1, b : 2 }]->(:B { c : 2 })]" +
38      "g1:B  { a : 2 } [(:A { a : 2, b : 2 })-[:a { a : 2, b : 2 }]->(:B { c : 3 })]" +
39      // full graph transformation
40      "g01:A { a : 2 } [(:A { a : 2, b : 1 })-->(:B { d : 2 })]" +
41      "g11:B { a : 3 } [(:A { a : 3, b : 1 })-->(:B { d : 3 })]" +
42      // graph head only transformation
43      "g02:A { a : 2 } [(:A { a : 1, b : 2 })-[:a { a : 1, b : 2 }]->(:B { c : 2 })]" +
44      "g12:B { a : 3 } [(:A { a : 2, b : 2 })-[:a { a : 2, b : 2 }]->(:B { c : 3 })]" +
45      // vertex only transformation
46      "g03:A { a : 1 } [(:A { a : 2, b : 1 })-[:a { a : 1, b : 2 }]->(:B { d : 2 })]" +
47      "g13:B { a : 2 } [(:A { a : 3, b : 1 })-[:a { a : 2, b : 2 }]->(:B { d : 3 })]" +
48      // edge only transformation
49      "g04:A { a : 1 } [(:A { a : 1, b : 2 })-->(:B { c : 2 })]" +
50      "g14:B { a : 2 } [(:A { a : 2, b : 2 })-->(:B { c : 3 })]";
51  
52    static EPGMGraphHead transformGraphHead(EPGMGraphHead current, EPGMGraphHead transformed) {
53      transformed.setLabel(current.getLabel());
54      transformed.setProperty("a", current.getPropertyValue("a").getInt() + 1);
55      return transformed;
56    }
57  
58    static EPGMVertex transformVertex(EPGMVertex current, EPGMVertex transformed) {
59      transformed.setLabel(current.getLabel());
60      if (current.getLabel().equals("A")) {
61        transformed.setProperty("a", current.getPropertyValue("a").getInt() + 1);
62        transformed.setProperty("b", current.getPropertyValue("b").getInt() - 1);
63      } else if (current.getLabel().equals("B")) {
64        transformed.setProperty("d", current.getPropertyValue("c"));
65      }
66      return transformed;
67    }
68  
69    static EPGMEdge transformEdge(EPGMEdge current, EPGMEdge transformed) {
70      return transformed;
71    }

78    @Test
79    public void testIdEquality() throws Exception {
80      FlinkAsciiGraphLoader loader = getLoaderFromString(testGraphString);
81  
82      List<GradoopId> expectedGraphHeadIds = Lists.newArrayList();
83      List<GradoopId> expectedVertexIds = Lists.newArrayList();
84      List<GradoopId> expectedEdgeIds = Lists.newArrayList();
85  
86      LogicalGraph inputGraph = loader.getLogicalGraphByVariable("g0");
87  
88      inputGraph.getGraphHead().map(new Id<>()).output(
89        new LocalCollectionOutputFormat<>(expectedGraphHeadIds));
90      inputGraph.getVertices().map(new Id<>()).output(
91        new LocalCollectionOutputFormat<>(expectedVertexIds));
92      inputGraph.getEdges().map(new Id<>()).output(
93        new LocalCollectionOutputFormat<>(expectedEdgeIds));
94  
95      LogicalGraph result = inputGraph
96        .transform(
97          TransformationTest::transformGraphHead,
98          TransformationTest::transformVertex,
99          TransformationTest::transformEdge
100       );
101 
102     List<GradoopId> resultGraphHeadIds = Lists.newArrayList();
103     List<GradoopId> resultVertexIds = Lists.newArrayList();
104     List<GradoopId> resultEdgeIds = Lists.newArrayList();
105 
106     result.getGraphHead()
107       .map(new Id<>())
108       .output(new LocalCollectionOutputFormat<>(resultGraphHeadIds));
109     result.getVertices()
110       .map(new Id<>())
111       .output(new LocalCollectionOutputFormat<>(resultVertexIds));
112     result.getEdges()
113       .map(new Id<>())
114       .output(new LocalCollectionOutputFormat<>(resultEdgeIds));
115 
116     getExecutionEnvironment().execute();
117 
118     validateIdEquality(expectedGraphHeadIds, resultGraphHeadIds);
119     validateIdEquality(expectedVertexIds, resultVertexIds);
120     validateIdEquality(expectedEdgeIds, resultEdgeIds);
121   }
