56     * Test MinProperty, MinVertexProperty and MinEdgeProperty with a graph collection.
57     *
58     * @throws Exception if the execution fails.
59     */
60    @Test
61    public void testCollectionMin() throws Exception {
62      FlinkAsciiGraphLoader loader = getLoaderFromString(
63        "g0[" +
64          "(va {p : 0.5})" +
65          "(vb {p : 0.3})" +
66          "(vc {p : 0.1})" +
67          "(va)-[ea {p : 2L}]->(vb)" +
68          "(vb)-[eb]->(vc)" +
69          "]" +
70          "g1[" +
71          "(va)-[ea]->(vb)" +
72          "]" +
73          "g2[]");
74  
75      GraphCollection inputCollection = loader
76        .getGraphCollectionByVariables("g0", "g1", "g2");
77  
78      MinVertexProperty minVertexProperty =
79        new MinVertexProperty(PROPERTY, VERTEX_AGGREGATE_PROPERTY);
80      MinEdgeProperty minEdgeProperty = new MinEdgeProperty(PROPERTY, EDGE_AGGREGATE_PROPERTY);
81      MinProperty minProperty = new MinProperty(PROPERTY, ELEMENT_AGGREGATE_PROPERTY);
82  
83      GraphCollection outputCollection = inputCollection
84        .apply(new ApplyAggregation<>(minVertexProperty))
85        .apply(new ApplyAggregation<>(minEdgeProperty))
86        .apply(new ApplyAggregation<>(minProperty));
87  
88      GradoopId g0Id = loader.getGraphHeadByVariable("g0").getId();
89      GradoopId g1Id = loader.getGraphHeadByVariable("g1").getId();
90      GradoopId g2Id = loader.getGraphHeadByVariable("g2").getId();
91  
92      for (GraphHead graphHead : outputCollection.getGraphHeads().collect()) {
93        assertTrue("edge minimum not set", graphHead.hasProperty
94          (minEdgeProperty.getAggregatePropertyKey()));
95        assertTrue("vertex minimum not set", graphHead.hasProperty
96          (minVertexProperty.getAggregatePropertyKey()));
97        assertTrue("element minimum not set", graphHead.hasProperty
98          (minProperty.getAggregatePropertyKey()));
99  
100       PropertyValue edgeAggregate =
101         graphHead.getPropertyValue(minEdgeProperty.getAggregatePropertyKey());
102       PropertyValue vertexAggregate =
103         graphHead.getPropertyValue(minVertexProperty.getAggregatePropertyKey());
104       PropertyValue elementAggregate =
105         graphHead.getPropertyValue(minProperty.getAggregatePropertyKey());
106 
107       if (graphHead.getId().equals(g0Id)) {
108         assertEquals(2, edgeAggregate.getLong());
109         assertEquals(0.1f, vertexAggregate.getFloat(), 0.00001);
110         assertEquals(0.1f, elementAggregate.getFloat(), 0.00001);
111       } else if (graphHead.getId().equals(g1Id)) {
112         assertEquals(2, edgeAggregate.getLong());
113         assertEquals(0.3f, vertexAggregate.getFloat(), 0.00001);
114         assertEquals(0.3f, elementAggregate.getFloat(), 0.00001);
115       } else if (graphHead.getId().equals(g2Id)) {
116         assertEquals(PropertyValue.NULL_VALUE, edgeAggregate);
117         assertEquals(PropertyValue.NULL_VALUE, vertexAggregate);
118         assertEquals(PropertyValue.NULL_VALUE, elementAggregate);
119       } else {
120         Assert.fail("unexpected graph head: " + graphHead);
121       }
122     }
123   }

360    * Test Count, VertexCount and EdgeCount with a graph collection.
361    *
362    * @throws Exception if the execution fails.
363    */
364   @Test
365   public void testCollectionCount() throws Exception {
366     FlinkAsciiGraphLoader loader = getLoaderFromString(
367       "g0[()-->()<--()]" +
368         "g1[()-->()-->()-->()]" +
369         "g2[()-->()]" +
370         "g3[]");
371 
372     GraphCollection inputCollection = loader
373       .getGraphCollectionByVariables("g0", "g1", "g2", "g3");
374 
375     VertexCount vertexCount = new VertexCount();
376     EdgeCount edgeCount = new EdgeCount();
377     Count count = new Count();
378 
379     GraphCollection outputCollection = inputCollection
380       .apply(new ApplyAggregation<>(vertexCount))
381       .apply(new ApplyAggregation<>(edgeCount))
382       .apply(new ApplyAggregation<>(count));
383 
384     GradoopId g0Id = loader.getGraphHeadByVariable("g0").getId();
385     GradoopId g1Id = loader.getGraphHeadByVariable("g1").getId();
386     GradoopId g2Id = loader.getGraphHeadByVariable("g2").getId();
387     GradoopId g3Id = loader.getGraphHeadByVariable("g3").getId();
388 
389     int graphHeadCount = 0;
390 
391     List<EPGMGraphHead> graphHeads = outputCollection.getGraphHeads().collect();
392 
393     for (GraphHead graphHead : graphHeads) {
394       graphHeadCount++;
395       assertTrue("vertex count not set", graphHead.hasProperty(
396         vertexCount.getAggregatePropertyKey()));
397       assertTrue("edge count not set", graphHead.hasProperty(
398         edgeCount.getAggregatePropertyKey()));
399       assertTrue("element count not set", graphHead.hasProperty(
400         count.getAggregatePropertyKey()));
401 
402       if (graphHead.getId().equals(g0Id)) {
403         assertCounts(graphHead, 3, 2);
404       } else if (graphHead.getId().equals(g1Id)) {
405         assertCounts(graphHead, 4, 3);
406       } else if (graphHead.getId().equals(g2Id)) {
407         assertCounts(graphHead, 2, 1);
408       } else if (graphHead
409         .getId().equals(g3Id)) {
410         assertCounts(graphHead, 0, 0);
411       } else {
412         Assert.fail("unexpected graph head: " + graphHead);
413       }
414     }
415 
416     assertTrue("wrong number of output graph heads", graphHeadCount == 4);
417   }
