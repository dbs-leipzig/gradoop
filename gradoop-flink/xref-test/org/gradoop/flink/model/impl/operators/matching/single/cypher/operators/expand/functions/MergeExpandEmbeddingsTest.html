144   private void testJoin(Embedding edge, List<Integer> distinctVertices, List<Integer>
145     distinctEdges,
146     int closingColumn, boolean isResult) throws Exception {
147 
148     Embedding base = new Embedding();
149     base.add(m);
150     base.add(e0);
151     base.add(n);
152 
153     EdgeWithTiePoint edgeTuple = new EdgeWithTiePoint(edge);
154 
155     ExpandEmbedding expandEmbedding = new ExpandEmbedding(
156       base, e1, a);
157 
158     MergeExpandEmbeddings op =
159       new MergeExpandEmbeddings(distinctVertices, distinctEdges, closingColumn);
160 
161     List<ExpandEmbedding> results = new ArrayList<>();
162     op.join(expandEmbedding, edgeTuple, new ListCollector<>(results));
163 
164     assertEquals(isResult ? 1 : 0, results.size());
165 
166     if (isResult) {
167       assertArrayEquals(new GradoopId[]{e1, a, edge.getId(1)}, results.get(0).getPath());
168       assertEquals(edge.getId(2), results.get(0).getEnd());
169     }
170   }
