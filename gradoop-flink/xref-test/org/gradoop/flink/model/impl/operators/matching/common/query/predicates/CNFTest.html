123   @Test
124   public void testEvaluationForProperties() {
125     String queryString = "MATCH (a), (b), (c)" +
126       "WHERE a=b AND a.name = \"Alice\" AND a.age > c.age";
127     QueryHandler query = new QueryHandler(queryString);
128     CNF cnf = query.getPredicates();
129 
130     GradoopId a = GradoopId.get();
131     GradoopId c = GradoopId.get();
132 
133     PropertyValue[] propertiesA = new PropertyValue[]{
134       PropertyValue.create("Alice"),
135       PropertyValue.create(42)
136     };
137 
138     Embedding embedding = new Embedding();
139     embedding.add(
140       a,
141       propertiesA
142     );
143 
144     PropertyValue[] propertiesB = new PropertyValue[]{PropertyValue.create(23)};
145     embedding.add(a);
146     embedding.add(
147       c,
148       propertiesB
149     );
150 
151     EmbeddingMetaData metaData = new EmbeddingMetaData();
152     metaData.setEntryColumn("a", EntryType.VERTEX, 0);
153     metaData.setEntryColumn("b", EntryType.VERTEX, 1);
154     metaData.setEntryColumn("c", EntryType.VERTEX, 2);
155     metaData.setPropertyColumn("a", "name", 0);
156     metaData.setPropertyColumn("a", "age", 1);
157     metaData.setPropertyColumn("c", "age", 2);
158 
159     assertTrue(cnf.evaluate(embedding, metaData));
160   }
