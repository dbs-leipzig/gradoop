124   @Test
125   public void testGraphContainment() throws Exception {
126     FlinkAsciiGraphLoader loader =
127       getSocialNetworkLoader();
128 
129     LogicalGraph g0 = loader.getLogicalGraphByVariable("g0");
130     LogicalGraph g2 = loader.getLogicalGraphByVariable("g2");
131 
132     // use collections as data sink
133     Collection<EPGMVertex> vertices0 = new HashSet<>();
134     Collection<EPGMEdge> edges0 = new HashSet<>();
135     Collection<EPGMVertex> vertices2 = new HashSet<>();
136     Collection<EPGMEdge> edges2 = new HashSet<>();
137     Collection<EPGMVertex> resVertices = new HashSet<>();
138     Collection<EPGMEdge> resEdges = new HashSet<>();
139 
140     LogicalGraph expected = g0.combine(g2);
141 
142     g0.getVertices().output(new LocalCollectionOutputFormat<>(vertices0));
143     g0.getEdges().output(new LocalCollectionOutputFormat<>(edges0));
144     g2.getVertices().output(new LocalCollectionOutputFormat<>(vertices2));
145     g2.getEdges().output(new LocalCollectionOutputFormat<>(edges2));
146     expected.getVertices().output(new LocalCollectionOutputFormat<>(resVertices));
147     expected.getEdges().output(new LocalCollectionOutputFormat<>(resEdges));
148 
149     getExecutionEnvironment().execute();
150 
151     Set<EPGMGraphElement> inVertices = new HashSet<>();
152     inVertices.addAll(vertices0);
153     inVertices.addAll(vertices2);
154     Set<EPGMGraphElement> inEdges = new HashSet<>();
155     inEdges.addAll(edges0);
156     inEdges.addAll(edges2);
157 
158     Set<EPGMGraphElement> outVertices = new HashSet<>();
159     inVertices.addAll(outVertices);
160     Set<EPGMGraphElement> outEdges = new HashSet<>();
161     inEdges.addAll(resEdges);
162 
163     checkElementMatches(inVertices, outVertices);
164     checkElementMatches(inEdges, outEdges);
165   }
