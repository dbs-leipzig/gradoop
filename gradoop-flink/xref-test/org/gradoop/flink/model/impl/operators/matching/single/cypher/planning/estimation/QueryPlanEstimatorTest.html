109       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);

112     estimator = new QueryPlanEstimator(plan, queryHandler, STATS);

190   @Test
191   public void testComplexCartesianProduct() throws Exception {
192     String query = "MATCH (a:Person)-[e1:knows]->(b:Person),(c:Forum)-[e2:hasTag]->(d:Tag)";
193     QueryHandler queryHandler = new QueryHandler(query);
194 
195     LeafNode aNode = new FilterAndProjectVerticesNode(null, "a",
196       queryHandler.getPredicates().getSubCNF("a"), Sets.newHashSet());
197     LeafNode bNode = new FilterAndProjectVerticesNode(null, "b",
198       queryHandler.getPredicates().getSubCNF("b"), Sets.newHashSet());
199     LeafNode cNode = new FilterAndProjectVerticesNode(null, "c",
200       queryHandler.getPredicates().getSubCNF("c"), Sets.newHashSet());
201     LeafNode dNode = new FilterAndProjectVerticesNode(null, "d",
202       queryHandler.getPredicates().getSubCNF("d"), Sets.newHashSet());
203 
204     LeafNode e1Node = new FilterAndProjectEdgesNode(null, "a", "e1", "b",
205       queryHandler.getPredicates().getSubCNF("e1"), Sets.newHashSet(), false);
206     LeafNode e2Node = new FilterAndProjectEdgesNode(null, "c", "e2", "d",
207       queryHandler.getPredicates().getSubCNF("e2"), Sets.newHashSet(), false);
208 
209     BinaryNode ae1 = new JoinEmbeddingsNode(aNode, e1Node, Lists.newArrayList("a"),
210       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
211     BinaryNode ae1b = new JoinEmbeddingsNode(ae1, bNode, Lists.newArrayList("b"),
212       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
213     BinaryNode ce2 = new JoinEmbeddingsNode(cNode, e2Node, Lists.newArrayList("c"),
214       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
215     BinaryNode ce2d = new JoinEmbeddingsNode(ce2, dNode, Lists.newArrayList("d"),
216       MatchStrategy.HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
217 
218     CartesianProductNode crossNode = new CartesianProductNode(ae1b, ce2d, MatchStrategy
219       .HOMOMORPHISM, MatchStrategy.HOMOMORPHISM);
220 
221     QueryPlan ae1bPlan = new QueryPlan(ae1b);
222     QueryPlan ce2dPlan = new QueryPlan(ce2d);
223     QueryPlan crossPlan = new QueryPlan(crossNode);
224 
225     QueryPlanEstimator ae1bEstimator = new QueryPlanEstimator(ae1bPlan, queryHandler, STATS);
226     QueryPlanEstimator ce2dEstimator = new QueryPlanEstimator(ce2dPlan, queryHandler, STATS);
227     QueryPlanEstimator crossEstimator = new QueryPlanEstimator(crossPlan, queryHandler, STATS);
228 
229     assertThat(crossEstimator.getCardinality(), is(ae1bEstimator.getCardinality() *
230       ce2dEstimator.getCardinality()));
231   }
