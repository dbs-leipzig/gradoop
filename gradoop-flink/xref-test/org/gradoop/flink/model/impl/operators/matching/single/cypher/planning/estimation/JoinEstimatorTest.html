35    @Test
36    public void testLabelFree() throws Exception {
37      String query = "MATCH (n)-[e]->(m)";
38  
39      QueryHandler queryHandler = new QueryHandler(query);
40  
41      LeafNode nNode = new FilterAndProjectVerticesNode(null, "n",
42        queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
43      LeafNode mNode = new FilterAndProjectVerticesNode(null, "m",
44        queryHandler.getPredicates().getSubCNF("m"), Sets.newHashSet());
45      LeafNode eNode = new FilterAndProjectEdgesNode(null,
46        "n", "e", "m",
47        queryHandler.getPredicates().getSubCNF("e"), Sets.newHashSet(), false);
48  
49      JoinEmbeddingsNode neJoin = new JoinEmbeddingsNode(nNode, eNode, Lists.newArrayList("n"),
50        MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
51      JoinEmbeddingsNode nemJoin = new JoinEmbeddingsNode(neJoin, mNode, Lists.newArrayList("m"),
52        MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
53  
54      JoinEstimator estimator = new JoinEstimator(queryHandler, STATS);
55      estimator.visit(neJoin);
56      estimator.visit(nemJoin);
57  
58      assertEquals(24L, estimator.getCardinality());
59    }

223       queryHandler.getPredicates().getSubCNF("n"), Sets.newHashSet());
224     LeafNode mNode = new FilterAndProjectVerticesNode(null, "m",
225       queryHandler.getPredicates().getSubCNF("m"), Sets.newHashSet());
226     LeafNode oNode = new FilterAndProjectVerticesNode(null, "o",
227       queryHandler.getPredicates().getSubCNF("o"), Sets.newHashSet());
228     LeafNode e1Node = new FilterAndProjectEdgesNode(null,
229       "n", "e1", "m",
230       queryHandler.getPredicates().getSubCNF("e1"), Sets.newHashSet(), false);
231     LeafNode e2Node = new FilterAndProjectEdgesNode(null,
232       "m", "e2", "o",
233       queryHandler.getPredicates().getSubCNF("e2"), Sets.newHashSet(), false);
234 
235     ExpandEmbeddingsNode ne1Join = new ExpandEmbeddingsNode(nNode, e1Node,
236       "n", "e", "m", 2, 2,
237       ExpandDirection.OUT, MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
238     JoinEmbeddingsNode ne1mJoin = new JoinEmbeddingsNode(ne1Join, mNode, Lists.newArrayList("m"),
239       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
240     JoinEmbeddingsNode ne1me2Join = new JoinEmbeddingsNode(ne1mJoin, e2Node, Lists.newArrayList("m"),
241       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
242     JoinEmbeddingsNode ne1me2oJoin = new JoinEmbeddingsNode(ne1me2Join, oNode, Lists.newArrayList("o"),
243       MatchStrategy.ISOMORPHISM, MatchStrategy.ISOMORPHISM);
244 
245     JoinEstimator estimator = new JoinEstimator(queryHandler, STATS);
246     estimator.visit(ne1me2oJoin);
247     estimator.visit(ne1me2Join);
248     estimator.visit(ne1mJoin);
249     estimator.visit(ne1Join);
250 
251     assertEquals(30L, estimator.getCardinality());
