36    public void emptyAndEmptyToEmpty() throws Exception {
37      FlinkAsciiGraphLoader loader = getLoaderFromString("empty:G[]");
38      LogicalGraph searchGraph = loader.getLogicalGraphByVariable("empty");
39      LogicalGraph patternGraph = loader.getLogicalGraphByVariable("empty");
40      VertexFusion f = new VertexFusion();
41      LogicalGraph output = f.execute(searchGraph, patternGraph);
42      LogicalGraph expected = loader.getLogicalGraphByVariable("empty");
43      collectAndAssertTrue(output.equalsByData(expected));
44      if (deepSearch) {
45        collectAndAssertTrue(VertexFusionUtils.myInducedEdgeSubgraphForFusion(searchGraph, expected)
46          .equalsByElementIds(VertexFusionUtils.myInducedEdgeSubgraphForFusion(expected, searchGraph)));
47      }
48    }
