26  public class BinaryGraphOperatorsTestBase extends GradoopFlinkTestBase {
27  
28    protected void checkElementMatches(Set<EPGMGraphElement> inElements,
29      Set<EPGMGraphElement> outElements) {
30      for (EPGMGraphElement outElement : outElements) {
31        boolean match = false;
32  
33        String elementClassName = outElement.getClass().getSimpleName();
34  
35        for (EPGMGraphElement inVertex : inElements) {
36          if (outElement.getId().equals(inVertex.getId())) {
37            assertEquals(
38              "wrong number of graphs for " + elementClassName,
39              inVertex.getGraphCount() + 1,
40              outElement.getGraphCount()
41            );
42            match = true;
43            break;
44          }
45        }
46        assertTrue("expected " + elementClassName + " not found", match);
47      }
48    }
