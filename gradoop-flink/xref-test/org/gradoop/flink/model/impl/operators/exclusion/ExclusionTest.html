107   @Test
108   public void testGraphContainment() throws Exception {
109     FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
110 
111     LogicalGraph g0 = loader.getLogicalGraphByVariable("g0");
112     LogicalGraph g2 = loader.getLogicalGraphByVariable("g2");
113 
114     // use collections as data sink
115     Collection<EPGMVertex> vertices0 = new HashSet<>();
116     Collection<EPGMEdge> edges0 = new HashSet<>();
117     Collection<EPGMVertex> vertices2 = new HashSet<>();
118     Collection<EPGMEdge> edges2 = new HashSet<>();
119     Collection<EPGMVertex> resVertices = new HashSet<>();
120     Collection<EPGMEdge> resEdges = new HashSet<>();
121 
122     LogicalGraph res = g2.exclude(g0);
123 
124     g0.getVertices().output(new LocalCollectionOutputFormat<>(vertices0));
125     g0.getEdges().output(new LocalCollectionOutputFormat<>(edges0));
126     g2.getVertices().output(new LocalCollectionOutputFormat<>(vertices2));
127     g2.getEdges().output(new LocalCollectionOutputFormat<>(edges2));
128     res.getVertices().output(new LocalCollectionOutputFormat<>(resVertices));
129     res.getEdges().output(new LocalCollectionOutputFormat<>(resEdges));
130 
131     getExecutionEnvironment().execute();
132 
133     Set<EPGMGraphElement> inVertices = new HashSet<>();
134     for (EPGMVertex vertex : vertices2) {
135       if (!vertices0.contains(vertex)) {
136         inVertices.add(vertex);
137       }
138     }
139     for (EPGMEdge edge : edges2) {
140       if (!edges0.contains(edge)) {
141         inVertices.add(edge);
142       }
143     }
144 
145     Set<EPGMGraphElement> outVertices = new HashSet<>();
146     inVertices.addAll(outVertices);
147     Set<EPGMGraphElement> outEdges = new HashSet<>();
148     Set<EPGMGraphElement> inEdges = new HashSet<>(resEdges);
149 
150     checkElementMatches(inVertices, outVertices);
151     checkElementMatches(inEdges, outEdges);
152   }
