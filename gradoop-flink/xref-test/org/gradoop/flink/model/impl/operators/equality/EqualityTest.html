324   private FlinkAsciiGraphLoader
325   getTestGraphLoader() {
326     String asciiGraphs = "gEmpty[]" +
327 
328       "gRef:G{dataDiff : false}[" +
329       // loop around a1 and edge from a1 to a2
330       "(a1:A{x : 1})-[loop:a{x : 1}]->(a1)-[aa:a{x : 1}]->(a2:A{x : 2})" +
331       // parallel edge from a1 to b1
332       "(a1)-[par1:p]->(b1:B),(a1)-[par2:p]->(b1)" +
333       // cycle of bs
334       "(b1)-[cyc1:c]->(b2:B)-[cyc2:c]->(b3:B)-[cyc3:c]->(b1)]" +
335 
336       // element id copy of gRef
337       "gClone:G{dataDiff : true}[" +
338       "(a1)-[loop]->(a1)-[aa]->(a2)" +
339       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
340       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]" +
341 
342       // element id copy of gRef with one different edge id
343       "gDiffId:G{dataDiff : false}[" +
344       "(a1)-[loop]->(a1)-[aa]->(a2)" +
345       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
346       "(b1)-[cyc1]->(b2)-[:c]->(b3)-[cyc3]->(b1)]" +
347 
348       // element id copy of gRef
349       // with each one different vertex and edge attribute
350       "gDiffData:G[" +
351       "(a1)-[loop]->(a1)-[:a{y : 1}]->(:A{x : \"diff\"})" +
352       "(a1)-[par1]->(b1),(a1)-[par2]->(b1)" +
353       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]" +
354 
355       // copy of gRef with partially reverse edges
356       "gRev:G{dataDiff : false}[" +
357       "(a1)-[loop]->(a1)<-[:a{x : 1}]-(a2)" +
358       "(a1)<-[:p]-(b1),(a1)-[:p]->(b1)" +
359       "(b1)-[cyc1]->(b2)-[cyc2]->(b3)-[cyc3]->(b1)]";
360 
361     return getLoaderFromString(asciiGraphs);
362   }
