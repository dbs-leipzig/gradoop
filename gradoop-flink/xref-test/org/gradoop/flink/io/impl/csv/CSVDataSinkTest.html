126   /**
127    * Test CSVDataSink to escape strings and labels that contain delimiter characters.
128    * Escape characters get inserted before delimiter characters that are part of strings or labels.
129    * Whitespace gets replaced by their control character sequence.
130    *
131    * @throws Exception if the execution or IO fails.
132    */
133   @Test
134   public void testWriteWithDelimiterCharacters() throws Exception {
135     String tmpPath = temporaryFolder.getRoot().getPath();
136     ExecutionEnvironment env = getExecutionEnvironment();
137 
138     String string1 = "abc;,|:\n=\\ def";
139     String string2 = "def;,|:\n=\\ ghi";
140 
141     List<PropertyValue> list = Arrays.asList(PropertyValue.create(string2), PropertyValue.create(string1));
142     Set<PropertyValue> set = new HashSet<>(list);
143     Map<PropertyValue, PropertyValue> map1 = new HashMap<>();
144     map1.put(PropertyValue.create(string1), PropertyValue.create(string2));
145     map1.put(PropertyValue.create("key"), PropertyValue.create(string1));
146 
147     Map<PropertyValue, PropertyValue> map2 = new HashMap<>();
148     map2.put(PropertyValue.create(string1), PropertyValue.create(1));
149     map2.put(PropertyValue.create("key"), PropertyValue.create(2));
150 
151     Map<PropertyValue, PropertyValue> map3 = new HashMap<>();
152     map3.put(PropertyValue.create(1), PropertyValue.create(string2));
153     map3.put(PropertyValue.create(2), PropertyValue.create(string1));
154 
155     Properties props = Properties.create();
156     props.set(string1, string2);
157     props.set(string2, GradoopTestUtils.BOOL_VAL_1);
158     props.set(GradoopTestUtils.KEY_2, string2);
159     props.set(GradoopTestUtils.KEY_3, list);
160     props.set(GradoopTestUtils.KEY_4, set);
161     props.set(GradoopTestUtils.KEY_5, map1);
162     props.set(GradoopTestUtils.KEY_5, map2);
163     props.set(GradoopTestUtils.KEY_6, map3);
164 
165     EPGMGraphHead graphHead = new EPGMGraphHeadFactory().createGraphHead(string1, props);
166     DataSet<EPGMGraphHead> graphHeads = env.fromElements(graphHead);
167 
168     EPGMVertex vertex = new EPGMVertexFactory().createVertex(string1, props);
169     DataSet<EPGMVertex> vertices = env.fromElements(vertex)
170       .map(new AddToGraph<>(graphHead))
171       .withForwardedFields("id;label;properties");
172 
173     DataSet<EPGMEdge> edges = env.fromElements(new EPGMEdgeFactory()
174       .createEdge(string1, vertex.getId(), vertex.getId(), props))
175       .map(new AddToGraph<>(graphHead))
176       .withForwardedFields("id;label;properties");
177 
178     LogicalGraph graph = getConfig().getLogicalGraphFactory()
179       .fromDataSets(graphHeads, vertices, edges);
180 
181     checkCSVWrite(tmpPath, graph);
182   }
